// ---- IMPORT side: fetch + decompress a cloud-stored session ----------------
async loadSessionFromUrl(url) =>
  if(!url) throw new Error("No URL provided.");
  // normalise a ?data=Name~hash share link to its uploads.dev file
  try {
    let u = new URL(url);
    let dataParam = u.searchParams.get("data");
    if(dataParam) {
      let fileName = dataParam.split("~").slice(-1)[0];
      if(fileName) url = "https://user.uploads.dev/file/" + fileName;
    }
  } catch(e) { }
  let resp = null;
  if(typeof superFetch === "function") {
    try { resp = await superFetch(url); } catch(e) { resp = null; }
  }
  if(!resp || !resp.ok) {
    resp = await fetch(url);
  }
  if(!resp || !resp.ok) throw new Error("Could not fetch that link (status " + (resp ? resp.status : "no response") + ").");
  let blob = await resp.blob();
  let bytes = new Uint8Array(await blob.arrayBuffer());
  let isGzip = bytes.length > 2 && bytes[0] === 31 && bytes[1] === 139;
  let text;
  if(isGzip) {
    let decompressedBlob = await decompressBlobWithGzip(blob);
    text = await decompressedBlob.text();
  } else {
    text = await blob.text();
  }
  return JSON.parse(text);


// ---- EXPORT side: gzip + upload a session, return a shareable respool link --
async generateShareLinkForSession(sessionJson) =>
  if(!window.CompressionStream) {
    return { error: "Your browser doesn't support the compression feature share links need. Try a recent Chrome, Edge or Firefox." };
  }
  try {
    let jsonString = JSON.stringify(sessionJson);
    let blob = new Blob([jsonString], { type: "application/json" });
    let compressedBlob = await compressBlobWithGzip(blob);
    let uploadResult = await uploadPlugin(compressedBlob);
    let uploadError = uploadResult && uploadResult.error;
    if(uploadError) return { error: String(uploadError) };
    let url = String((uploadResult && uploadResult.url) ? uploadResult.url : uploadResult);
    if(!url || url === "undefined") return { error: "Upload failed — no URL was returned." };
    let fileName = url.replace("https://user.uploads.dev/file/", "");
    let label = "session";
    try {
      let tables = sessionJson.data.data;
      let threadsT = tables.find(t => t.tableName === "threads");
      let nm = threadsT && threadsT.rows[0] && (threadsT.rows[0].name || (threadsT.rows[0].character && threadsT.rows[0].character.name));
      if(nm) label = nm.replace(/\s+/g, "_").replaceAll("~", "");
    } catch(e) { }
    let shareUrl = `https://perchance.org/${window.generatorName}?data=${label}~${fileName}`;
    return { shareUrl: shareUrl, url: url };
  } catch(e) {
    console.error("respool export:", e);
    return { error: String(e && e.message ? e.message : e) };
  }


// ---- gzip helpers (mirror the ai-character-chat originals) -----------------
async compressBlobWithGzip(blob) =>
  const cs = new CompressionStream("gzip");
  const compressedStream = blob.stream().pipeThrough(cs);
  let outputBlob = await new Response(compressedStream).blob();
  return new Blob([outputBlob], { type: "application/gzip" });

async decompressBlobWithGzip(blob) =>
  const ds = new DecompressionStream("gzip");
  const decompressedStream = blob.stream().pipeThrough(ds);
  return await new Response(decompressedStream).blob();
