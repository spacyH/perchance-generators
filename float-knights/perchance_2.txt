function R(min, max) { return Math.random() * (max - min) + min; }
function RI(min, max) { return Math.floor(R(min, max + 1)); }
function pick(arr) { return arr[Math.floor(Math.random() * arr.length)]; }
function pickN(arr, n) {
  const c = arr.slice();
  const out = [];
  for (let i = 0; i < n && c.length; i++) {
    out.push(c.splice(Math.floor(Math.random() * c.length), 1)[0]);
  }
  return out;
}
function shuffle(arr) {
  const a = arr.slice();
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}
function clamp(v, lo, hi) { return Math.max(lo, Math.min(hi, v)); }
function distance(a, b) {
  const dx = a.x - b.x, dy = a.y - b.y;
  return Math.sqrt(dx*dx + dy*dy);
}

// "Score" a piece of gear so the AI can compare its current piece to
// a scavengable upgrade. Higher = better for that slot.
// Crucially, this is now POSITIONAL — a Sniper values a Railgun more than
// a stronger-on-paper weapon they don't know how to use, so we don't get
// every survivor swapping to the absolute strongest piece.
function gearScore(slot, gear, ownerPosition) {
  if (!gear) return 0;
  if (slot === "weapon") {
    let s = gear.dmgMod * 2.5 + gear.defMod * 1.2 + (-gear.cdMod) * 6 + gear.critMod * 80 + gear.stun * 25;
    // +5 if this weapon matches the owner's natural archetype
    if (ownerPosition && gear.flavorPos === ownerPosition) s += 5;
    // Snipers like long-range; Tanks/Brawlers like melee
    if (ownerPosition === "sniper" && gear.range > 200) s += 4;
    if (ownerPosition === "speed"  && gear.range > 100) s += 2;
    if ((ownerPosition === "tank" || ownerPosition === "brawl") && gear.range < 50) s += 3;
    return s;
  } else if (slot === "armor") {
    let s = gear.defMod * 1.5 + gear.spdMod * 6 + gear.critMod * 80;
    if (ownerPosition && gear.flavorPos === ownerPosition) s += 4;
    return s;
  } else if (slot === "item") {
    return 5;
  }
  return 0;
}

function initLeague() {
  // Build initial 4-team x 5-position rosters
  for (const team of TEAMS) {
    LEAGUE.rosters[team.id] = POSITIONS.map(p => makePlayer(team.id, p));
    LEAGUE.records[team.id] = { wins: 0, losses: 0, championships: 0 };
  }
  LEAGUE.season = 1;
  LEAGUE.bits = 0;
  LEAGUE.graveyard = [];
  LEAGUE.history = [];
  LEAGUE.bets = [];
  LEAGUE.pendingBoosts = {};
  buildBracket();
}

/* ============================================================
   BRACKET SCHEDULER — random seeding each season
   4 teams → 2 semifinals → 1 final
   ============================================================ */
function buildBracket() {
  const seeded = shuffle(TEAMS.map(t => t.id));
  LEAGUE.bracket = {
    semi1: { teamA: seeded[0], teamB: seeded[1], winner: null, casualties: [] },
    semi2: { teamA: seeded[2], teamB: seeded[3], winner: null, casualties: [] },
    final: { teamA: null, teamB: null, winner: null, casualties: [] },
  };
  LEAGUE.currentSlot = "semi1";
}

function advanceBracket() {
  // Called when a match resolves. Sets up the next match or marks season complete.
  const b = LEAGUE.bracket;
  if (LEAGUE.currentSlot === "semi1") {
    LEAGUE.currentSlot = "semi2";
    b.final.teamA = b.semi1.winner;
  } else if (LEAGUE.currentSlot === "semi2") {
    LEAGUE.currentSlot = "final";
    b.final.teamB = b.semi2.winner;
  } else if (LEAGUE.currentSlot === "final") {
    LEAGUE.currentSlot = null; // season complete
  }
}

function activeMatchTeams() {
  if (!LEAGUE.currentSlot) return null;
  const m = LEAGUE.bracket[LEAGUE.currentSlot];
  return { slot: LEAGUE.currentSlot, teamA: m.teamA, teamB: m.teamB, match: m };
}

/* ============================================================
   ROSTER MAINTENANCE — replacement of deceased between matches
   Mortality is permanent; rookies replace dead at the SAME position
   so each team always has one of each archetype.
   ============================================================ */
function reapDeceased() {
  // Move dead players from active rosters to graveyard, then call up rookies.
  const callups = [];
  for (const team of TEAMS) {
    const roster = LEAGUE.rosters[team.id];
    for (let i = 0; i < roster.length; i++) {
      const p = roster[i];
      if (p.deceased) {
        // Already dead — make sure they're in graveyard, then replace
        const pos = POS_BY_ID[p.position];
        const rookie = makePlayer(team.id, pos);
        rookie.debutSeason = LEAGUE.season;
        roster[i] = rookie;
        callups.push({ team: team.id, rookie, replaced: p });
      }
    }
  }
  return callups;
}

function applyBoosts(teamId, players) {

  const boost = LEAGUE.pendingBoosts[teamId];

  for (const p of players) {
    if (boost.hp)   p.hp = Math.min(CAPS.hp, p.hp + boost.hp);
    if (boost.dmg)  p.stats = { ...p.stats, dmg:  clamp(p.stats.dmg + boost.dmg,  1, CAPS.dmg) };
    if (boost.def)  p.stats = { ...p.stats, def:  clamp(p.stats.def + boost.def,  0, CAPS.def) };
    if (boost.spd)  p.stats = { ...p.stats, spd:  clamp(+(p.stats.spd + boost.spd).toFixed(2), 0.6, CAPS.spd) };
    if (boost.crit) p.stats = { ...p.stats, crit: clamp(p.stats.crit + boost.crit, 0.02, 0.50) };
  }
  // Boosts consumed
  delete LEAGUE.pendingBoosts[teamId];
}

/* ============================================================
   MATCH INITIALIZATION (best-of-N rounds)
   - startMatch() sets up match-level state, resets lives, then
     immediately starts round 1.
   - startRound() fields all eligible players (lives > 0, !deceased),
     positions them, fires "start" items on round 1, sets phase = "round".
   ============================================================ */
function startMatch() {
  if (!LEAGUE.currentSlot) return;
  const b = LEAGUE.bracket[LEAGUE.currentSlot];
  if (!b.teamA || !b.teamB) return;

  const teamA = LEAGUE.rosters[b.teamA].filter(p => !p.deceased);
  const teamB = LEAGUE.rosters[b.teamB].filter(p => !p.deceased);

  // Reset match-level state on every player going into this match
  for (const p of [...teamA, ...teamB]) {
    p.lives = MATCH.startingLives;
    p.itemUsed = false;
    p.boostMods = null;
    p.balloonsPopped = 0;
  }

  // Apply pending shop boosts (consumed once per match)
  applyBoosts(b.teamA, teamA);
  applyBoosts(b.teamB, teamB);

  LEAGUE.match = {
    slot: LEAGUE.currentSlot,
    teamAId: b.teamA, teamBId: b.teamB,
    teamAplayers: teamA, teamBplayers: teamB,
    bestOf: MATCH.bestOf,
    roundsToWin: MATCH.roundsToWin,
    roundsWonA: 0, roundsWonB: 0,
    rounds: [],
    currentRoundIndex: 0,
    phase: "preRound",
    frame: 0,
    roundFrame: 0,
    killsByTeam: { [b.teamA]: 0, [b.teamB]: 0 },
    blood: [],
    corpses: [],
    floats: [],
    drops: [],
    projectiles: [],
    terrain: generateTerrain(),  // randomized obstacles for this match
    winnerId: null,
  };

  feed("match", phrasePick(PHRASE_MATCH_START, {
    a: TEAM_BY_ID[b.teamA].emoji + " " + TEAM_BY_ID[b.teamA].short,
    b: TEAM_BY_ID[b.teamB].emoji + " " + TEAM_BY_ID[b.teamB].short,
  }));

  // Begin round 1 immediately
  startRound();
}

function startRound() {
  const m = LEAGUE.match;
  if (!m) return;
  m.currentRoundIndex++;
  m.phase = "round";
  m.roundFrame = 0;
  m.projectiles = m.projectiles || []; // ensure projectile array exists

  // Eligible players: not deceased AND have lives remaining
  const eligibleA = m.teamAplayers.filter(p => !p.deceased && p.lives > 0);
  const eligibleB = m.teamBplayers.filter(p => !p.deceased && p.lives > 0);

  if (!eligibleA.length || !eligibleB.length) {
    finalizeRound(/*forfeit*/ true);
    return;
  }

  layoutTeam(eligibleA, "left");
  layoutTeam(eligibleB, "right");

  // Spawn neutral items on the field. Round 1 gets the full set (3 Health
  // Packs + 2 Armor + 3 random); later rounds refresh if the field is
  // sparse (re-stock so survivors don't fight over scraps).
  if (m.currentRoundIndex === 1) {
    spawnNeutralItems(m, 8);
  } else {
    if (m.drops.length < 4) spawnNeutralItems(m, 4);
  }

  if (m.currentRoundIndex === 1) {
    for (const p of [...eligibleA, ...eligibleB]) {
      if (p.item && p.item.trigger === "start" && !p.itemUsed) {
        applyItemTrigger(p);
      }
    }
  }

  feed("match", `🔔 Round ${m.currentRoundIndex} of ${m.bestOf} — fight!`);
}

/* Spawn neutral items at match start. Mix:
   - Health Packs (instant +30 HP on pickup, NOT a gear slot)
   - Armor pieces (random armor — instant equip if upgrade)
   - General loot (random weapon/armor/item)
   This gives the AI reasons to scout the map besides chasing each other. */
function spawnNeutralItems(m, n) {
  const margin = 60;
  const safeSpawn = () => {
    let x, y, attempts = 0;
    do {
      x = R(margin, COMBAT.arenaW - margin);
      y = R(margin, COMBAT.arenaH - margin);
      attempts++;
    } while (m.terrain && collidesTerrain(x, y, m.terrain) && attempts < 12);
    return { x, y };
  };

  // Always spawn 2-3 Health Packs (consumable HP boost)
  const healthCount = Math.min(n, 3);
  for (let i = 0; i < healthCount; i++) {
    const { x, y } = safeSpawn();
    m.drops.push({
      x, y,
      slot: "health",
      gear: { id: "healthpack", ico: "❤", name: "Health Pack", heal: 30 },
      fromName: null, neutral: true,
    });
  }

  // Always spawn 2 Armor pieces (equip if upgrade)
  for (let i = 0; i < 2; i++) {
    const { x, y } = safeSpawn();
    m.drops.push({
      x, y,
      slot: "armor",
      gear: { ...pick(ARMORS) },
      fromName: null, neutral: true,
    });
  }

  // Fill the rest with general loot (random slot)
  const rest = Math.max(0, n - healthCount - 2);
  for (let i = 0; i < rest; i++) {
    const { x, y } = safeSpawn();
    const slot = pick(["weapon", "armor", "item"]);
    const catalog = slot === "weapon" ? WEAPONS : slot === "armor" ? ARMORS : ITEMS;
    m.drops.push({
      x, y, slot,
      gear: { ...pick(catalog) },
      fromName: null, neutral: true,
    });
  }
}

/* ============================================================
   TERRAIN GENERATION
   3-7 square obstacle blocks clustered near the center vertical
   line of the arena. Each block is randomly nudged left/right so
   no two matches feel identical. Blocks are ~36×36px (roughly 2×
   a player's footprint) and AI must path around them.
   ============================================================ */
function generateTerrain() {
  const blocks = [];
  const count = RI(3, 7);
  const SIZE = 36;
  const cx = COMBAT.arenaW / 2;
  const usable = COMBAT.arenaH - 80;
  // Distribute vertically with even spacing, then jitter
  for (let i = 0; i < count; i++) {
    const yFrac = (i + 1) / (count + 1);
    const y = Math.round(40 + usable * yFrac - SIZE / 2);
    // Cluster near center line, with x-nudge of ±90px
    const x = Math.round(cx - SIZE / 2 + R(-90, 90));
    blocks.push({ x, y, size: SIZE });
  }
  return blocks;
}

function layoutTeam(players, side) {
  const cx = COMBAT.arenaW / 2;
  const cy = COMBAT.arenaH / 2;
  const baseX = side === "left" ? cx - COMBAT.spawnRadius : cx + COMBAT.spawnRadius;
  const spacing = 70;
  const startY = cy - (players.length - 1) * spacing / 2;
  for (let i = 0; i < players.length; i++) {
    const p = players[i];
    p.hp = p.stats.hp;
    p.alive = true;
    p.x = baseX + R(-12, 12);
    p.y = startY + i * spacing + R(-8, 8);
    p.vx = 0; p.vy = 0;
    p.target = null;
    p.attackCooldown = 0;
    p.facing = side === "left" ? 1 : -1;
    p.stunned = 0;
    p.balloonsPopped = 0;
    p.aiState = "ENGAGE";   // SEEK_ITEM | ENGAGE | FLEE | FINISH_OFF
    p.attackedBy = null;    // who hit me last (for mercy/aggro)
    p.attackedByExpires = 0;
  }
}

/* ============================================================
   ITEM TRIGGERS — fired by "start" or "lowHP" conditions
   Sets temporary boostMods or applies one-shot effects.
   ============================================================ */
function applyItemTrigger(p) {
  if (!p.item || p.itemUsed) return;
  const id = p.item.id || (p.item.name && ITEMS.find(i => i.name === p.item.name)?.id);
  switch (id) {
    case "booster":
      p.boostMods = { spd: 0.5, life: 240 }; // 12s @ 50ms tick
      feed("item", `💨 ${p.name} engages booster pack!`);
      break;
    case "powercell":
      p.boostMods = { dmgMult: 1.25, life: 300 }; // 15s
      feed("item", `🔋 ${p.name} overcharges power cell!`);
      break;
    case "stim":
      p.hp = Math.min(p.stats.hp, p.hp + 25);
      feed("item", `🍶 ${p.name} jabs a healing stim — ${p.hp}/${p.stats.hp} HP.`);
      break;
    case "grenade": {
      // Hit nearest living enemy for 18
      const m = LEAGUE.match;
      if (!m) break;
      const enemies = (p.teamId === m.teamAId ? m.teamBplayers : m.teamAplayers).filter(e => e.alive);
      let best = null, bestD = Infinity;
      for (const e of enemies) { const d = distance(p, e); if (d < bestD) { bestD = d; best = e; } }
      if (best) {
        const grenadeDmg = 18;
        best.hp -= grenadeDmg;
        m.floats.push({ x: best.x, y: best.y - 14, text: `-${grenadeDmg}💥`, life: 30, color: "#f7c948" });
        feed("item", `🧨 ${p.name} chucks a frag grenade at ${best.name}!`);
        if (best.hp <= 0) {
          best.hp = 0;
          best.alive = false;
          best.balloonsPopped = 3;
          best.lives = Math.max(0, (best.lives || 0) - 1);
          p.career.kills++;
          m.killsByTeam[p.teamId]++;
          if (best.lives > 0) {
            feed("kill", `💥 ${p.name}'s grenade KOs ${best.name}! (${best.lives} ${best.lives===1?"life":"lives"} left)`);
            m.floats.push({ x: best.x, y: best.y - 16, text: "DOWN", life: 24, color: "#ef4d4d" });
          } else {
            best.deceased = true;
            best.deceasedSeason = LEAGUE.season;
            best.deceasedBy = p.name + " (grenade)";
            LEAGUE.graveyard.push({
              name: best.name, teamId: best.teamId, position: best.position,
              season: LEAGUE.season, killer: p.name + " (grenade)",
              killerTeam: p.teamId, slot: m.slot, round: m.currentRoundIndex,
            });
            m.corpses.push({ x: best.x, y: best.y, color: TEAM_BY_ID[best.teamId].color });
            dropGear(best, m);
            feed("kill", `💥 ${p.name} EXPLODES ${best.name} with the grenade!`);
          }
        }
      }
      break;
    }
    case "visor":
      // passive, handled in damage calc — just mark as "active" for the match
      // (no boostMods — crit bonus is read directly from the equipped item)
      break;
  }
  p.itemUsed = true;
}

/* ============================================================
   AI / TICK — team-aware
   - Targets only living opponents
   - Approaches & attacks within hit range
   - Crit + RNG damage roll
   ============================================================ */
/* ============================================================
   AI / TICK — state-machine driven
   States per player:
     ENGAGE      — pursue target, attack when in range
     SEEK_ITEM   — walk toward a desirable nearby drop
     FLEE        — low HP, run away from threats
     FINISH_OFF  — stick on a target that's actively attacking us (mercy)
   Combat:
     - Melee weapons (range <= 30) hit instantly when in range
     - Ranged weapons spawn projectiles with accuracy = f(distance, target velocity)
     - Projectiles are fire-and-forget; targets can dodge
   ============================================================ */
function tickMatch() {
  const m = LEAGUE.match;
  if (!m || m.phase !== "round") return;

  m.frame++;
  m.roundFrame++;

  // Decay floats and blood
  for (let i = m.floats.length - 1; i >= 0; i--) {
    m.floats[i].life--;
    m.floats[i].y -= 0.7;
    if (m.floats[i].life <= 0) m.floats.splice(i, 1);
  }
  for (let i = m.blood.length - 1; i >= 0; i--) {
    m.blood[i].life--;
    if (m.blood[i].life <= 0) m.blood.splice(i, 1);
  }

  // Tick projectiles BEFORE players move so dodging is fair
  tickProjectiles(m);

  const allA = m.teamAplayers, allB = m.teamBplayers;

  // Per-player AI tick
  for (const list of [allA, allB]) {
    const enemies = list === allA ? allB : allA;
    const allies = list;
    const enemyAlive = enemies.filter(e => e.alive);
    for (const p of list) {
      if (!p.alive) continue;

      // Decay timers
      if (p.stunned > 0) p.stunned--;
      if (p.boostMods) {
        p.boostMods.life--;
        if (p.boostMods.life <= 0) p.boostMods = null;
      }
      if (p.attackedByExpires > 0) p.attackedByExpires--;
      else p.attackedBy = null;

      // Low-HP item trigger
      if (p.item && !p.itemUsed && p.item.trigger === "lowHP") {
        const hpFrac = p.hp / p.stats.hp;
        const itemId = p.item.id || (p.item.name && ITEMS.find(i => i.name === p.item.name)?.id);
        const threshold = itemId === "grenade" ? 0.25 : 0.30;
        if (hpFrac < threshold && hpFrac > 0) applyItemTrigger(p);
      }

      // Decide AI state
      decideAIState(p, enemyAlive, m);

      // Execute state
      executeAIState(p, enemyAlive, m);
    }
  }

  // Round end condition
  const aAlive = allA.filter(p => p.alive).length;
  const bAlive = allB.filter(p => p.alive).length;
  if (aAlive === 0 || bAlive === 0 || m.roundFrame >= COMBAT.roundMaxFrames) {
    finalizeRound();
  }
}

/* ============================================================
   AI STATE SELECTION
   Picks the highest-priority state for this tick. Preference:
     1. Stunned → stay put
     2. Low HP (<25%) AND no one closely engaging me → FLEE
     3. Recently attacked (within 90 frames) → FINISH_OFF the attacker (mercy)
     4. Useful drop within 120px AND no enemy within 80px → SEEK_ITEM
     5. Default → ENGAGE nearest enemy
   ============================================================ */
function decideAIState(p, enemyAlive, m) {
  if (p.stunned > 0) { p.aiState = "ENGAGE"; return; }

  const hpFrac = p.hp / p.stats.hp;
  const myWpn = p.weapon || {};
  const ranged = (myWpn.range || 0) > 30;

  // Find nearest enemy
  let nearest = null, nearestD = Infinity;
  for (const e of enemyAlive) {
    const d = distance(p, e);
    if (d < nearestD) { nearestD = d; nearest = e; }
  }

  // Mercy / "finish-off the attacker" — if someone is actively shooting us,
  // and they're close, prefer engaging THEM over chasing a fleeing weakling.
  if (p.attackedBy) {
    const attacker = enemyAlive.find(e => e.id === p.attackedBy);
    if (attacker && distance(p, attacker) < (ranged ? myWpn.range : 220)) {
      p.aiState = "FINISH_OFF";
      p.target = attacker;
      return;
    }
  }

  // FLEE — low HP and not currently in melee range of any enemy
  if (hpFrac < 0.25 && nearest && nearestD > 40) {
    p.aiState = "FLEE";
    p.target = nearest; // remember who we're fleeing FROM
    return;
  }

  // SEEK_ITEM — useful drop within reach AND no immediate threat
  const wantedDrop = findBestDropFor(p, m);
  if (wantedDrop && (!nearest || nearestD > 90)) {
    p.aiState = "SEEK_ITEM";
    p.itemTarget = wantedDrop;
    return;
  }

  // Default: engage the nearest enemy
  p.aiState = "ENGAGE";
  if (nearest) p.target = nearest;
}

function findBestDropFor(p, m) {
  if (!m.drops.length) return null;
  // Vector toward current target (used to prefer drops along the path)
  let tx = 0, ty = 0, hasDir = false;
  if (p.target && p.target.alive) {
    const dx = p.target.x - p.x, dy = p.target.y - p.y;
    const len = Math.sqrt(dx*dx + dy*dy) || 1;
    tx = dx / len; ty = dy / len; hasDir = true;
  }
  let best = null, bestScore = 0;
  for (const d of m.drops) {
    const dx = d.x - p.x, dy = d.y - p.y;
    const dist2 = dx*dx + dy*dy;
    if (dist2 > 220*220) continue;
    const dist = Math.sqrt(dist2);

    let upgrade;
    if (d.slot === "health") {
      // Skip if at full HP. Otherwise the value scales with how hurt we are.
      if (p.hp >= p.stats.hp) continue;
      const hpFrac = p.hp / p.stats.hp;
      upgrade = 4 + (1 - hpFrac) * 14; // 4 at full → 18 at near-dead
    } else {
      const cur = gearScore(d.slot, p[d.slot], p.position);
      const newScore = gearScore(d.slot, d.gear, p.position);
      upgrade = newScore - cur;
      if (upgrade <= 1.5) continue;
    }

    // Bonus for "along the way" — drops in the direction of our target
    // are worth detouring for; drops behind us require a real detour.
    let alignment = 0;
    if (hasDir) {
      const ux = dx / dist, uy = dy / dist;
      alignment = Math.max(0, ux * tx + uy * ty); // 0..1
    }
    const utility = upgrade + alignment * 6 - dist * 0.025;
    if (utility > bestScore) { bestScore = utility; best = d; }
  }
  return best;
}

/* ============================================================
   AI STATE EXECUTION
   ============================================================ */
function executeAIState(p, enemyAlive, m) {
  const effSpd = (p.stats.spd + (p.boostMods?.spd || 0)) * 1.4;
  const wpn = p.weapon || {};
  const ranged = (wpn.range || 0) > 30;

  switch (p.aiState) {
    case "FLEE": {
      // First try to flee TOWARD a health pickup — far better than just running
      let bestHP = null, bestHPd = Infinity;
      for (const d of m.drops) {
        if (d.slot !== "health") continue;
        const dx = p.x - d.x, dy = p.y - d.y;
        const dist2 = dx*dx + dy*dy;
        if (dist2 < 320*320 && dist2 < bestHPd) { bestHPd = dist2; bestHP = d; }
      }
      if (bestHP) {
        const ang = Math.atan2(bestHP.y - p.y, bestHP.x - p.x);
        moveAlong(p, ang, effSpd * 1.15); // adrenaline + medkit hope
      } else if (p.target && p.target.alive) {
        // No health nearby — run AWAY from threat
        const ang = Math.atan2(p.y - p.target.y, p.x - p.target.x);
        moveAlong(p, ang, effSpd * 1.1);
      } else {
        p.aiState = "ENGAGE";
        break;
      }
      // Maybe fire a desperate parting shot
      if (ranged && p.target && p.attackCooldown <= 0 && p.stunned <= 0
          && hasLineOfSight(p.x, p.y, p.target.x, p.target.y, m.terrain)) {
        fireAttack(p, p.target, m);
        p.attackCooldown = computeCooldown(p, effSpd);
      }
      break;
    }
    case "SEEK_ITEM": {
      const drop = p.itemTarget;
      if (!drop || !m.drops.includes(drop)) {
        p.aiState = "ENGAGE";
        break;
      }
      const ang = Math.atan2(drop.y - p.y, drop.x - p.x);
      moveAlong(p, ang, effSpd);
      break;
    }
    case "FINISH_OFF":
    case "ENGAGE":
    default: {
      if (!p.target || !p.target.alive) {
        let nearest = null, nearestD = Infinity;
        for (const e of enemyAlive) {
          const d = distance(p, e);
          if (d < nearestD) { nearestD = d; nearest = e; }
        }
        p.target = nearest;
      }
      if (!p.target) break;

      const d = distance(p, p.target);
      // Ranged knights prefer to maintain a comfortable distance — UNLESS
      // there's terrain between them and the target, in which case they
      // need to close to get a shot.
      const losClear = !ranged || hasLineOfSight(p.x, p.y, p.target.x, p.target.y, m.terrain);
      const idealDist = ranged ? Math.min(wpn.range * 0.6, 220) : 18;
      const tooClose = ranged && d < 80 && losClear;
      if (tooClose) {
        // Back-step (only if we have line of sight, otherwise close instead)
        const ang = Math.atan2(p.y - p.target.y, p.x - p.target.x);
        moveAlong(p, ang, effSpd * 0.7);
      } else if (d > idealDist || !losClear) {
        // Move toward (closing the distance, also tries to clear sight lines)
        const ang = Math.atan2(p.target.y - p.y, p.target.x - p.x);
        const moved = moveAlong(p, ang, effSpd);
        // If movement was blocked by terrain (corner-stuck), strafe perpendicular
        // for ~15 frames to round the corner.
        if (!moved && m.terrain) {
          if (!p.navAvoidanceFrames || p.navAvoidanceFrames <= 0) {
            p.navAvoidanceFrames = 18;
            p.navAvoidanceDir = Math.random() < 0.5 ? 1 : -1;
          }
        }
      }
      if (p.navAvoidanceFrames > 0) {
        const baseAng = Math.atan2(p.target.y - p.y, p.target.x - p.x);
        const perpAng = baseAng + (Math.PI / 2) * p.navAvoidanceDir;
        moveAlong(p, perpAng, effSpd);
        p.navAvoidanceFrames--;
      }

      // Attempt attack if in range AND we have line of sight
      const inRange = ranged
        ? (d <= wpn.range && hasLineOfSight(p.x, p.y, p.target.x, p.target.y, m.terrain))
        : d <= COMBAT.hitRange;
      if (inRange && p.attackCooldown <= 0 && p.stunned <= 0) {
        fireAttack(p, p.target, m);
        p.attackCooldown = computeCooldown(p, effSpd);
      }
      break;
    }
  }
  if (p.attackCooldown > 0) p.attackCooldown--;

  // Auto-pickup if walking over a drop (any slot, any score — if you're on
  // top of it, take it and worry about the swap later)
  attemptScavenge(p, m);
}

function moveAlong(p, ang, speed) {
  const startX = p.x, startY = p.y;
  let nx = p.x + Math.cos(ang) * speed;
  let ny = p.y + Math.sin(ang) * speed;

  // Apply separation push from nearby allies + enemies (boids-style)
  const m = LEAGUE.match;
  if (m) {
    const all = m.teamAplayers.concat(m.teamBplayers);
    const SEP_RADIUS = 22;
    let pushX = 0, pushY = 0;
    for (const o of all) {
      if (o === p || !o.alive) continue;
      const dx = nx - o.x, dy = ny - o.y;
      const d2 = dx*dx + dy*dy;
      if (d2 < SEP_RADIUS * SEP_RADIUS && d2 > 0.01) {
        const d = Math.sqrt(d2);
        const force = (SEP_RADIUS - d) / SEP_RADIUS;
        pushX += (dx / d) * force * speed * 0.6;
        pushY += (dy / d) * force * speed * 0.6;
      }
    }
    nx += pushX;
    ny += pushY;
  }

  // Terrain collision — wall-slide if blocked
  if (m && m.terrain) {
    if (collidesTerrain(nx, ny, m.terrain)) {
      // Try X-only
      if (!collidesTerrain(nx, p.y, m.terrain)) {
        ny = p.y;
      } else if (!collidesTerrain(p.x, ny, m.terrain)) {
        nx = p.x;
      } else {
        nx = p.x; ny = p.y;
      }
    }
  }

  p.vx = nx - p.x;
  p.vy = ny - p.y;
  p.x = clamp(nx, 12, COMBAT.arenaW - 12);
  p.y = clamp(ny, 12, COMBAT.arenaH - 12);
  p.facing = p.vx >= 0 ? 1 : -1;

  // Did we actually move? (Returns false if fully stuck — caller can fall
  // back to perpendicular strafe to round corners.)
  return Math.abs(p.x - startX) > 0.1 || Math.abs(p.y - startY) > 0.1;
}

// Test whether a point (with player half-size ~10px) collides with any
// terrain block. Terrain blocks are squares with x,y as top-left + size.
function collidesTerrain(x, y, terrain) {
  const PAD = 10;
  for (const t of terrain) {
    if (x + PAD > t.x && x - PAD < t.x + t.size &&
        y + PAD > t.y && y - PAD < t.y + t.size) {
      return true;
    }
  }
  return false;
}

// Liang-Barsky segment-vs-AABB test. Returns true if the segment from
// (x0,y0) to (x1,y1) intersects the axis-aligned rect at (rx,ry,rw,rh).
// Used for swept projectile collision and line-of-sight checks.
function segmentHitsRect(x0, y0, x1, y1, rx, ry, rw, rh) {
  const dx = x1 - x0, dy = y1 - y0;
  let tEnter = 0, tExit = 1;
  const tests = [[-dx, x0 - rx], [dx, rx + rw - x0], [-dy, y0 - ry], [dy, ry + rh - y0]];
  for (const [p, q] of tests) {
    if (p === 0) {
      if (q < 0) return false;
    } else {
      const t = q / p;
      if (p < 0) {
        if (t > tExit) return false;
        if (t > tEnter) tEnter = t;
      } else {
        if (t < tEnter) return false;
        if (t < tExit) tExit = t;
      }
    }
  }
  return tEnter <= tExit;
}

// Does the line from a→b cross any terrain block?
function hasLineOfSight(ax, ay, bx, by, terrain) {
  if (!terrain || !terrain.length) return true;
  for (const t of terrain) {
    if (segmentHitsRect(ax, ay, bx, by, t.x, t.y, t.size, t.size)) return false;
  }
  return true;
}

function computeCooldown(p, effSpd) {
  const baseCd = Math.max(6, Math.round(18 / Math.max(0.6, effSpd / 1.4)));
  const cdScale = 1 + (p.weapon?.cdMod || 0);
  return Math.max(4, Math.round(baseCd * cdScale));
}

/* ============================================================
   FIRE ATTACK — branches melee vs projectile
   ============================================================ */
function fireAttack(attacker, target, m) {
  const wpn = attacker.weapon || {};
  const range = wpn.range || 0;

  if (range <= 30) {
    // Melee — instant resolve as before
    resolveAttack(attacker, target, m);
    return;
  }

  // Ranged — spawn a projectile with accuracy spread
  const dx = target.x - attacker.x, dy = target.y - attacker.y;
  const dist = Math.sqrt(dx*dx + dy*dy);
  const baseAng = Math.atan2(dy, dx);

  // Inaccuracy scales with: weapon baseline, distance, target velocity
  const targetSpeed = Math.sqrt(target.vx*target.vx + target.vy*target.vy);
  const distFactor = (dist / wpn.range) * 0.18;             // out to weapon range adds spread
  const velFactor = Math.min(targetSpeed * 0.04, 0.15);     // fast targets harder to track
  const spread = (wpn.inacc || 0.05) + distFactor + velFactor;
  const aimAng = baseAng + (Math.random() - 0.5) * 2 * spread;

  const speed = wpn.projSpeed || 12;
  m.projectiles.push({
    x: attacker.x + Math.cos(baseAng) * 8,
    y: attacker.y + Math.sin(baseAng) * 8 - 4,
    vx: Math.cos(aimAng) * speed,
    vy: Math.sin(aimAng) * speed,
    dmg: attacker.stats.dmg + (wpn.dmgMod || 0),
    fromId: attacker.id,
    teamId: attacker.teamId,
    ttl: Math.ceil(wpn.range / speed) + 4,
    wpn,
    isCrit: Math.random() < clamp(attacker.stats.crit + (wpn.critMod || 0) + (attacker.armor?.critMod || 0) + ((attacker.item?.id === "visor" || attacker.item?.name === "Targeting Visor") ? 0.05 : 0), 0.02, 0.5),
  });
  // Muzzle flash
  m.floats.push({ x: attacker.x, y: attacker.y - 2, text: "•", life: 6, color: "#ffd96b" });
}

/* ============================================================
   PROJECTILE PHYSICS — fly along path, collide with first enemy
   ============================================================ */
function tickProjectiles(m) {
  if (!m.projectiles || !m.projectiles.length) return;
  const enemies = (teamId) => teamId === m.teamAId ? m.teamBplayers : m.teamAplayers;

  for (let i = m.projectiles.length - 1; i >= 0; i--) {
    const pr = m.projectiles[i];
    const oldX = pr.x, oldY = pr.y;
    pr.x += pr.vx;
    pr.y += pr.vy;
    pr.ttl--;

    // Off-arena or expired
    if (pr.ttl <= 0 || pr.x < 0 || pr.x > COMBAT.arenaW || pr.y < 0 || pr.y > COMBAT.arenaH) {
      m.projectiles.splice(i, 1);
      continue;
    }

    // Terrain — SWEPT check against the segment (oldX,oldY)→(x,y) so fast
    // projectiles can't tunnel through a 36px block in a single tick.
    if (m.terrain) {
      let blocked = null;
      for (const t of m.terrain) {
        if (segmentHitsRect(oldX, oldY, pr.x, pr.y, t.x, t.y, t.size, t.size)) {
          blocked = t; break;
        }
      }
      if (blocked) {
        // Spark visual at the entry point of the block (approx)
        m.floats.push({ x: pr.x, y: pr.y, text: "✦", life: 8, color: "#f7c948" });
        m.projectiles.splice(i, 1);
        continue;
      }
    }

    // Collision with any enemy of the firing team
    let hit = null;
    for (const e of enemies(pr.teamId)) {
      if (!e.alive) continue;
      const dx = e.x - pr.x, dy = e.y - pr.y;
      if (dx*dx + dy*dy < 11*11) { hit = e; break; }
    }
    if (hit) {
      const attacker = (m.teamAplayers.concat(m.teamBplayers)).find(x => x.id === pr.fromId);
      if (attacker) applyProjectileHit(attacker, hit, pr, m);
      m.projectiles.splice(i, 1);
    }
  }
}

function applyProjectileHit(attacker, defender, pr, m) {
  // Defense: scale damage down by def/(def+50)
  const arm = defender.armor || {};
  const effDef = clamp(defender.stats.def + (arm.defMod || 0) + (defender.weapon?.defMod || 0), 0, CAPS.def + 20);
  const dmgMult = attacker.boostMods?.dmgMult || 1;
  const rng = R(1 - COMBAT.rngSpread, 1 + COMBAT.rngSpread);
  const raw = pr.dmg * dmgMult * rng * (pr.isCrit ? 2.0 : 1.0);
  const reduction = effDef / (effDef + 50);
  const dmg = Math.max(1, Math.round(raw * (1 - reduction)));

  // Track attacker-of-record on defender (for mercy state)
  defender.attackedBy = attacker.id;
  defender.attackedByExpires = 90;

  applyDamage(attacker, defender, dmg, pr.isCrit, m);
}

/* Shared damage / death pipeline for both melee and projectile hits */
function applyDamage(attacker, defender, dmg, isCrit, m) {
  const hpFracBefore = defender.hp / defender.stats.hp;
  defender.hp -= dmg;
  const hpFracAfter = defender.hp / defender.stats.hp;
  // Balloon pop thresholds at 2/3 and 1/3
  for (const t of [2/3, 1/3]) {
    if (hpFracBefore > t && hpFracAfter <= t && defender.hp > 0) {
      defender.balloonsPopped = (defender.balloonsPopped || 0) + 1;
      m.floats.push({ x: defender.x, y: defender.y - 18, text: "POP!", life: 22, color: "#fff" });
    }
  }
  m.floats.push({
    x: defender.x, y: defender.y - 10,
    text: isCrit ? `${dmg}!` : String(dmg),
    life: 20,
    color: isCrit ? "#f7c948" : "#ef4d4d",
  });
  for (let i = 0; i < (isCrit ? 6 : 3); i++) {
    m.blood.push({
      x: defender.x + R(-6, 6), y: defender.y + R(-6, 6),
      life: 60 + RI(0, 30),
      color: pick(["#a01818", "#831010", "#7a0c0c", "#bf2020"]),
    });
  }
  // Stun chance from attacker's weapon
  const wpn = attacker.weapon || {};
  if ((wpn.stun || 0) > 0 && Math.random() < wpn.stun) {
    defender.stunned = 30;
    m.floats.push({ x: defender.x, y: defender.y - 14, text: "STUN", life: 18, color: "#88c0d0" });
  }
  if (isCrit) feed("kill", phrasePick(PHRASE_CRIT, { a: attacker.name, b: defender.name }));

  if (defender.hp <= 0) handleKO(attacker, defender, m);
}

function handleKO(attacker, defender, m) {
  defender.alive = false;
  defender.hp = 0;
  defender.balloonsPopped = 3;
  defender.lives = Math.max(0, (defender.lives || 0) - 1);
  attacker.career.kills++;
  m.killsByTeam[attacker.teamId]++;

  if (defender.lives > 0) {
    feed("kill", `💢 ${attacker.name} KOs ${defender.name}! (${defender.lives} ${defender.lives === 1 ? "life" : "lives"} left)`);
    m.floats.push({ x: defender.x, y: defender.y - 14, text: "DOWN", life: 24, color: "#ef4d4d" });
  } else {
    defender.deceased = true;
    defender.deceasedSeason = LEAGUE.season;
    defender.deceasedBy = attacker.name;
    LEAGUE.graveyard.push({
      name: defender.name, teamId: defender.teamId, position: defender.position,
      season: LEAGUE.season, killer: attacker.name, killerTeam: attacker.teamId,
      slot: m.slot, round: m.currentRoundIndex,
    });
    m.corpses.push({ x: defender.x, y: defender.y, color: TEAM_BY_ID[defender.teamId].color });
    dropGear(defender, m);
    feed("kill", phrasePick(PHRASE_KILL, { a: attacker.name, b: defender.name }));
  }
}

/* Melee instant-hit path — kept for backwards-compat with applyItemTrigger
   (grenade) and for melee weapons. */
function resolveAttack(attacker, defender, m) {
  const wpn = attacker.weapon || {};
  const arm = defender.armor || {};
  const visorBonus = (attacker.item && (attacker.item.id === "visor" || attacker.item.name === "Targeting Visor")) ? 0.05 : 0;
  const dmgMult = attacker.boostMods?.dmgMult || 1;
  const effDmg  = clamp(attacker.stats.dmg + (wpn.dmgMod || 0), 1, CAPS.dmg + 20) * dmgMult;
  const effDef  = clamp(defender.stats.def + (arm.defMod || 0) + (defender.weapon?.defMod || 0), 0, CAPS.def + 20);
  const effCrit = clamp(attacker.stats.crit + (wpn.critMod || 0) + (arm.critMod || 0) + visorBonus, 0.02, 0.50);
  const rng = R(1 - COMBAT.rngSpread, 1 + COMBAT.rngSpread);
  const isCrit = Math.random() < effCrit;
  const raw = effDmg * rng * (isCrit ? 2.0 : 1.0);
  const reduction = effDef / (effDef + 50);
  const dmg = Math.max(1, Math.round(raw * (1 - reduction)));

  // Track attacker-of-record (mercy state)
  defender.attackedBy = attacker.id;
  defender.attackedByExpires = 90;

  applyDamage(attacker, defender, dmg, isCrit, m);
}

/* ============================================================
   GEAR DROPS & SCAVENGING
   When a knight falls, their loadout (weapon/armor/item) drops
   at the corpse. Living players who walk within range auto-pickup
   any upgrade. Picked-up gear is renamed
     "<owner>'s <gear> (scavenged)"
   and re-drops with that scavenged name if its new owner dies.
   ============================================================ */
function dropGear(deadPlayer, m) {
  const owner = deadPlayer.name;
  // Drop weapon
  if (deadPlayer.weapon) {
    m.drops.push({
      x: deadPlayer.x + R(-6, 6),
      y: deadPlayer.y + R(-6, 6),
      slot: "weapon",
      gear: { ...deadPlayer.weapon, displayName: scavengedName(deadPlayer.weapon, owner), originalOwner: deadPlayer.weapon.originalOwner || owner },
      fromName: owner,
    });
  }
  // Drop armor
  if (deadPlayer.armor) {
    m.drops.push({
      x: deadPlayer.x + R(-6, 6),
      y: deadPlayer.y + R(-6, 6),
      slot: "armor",
      gear: { ...deadPlayer.armor, displayName: scavengedName(deadPlayer.armor, owner), originalOwner: deadPlayer.armor.originalOwner || owner },
      fromName: owner,
    });
  }
  // Drop item only if not yet used (a used consumable is gone)
  if (deadPlayer.item && !deadPlayer.itemUsed) {
    m.drops.push({
      x: deadPlayer.x + R(-6, 6),
      y: deadPlayer.y + R(-6, 6),
      slot: "item",
      gear: { ...deadPlayer.item, displayName: scavengedName(deadPlayer.item, owner), originalOwner: deadPlayer.item.originalOwner || owner },
      fromName: owner,
    });
  }
}

function scavengedName(gear, owner) {
  // Strip any existing possessive prefix so we don't get "X's Y's Z's ... (scavenged)"
  const base = gear.name; // canonical name from catalog (never mutated)
  return `${owner}'s ${base} (scavenged)`;
}

function attemptScavenge(p, m) {
  if (!m.drops.length) return;
  const RANGE = 22;
  for (let i = 0; i < m.drops.length; i++) {
    const d = m.drops[i];
    const dx = p.x - d.x, dy = p.y - d.y;
    if (dx*dx + dy*dy > RANGE*RANGE) continue;

    // Health Pack — instant heal, no slot swap
    if (d.slot === "health") {
      // Skip if already at full HP
      if (p.hp >= p.stats.hp) continue;
      const heal = d.gear.heal || 30;
      const before = p.hp;
      p.hp = Math.min(p.stats.hp, p.hp + heal);
      const actual = p.hp - before;
      feed("item", `❤ ${p.name} grabs a Health Pack (+${actual} HP → ${p.hp}/${p.stats.hp})`);
      m.floats.push({ x: p.x, y: p.y - 16, text: `+${actual}♥`, life: 22, color: "#ef4d4d" });
      m.drops.splice(i, 1);
      return;
    }

    // Compare scores using THIS player's archetype preference
    const slot = d.slot;
    const currentScore = gearScore(slot, p[slot], p.position);
    const newScore     = gearScore(slot, d.gear,   p.position);

    // Pick up if it's an upgrade for THIS player. Items always swap.
    const shouldPickUp = (slot === "item" && !p.itemUsed) ? newScore >= currentScore
                                                          : newScore > currentScore + 0.5;
    if (!shouldPickUp) continue;

    // Pick it up — replace gear, mark scavenge stat
    d.gear.scavengedCount = (d.gear.scavengedCount || 0) + 1;
    p[slot] = d.gear;
    if (slot === "item") p.itemUsed = false;
    p.career.scavenged = (p.career.scavenged || 0) + 1;
    const legendary = d.gear.scavengedCount >= 3 ? " ⭐LEGENDARY⭐" : "";
    feed("item", `🎒 ${p.name} scavenges ${gearDisplayName(d.gear)}!${legendary}`);
    m.floats.push({ x: p.x, y: p.y - 16, text: "+gear", life: 22, color: "#4dd58a" });
    m.drops.splice(i, 1);
    return;
  }
}

/* ============================================================
   ROUND FINALIZATION
   Determines round winner (the team with at least one alive player).
   Increments round score. Checks for match end (best-of-N reached
   OR a team has no fielded players for the next round). Otherwise
   transitions to "betweenRounds" so the user can place bets / shop.
   ============================================================ */
function finalizeRound(forfeit) {
  const m = LEAGUE.match;
  if (!m) return;

  const aAlive = m.teamAplayers.filter(p => p.alive).length;
  const bAlive = m.teamBplayers.filter(p => p.alive).length;
  let roundWinnerId, roundLoserId;
  if (forfeit) {
    // One team can't field anyone — the other team wins by forfeit
    const eligibleA = m.teamAplayers.filter(p => !p.deceased && p.lives > 0);
    const eligibleB = m.teamBplayers.filter(p => !p.deceased && p.lives > 0);
    roundWinnerId = eligibleA.length ? m.teamAId : m.teamBId;
    roundLoserId  = eligibleA.length ? m.teamBId : m.teamAId;
    feed("match", `⚐ ${TEAM_BY_ID[roundLoserId].short} cannot field a roster — ${TEAM_BY_ID[roundWinnerId].short} wins by forfeit!`);
  } else if (aAlive > bAlive) {
    roundWinnerId = m.teamAId; roundLoserId = m.teamBId;
  } else if (bAlive > aAlive) {
    roundWinnerId = m.teamBId; roundLoserId = m.teamAId;
  } else {
    // Mutual wipe / timeout — coin flip
    if (Math.random() < 0.5) { roundWinnerId = m.teamAId; roundLoserId = m.teamBId; }
    else { roundWinnerId = m.teamBId; roundLoserId = m.teamAId; }
  }

  if (roundWinnerId === m.teamAId) m.roundsWonA++;
  else m.roundsWonB++;

  // Settle "round winner" bets immediately
  settleRoundBets(roundWinnerId);

  // Round log
  m.rounds.push({
    index: m.currentRoundIndex,
    winner: roundWinnerId,
    aAlive, bAlive,
  });

  feed("match", `🏁 Round ${m.currentRoundIndex} won by ${TEAM_BY_ID[roundWinnerId].emoji} ${TEAM_BY_ID[roundWinnerId].short} (${m.roundsWonA}–${m.roundsWonB} match score).`);

  // Check for match end
  const matchOver =
    m.roundsWonA >= m.roundsToWin ||
    m.roundsWonB >= m.roundsToWin ||
    forfeit;
  if (matchOver) {
    finalizeMatch();
    return;
  }

  // Otherwise: between-rounds intermission
  m.phase = "betweenRounds";
  if (typeof showRoundIntermission === "function") showRoundIntermission();
}

function continueAfterRound() {
  // Called from the round-intermission "Begin Next Round" button.
  const m = LEAGUE.match;
  if (!m) return;
  startRound();
}

/* ============================================================
   MATCH FINALIZATION
   Reached when one team wins `roundsToWin` rounds OR by forfeit.
   ============================================================ */
function finalizeMatch() {
  const m = LEAGUE.match;
  if (!m) return;

  const winnerId = m.roundsWonA > m.roundsWonB ? m.teamAId : m.teamBId;
  const loserId = winnerId === m.teamAId ? m.teamBId : m.teamAId;
  m.winnerId = winnerId;
  m.phase = "post";

  // Career stats: matchesPlayed for everyone, matchesWon for winners who survived
  for (const p of [...m.teamAplayers, ...m.teamBplayers]) {
    p.career.matchesPlayed++;
  }
  for (const p of (winnerId === m.teamAId ? m.teamAplayers : m.teamBplayers)) {
    if (!p.deceased) p.career.matchesWon++;
  }

  // Records
  LEAGUE.records[winnerId].wins++;
  LEAGUE.records[loserId].losses++;

  // Bracket result
  LEAGUE.bracket[m.slot].winner = winnerId;
  LEAGUE.bracket[m.slot].casualties = LEAGUE.graveyard
    .filter(g => g.season === LEAGUE.season && g.slot === m.slot);

  // Settle match-level bets
  settleBets(winnerId);

  feed("match", `🏆 ${TEAM_BY_ID[winnerId].emoji} ${TEAM_BY_ID[winnerId].name} WIN the match ${m.roundsWonA}–${m.roundsWonB}!`);

  if (typeof showIntermission === "function") showIntermission();
}

function continueAfterMatch() {
  // Called from the intermission "Continue" button.
  const wasFinal = LEAGUE.currentSlot === "final";
  advanceBracket();

  if (wasFinal || !LEAGUE.currentSlot) {
    // Season complete
    finalizeSeason();
    return;
  }

  // Reap deceased and build the next match
  const callups = reapDeceased();
  for (const c of callups) {
    feed("spawn", `📣 Rookie call-up: ${c.rookie.name} replaces ${c.replaced.name} (${TEAM_BY_ID[c.team].name}).`);
  }
  LEAGUE.match = null;
  startMatch();
}

/* ============================================================
   SEASON FINALIZATION
   - Champion crowned
   - History recorded
   - All deceased reaped, rookies called up
   - New bracket built for next season
   ============================================================ */
function finalizeSeason() {
  const champion = LEAGUE.bracket.final.winner;
  const runnerup = champion === LEAGUE.bracket.final.teamA
    ? LEAGUE.bracket.final.teamB
    : LEAGUE.bracket.final.teamA;
  if (champion) {
    LEAGUE.records[champion].championships++;
    feed("season",
      `🏆 SEASON ${LEAGUE.season} CHAMPION: ${TEAM_BY_ID[champion].emoji} ${TEAM_BY_ID[champion].name}!`);
  }

  const seasonCasualties = LEAGUE.graveyard.filter(g => g.season === LEAGUE.season);
  LEAGUE.history.push({
    season: LEAGUE.season,
    champion, runnerup,
    semi1Winner: LEAGUE.bracket.semi1.winner,
    semi2Winner: LEAGUE.bracket.semi2.winner,
    casualties: seasonCasualties.length,
  });

  // Show season-end modal; user clicks "Begin Next Season"
  if (typeof showSeasonEnd === "function") showSeasonEnd(champion, runnerup, seasonCasualties.length);
}

function beginNextSeason() {
  // Reap and replace
  const callups = reapDeceased();
  for (const p of Object.values(LEAGUE.rosters).flat()) {
    p.career.seasonsPlayed++;
  }
  if (callups.length) {
    feed("season", `📣 Off-season call-ups: ${callups.length} rookies join the league.`);
  }
  LEAGUE.season++;
  LEAGUE.match = null;
  buildBracket();
  startMatch();
}

/* ============================================================
   BETTING — both team-level and per-player (last-standing)
   Odds are computed from rough roster strength.
   ============================================================ */
function teamStrength(teamId) {
  const r = (LEAGUE.rosters[teamId] || []).filter(p => !p.deceased);
  let s = 0;
  for (const p of r) {
    s += p.stats.hp + p.stats.dmg * 4 + p.stats.def + p.stats.spd * 8;
  }
  return Math.max(1, s);
}

function teamOdds(teamA, teamB) {
  const sA = teamStrength(teamA), sB = teamStrength(teamB);
  const total = sA + sB;
  const probA = sA / total, probB = sB / total;
  // Convert to payout multiplier (with a small house edge of 5%)
  const margin = 0.95;
  return {
    [teamA]: +(margin / probA).toFixed(2),
    [teamB]: +(margin / probB).toFixed(2),
  };
}

function playerOdds(teamA, teamB) {
  // Probability of "last standing" (or last-killed) for each player.
  // Crude: stat strength relative to the entire match pool, weighted toward survivors.
  const all = [
    ...(LEAGUE.rosters[teamA] || []).filter(p => !p.deceased),
    ...(LEAGUE.rosters[teamB] || []).filter(p => !p.deceased),
  ];
  const strengths = all.map(p =>
    p.stats.hp / 100 + p.stats.dmg / 8 + p.stats.def / 20 + p.stats.spd / 1.5);
  const total = strengths.reduce((a,b)=>a+b, 0);
  const odds = {};
  for (let i = 0; i < all.length; i++) {
    const prob = strengths[i] / total;
    odds[all[i].id] = +(0.85 / Math.max(0.01, prob)).toFixed(2);
  }
  return odds;
}

function placeBet(kind, target, amount) {
  // kind: "team"  → target is teamId, resolves at MATCH end
  // kind: "round" → target is teamId, resolves at end of NEXT round
  // kind: "player" → target is playerId, resolves at MATCH end
  if (amount <= 0 || amount > LEAGUE.bits) return false;
  if (!LEAGUE.currentSlot) return false;
  const slotData = LEAGUE.bracket[LEAGUE.currentSlot];
  if (!slotData || !slotData.teamA || !slotData.teamB) return false;

  let payoutMult = 1;
  if (kind === "team" || kind === "round") {
    const o = teamOdds(slotData.teamA, slotData.teamB);
    payoutMult = o[target] || 0;
    if (!payoutMult) return false;
    // Round bets pay slightly less (resolves faster, less risk)
    if (kind === "round") payoutMult = +(payoutMult * 0.85).toFixed(2);
  } else if (kind === "player") {
    const o = playerOdds(slotData.teamA, slotData.teamB);
    payoutMult = o[target] || 0;
    if (!payoutMult) return false;
  } else return false;

  LEAGUE.bits -= amount;
  // For round bets, tag them with the upcoming round index so they only resolve once
  const m = LEAGUE.match;
  const roundIndex = (kind === "round" && m) ? (m.currentRoundIndex + (m.phase === "betweenRounds" || m.phase === "preRound" ? 1 : 0)) : null;
  LEAGUE.bets.push({ kind, target, amount, payoutMult, slot: LEAGUE.currentSlot, roundIndex });
  feed("item", `💰 Bet placed: ${amount} bits on ${kind}=${target} @ ${payoutMult}x`);
  return true;
}

// Resolves "round winner" bets after each round.
function settleRoundBets(roundWinnerTeamId) {
  const m = LEAGUE.match;
  if (!m) return;
  const remaining = [];
  for (const bet of LEAGUE.bets) {
    if (bet.kind !== "round" || bet.slot !== m.slot) {
      remaining.push(bet);
      continue;
    }
    if (bet.roundIndex !== m.currentRoundIndex) {
      remaining.push(bet);
      continue;
    }
    const tName = TEAM_BY_ID[bet.target]?.short || "?";
    if (bet.target === roundWinnerTeamId) {
      const payout = Math.round(bet.amount * bet.payoutMult);
      LEAGUE.bits += payout;
      feed("item", `💸 Round bet won! +${payout} bits (round ${m.currentRoundIndex})`);
      if (typeof window !== "undefined" && window.toast)
        window.toast(`Round ${m.currentRoundIndex} bet WON: <b>+${payout}</b> bits on ${tName}`, "win");
    } else {
      feed("item", `💔 Round bet lost: ${bet.amount} bits.`);
      if (typeof window !== "undefined" && window.toast)
        window.toast(`Round ${m.currentRoundIndex} bet lost: <b>-${bet.amount}</b> bits on ${tName}`, "lose");
    }
  }
  LEAGUE.bets = remaining;
}

// Resolves match-level bets ("team" + "player") after a match concludes.
function settleBets(winnerTeamId) {
  const m = LEAGUE.match;
  if (!m) return;
  const lastStanding = (() => {
    const all = [...m.teamAplayers, ...m.teamBplayers]
      .filter(p => !p.deceased && p.lives > 0);
    if (!all.length) return null;
    all.sort((a,b) => (b.lives - a.lives) || (b.career.kills - a.career.kills));
    return all[0];
  })();

  const remaining = [];
  for (const bet of LEAGUE.bets) {
    if (bet.slot !== m.slot) {
      remaining.push(bet);
      continue;
    }
    if (bet.kind === "round") {
      LEAGUE.bits += bet.amount;
      feed("item", `↩️ Unsettled round bet refunded: ${bet.amount} bits.`);
      if (typeof window !== "undefined" && window.toast)
        window.toast(`Round bet refunded: <b>+${bet.amount}</b> bits (match ended early)`, "warn");
      continue;
    }
    let win = false;
    let label = "";
    if (bet.kind === "team") {
      label = TEAM_BY_ID[bet.target]?.short || "?";
      if (bet.target === winnerTeamId) win = true;
    }
    if (bet.kind === "player") {
      const p = Object.values(LEAGUE.rosters).flat().find(x => x.id === bet.target);
      label = p ? p.name : "?";
      if (lastStanding && bet.target === lastStanding.id) win = true;
    }
    if (win) {
      const payout = Math.round(bet.amount * bet.payoutMult);
      LEAGUE.bits += payout;
      feed("item", `💸 Bet won! +${payout} bits (${bet.kind})`);
      if (typeof window !== "undefined" && window.toast)
        window.toast(`Match bet WON: <b>+${payout}</b> bits on ${label}`, "win");
    } else {
      feed("item", `💔 Bet lost: ${bet.amount} bits on ${bet.kind}.`);
      if (typeof window !== "undefined" && window.toast)
        window.toast(`Match bet lost: <b>-${bet.amount}</b> bits on ${label}`, "lose");
    }
  }
  LEAGUE.bets = remaining;

  // Always award some bits for watching
  const watchReward = 5 + RI(0, 5);
  LEAGUE.bits += watchReward;
  feed("item", `🪙 +${watchReward} bits (broadcast tip)`);
}

/* ============================================================
   SHOP
   ============================================================ */
function buyItem(itemId, teamId) {
  const item = SHOP_BY_ID[itemId];
  if (!item) return false;
  if (LEAGUE.bits < item.cost) return false;
  if (LEAGUE.match && LEAGUE.match.phase === "round") return false; // can't buy during a live round

  // Saboteur targets the OPPONENT in the next active match
  let appliedToTeam = teamId;
  if (item.target === "opp") {
    if (!LEAGUE.currentSlot) return false;
    const m = LEAGUE.bracket[LEAGUE.currentSlot];
    appliedToTeam = (teamId === m.teamA) ? m.teamB : m.teamA;
  }

  const boost = LEAGUE.pendingBoosts[appliedToTeam] || (LEAGUE.pendingBoosts[appliedToTeam] = {});
  item.apply(boost);
  LEAGUE.bits -= item.cost;
  feed("item", `🛒 Purchased ${item.nm} for ${TEAM_BY_ID[appliedToTeam].name}.`);
  return true;
}

function lighten(hex, amt) {
  const c = hex.replace("#", "");
  const r = parseInt(c.substr(0,2), 16);
  const g = parseInt(c.substr(2,2), 16);
  const b = parseInt(c.substr(4,2), 16);
  const f = (v) => clamp(Math.round(v + (255 - v) * amt), 0, 255);
  return `rgb(${f(r)}, ${f(g)}, ${f(b)})`;
}

    $("#ticker").textContent = `Season ${LEAGUE.season} complete. Begin next season →`;
  }
}

function renderTeamPanel(side) {
  const am = activeMatchTeams();
  if (!am) return;
  const teamId = side === "A" ? am.teamA : am.teamB;
  const team = TEAM_BY_ID[teamId];
  const players = LEAGUE.rosters[teamId];

  $("#team" + side + "emoji").textContent = team.emoji;
  $("#team" + side + "name").textContent = team.name;
  const rec = LEAGUE.records[teamId];
  $("#team" + side + "record").textContent =
    `${rec.wins}–${rec.losses} · ${rec.championships} 🏆`;

  // Alive count (active match only; otherwise show roster size)
  let alive = 0;
  if (LEAGUE.match && (LEAGUE.match.teamAId === teamId || LEAGUE.match.teamBId === teamId)) {
    const list = LEAGUE.match.teamAId === teamId ? LEAGUE.match.teamAplayers : LEAGUE.match.teamBplayers;
    alive = list.filter(p => p.alive).length;
  } else {
    alive = players.filter(p => !p.deceased).length;
  }
  const aliveEl = $("#team" + side + "alive");
  aliveEl.textContent = alive;
  aliveEl.className = "alive-count " + (alive >= 4 ? "green" : alive >= 2 ? "amber" : "red");

  // Player chips
  const list = $("#team" + side + "list");
  list.innerHTML = "";
  for (const p of players) {
    const pos = POS_BY_ID[p.position];
    const chip = document.createElement("div");
    chip.className = "player-chip" + (p.deceased ? " deceased" : "");
    const liveP = LEAGUE.match
      ? (LEAGUE.match.teamAplayers.concat(LEAGUE.match.teamBplayers)).find(x => x.id === p.id)
      : null;
    // KO'd this round but still has lives → "dead" class (faded, not deceased)
    if (liveP && !liveP.alive && !p.deceased) chip.classList.add("dead");
    const hpFrac = liveP ? clamp(liveP.hp / liveP.stats.hp, 0, 1) : 1;
    const hpCls = hpFrac > 0.6 ? "" : hpFrac > 0.3 ? "med" : "low";

    // Lives pips (only show when in an active match)
    let livesHTML = "";
    if (liveP && !p.deceased && LEAGUE.match) {
      const totalLives = MATCH.startingLives;
      const remaining = liveP.lives;
      let pips = "";
      for (let i = 0; i < totalLives; i++) {
        pips += `<div class="pip${i >= remaining ? " spent" : ""}"></div>`;
      }
      livesHTML = `<div class="lives" title="${remaining}/${totalLives} lives">${pips}</div>`;
    }

    chip.innerHTML = `
      <div class="pos-icon ${pos.cls}" title="${pos.label}">${pos.icon}</div>
      <div class="name-block">
        <div class="pname">${escapeHTML(p.name)}</div>
        <div class="psub">${pos.label} · K:${p.career.kills}${livesHTML ? "" : ""}</div>
      </div>
      ${livesHTML || `<div class="hpbar ${hpCls}"><div style="width:${Math.round(hpFrac * 100)}%"></div></div>`}
    `;
    chip.addEventListener("click", () => showPlayerProfile(p));
    list.appendChild(chip);
  }
}

function renderBracketStrip() {
  const slots = ["semi1", "semi2", "final"];
  const labels = { semi1: "Semifinal 1", semi2: "Semifinal 2", final: "Final" };
  for (const slot of slots) {
    const el = $(`.bracket-match[data-slot="${slot}"]`);
    if (!el) continue;
    const m = LEAGUE.bracket[slot];
    el.classList.toggle("current", LEAGUE.currentSlot === slot);
    el.classList.toggle("done", !!m.winner);

    const aId = m.teamA, bId = m.teamB;
    const aTeam = aId ? TEAM_BY_ID[aId] : null;
    const bTeam = bId ? TEAM_BY_ID[bId] : null;
    const aClass = m.winner === aId ? "winner" : (m.winner && m.winner !== aId ? "loser" : "");
    const bClass = m.winner === bId ? "winner" : (m.winner && m.winner !== bId ? "loser" : "");

    // Score: kills if match was played, else "—"
    let aScore = "—", bScore = "—";
    if (m.winner && LEAGUE.match && LEAGUE.match.slot === slot) {
      aScore = LEAGUE.match.killsByTeam[aId] || 0;
      bScore = LEAGUE.match.killsByTeam[bId] || 0;
    }

    el.innerHTML = `
      <div class="label">${labels[slot]}</div>
      <div class="team-row ${aClass}">
        <span class="em">${aTeam?.emoji || "❓"}</span>
        <span class="nm">${aTeam?.short || "TBD"}</span>
        <span class="sc">${aScore}</span>
      </div>
      <div class="vs-divider">VS</div>
      <div class="team-row ${bClass}">
        <span class="em">${bTeam?.emoji || "❓"}</span>
        <span class="nm">${bTeam?.short || "TBD"}</span>
        <span class="sc">${bScore}</span>
      </div>
    `;
  }
}

function renderStandings() {
  const list = $("#standingsList");
  list.innerHTML = "";
  const sorted = TEAMS.slice().sort((a, b) => {
    const ra = LEAGUE.records[a.id], rb = LEAGUE.records[b.id];
    if (rb.championships !== ra.championships) return rb.championships - ra.championships;
    if (rb.wins !== ra.wins) return rb.wins - ra.wins;
    return ra.losses - rb.losses;
  });
  for (let i = 0; i < sorted.length; i++) {
    const t = sorted[i];
    const r = LEAGUE.records[t.id];
    const row = document.createElement("div");
    row.style.cssText = "display:grid;grid-template-columns:18px 1fr auto;gap:6px;padding:5px 10px;font-size:11px;border-bottom:1px solid var(--bg-2);";
    row.innerHTML = `
      <span style="color:var(--mute);">${i+1}</span>
      <span>${t.emoji} ${escapeHTML(t.short)}</span>
      <span style="font-family:'Courier New',monospace;color:var(--dim);">${r.wins}-${r.losses} · ${r.championships}🏆</span>
    `;
    list.appendChild(row);
  }
}

function renderPhasePill() {
  const phaseEl = $("#matchPhase");
  if (!phaseEl) return;
  const m = LEAGUE.match;
  const baseLabel = m
    ? ({"preRound":"Pre-round", "round":"LIVE", "betweenRounds":"INTERMISSION", "post":"POST"}[m.phase] || m.phase)
    : "Pre-match";

  // Add the autoplay countdown pill if there is one
  let pill = "";
  if (typeof AUTOPLAY_PENDING !== "undefined" && AUTOPLAY_PENDING) {
    const ms = Math.max(0, AUTOPLAY_PENDING.remainingMs);
    const sec = Math.ceil(ms / 1000);
    const urgent = sec <= 5 ? " urgent" : "";
    const labelMap = { round: "Next round", match: "Next match", season: "Next season" };
    const label = labelMap[AUTOPLAY_PENDING.kind] || "Next";
    const apOff = (typeof AUTOPLAY !== "undefined" && !AUTOPLAY.enabled) ? "(paused)" : `${sec}s`;
    pill = ` <span class="countdown-pill${urgent}" id="cdPill" title="Click to skip">${label}: ${apOff} <button class="countdown-skip" id="cdSkipBtn">skip ▸</button></span>`;
  }
  phaseEl.innerHTML = baseLabel + pill;
  // Re-bind skip button each frame (cheap)
  const skipBtn = $("#cdSkipBtn");
  if (skipBtn) skipBtn.onclick = (e) => { e.stopPropagation(); skipCountdown(); };
}

function renderRoundStrip() {
  const stripEl = $("#roundStrip");
  if (!stripEl) return;
  const m = LEAGUE.match;
  if (!m) {
    stripEl.style.display = "none";
    return;
  }
  stripEl.style.display = "";
  const pipsEl = $("#roundPips");
  pipsEl.innerHTML = "";
  // Build pips for each round in best-of-N
  // Pip filled with team color if won by that team; current round pulses gold.
  for (let i = 1; i <= m.bestOf; i++) {
    const pip = document.createElement("div");
    pip.className = "pip";
    const round = m.rounds.find(r => r.index === i);
    if (round) {
      pip.classList.add(round.winner === m.teamAId ? "won-a" : "won-b");
      pip.style.backgroundColor = TEAM_BY_ID[round.winner].color;
      pip.style.borderColor = TEAM_BY_ID[round.winner].color;
    }
    if (m.phase === "round" && i === m.currentRoundIndex) {
      pip.classList.add("current");
    }
    pipsEl.appendChild(pip);
  }
  $("#roundScore").textContent = `${m.roundsWonA}–${m.roundsWonB}`;
}

function betLabelFor(bet) {
  if (bet.kind === "team" || bet.kind === "round") {
    const t = TEAM_BY_ID[bet.target];
    return t ? `${t.emoji} ${escapeHTML(t.short)}` : "?";
  }
  if (bet.kind === "player") {
    const p = Object.values(LEAGUE.rosters).flat().find(x => x.id === bet.target);
    return p ? escapeHTML(p.name) : "?";
  }
  return "?";
}

    const roundPay = +(odds[tid] * 0.85).toFixed(2);

  const labelEl = $("#betSelectionLabel");
  const placeBtn = $("#betPagePlace");
  if (!BET_PICK.kind) {
    labelEl.textContent = "Pick a market above";
    placeBtn.disabled = true;
    return;
  }
  if (BET_PICK.kind === "team") {
    const t = TEAM_BY_ID[BET_PICK.target];
    labelEl.innerHTML = `Match Winner: ${t.emoji} ${escapeHTML(t.name)}`;
  } else if (BET_PICK.kind === "round") {
    const t = TEAM_BY_ID[BET_PICK.target];

    labelEl.innerHTML = `Last Standing: ${escapeHTML(p?.name || "?")}`;
  }
  placeBtn.disabled = false;
}

/* ============================================================
   MATCH INTERMISSION — non-blocking
   ============================================================ */
function showIntermission() {
  const m = LEAGUE.match;
  if (!m) return;
  const winner = TEAM_BY_ID[m.winnerId];
  const isFinal = m.slot === "final";
  const matchLabel = m.slot === "semi1" ? "Semifinal 1" : m.slot === "semi2" ? "Semifinal 2" : "Final";

  // Match-end toast: green if user won bets on this team
  toast(`🏆 ${winner.emoji} <b>${escapeHTML(winner.name)}</b> win ${matchLabel} ${m.roundsWonA}–${m.roundsWonB}.`, "info", { ico: "🏆", ttl: 8000 });

  if (isFinal) {
    // Season end will trigger from finalizeSeason; auto-schedule there.
    scheduleAutoAdvance("match", () => {

    });

      renderAll();
    });
  }
}

/* ============================================================
   CLOUD SYNC (Perchance uploadPlugin)
   ============================================================ */
function serializeLeague() {
  // Strip transient match-time fields (target, x, y, etc.) — they reset per match
  // and `target` causes circular references between live players.
  // Weapon/armor/item gear IS persistent (scavenged names cascade across matches).
  const cleanRosters = {};
  for (const tid of Object.keys(LEAGUE.rosters)) {
    cleanRosters[tid] = LEAGUE.rosters[tid].map(p => ({
      id: p.id,
      name: p.name,
      teamId: p.teamId,
      position: p.position,
      stats: p.stats,
      weapon: p.weapon,
      armor: p.armor,
      item: p.item,
      career: p.career,
      deceased: p.deceased,
      deceasedSeason: p.deceasedSeason,
      deceasedBy: p.deceasedBy,
      debutSeason: p.debutSeason,
    }));
  }
  return JSON.stringify({
    v: 4,
    season: LEAGUE.season,
    bits: LEAGUE.bits,
    rosters: cleanRosters,
    graveyard: LEAGUE.graveyard,
    records: LEAGUE.records,
    bracket: LEAGUE.bracket,
    currentSlot: LEAGUE.currentSlot,
    history: LEAGUE.history,
    pendingBoosts: LEAGUE.pendingBoosts,
    bets: LEAGUE.bets,
    syncName: ($("#syncName") && $("#syncName").value) || "",
  });
}

function deserializeLeague(jsonStr) {
  const data = JSON.parse(jsonStr);
  LEAGUE.season = data.season || 1;
  LEAGUE.bits = data.bits || 0;
  // Rehydrate rosters with default match-time fields
  LEAGUE.rosters = {};
  for (const tid of Object.keys(data.rosters || {})) {
    LEAGUE.rosters[tid] = data.rosters[tid].map(p => ({
      ...p,
      // Match-time fields reset on load
      x: 0, y: 0, vx: 0, vy: 0,
      hp: p.stats ? p.stats.hp : 100,
      alive: !p.deceased,
      target: null,
      attackCooldown: 0,
      facing: 1,
      stunned: 0,
      itemUsed: false,
      boostMods: null,
      balloonsPopped: 0,
      // Backfill gear if a v3 save is loaded into v4
      weapon: p.weapon || { ...(WEAPONS.find(w => w.flavorPos === p.position) || WEAPONS[0]) },
      armor:  p.armor  || { ...(ARMORS.find(a => a.flavorPos === p.position)  || ARMORS[0]) },
      item:   p.item   || { ...pick(ITEMS) },
      career: p.career
        ? { scavenged: 0, ...p.career }
        : { kills: 0, matchesPlayed: 0, matchesWon: 0, seasonsPlayed: 0, scavenged: 0 },
    }));
  }
  LEAGUE.graveyard = data.graveyard || [];
  LEAGUE.records = data.records || {};
  LEAGUE.bracket = data.bracket || null;
  LEAGUE.currentSlot = data.currentSlot || null;
  LEAGUE.history = data.history || [];
  LEAGUE.pendingBoosts = data.pendingBoosts || {};
  LEAGUE.bets = data.bets || [];
  LEAGUE.match = null;
  // Recompute _idCounter from highest player id
  let maxId = 0;
  for (const r of Object.values(LEAGUE.rosters)) {
    for (const p of r) if (p.id > maxId) maxId = p.id;
  }
  _idCounter = maxId + 1;
  if ($("#syncName") && data.syncName) $("#syncName").value = data.syncName;
}

function setSyncStatus(msg, kind) {
  const el = $("#syncStatus");
  el.textContent = msg;
  el.className = "status" + (kind ? " " + kind : "");
}

async function pushSave() {
  if (typeof uploadPlugin === "undefined") {
    setSyncStatus("uploadPlugin not loaded — top editor needs uploadPlugin import", "err");
    return;
  }
  const name = $("#syncName").value.trim();
  if (!name) {
    setSyncStatus("Pick a username first", "err");
    return;
  }
  setSyncStatus("Saving…");
  try {
    const blob = serializeLeague();
    const filename = `float-knights-${name.replace(/[^a-z0-9]/gi,"_")}.json`;
    const result = await uploadPlugin.upload(blob, filename);
    if (result.error) {
      setSyncStatus("Upload failed: " + result.error, "err");
      return;
    }
    $("#syncCode").value = result.url;
    try { localStorage.setItem("fk_syncUrl", result.url); } catch {}
    try { localStorage.setItem("fk_syncName", name); } catch {}
    setSyncStatus(`Saved (${result.size} bytes). Sync code is the URL.`, "ok");
  } catch (e) {
    setSyncStatus("Save error: " + (e.message || e), "err");
  }
}

async function pullSave() {
  const url = $("#syncCode").value.trim();
  if (!url) {
    setSyncStatus("Paste a sync code (URL) first", "err");
    return;
  }
  setSyncStatus("Loading…");
  try {
    const fetchFn = (typeof superFetch === "function") ? superFetch : fetch;
    const res = await fetchFn(url);
    const text = await res.text();
    deserializeLeague(text);
    try { localStorage.setItem("fk_syncUrl", url); } catch {}
    setSyncStatus("Loaded successfully.", "ok");
    LEAGUE.match = null;
    if (LEAGUE.currentSlot) startMatch();
    renderAll();
  } catch (e) {
    setSyncStatus("Load failed: " + (e.message || e), "err");
  }
}

   EVENT BINDINGS
   ============================================================ */
function bindEvents() {
  // Speed buttons
  $$(".controls .grp button").forEach(btn => {
    btn.addEventListener("click", () => setSpeed(parseFloat(btn.dataset.spd)));
  });

  // Skip / reset — skip fast-forwards the CURRENT round only (matches are
  // best-of-N now; user gets a chance to bet between rounds).
  $("#btnSkip").addEventListener("click", () => {
    if (!LEAGUE.match) return;
    if (LEAGUE.match.phase === "round") {
      let safety = 0;
      while (LEAGUE.match && LEAGUE.match.phase === "round" && safety < COMBAT.roundMaxFrames) {
        tickMatch();
        safety++;
      }
      renderAll();
    }
  });
  $("#btnReset").addEventListener("click", () => {
    if (!confirm("Reset the entire league? All rosters, history, and bits will be wiped.")) return;
    initLeague();
    LEAGUE.match = null;
    FEED.length = 0;
    feed("season", "🚨 League reset. New season begins.");
    startMatch();
    renderAll();
  });

  // Top-level tab navigation (Game / Bet / Store / About)
  $$(".tab-nav button").forEach(btn => {
    btn.addEventListener("click", () => switchTab(btn.dataset.tab));
  });
  // Any element with data-go-tab="..." also switches tab on click
  document.addEventListener("click", (e) => {
    const t = e.target.closest("[data-go-tab]");
    if (t) { e.preventDefault(); switchTab(t.dataset.goTab); }
  });

  // Place bet (Bet tab)
  const placeBtn = $("#betPagePlace");
  if (placeBtn) {
    placeBtn.addEventListener("click", () => {
      const amt = parseInt($("#betPageAmount").value, 10);
      if (!amt || amt <= 0) {
        setSyncStatus("Invalid bet amount", "err");
        return;
      }
      if (!BET_PICK.kind || BET_PICK.target == null) return;
      const ok = placeBet(BET_PICK.kind, BET_PICK.target, amt);
      if (ok) {
        BET_PICK = { kind: null, target: null };
        placeBtn.disabled = true;
        renderAll();
      }
    });
  }

  // Modal triggers
  $("#btnTeams").addEventListener("click", showTeams);
  $("#btnHistory").addEventListener("click", showHistory);
  $("#btnGraveyard").addEventListener("click", showGraveyard);
  $("#btnBracket").addEventListener("click", showBracket);

  // Autoplay toggle
  const apToggle = $("#autoplayToggle");
  if (apToggle) {
    apToggle.addEventListener("click", () => {
      AUTOPLAY.enabled = !AUTOPLAY.enabled;
      apToggle.classList.toggle("on", AUTOPLAY.enabled);
      toast(AUTOPLAY.enabled ? "Autoplay <b>ON</b> — game advances on its own." : "Autoplay <b>OFF</b> — countdown paused.", AUTOPLAY.enabled ? "info" : "warn");
    });
  }

  // Sync
  $("#btnPush").addEventListener("click", pushSave);
  $("#btnPull").addEventListener("click", pullSave);

  // Restore last sync URL/name from localStorage
  try {
    const u = localStorage.getItem("fk_syncUrl");
    const n = localStorage.getItem("fk_syncName");
    if (u) $("#syncCode").value = u;
    if (n) $("#syncName").value = n;
  } catch {}
}

/* ============================================================
   TOAST NOTIFICATION SYSTEM
   Bottom-right stack. Kinds: "win" (green), "lose" (red),
   "warn" / "info" (yellow / blue). Auto-dismisses after 6s.
   ============================================================ */
function toast(msg, kind, opts) {
  kind = kind || "info";
  opts = opts || {};
  const stack = $("#toastStack");
  if (!stack) return;
  const el = document.createElement("div");
  el.className = "toast " + kind;
  const icoMap = { win: "💸", lose: "💔", warn: "⚠️", info: "ℹ️" };
  el.innerHTML = `
    <span class="ico">${opts.ico || icoMap[kind] || "ℹ️"}</span>
    <div class="body">${msg}</div>
    <button class="close-x" aria-label="dismiss">×</button>
  `;
  const dismiss = () => {
    el.classList.add("fade-out");
    setTimeout(() => { if (el.parentNode) el.parentNode.removeChild(el); }, 380);
  };
  el.querySelector(".close-x").addEventListener("click", dismiss);
  stack.appendChild(el);
  const ttl = opts.ttl || 6000;
  setTimeout(dismiss, ttl);
}
// Make available to the simulation layer
window.toast = toast;

/* ============================================================
   AUTO-PLAY — countdown between rounds/matches/seasons
   ============================================================ */
const AUTOPLAY = {
  enabled: true,
  cooldownRoundMs: 60000,   // 1 minute between rounds
  cooldownMatchMs: 60000,   // 1 minute between matches
  cooldownSeasonMs: 60000,  // 1 minute between seasons
};
// Pending auto-action: {kind: "round"|"match"|"season", remainingMs, action}
let AUTOPLAY_PENDING = null;

function scheduleAutoAdvance(kind, action) {
  const cd = kind === "round"  ? AUTOPLAY.cooldownRoundMs  :
             kind === "match"  ? AUTOPLAY.cooldownMatchMs  :
                                 AUTOPLAY.cooldownSeasonMs;
  AUTOPLAY_PENDING = { kind, remainingMs: cd, totalMs: cd, action };
  renderArenaHeader();
}
function skipCountdown() {
  if (!AUTOPLAY_PENDING) return;
  const a = AUTOPLAY_PENDING.action;
  AUTOPLAY_PENDING = null;
  try { a(); } catch (e) { console.error(e); }
  renderAll();
}
function tickAutoplay(dt) {
  if (!AUTOPLAY_PENDING) return;
  if (!AUTOPLAY.enabled) return; // user turned it off — wait for click
  AUTOPLAY_PENDING.remainingMs -= dt;
  if (AUTOPLAY_PENDING.remainingMs <= 0) skipCountdown();
}

/* ============================================================
   TAB SWITCHING — Game / Bet / Store / About
   ============================================================ */
function switchTab(tabId) {
  $$(".tab-content").forEach(el => el.classList.toggle("active", el.dataset.tab === tabId));
  $$(".tab-nav button").forEach(btn => btn.classList.toggle("active", btn.dataset.tab === tabId));
  if (tabId === "bet" && typeof renderBetTab === "function") renderBetTab();
  if (tabId === "store" && typeof renderStore === "function") renderStore();
  if (tabId === "game" && typeof renderAll === "function") renderAll();
}

/* ============================================================
   MAIN GAME LOOP
   ============================================================ */
function loop(ts) {
  if (!lastTick) lastTick = ts;
  const dt = ts - lastTick;
  lastTick = ts;

  // Auto-advance countdown ticks in real-time, regardless of match state
  tickAutoplay(dt);

  if (SPEED > 0 && LEAGUE.match && LEAGUE.match.phase === "round") {
    pendingTickAcc += dt * SPEED;
    while (pendingTickAcc >= COMBAT.frameMs) {
      tickMatch();
      pendingTickAcc -= COMBAT.frameMs;
      if (!LEAGUE.match || LEAGUE.match.phase !== "round") {
        pendingTickAcc = 0;
        break;
      }
    }
  } else {
    pendingTickAcc = 0;
  }

  drawArena();
  // Lightweight UI updates each frame
  renderArenaHeader();
  // Throttled full UI render (every ~250ms)
  if (!loop._lastFull || ts - loop._lastFull > 250) {
    loop._lastFull = ts;
    renderTeamPanel("A");
    renderTeamPanel("B");
    renderHeader();
  }

  requestAnimationFrame(loop);
}
