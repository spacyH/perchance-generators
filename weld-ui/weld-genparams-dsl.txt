$meta
  title       = weld.genparams
  description = A normalized schema for image-generation parameters, so apps, weld.image, and weld.promptforge all speak one shape. Defines the full vocabulary a modern image app uses -- prompt, negativePrompt, aspect ratio, steps, cfg/guidance, sampler, seed, batch, clip-skip, strength, and a conditioning block -- with sane defaults, validation, named aspect ratios that resolve to Perchance-legal sizes, and merge/remix helpers. It does not call any backend; it produces a clean params object you hand to a generator, degrading gracefully when a backend supports less. Stash on window.weld.genparams.
  tags        = weld, genparams, image, parameters, schema, plugin

$output() =>
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.genparams) return window.weld.genparams;

  // -----------------------------------------------------------------
  // weld.genparams: one shape for "how should this image be generated".
  //
  // Every image front-end ends up with the same sprawling parameter set --
  // prompt, negative prompt, aspect ratio, steps, guidance scale, sampler,
  // seed, batch size, and so on. Apps reinvent it, weld.promptforge emits a
  // prompt with no home for the rest, and weld.image just forwards whatever
  // opts it's handed. weld.genparams gives that set a name and a default:
  //
  //   var p = weld.genparams.create({ prompt: 'a fox in snow', steps: 30 });
  //   var fixed = weld.genparams.validate(p);          // clamps + warns
  //   var forImage = weld.genparams.toImageOpts(fixed); // -> weld.image.generate
  //   var remixed = weld.genparams.remix(p, { seed: 12345, cfgScale: 9 });
  //
  // It calls NO backend. It just normalizes, so the rest of the suite agrees
  // on one schema and an app can degrade gracefully: a backend that ignores
  // cfgScale still gets a valid prompt + size, never a crash.
  // -----------------------------------------------------------------

  // Named aspect ratios -> Perchance-legal t2i sizes (512/768 grid).
  // (Perchance's text-to-image-plugin only accepts these four boxes.)
  var ASPECTS = {
    square:    { ratio: '1:1',  width: 512, height: 512 },
    portrait:  { ratio: '2:3',  width: 512, height: 768 },
    landscape: { ratio: '3:2',  width: 768, height: 512 },
    large:     { ratio: '1:1',  width: 768, height: 768 }
  };

  // The full default schema. Fields a backend doesn't understand are simply
  // ignored downstream; nothing here forces a particular generator.
  function defaults() {
    return {
      prompt: '',
      negativePrompt: '',
      aspect: 'square',          // a key of ASPECTS, or set width/height directly
      width: null,               // if set, overrides aspect
      height: null,
      steps: 25,                 // sampling steps
      cfgScale: 7,               // guidance / how hard to follow the prompt
      sampler: 'default',        // sampler/scheduler name (backend-specific)
      seed: null,                // null = random each run
      batchSize: 1,              // how many images to request
      clipSkip: 1,
      strength: null,            // img2img/denoise strength, 0..1 (null = n/a)
      style: null,               // a weld.image style preset name, optional
      conditioning: null,        // optional ControlNet-style block (see below)
      extra: null                // escape hatch for backend-specific fields
    };
  }

  // Allowed ranges for clamping. Kept deliberately generous; a backend may
  // narrow further. Each range is a two-element min/max pair.
  var RANGES = {
    steps:     [1, 150],
    cfgScale:  [1, 30],
    batchSize: [1, 8],
    clipSkip:  [1, 12],
    strength:  [0, 1]
  };

  function clampNum(v, lohi, fallback) {
    var n = Number(v);
    if (!isFinite(n)) return fallback;
    if (n < lohi[0]) return lohi[0];
    if (n > lohi[1]) return lohi[1];
    return n;
  }

  // create(overrides): a fully-populated params object (defaults + your keys).
  function create(overrides) {
    var p = defaults();
    if (overrides && typeof overrides === 'object') {
      for (var k in overrides) {
        if (Object.prototype.hasOwnProperty.call(overrides, k)) Reflect.set(p, k, Reflect.get(overrides, k));
      }
    }
    return p;
  }

  // resolveSize(p): figure out the final width/height. Explicit width+height
  // win; otherwise the named aspect resolves to a legal box.
  function resolveSize(p) {
    if (p && p.width && p.height) return { width: p.width | 0, height: p.height | 0, ratio: null };
    var a = (p && Reflect.get(ASPECTS, p.aspect)) ? Reflect.get(ASPECTS, p.aspect) : ASPECTS.square;
    return { width: a.width, height: a.height, ratio: a.ratio };
  }

  // validate(p): returns { params, warnings }. Clamps numbers into range,
  // fills missing fields, and notes anything it had to correct. Never throws
  // on bad input; it repairs and warns, because a generator call shouldn't
  // die over a too-high step count.
  function validate(input) {
    var p = create(input);
    var warnings = new Array();

    if (!String(p.prompt || '').trim()) {
      warnings.push('empty prompt -- most backends will hang or error on this');
    }

    var before;
    before = p.steps;     p.steps = clampNum(p.steps, RANGES.steps, 25);          if (p.steps !== Number(before)) warnings.push('steps clamped to ' + p.steps);
    before = p.cfgScale;  p.cfgScale = clampNum(p.cfgScale, RANGES.cfgScale, 7);  if (p.cfgScale !== Number(before)) warnings.push('cfgScale clamped to ' + p.cfgScale);
    before = p.batchSize; p.batchSize = clampNum(p.batchSize, RANGES.batchSize, 1) | 0; if (p.batchSize !== (Number(before) | 0)) warnings.push('batchSize clamped to ' + p.batchSize);
    before = p.clipSkip;  p.clipSkip = clampNum(p.clipSkip, RANGES.clipSkip, 1) | 0; if (p.clipSkip !== (Number(before) | 0)) warnings.push('clipSkip clamped to ' + p.clipSkip);

    if (p.strength != null) {
      before = p.strength; p.strength = clampNum(p.strength, RANGES.strength, null);
      if (p.strength !== Number(before)) warnings.push('strength clamped to ' + p.strength);
    }

    if (p.seed != null) {
      var s = Number(p.seed);
      p.seed = isFinite(s) ? (s | 0) : null;
      if (p.seed == null) warnings.push('seed was not a number -- set to random (null)');
    }

    if (!p.width && !p.height && !Reflect.get(ASPECTS, p.aspect)) {
      warnings.push('unknown aspect "' + p.aspect + '" -- defaulting to square');
      p.aspect = 'square';
    }

    if (p.conditioning) {
      var c = normalizeConditioning(p.conditioning);
      if (!c) { warnings.push('conditioning block was malformed -- dropped'); p.conditioning = null; }
      else p.conditioning = c;
    }

    return { params: p, warnings: warnings };
  }

  // Optional ControlNet-style conditioning. Normalized but never required.
  function normalizeConditioning(c) {
    if (!c || typeof c !== 'object') return null;
    if (!c.image && !c.type) return null;
    return {
      type: String(c.type || 'reference'),     // pose | depth | canny | reference | ...
      image: c.image || null,                  // url/dataURL the backend understands
      strength: c.strength != null ? clampNum(c.strength, [0, 2], 1) : 1,
      scale: c.scale != null ? Number(c.scale) : 1,
      endPercent: c.endPercent != null ? clampNum(c.endPercent, [0, 1], 1) : 1
    };
  }

  // merge(base, overrides): shallow merge producing a NEW params object.
  function merge(base, overrides) {
    var p = create(base);
    if (overrides && typeof overrides === 'object') {
      for (var k in overrides) {
        if (Object.prototype.hasOwnProperty.call(overrides, k)) Reflect.set(p, k, Reflect.get(overrides, k));
      }
    }
    return p;
  }

  // remix(base, overrides): like merge, but tuned for "take this generation
  // and tweak it"; by default it randomizes the seed (a remix should look
  // different) unless you pin one in the overrides.
  function remix(base, overrides) {
    var p = merge(base, overrides);
    var pinnedSeed = overrides && Object.prototype.hasOwnProperty.call(overrides, 'seed');
    if (!pinnedSeed) p.seed = randomSeed();
    return p;
  }

  function randomSeed() { return (Math.floor(Math.random() * 2147483647)) | 0; }

  // toImageOpts(p): produce the opts object weld.image.generate expects.
  // Resolves the size, maps names, and forwards the rest so a richer backend
  // can pick up what it understands. weld.image only needs prompt+resolution,
  // so this always yields something it can run.
  function toImageOpts(input) {
    var v = validate(input);
    var p = v.params;
    var size = resolveSize(p);
    var opts = {
      prompt: p.prompt,
      negativePrompt: p.negativePrompt || undefined,
      resolution: size.width + 'x' + size.height,
      width: size.width,
      height: size.height,
      steps: p.steps,
      cfgScale: p.cfgScale,
      guidanceScale: p.cfgScale,     // alias many backends use
      sampler: p.sampler !== 'default' ? p.sampler : undefined,
      seed: p.seed != null ? p.seed : undefined,
      batchSize: p.batchSize,
      clipSkip: p.clipSkip,
      strength: p.strength != null ? p.strength : undefined,
      style: p.style || undefined
    };
    if (p.conditioning) opts.conditioning = p.conditioning;
    if (p.extra && typeof p.extra === 'object') {
      for (var k in p.extra) { if (Object.prototype.hasOwnProperty.call(p.extra, k)) Reflect.set(opts, k, Reflect.get(p.extra, k)); }
    }
    return opts;
  }

  // A compact human-readable description of a params set, for UIs/tooltips.
  function describe(input) {
    var p = create(input);
    var size = resolveSize(p);
    var bits = new Array();
    bits.push(size.width + 'x' + size.height + (size.ratio ? ' (' + size.ratio + ')' : ''));
    bits.push(p.steps + ' steps');
    bits.push('cfg ' + p.cfgScale);
    if (p.seed != null) bits.push('seed ' + p.seed);
    if (p.batchSize > 1) bits.push('x' + p.batchSize);
    if (p.style) bits.push('style: ' + p.style);
    if (p.conditioning) bits.push('+' + p.conditioning.type);
    return bits.join(' / ');
  }

  function aspectNames() { return Object.keys(ASPECTS); }

  window.weld.genparams = {
    defaults: defaults,
    create: create,
    validate: validate,
    merge: merge,
    remix: remix,
    toImageOpts: toImageOpts,
    resolveSize: resolveSize,
    describe: describe,
    aspects: ASPECTS,
    aspectNames: aspectNames,
    ranges: RANGES,
    randomSeed: randomSeed,
    version: '1.0.0'
  };

  return window.weld.genparams;
