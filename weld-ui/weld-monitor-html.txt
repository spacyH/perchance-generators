<h2>What it is</h2>
<p>weld.monitor is a repeatable regression-test suite for the Perchance platform itself. It runs about a hundred assertions across eight groups and rolls the results into a pass / fail / drift / skip tally plus a stable <strong>fingerprint</strong>. When something breaks in your app, it answers the first question instantly: <em>is it me, Perchance, or the wider internet?</em></p>
<p>It is exposed as a plugin so any app can self-check &mdash; call <code>run()</code> and render the results however you like. The <a href="https://perchance.org/weld-monitor-example">live example</a> is one such UI.</p>
<pre data-codeid="install"></pre>

<h2>run(opts)</h2>
<div class="sig" data-codeid="sig_run"></div>
<p>Runs all eight groups in order and resolves with a report. Results stream as they land via the callbacks, so a UI can render progressively:</p>
<pre data-codeid="run_ex"></pre>
<p>The resolved <strong>report</strong> is <code>&#123; suite, version, startedAt, finishedAt, userAgent, counts, fingerprint, tests, infrastructure &#125;</code>. Each test row is <code>&#123; group, name, status, expected, actual, note &#125;</code>.</p>

<h2>The eight groups</h2>
<p>Ordered from "works even if Perchance is down" to "exercises the heaviest backends," so a failure's position localizes the problem.</p>
<table>
<tr><th>#</th><th>group</th><th>pins</th></tr>
<tr><td>0</td><td><code>infra</code></td><td>Cloudflare / GitHub status (proxied and direct) &mdash; an upstream outage vs a Perchance bug. Kept out of the fingerprint.</td></tr>
<tr><td>1</td><td><code>api</code></td><td>the <code>perchance.org/api/*</code> + fileInfo endpoints return expected shapes.</td></tr>
<tr><td>2</td><td><code>env</code></td><td>sandbox posture and the identity globals (<code>generatorPublicId</code>, etc.).</td></tr>
<tr><td>3</td><td><code>root</code></td><td>the proxy semantics, the DSL list handle, the one-way function bridge, a kv round-trip.</td></tr>
<tr><td>4</td><td><code>upload</code></td><td>the file lifecycle: upload, read back, metadata, delete.</td></tr>
<tr><td>5</td><td><code>fetch</code></td><td>superFetch handles GET/POST/headers/redirects/status correctly.</td></tr>
<tr><td>6</td><td><code>text</code></td><td>the ai-text invariants: 6000-token budget, the boxed-String trap, stopReason vocab, the 20-vs-21 stop-sequence edge, serial broker.</td></tr>
<tr><td>7</td><td><code>image</code></td><td>text-to-image: sync vs async, boxed result, supported resolutions.</td></tr>
</table>
<div class="note">Each group pins a <strong>contract</strong> that Weld plugins assume &mdash; <code>weld.stream</code> on the stopReason vocab and serial broker, <code>weld.image</code> on the four resolutions, <code>weld.tokens</code> on the 6000-token budget. The monitor is the executable proof those assumptions still hold. See the <a href="https://perchance.org/weld-platform">Platform reference</a> for the full architecture.</div>

<h2>Drift, skip &amp; the fingerprint</h2>
<div class="sig" data-codeid="sig_fp"></div>
<p>A hard mismatch is a <code>FAIL</code> (the contract broke); a soft one is a <code>DRIFT</code> (something changed but didn't necessarily break); a <code>SKIP</code> records a probe that couldn't run. At the end, every <code>group:name=status</code> is hashed (FNV-1a) into a short hex <strong>fingerprint</strong> &mdash; so you compare one string against the last known-good run instead of diffing a hundred rows. Infrastructure is excluded from the fingerprint by design, so a Cloudflare blip doesn't change the hash that represents Perchance's behavior.</p>

  }else if(proxyOk>0 && directOk>0){
    diag='ALL CLEAR — both Cloudflare proxy and direct (non-CF) paths are working.';
  }else if(proxyOk>0 && directFail>0){
    diag='PARTIAL — CF proxy works but some direct fetches failed. Non-CF CDNs may have issues.';
  }else{
    diag='INCONCLUSIVE — not enough data points to determine infrastructure state.';
  }
  var diagStatus=(proxyFail>0||directFail>1)?'DRIFT':'PASS';
  infraRec('DIAGNOSTIC',diagStatus,'ALL CLEAR',diag,
    'proxy: '+proxyOk+' ok / '+proxyFail+' fail | direct: '+directOk+' ok / '+directFail+' fail');
}

  }catch(e){assertSkip('fetch','cache','failed');}
  // Redirect
  try{var rr=await T(sf('https://httpbin.org/redirect-to?url='+encodeURIComponent('https://httpbin.org/get')+'&status_code=302'),20000,'redir');
    assertEq('fetch','302 redirect → 200',200,rr.status);
  }catch(e){assertSkip('fetch','redirect','failed');}
  // v3: data: URL
  try{var dr=await T(sf('data:text/plain;charset=utf-8,hello%20world'),10000,'data');
    var dt=await dr.text();
    assertEq('fetch','data: URL → "hello world"','hello world',dt);
  }catch(e){assertSkip('fetch','data: URL','failed');}
  // v3: status passthrough (418)
  try{var tr=await T(sf('https://httpbin.org/status/418'),15000,'418');
    assertEq('fetch','status 418 passthrough',418,tr.status);
  }catch(e){assertSkip('fetch','418','failed');}
}

// FNV-1a fingerprint over group:name=status (excludes infra by design).
function fingerprint(tests) {
  var sig = (tests || TESTS || []).map(function (t) { return t.group + ':' + t.name + '=' + t.status; }).join('|');
  return fnv(sig);
}

var RUNNING = false;
function run(opts) {
  if (RUNNING) return Promise.reject(new Error('weld.monitor: a run is already in progress'));
  RUNNING = true;
  SINK = opts || {};
  TESTS = []; COUNTS = { PASS: 0, FAIL: 0, DRIFT: 0, SKIP: 0 };
  INFRA = []; ICOUNTS = { PASS: 0, FAIL: 0, DRIFT: 0, SKIP: 0 };
  var startedAt = new Date().toISOString();
  var groups = [ ['infra', testInfra], ['api', testApi], ['env', testEnv], ['root', testRoot],
                ['upload', testUpload], ['fetch', testFetch], ['text', testText], ['image', testImage]];

  function step(i) {
    if (i >= groups.length) return Promise.resolve();
    return Promise.resolve().then(function () { return groups[i][1](); })
      .catch(function (e) { rec(groups[i][0], 'orchestrator', 'FAIL', 'no-throw', String((e && e.message) || e), null); })
      .then(function () { return step(i + 1); });
  }

  return step(0).then(function () {
    setProg(100, 'complete');
    var fp = fingerprint(TESTS);
    var report = {
      suite: 'weld.monitor', version: '1.0.0',
      startedAt: startedAt, finishedAt: new Date().toISOString(),
      userAgent: (typeof navigator !== 'undefined') ? navigator.userAgent : '',
      counts: COUNTS, fingerprint: fp,
      tests: TESTS, infrastructure: { counts: ICOUNTS, checks: INFRA }
    };
    RUNNING = false;
    return report;
  }, function (e) { RUNNING = false; throw e; });
}

var GROUPS = [
  { id: 'infra',  title: 'Infrastructure Status', note: 'separate from platform fingerprint' },
  { id: 'api',    title: 'Public HTTP API' },
  { id: 'env',    title: 'Environment' },
  { id: 'root',   title: 'root Proxy & DSL Bridge' },
  { id: 'upload', title: 'upload-plugin' },
  { id: 'fetch',  title: 'superFetch' },
  { id: 'text',   title: 'ai-text-plugin', note: 'slow — multiple generations' },
  { id: 'image',  title: 'text-to-image-plugin', note: 'slowest — multiple images' }
];

window.weld.monitor = {
  run: run,
  fingerprint: fingerprint,
  isRunning: function () { return RUNNING; },
  groups: GROUPS,
  version: '1.0.0'
};

return window.weld.monitor;
})();
