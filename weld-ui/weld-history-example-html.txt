// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

function input(opts) {
  opts = opts || {};
  return el('input', {
    class: 'w-input',
    type: opts.type || 'text',

    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
    readonly: opts.readonly ? 'readonly' : null,
  });
}

function select(opts) {
  opts = opts || {};
  var options = (opts.options || []).map(function(o){
    if (typeof o === 'string') o = { value: o, label: o };
    return el('option', {
      value: o.value,
      selected: o.value === opts.value ? 'selected' : null,
    }, o.label);
  });
  return el('select', {
    class: 'w-select',
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
  }, options);
}

// weld.persist: Collection-based CRUD over IndexedDB.
// One database, one object store, logical collections via a compound
// key of (collection + id). No schema versioning needed when collections
// are added (they're just key prefixes). No Dexie dependency.
//
// Usage:
//   var db = weld.persist.open('my-app');
//   var chars = db.collection('characters');
//   await chars.put({ id: 'c1', name: 'Alice' });
//   var alice = await chars.get('c1');
//   var all = await chars.getAll();
//   await chars.delete('c1');

var STORE_NAME = 'records';
var DB_VERSION = 1;

// Cache of opened databases: dbName -> Promise<IDBDatabase>
var dbCache = {};

function openDatabase(dbName) {
  if (dbCache[dbName]) return dbCache[dbName];
  dbCache[dbName] = new Promise(function(resolve, reject) {
    var request = indexedDB.open(dbName, DB_VERSION);
    request.onupgradeneeded = function(e) {
      var db = e.target.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        var store = db.createObjectStore(STORE_NAME, { keyPath: ['_col', '_id'] });
        store.createIndex('by_col', '_col', { unique: false });
      }
    };
    request.onsuccess = function(e) { resolve(e.target.result); };
    request.onerror = function(e) {
      delete dbCache[dbName];
      reject(new Error('weld.persist: failed to open "' + dbName + '": ' + (e.target.error || 'unknown')));
    };
  });
  return dbCache[dbName];
}

function uid() {
  return Date.now().toString(36) + '_' + Math.random().toString(36).slice(2, 8);
}

// -----------------------------------------------------------------
// Collection handle
// -----------------------------------------------------------------
function makeCollection(dbName, colName) {
  function getStore(mode) {
    return openDatabase(dbName).then(function(db) {
      var tx = db.transaction(STORE_NAME, mode);
      return tx.objectStore(STORE_NAME);
    });
  }

  function wrapRequest(req) {
    return new Promise(function(resolve, reject) {
      req.onsuccess = function() { resolve(req.result); };
      req.onerror = function() { reject(req.error); };
    });
  }

  // Put: upsert a record. If record has no .id, one is generated.
  function put(record) {
    if (!record || typeof record !== 'object') {
      return Promise.reject(new Error('persist.put: record must be an object'));
    }
    var rec = Object.assign({}, record);
    if (rec.id == null) rec.id = uid();
    rec._col = colName;
    rec._id = String(rec.id);
    return getStore('readwrite').then(function(store) {
      return wrapRequest(store.put(rec));
    }).then(function() {
      return rec.id;
    });
  }

  // Put many records in a single transaction
  function putMany(records) {
    if (!Array.isArray(records)) return Promise.reject(new Error('persist.putMany: expected array'));
    return openDatabase(dbName).then(function(db) {
      return new Promise(function(resolve, reject) {
        var tx = db.transaction(STORE_NAME, 'readwrite');
        var store = tx.objectStore(STORE_NAME);
        var ids = [];
        for (var i = 0; i < records.length; i++) {
          var rec = Object.assign({}, records[i]);
          if (rec.id == null) rec.id = uid();
          rec._col = colName;
          rec._id = String(rec.id);
          ids.push(rec.id);
          store.put(rec);
        }
        tx.oncomplete = function() { resolve(ids); };
        tx.onerror = function() { reject(tx.error); };
      });
    });
  }

  // Get a single record by id
  function get(id) {
    return getStore('readonly').then(function(store) {
      return wrapRequest(store.get([colName, String(id)]));
    }).then(function(rec) {
      if (!rec) return null;
      var out = Object.assign({}, rec);
      delete out._col;
      delete out._id;
      return out;
    });
  }

  // Get all records in this collection
  function getAll() {
    return getStore('readonly').then(function(store) {
      var index = store.index('by_col');
      return wrapRequest(index.getAll(colName));
    }).then(function(recs) {
      return (recs || []).map(function(rec) {
        var out = Object.assign({}, rec);
        delete out._col;
        delete out._id;
        return out;
      });
    });
  }

  // Query: get all records matching a filter function
  function query(filterFn) {
    return getAll().then(function(all) {
      if (typeof filterFn !== 'function') return all;
      return all.filter(filterFn);
    });
  }

  // Delete a single record by id
  function del(id) {
    return getStore('readwrite').then(function(store) {
      return wrapRequest(store.delete([colName, String(id)]));
    }).then(function() { return true; });
  }

  // Delete many records by id array
  function delMany(ids) {
    if (!Array.isArray(ids)) return Promise.reject(new Error('persist.delMany: expected array'));
    return openDatabase(dbName).then(function(db) {
      return new Promise(function(resolve, reject) {
        var tx = db.transaction(STORE_NAME, 'readwrite');
        var store = tx.objectStore(STORE_NAME);
        for (var i = 0; i < ids.length; i++) {
          store.delete([colName, String(ids[i])]);
        }
        tx.oncomplete = function() { resolve(ids.length); };
        tx.onerror = function() { reject(tx.error); };
      });
    });
  }

  // Clear all records in this collection
  function clear() {
    return getAll().then(function(all) {
      var ids = all.map(function(r) { return r.id; });
      return delMany(ids);
    });
  }

  // Count records in this collection
  function count() {
    return getStore('readonly').then(function(store) {
      var index = store.index('by_col');
      return wrapRequest(index.count(colName));
    });
  }

  // Update: merge a patch into an existing record
  function update(id, patch) {
    return get(id).then(function(existing) {
      if (!existing) return Promise.reject(new Error('persist.update: record not found: ' + id));
      var merged = Object.assign({}, existing, patch, { id: id });
      return put(merged);
    });
  }

  return {
    put: put,
    putMany: putMany,
    get: get,
    getAll: getAll,
    query: query,
    delete: del,
    deleteMany: delMany,
    clear: clear,
    count: count,
    update: update
  };
}

            delete out._id;

      });
    });
  }

  function importAll(data) {

      return Promise.reject(new Error('persist.importAll: invalid data'));
    }
    var colNames = Object.keys(data.collections);
    var chain = Promise.resolve();
    for (var i = 0; i < colNames.length; i++) {

        var records = data.collections[colName];
        chain = chain.then(function() {
          return collection(colName).putMany(records);
        });
      })(colNames[i]);
    }
    return chain;
  }

  // Delete the entire database
  function destroy() {
    delete dbCache[dbName];
    return new Promise(function(resolve, reject) {
      var req = indexedDB.deleteDatabase(dbName);
      req.onsuccess = function() { resolve(true); };
      req.onerror = function() { reject(req.error); };
    });
  }

  return {
    collection: collection,
    collections: collections,
    exportAll: exportAll,
    importAll: importAll,
    destroy: destroy,
    name: dbName
  };
}

function uid() {

// -----------------------------------------------------------------
// Tree instance factory
// -----------------------------------------------------------------
function createTree(opts) {
  opts = opts || {};
  var storage = opts.storage;
  var storageKey = opts.storageKey || 'weld-branching';

  // Internal state
  var nodes = {};      // id -> node
  var rootIds = [];    // ordered list of root node ids
  var activeId = null; // currently active leaf

  function save() {
    if (!storage) return;
    try {
      storage.setItem(storageKey, JSON.stringify({
        nodes: nodes,
        rootIds: rootIds,
        activeId: activeId
      }));
    } catch (e) { /* swallow */ }
  }

  function load() {
    if (!storage) return false;
    try {
      var raw = storage.getItem(storageKey);
      if (!raw) return false;
      var parsed = JSON.parse(raw);
      if (parsed && typeof parsed === 'object') {
        nodes = parsed.nodes || {};
        rootIds = parsed.rootIds || [];
        activeId = parsed.activeId || null;
        return true;
      }
    } catch (e) { /* swallow */ }
    return false;
  }

  // Try loading initial state
  load();

  // -----------------------------------------------------------------
  // Core: append a node
  // -----------------------------------------------------------------
  function append(data, parentId) {
    if (data === undefined) data = null;
    if (parentId === undefined) {
      // Default parent: the currently active leaf, if any
      parentId = activeId;
    }
    var id = uid();
    var node = {
      id: id,
      parentId: parentId || null,
      childIds: [],
      data: data,
      createdAt: Date.now()
    };
    nodes[id] = node;

    if (parentId && nodes[parentId]) {
      nodes[parentId].childIds.push(id);
    } else {
      // New root
      rootIds.push(id);
    }
    activeId = id;
    save();
    return id;
  }

  // -----------------------------------------------------------------
  // Branch -- explicit alias for append with parent
  // -----------------------------------------------------------------
  function branch(fromId, data) {
    if (!nodes[fromId]) throw new Error('branching: unknown node ' + fromId);
    var parentId = nodes[fromId].parentId;
    var id = append(data, parentId);
    return id;
  }

  // -----------------------------------------------------------------
  // Get / list
  // -----------------------------------------------------------------
  function get(id) {
    var n = nodes[id];
    if (!n) return null;
    return {
      id: n.id,
      parentId: n.parentId,
      childIds: n.childIds.slice(),
      data: n.data,
      createdAt: n.createdAt
    };
  }

  function getAll() {
    var out = [];
    for (var id in nodes) out.push(get(id));
    return out;
  }

  function getRoots() {
    return rootIds.map(get);
  }

  // -----------------------------------------------------------------
  // Path: from root to a given node (or active leaf)
  // -----------------------------------------------------------------
  function path(toId) {
    var target = toId || activeId;
    if (!target || !nodes[target]) return [];
    var chain = [];
    var cursor = nodes[target];
    while (cursor) {
      chain.unshift(get(cursor.id));
      cursor = cursor.parentId ? nodes[cursor.parentId] : null;
    }
    return chain;
  }

  function activeBranch() {
    return path(activeId);
  }

  function setActive(id) {
    if (!nodes[id]) throw new Error('branching: unknown node ' + id);
    activeId = id;
    save();
    return id;
  }

  function getActiveId() {
    return activeId;
  }

  // -----------------------------------------------------------------
  // Siblings -- nodes with the same parent
  // -----------------------------------------------------------------
  function siblings(id) {
    var n = nodes[id];
    if (!n) return [];
    var parentChildren;
    if (n.parentId && nodes[n.parentId]) {
      parentChildren = nodes[n.parentId].childIds;
    } else {
      parentChildren = rootIds;
    }
    return parentChildren.map(get);
  }

  // -----------------------------------------------------------------
  // Switching to a sibling -- pick by index relative to current
  // -----------------------------------------------------------------
  function switchSibling(id, direction) {
    var sibs = siblings(id);
    if (sibs.length === 0) return null;
    var idx = sibs.findIndex(function(s) { return s.id === id; });
    if (idx === -1) return null;
    var newIdx;
    if (direction === 'next' || direction === 1) {
      newIdx = (idx + 1) % sibs.length;
    } else if (direction === 'prev' || direction === -1) {
      newIdx = (idx - 1 + sibs.length) % sibs.length;
    } else if (typeof direction === 'number') {
      newIdx = Math.max(0, Math.min(sibs.length - 1, direction));
    } else {
      return null;
    }
    return sibs[newIdx].id;
  }

  // -----------------------------------------------------------------
  // Switch branch by selecting a node; updates active to that node's deepest descendant
  // along its primary (first-child) path
  // -----------------------------------------------------------------
  function switchTo(id) {
    if (!nodes[id]) throw new Error('branching: unknown node ' + id);
    // Walk down following first child until leaf
    var cursor = nodes[id];
    while (cursor.childIds.length > 0) {
      cursor = nodes[cursor.childIds[0]];
    }
    activeId = cursor.id;
    save();
    return activeId;
  }

  // -----------------------------------------------------------------
  // Remove a node and all its descendants
  // -----------------------------------------------------------------
  function remove(id) {
    var n = nodes[id];
    if (!n) return false;
    // Collect descendants
    var toRemove = [];
    var stack = [id];
    while (stack.length) {
      var cur = stack.pop();
      toRemove.push(cur);
      var node = nodes[cur];
      if (node) {
        for (var i = 0; i < node.childIds.length; i++) stack.push(node.childIds[i]);
      }
    }
    // Detach from parent
    if (n.parentId && nodes[n.parentId]) {
      nodes[n.parentId].childIds = nodes[n.parentId].childIds.filter(function(c) { return c !== id; });
    } else {
      rootIds = rootIds.filter(function(r) { return r !== id; });
    }
    // Delete all
    for (var k = 0; k < toRemove.length; k++) delete nodes[toRemove[k]];
    // Reset active if needed
    if (toRemove.indexOf(activeId) !== -1) {
      activeId = (n.parentId && nodes[n.parentId]) ? n.parentId : (rootIds[0] || null);
    }
    save();
    return true;
  }

  function clear() {
    nodes = {};
    rootIds = [];
    activeId = null;
    save();
  }

  // -----------------------------------------------------------------
  // Serialize / deserialize
  // -----------------------------------------------------------------
  function serialize() {
    return JSON.parse(JSON.stringify({
      nodes: nodes,
      rootIds: rootIds,
      activeId: activeId
    }));
  }

  function deserialize(state) {
    if (!state || typeof state !== 'object') throw new Error('branching: invalid state');
    nodes = state.nodes ? JSON.parse(JSON.stringify(state.nodes)) : {};
    rootIds = state.rootIds ? state.rootIds.slice() : [];
    activeId = state.activeId || null;
    save();
  }

  function size() {
    var n = 0;
    for (var k in nodes) n++;
    return n;
  }

  function isLeaf(id) {
    var n = nodes[id];
    return n && n.childIds.length === 0;
  }

  return {
    append: append,
    branch: branch,
    get: get,
    getAll: getAll,
    getRoots: getRoots,
    path: path,
    activeBranch: activeBranch,
    setActive: setActive,
    getActiveId: getActiveId,
    siblings: siblings,
    switchSibling: switchSibling,
    switchTo: switchTo,
    remove: remove,
    clear: clear,
    serialize: serialize,
    deserialize: deserialize,
    size: size,
    isLeaf: isLeaf
  };
}

window.weld.branching = {
  create: createTree,
  version: '1.0.0'

      try { changeListeners[i](event, data); } catch (e) { /* swallow */ }
    }
  }

  function scheduleSave() {
    if (!autoSave || !activeSessionId) return;
    if (saveTimer) clearTimeout(saveTimer);
    saveTimer = setTimeout(function() {
      saveTimer = null;
      save().catch(function(e) {
        console.error('[weld.history] auto-save failed:', e);
      });
    }, autoSaveDebounce);
  }

  // -------- Create a new session --------
  function create(meta) {
    meta = meta || {};
    var id = meta.id || uid();
    var now = Date.now();
    var session = {
      id: id,
      title: meta.title || 'New conversation',
      characterId: meta.characterId || null,
      characterName: meta.characterName || null,
      createdAt: now,
      updatedAt: now,
      messageCount: 0,
      metadata: meta.metadata || null
    };

    // Initialize tree or message array
    if (hasBranching()) {
      activeTree = window.weld.branching.create();
      activeMessages = null;
    } else {
      activeTree = null;
      activeMessages = [];
    }
    activeSessionId = id;

    return sessions.put(session).then(function() {
      return saveTree();
    }).then(function() {
      emit('create', { sessionId: id });
      return id;
    });
  }

  // -------- Load a session --------
  function load(sessionId) {
    activeSessionId = null;
    activeTree = null;
    activeMessages = null;

    return sessions.get(sessionId).then(function(session) {
      if (!session) throw new Error('weld.history: session not found: ' + sessionId);
      return trees.get(sessionId).then(function(treeData) {
        activeSessionId = sessionId;
        if (treeData && treeData.tree && hasBranching()) {
          activeTree = window.weld.branching.create();
          activeTree.deserialize(treeData.tree);
          activeMessages = null;
        } else if (treeData && treeData.messages) {
          activeTree = null;
          activeMessages = treeData.messages;
        } else {
          // Empty session
          if (hasBranching()) {
            activeTree = window.weld.branching.create();
            activeMessages = null;
          } else {
            activeTree = null;
            activeMessages = [];
          }
        }
        // Update view time
        return sessions.update(sessionId, { updatedAt: Date.now() });
      });
    }).then(function() {
      emit('load', { sessionId: sessionId });
      return sessionId;
    });
  }

  // -------- Save the active session --------
  function save() {
    if (!activeSessionId) return Promise.reject(new Error('weld.history: no active session'));
    var treeData = { id: activeSessionId };
    if (activeTree) {
      treeData.tree = activeTree.serialize();
      treeData.messages = null;
    } else {
      treeData.tree = null;
      treeData.messages = activeMessages ? activeMessages.slice() : [];
    }
    var msgCount = activeTree ? activeTree.size() : (activeMessages ? activeMessages.length : 0);
    return Promise.all([
      trees.put(treeData),
      sessions.update(activeSessionId, {
        updatedAt: Date.now(),
        messageCount: msgCount
      })
    ]).then(function() {
      emit('save', { sessionId: activeSessionId });
    });
  }

  function saveTree() {
    var treeData = { id: activeSessionId };
    if (activeTree) {
      treeData.tree = activeTree.serialize();
      treeData.messages = null;
    } else {
      treeData.tree = null;
      treeData.messages = activeMessages ? activeMessages.slice() : [];
    }
    return trees.put(treeData);
  }

  // -------- Append a message to the active session --------
  function append(message) {
    if (!activeSessionId) throw new Error('weld.history: no active session');
    if (activeTree) {
      var id = activeTree.append(message);
      scheduleSave();
      emit('append', { sessionId: activeSessionId, nodeId: id });
      return id;
    } else {
      if (!message.id) message.id = uid();
      activeMessages.push(message);
      scheduleSave();
      emit('append', { sessionId: activeSessionId, messageId: message.id });
      return message.id;
    }
  }

  // -------- Get the active conversation --------
  function getMessages() {
    if (!activeSessionId) return [];
    if (activeTree) {
      return activeTree.activeBranch().map(function(n) { return n.data; });
    }
    return activeMessages ? activeMessages.slice() : [];
  }

  function getTree() {
    return activeTree;
  }

  // -------- List sessions --------
  function list(listOpts) {
    listOpts = listOpts || {};
    return sessions.getAll().then(function(all) {
      // Filter
      if (listOpts.characterId) {
        all = all.filter(function(s) { return s.characterId === listOpts.characterId; });
      }
      if (listOpts.search) {
        var q = String(listOpts.search).toLowerCase();
        all = all.filter(function(s) {
          var t = s.title ? s.title.toLowerCase() : '';
          var cn = s.characterName ? s.characterName.toLowerCase() : '';
          return t.indexOf(q) !== -1 || cn.indexOf(q) !== -1;
        });
      }
      // Sort by updatedAt descending (most recent first)
      all.sort(function(a, b) { return (b.updatedAt || 0) - (a.updatedAt || 0); });
      // Limit
      if (listOpts.limit && listOpts.limit > 0) {
        all = all.slice(0, listOpts.limit);
      }
      return all;
    });
  }

  // -------- Session metadata operations --------
  function rename(sessionId, title) {
    return sessions.update(sessionId, { title: String(title), updatedAt: Date.now() }).then(function() {
      emit('rename', { sessionId: sessionId, title: title });
    });
  }

  function getMeta(sessionId) {
    return sessions.get(sessionId || activeSessionId);
  }

  function updateMeta(sessionId, patch) {
    return sessions.update(sessionId, Object.assign({}, patch, { updatedAt: Date.now() }));
  }

  // -------- Delete --------
  function del(sessionId) {
    if (sessionId === activeSessionId) {
      activeSessionId = null;
      activeTree = null;
      activeMessages = null;
    }
    return Promise.all([
      sessions.delete(sessionId),
      trees.delete(sessionId)
    ]).then(function() {
      emit('delete', { sessionId: sessionId });
      return true;
    });
  }

  function clear() {
    activeSessionId = null;
    activeTree = null;
    activeMessages = null;
    return Promise.all([
      sessions.clear(),
      trees.clear()
    ]).then(function() {
      emit('clear', {});
    });
  }

  // -------- Listeners --------
  function onChange(fn) {
    changeListeners.push(fn);
    return function() {
      var idx = changeListeners.indexOf(fn);
      if (idx >= 0) changeListeners.splice(idx, 1);
    };
  }

  function getActiveId() {
    return activeSessionId;
  }

  function isActive() {
    return activeSessionId !== null;
  }

  return {
    create: create,
    load: load,
    save: save,
    append: append,
    getMessages: getMessages,
    getTree: getTree,
    list: list,
    rename: rename,
    getMeta: getMeta,
    updateMeta: updateMeta,
    delete: del,
    clear: clear,
    onChange: onChange,
    getActiveId: getActiveId,
    isActive: isActive
  };
}

window.weld.history = {
  create: createHistory,
  version: '1.0.0'
};

})();
</script>
