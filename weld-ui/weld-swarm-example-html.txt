<script>
(function () {
  try { if (typeof root !== 'undefined' && typeof root.weldSkybridge === 'function') root.weldSkybridge(); } catch (e) {}
  try { if (typeof root !== 'undefined' && typeof root.weldSync === 'function') root.weldSync(); } catch (e) {}
  try { if (typeof root !== 'undefined' && typeof root.weldCrew === 'function') root.weldCrew(); } catch (e) {}
  try { if (typeof root !== 'undefined' && typeof root.weldSwarm === 'function') root.weldSwarm(); } catch (e) {}

  function $(id) { return document.getElementById(id); }
  var SW = (window.weld && window.weld.swarm) || null;
  var CR = (window.weld && window.weld.crew) || null;
  var sb = (window.weld && window.weld.skybridge) || null;

  var MODES = [
    { id: 'local', label: 'Local demo (driver + workers in this tab)' },
    { id: 'worker', label: 'This tab: Worker (serve tasks)' },
    { id: 'user', label: 'This tab: User / Driver (dispatch to other tabs)' }
  ];
  var PRESETS = [
    { label: 'Three tavern names', tasks: ['Invent a fantasy tavern name', 'Invent a second tavern name', 'Invent a third tavern name'] },
    { label: 'Summarize three topics', tasks: ['Summarize the topic: dragons', 'Summarize the topic: alchemy', 'Summarize the topic: sea trade'] },
    { label: 'Roll four initiatives', tasks: ['Roll initiative for the rogue', 'Roll initiative for the mage', 'Roll initiative for the fighter', 'Roll initiative for the cleric'] }
  ];

  var t0 = 0;
  function feedLine(kind, text) {
    var el = document.createElement('span'); el.className = 'l';
    var ts = document.createElement('span'); ts.className = 'ts'; ts.textContent = t0 ? ('+' + String(Date.now() - t0) + 'ms ') : '';
    var k = document.createElement('span'); k.className = 'k-' + kind; k.textContent = kind.toUpperCase() + ' ';
    var b = document.createElement('span'); b.className = 'body'; b.textContent = text;
    el.appendChild(ts); el.appendChild(k); el.appendChild(b); $('feed').appendChild(el);
    $('feed').scrollTop = $('feed').scrollHeight;
  }

  // in-page shared bus for the local demo (delivers to all peers but sender; crews ignore their own)
  function sharedBus() {
    var subs = [];
    return function endpoint(id) {
      return { kind: 'demo', send: function (m) { setTimeout(function () { subs.forEach(function (s) { if (s.id !== id) { try { s.fn(m); } catch (e) {} } }); }, 0); }, onMessage: function (fn) { subs.push({ id: id, fn: fn }); } };
    };
  }

  // a worker's brain: run weld.crew on the task; demo reasoner echoes the goal so results differ
  function workerReasoner(workerId) {
    var step = 0;
    return function (prompt) {
      step++;
      var g = (prompt.match(/GOAL:\s*(.*)/) || [])[1] || 'task';
      if (step === 1) return Promise.resolve(JSON.stringify({ thought: 'Working on it.', tool: 'think', args: { note: 'handling ' + g } }));
      return Promise.resolve(JSON.stringify({ thought: 'Done.', final: workerId + ' completed: ' + g }));
    };
  }
  function makeServe(workerId, ownModel) {
    return async function (payload) {
      var a = CR.create({ workspace: 'swarm-' + workerId, reason: ownModel ? undefined : workerReasoner(workerId) });
      feedLine('action', workerId + ' \u2190 ' + payload.goal);
      var out = await a.run(payload.goal, {});
      feedLine('obs', workerId + ' \u2192 ' + (out.answer || JSON.stringify(out)));
      return { worker: workerId, answer: out.answer, ok: out.ok };
    };
  }

  async function ownModelAvailable() {
    if (sb && sb.connected && typeof sb.has === 'function' && sb.has('ai') && typeof sb.ai === 'function') {
      try { var r = await sb.ai('Reply with: ok', { system: 'Reply with the single word ok.' }); return !!(r && r.ok); } catch (e) {}
    }
    return false;
  }

  function readTasks() { return $('tasks').value.split('\n').map(function (s) { return s.trim(); }).filter(Boolean).map(function (g) { return { goal: g }; }); }
  function renderResults(rs) { $('results').textContent = rs.length ? rs.map(function (r, i) { return (i + 1) + '. ' + (r && r.error ? ('ERROR: ' + r.error) : (r.answer + '   (' + r.worker + ')')); }).join('\n') : '(none)'; }

  var active = [];            // crews to tear down between runs
  function teardown() { active.forEach(function (c) { try { c.leave(); } catch (e) {} }); active = []; }

  // ---- LOCAL DEMO: spin up driver + N workers over an in-page bus, dispatch the batch ----
  async function runLocal() {
    teardown();
    var tasks = readTasks(); if (!tasks.length) { feedLine('error', 'no subtasks'); return; }
    var nWorkers = parseInt($('workers').value, 10) || 3;
    var conc = parseInt($('conc').value, 10) || 2;
    var ownModel = await ownModelAvailable();
    $('role').textContent = 'role: local demo'; $('transport').textContent = 'transport: in-page'; paintModelPill(ownModel);
    var bus = sharedBus();
    for (var i = 1; i <= nWorkers; i++) {
      var id = 'worker-' + i;
      var w = SW.create({ role: 'worker', id: id, transport: bus(id), heartbeat: false });
      w.serve(makeServe(id, ownModel)); active.push(w);
    }
    var driver = SW.create({ role: 'user', id: 'driver', transport: bus('driver'), heartbeat: false }); active.push(driver);
    await new Promise(function (r) { setTimeout(r, 40); });          // let registrations land
    refreshPresence(driver);
    feedLine('start', 'dispatching ' + tasks.length + ' subtasks to ' + nWorkers + ' workers (concurrency ' + conc + ')');
    var results = await driver.dispatchMany(tasks, { concurrency: conc, timeoutMs: 8000 });
    feedLine('final', 'collected ' + results.length + ' results');
    renderResults(results); refreshPresence(driver);
  }

  // ---- WORKER MODE: register over weld.sync, serve tasks from other tabs ----
  var roleCrew = null;
  async function startServing() {
    teardown();
    var ownModel = await ownModelAvailable(); paintModelPill(ownModel);
    roleCrew = SW.create({ role: 'worker' }); active.push(roleCrew);
    roleCrew.serve(makeServe('worker@' + roleCrew.id.slice(-4), ownModel));
    $('transport').textContent = 'transport: ' + roleCrew.transport; $('role').textContent = 'role: worker (serving)';
    feedLine('start', 'serving as ' + roleCrew.id + ' over ' + roleCrew.transport);
    pollPresence(roleCrew);
  }
  // ---- USER MODE: register over weld.sync, dispatch to workers in other tabs ----
  async function runUser() {
    var tasks = readTasks(); if (!tasks.length) { feedLine('error', 'no subtasks'); return; }
    if (!roleCrew) { roleCrew = SW.create({ role: 'user' }); active.push(roleCrew); $('transport').textContent = 'transport: ' + roleCrew.transport; $('role').textContent = 'role: user'; pollPresence(roleCrew); await new Promise(function (r) { setTimeout(r, 300); }); }
    var conc = parseInt($('conc2').value, 10) || 2;
    if (!roleCrew.workers().length) { feedLine('error', 'no workers found \u2014 open a second tab, set it to Worker, and Start serving'); return; }
    feedLine('start', 'dispatching ' + tasks.length + ' subtasks to ' + roleCrew.workers().length + ' worker(s)');
    var results = await roleCrew.dispatchMany(tasks, { concurrency: conc, timeoutMs: 15000 });
    feedLine('final', 'collected ' + results.length + ' results'); renderResults(results);
  }

  function refreshPresence(crew) {
    var ps = crew.peers(); $('peers').textContent = 'peers: ' + ps.length;
    $('presence').textContent = ps.length ? ps.map(function (p) { return '\u2022 ' + p.role + '  ' + p.peerId; }).join('\n') : '(none yet)';
  }
  var pollTimer = null;
  function pollPresence(crew) { if (pollTimer) clearInterval(pollTimer); pollTimer = setInterval(function () { refreshPresence(crew); }, 1000); }

  function paintModelPill(own) { /* reuse role pill area */ }

  function buildSelect(sel, arr, fn) { arr.forEach(function (x, i) { var o = document.createElement('option'); o.value = String(i); o.textContent = x.label; sel.appendChild(o); }); sel.onchange = function () { fn(arr[parseInt(sel.value, 10)]); }; sel.value = '0'; sel.onchange(); }

  function applyMode(m) {
    $('cfg-local').style.display = (m === 'local') ? '' : 'none';
    $('cfg-user').style.display = (m === 'user') ? '' : 'none';
    $('cfg-tasks').style.display = (m === 'worker') ? 'none' : '';
    $('go').style.display = (m === 'worker') ? 'none' : '';
    $('serve').style.display = (m === 'worker') ? '' : 'none';
    teardown(); roleCrew = null; if (pollTimer) { clearInterval(pollTimer); pollTimer = null; }
    $('role').textContent = 'role: ' + m; $('peers').textContent = 'peers: 0'; $('presence').textContent = '\u2014';
  }

  // init
  if (!SW || !CR) { feedLine('error', 'weld.swarm or weld.crew not loaded \u2014 check imports + triggers'); }
  buildSelect($('mode'), MODES, function (m) { applyMode(m.id); });
  buildSelect($('preset'), PRESETS, function (p) { $('tasks').value = p.tasks.join('\n'); });
  $('workers').oninput = function () { $('workers-v').textContent = $('workers').value; };
  $('conc').oninput = function () { $('conc-v').textContent = $('conc').value; };
  $('conc2').oninput = function () { $('conc2-v').textContent = $('conc2').value; };
  $('go').onclick = function () { t0 = Date.now(); $('feed').innerHTML = ''; var m = MODES[parseInt($('mode').value, 10)].id; if (m === 'local') runLocal(); else runUser(); };
  $('serve').onclick = function () { t0 = Date.now(); $('feed').innerHTML = ''; startServing(); };
  $('clear').onclick = function () { $('results').textContent = '\u2014'; $('feed').innerHTML = ''; };
  $('feed-clear').onclick = function () { $('feed').innerHTML = ''; };
})();
</script>
