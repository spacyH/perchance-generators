$meta
  title       = weld.import
  description = A format-detecting importer for character and chat data. Sniffs the format of pasted text, a parsed object, or an uploaded file -- TavernAI / SillyTavern character cards (v1/v2/v3, JSON or PNG), FurAI character and chat exports, ai-character-chat dumps, and JSONL chat logs -- and normalizes each to one predictable shape. Routes PNG cards through weld.cards. Generalizes the Smart-story import suite. Stash on window.weld.import.
  tags        = weld, import, character, chat, tavern, sillytavern, furai, jsonl, plugin

$output() =>
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.import) return window.weld.import;

  // -----------------------------------------------------------------
  // weld.import: one front door for "here's a character/chat from
  // somewhere else, load it." Every community tool has its own export
  // shape; this detects which one you handed it and returns a normalized
  // result so the rest of your app never branches on format:
  //
  //   var r = weld.import.detect(pastedText);     // { format, kind, confidence }
  //   var out = await weld.import.parse(input);   // { format, character?, chat?, warnings }
  //   var out = await weld.import.parseFile(file);// same, for an uploaded File/Blob
  //
  // kind is 'character' | 'chat' | 'unknown'. A character normalizes to the
  // same fields weld.cards.toCharacter produces; a chat normalizes to
  // { messages:[{ role, name, content }], title }. PNG character cards are
  // delegated to weld.cards when it's present.
  // -----------------------------------------------------------------

  var deps = { cards: null };
  function attach(d) { if (d && d.cards) deps.cards = d.cards; }
  function cards() { return deps.cards || (window.weld && window.weld.cards) || null; }

  // ---- defensive limits -------------------------------------------------
  // weld.import takes UNTRUSTED files (character cards, chat logs from other
  // apps). Without bounds, a malicious or accidentally-huge input -- a 50MB
  // JSON, a card with a megabyte-long name, a chat log with a million
  // messages -- would be parsed straight into memory. These caps reject
  // oversized input before it can do damage. They are generous for real
  // cards/logs and tunable via weld.import.limits(). (Pattern adapted from
  // the import-validator in Adaptive Session Studio.)
  var LIMITS = {
    INPUT_BYTES:   8000000,   // 8 MB raw text/JSON (a v2 card is ~tens of KB; logs larger)
    PNG_BYTES:    16000000,   // 16 MB PNG (embedded-card images run large)
    MESSAGES:        50000,   // max messages in an imported chat
    STRING_LEN:     500000,   // max length of any single normalized string field
    JSONL_LINES:    100000    // max lines scanned in a JSONL chat
  };
  function limits(patch) {
    if (patch && typeof patch === 'object') {
      for (var k in patch) { if (Object.prototype.hasOwnProperty.call(patch, k) && typeof Reflect.get(LIMITS, k) === 'number') Reflect.set(LIMITS, k, Reflect.get(patch, k) | 0); }
    }
    return Object.assign({}, LIMITS);
  }
  // Approx byte length of a string input (chars are close enough for the cap).
  function rawSize(input) {
    if (typeof input === 'string') return input.length;
    if (input && (input.byteLength !== undefined)) return input.byteLength;     // ArrayBuffer / typed array
    if (input && typeof input === 'object') { try { return JSON.stringify(input).length; } catch (e) { return 0; } }
    return 0;
  }
  function guardInput(input) {
    var isBytes = input && (input.byteLength !== undefined || (typeof Uint8Array !== 'undefined' && input instanceof Uint8Array));
    var cap = isBytes ? LIMITS.PNG_BYTES : LIMITS.INPUT_BYTES;
    var size = rawSize(input);
    if (size > cap) {
      throw new Error('weld.import: input too large (' + Math.round(size / 1e6) + ' MB; max ' + Math.round(cap / 1e6) + ' MB). Tune via weld.import.limits() if intentional.');
    }
  }

  // ---- helpers ----
  function asObj(input) {
    if (input && typeof input === 'object') return input;
    if (typeof input === 'string') { try { return JSON.parse(input); } catch (e) { return null; } }
    return null;
  }
  function has(o) { var keys = Array.prototype.slice.call(arguments, 1); for (var i = 0; i < keys.length; i++) { if (o && Reflect.get(o, Reflect.get(keys, i)) !== undefined) return true; } return false; }
  function str(v) {
    if (v == null) return '';
    var s = String(v);
    return s.length > LIMITS.STRING_LEN ? s.slice(0, LIMITS.STRING_LEN) : s;   // cap runaway string fields
  }

  // ---- format detection -------------------------------------------------
  // Returns { format, kind, confidence } without fully parsing.
  function detect(input) {
    // PNG bytes (character card embedded in image)
    if (looksLikePng(input)) return { format: 'tavern-png', kind: 'character', confidence: 0.95 };

    var text = (typeof input === 'string') ? input : null;
    var o = asObj(input);

    // JSONL: multiple lines, each a JSON object (chat log)
    if (text && isJsonl(text)) return { format: 'jsonl-chat', kind: 'chat', confidence: 0.8 };

    if (!o) return { format: 'unknown', kind: 'unknown', confidence: 0 };

    // TavernAI / SillyTavern character card v2/v3 (spec + data envelope)
    if (o.spec === 'chara_card_v3' || (o.spec_version && /3/.test(str(o.spec_version)) && o.data)) {
      return { format: 'tavern-v3', kind: 'character', confidence: 0.95 };
    }
    if (o.spec === 'chara_card_v2' || (o.spec_version && o.data)) {
      return { format: 'tavern-v2', kind: 'character', confidence: 0.95 };
    }
    // ai-character-chat export (its own envelope)
    if (has(o, 'addCharacter', 'characterName') || (o.characters && o.messages && o.threadName !== undefined)) {
      return { format: 'ai-character-chat', kind: (o.messages ? 'chat' : 'character'), confidence: 0.7 };
    }
    // FurAI character (flat fields, FurAI-specific keys)
    if (has(o, 'persona', 'greeting') && (o.char_name || o.name)) {
      return { format: 'furai-character', kind: 'character', confidence: 0.6 };
    }
    // FurAI / generic chat dump (array of messages or {messages:[...]})
    if (Array.isArray(o) && o.length && (o[0].role || o[0].is_user !== undefined || o[0].mes !== undefined)) {
      return { format: 'furai-chat', kind: 'chat', confidence: 0.6 };
    }
    if (Array.isArray(o.messages) && o.messages.length) {
      return { format: 'generic-chat', kind: 'chat', confidence: 0.55 };
    }
    // TavernAI v1 / generic flat character card
    if (has(o, 'name', 'char_name') && has(o, 'description', 'personality', 'first_mes', 'persona')) {
      return { format: 'tavern-v1', kind: 'character', confidence: 0.55 };
    }
    return { format: 'unknown', kind: 'unknown', confidence: 0 };
  }

  function looksLikePng(input) {
    var c = cards();
    if (c && typeof c.isPng === 'function') { try { return c.isPng(input); } catch (e) {} }
    // fallback signature check on a byte-ish input
    try {
      var b = input;
      if (b && b.length >= 8 && b[0] === 0x89 && b[1] === 0x50 && b[2] === 0x4E && b[3] === 0x47) return true;
    } catch (e) {}
    return false;
  }
  function isJsonl(text) {
    var lines = text.split(/\r?\n/).filter(function (l) { return l.trim(); });
    if (lines.length < 2) return false;
    var okCount = 0, checked = 0;
    for (var i = 0; i < lines.length && checked < 6; i++) {
      checked++;
      try { var o = JSON.parse(Reflect.get(lines, i)); if (o && typeof o === 'object') okCount++; } catch (e) { /* tolerate a bad line */ }
    }
    // It's JSONL if most sampled lines parse as objects (allows a stray bad line).
    return okCount >= 2 && okCount >= (checked - 1);
  }

  // ---- normalizers ------------------------------------------------------
  function normCharacterFlat(d) {
    return {
      name: str(d.name || d.char_name),
      description: str(d.description || d.persona),
      personality: str(d.personality),
      scenario: str(d.scenario || d.world_scenario),
      first_message: str(d.first_mes || d.first_message || d.greeting),
      example_dialogue: str(d.mes_example || d.example_dialogue),
      system_prompt: str(d.system_prompt),
      creator_notes: str(d.creator_notes),
      tags: Array.isArray(d.tags) ? d.tags : new Array(),
      creator: str(d.creator)
    };
  }
  function normMessage(m, idx) {
    // map the many "who said it" conventions to role/name/content
    var role, name = '', content = '';
    if (m.role) role = (m.role === 'assistant' || m.role === 'char') ? 'assistant' : (m.role === 'user' ? 'user' : m.role);
    else if (m.is_user !== undefined) role = m.is_user ? 'user' : 'assistant';
    else if (m.name && /you|user/i.test(m.name)) role = 'user';
    else role = 'assistant';
    name = str(m.name || m.author || m.speaker);
    content = str(m.content || m.mes || m.text || m.message);
    return { role: role, name: name, content: content };
  }
  function normChatMessages(arr) {
    var list = arr || new Array();
    if (list.length > LIMITS.MESSAGES) list = list.slice(0, LIMITS.MESSAGES);   // cap runaway logs
    return list.map(normMessage).filter(function (m) { return m.content; });
  }

  // ---- parse ------------------------------------------------------------
  // parse(input) -> Promise<{ format, kind, character?, chat?, warnings }>
  function parse(input) {
    try { guardInput(input); } catch (e) { return Promise.reject(e); }
    var det = detect(input);
    var warnings = new Array();
    try {
      switch (det.format) {
        case 'tavern-png': {
          var c = cards();
          if (!c || typeof c.read !== 'function') return Promise.reject(new Error('weld.import: PNG card needs weld.cards (import it / attach it)'));
          var spec = c.read(input);
          if (!spec) { warnings.push('PNG had no embedded character data'); return Promise.resolve({ format: det.format, kind: 'unknown', warnings: warnings }); }
          return Promise.resolve({ format: det.format, kind: 'character', character: c.toCharacter(spec.spec || spec), warnings: warnings });
        }
        case 'tavern-v3':
        case 'tavern-v2': {
          var o = asObj(input);
          var d = (o && o.data) ? o.data : o;
          // prefer weld.cards.toCharacter when present (handles spec quirks)
          var c2 = cards();
          var ch = (c2 && c2.toCharacter) ? c2.toCharacter(o) : normCharacterFlat(d);
          return Promise.resolve({ format: det.format, kind: 'character', character: ch, warnings: warnings });
        }
        case 'tavern-v1':
        case 'furai-character': {
          return Promise.resolve({ format: det.format, kind: 'character', character: normCharacterFlat(asObj(input)), warnings: warnings });
        }
        case 'ai-character-chat': {
          var acc = asObj(input);
          if (acc && Array.isArray(acc.messages)) {
            return Promise.resolve({ format: det.format, kind: 'chat', chat: { title: str(acc.threadName || acc.title), messages: normChatMessages(acc.messages) }, warnings: warnings });
          }
          return Promise.resolve({ format: det.format, kind: 'character', character: normCharacterFlat(acc.character || acc), warnings: warnings });
        }
        case 'furai-chat':
        case 'generic-chat': {
          var go = asObj(input);
          var arr = Array.isArray(go) ? go : go.messages;
          return Promise.resolve({ format: det.format, kind: 'chat', chat: { title: str(go.title || go.threadName), messages: normChatMessages(arr) }, warnings: warnings });
        }
        case 'jsonl-chat': {
          var lines = String(input).split(/\r?\n/).filter(function (l) { return l.trim(); });
          if (lines.length > LIMITS.JSONL_LINES) { warnings.push('truncated to first ' + LIMITS.JSONL_LINES + ' lines'); lines = lines.slice(0, LIMITS.JSONL_LINES); }
          var msgs = new Array();
          for (var i = 0; i < lines.length; i++) { try { msgs.push(JSON.parse(Reflect.get(lines, i))); } catch (e) { warnings.push('skipped malformed line ' + (i + 1)); } }
          return Promise.resolve({ format: det.format, kind: 'chat', chat: { title: '', messages: normChatMessages(msgs) }, warnings: warnings });
        }
        default:
          return Promise.resolve({ format: 'unknown', kind: 'unknown', warnings: ['weld.import: unrecognized format'] });
      }
    } catch (e) {
      return Promise.reject(new Error('weld.import: parse failed (' + (e && e.message) + ')'));
    }
  }

  // parseFile(file) -> reads a File/Blob then parses (PNG as bytes, else text).
  function parseFile(file) {
    if (!file) return Promise.reject(new Error('weld.import: no file'));
    var isPngName = /\.png$/i.test(file.name || '') || (file.type === 'image/png');
    var sizeCap = isPngName ? LIMITS.PNG_BYTES : LIMITS.INPUT_BYTES;
    if (typeof file.size === 'number' && file.size > sizeCap) {
      return Promise.reject(new Error('weld.import: file too large (' + Math.round(file.size / 1e6) + ' MB; max ' + Math.round(sizeCap / 1e6) + ' MB).'));
    }
    return new Promise(function (resolve, reject) {
      if (typeof FileReader === 'undefined') { reject(new Error('weld.import: FileReader unavailable')); return; }
      var fr = new FileReader();
      fr.onload = function () { resolve(fr.result); };
      fr.onerror = function () { reject(new Error('weld.import: file read failed')); };
      if (isPngName) fr.readAsArrayBuffer(file); else fr.readAsText(file);
    }).then(function (data) {
      if (isPngName) return parse(new Uint8Array(data));
      return parse(data);
    });
  }

  function formats() {
    return ['tavern-png', 'tavern-v3', 'tavern-v2', 'tavern-v1', 'furai-character', 'furai-chat', 'ai-character-chat', 'generic-chat', 'jsonl-chat'];
  }

  window.weld.import = {
    detect: detect,
    parse: parse,
    parseFile: parseFile,
    attach: attach,
    formats: formats,
    limits: limits,
    version: '1.1.0'
  };

  return window.weld.import;
