(function(){

var sync = null;

function el(id){return document.getElementById(id);}
function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');}

    '</section>' +

    // 02 Shared text
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">02</span><h2>Shared text area</h2></div>' +
      '<p>Type here. Every other tab sees your text live.</p>' +
      '<textarea id="shared-text" rows="4" placeholder="Type to broadcast\u2026"></textarea>' +

    '</section>' +

    // 03 Ephemeral broadcast
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">03</span><h2>Broadcast messages (no storage)</h2></div>' +
      '<p>Fire-and-forget messages to all other tabs. Useful for notifications, presence, "user is typing" indicators.</p>' +
      '<div class="row">' +
        '<input type="text" id="msg-input" placeholder="Type a message\u2026">' +
        '<button id="msg-send" class="primary">Broadcast</button>' +
      '</div>' +
      '<div id="msg-log" style="margin-top:12px;max-height:240px;overflow-y:auto"></div>' +
    '</section>' +

    // 04 Reactive state
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">04</span><h2>Reactive state primitive</h2></div>' +
      '<p>The <code>sync.state(key)</code> helper gives you an observable value with .get(), .set(), .onChange(). Pick a theme color and watch it propagate.</p>' +
      '<div class="row">' +
        '<button data-color="#4493f8" class="theme-btn">Blue</button>' +
        '<button data-color="#3fb950" class="theme-btn">Green</button>' +
        '<button data-color="#f85149" class="theme-btn">Red</button>' +
        '<button data-color="#d29922" class="theme-btn">Yellow</button>' +
      '</div>' +
      '<div id="theme-preview" style="margin-top:12px;height:80px;border-radius:8px;display:flex;align-items:center;justify-content:center;color:#0d1117;font-weight:600;background:var(--accent);transition:background 0.3s">Current theme</div>' +

    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">05</span><h2>Storage inspector</h2></div>' +

      '<div class="row">' +
        '<button id="storage-refresh">Refresh</button>' +

      '</div>' +
      '<div id="storage-list" style="margin-top:8px"></div>' +

  // Tab info
  var info = sync.info();
  el('my-id').textContent = info.peerId.slice(0, 12) + '...';
  el('backend').textContent = info.broadcastSupported ? 'BroadcastChannel' : 'localStorage events';
  
  // Counter
  el('counter-value').textContent = sync.get('counter', 0);
  el('counter-plus').onclick = function(){

  };
  el('counter-reset').onclick = function(){ sync.set('counter', 0); };
  sync.onKeyChange('counter', function(value, source){
    el('counter-value').textContent = value || 0;
    var time = new Date().toLocaleTimeString();
    el('counter-history').textContent = 'Last update at ' + time + ' from ' + source + ' tab';
  });
  
  // Shared text
  el('shared-text').value = sync.get('shared-text', '');
  el('shared-text').oninput = function(){ sync.set('shared-text', this.value); };
  sync.onKeyChange('shared-text', function(value, source){
    if (source === 'remote') {
      var elem = el('shared-text');
      var pos = elem.selectionStart;
      elem.value = value || '';
      el('shared-info').textContent = 'Last update at ' + new Date().toLocaleTimeString() + ' from another tab';
    }
  });
  
  // Broadcast
  el('msg-send').onclick = sendMessage;
  el('msg-input').onkeydown = function(e){ if(e.key==='Enter') sendMessage(); };
  sync.subscribe('chat', function(data, fromPeer){
    addMessageToLog(data, fromPeer, 'remote');
  });
  
  // Theme
  var themeState = sync.state('theme', '#4493f8');
  applyTheme(themeState.get());
  document.querySelectorAll('.theme-btn').forEach(function(btn){
    btn.onclick = function(){ themeState.set(btn.dataset.color); };
  });
  themeState.onChange(function(value){ applyTheme(value); });
  
  // Storage inspector
  el('storage-refresh').onclick = refreshStorage;
  el('storage-clear').onclick = function(){
    if (confirm('Clear all stored data for this generator?')) {
      var n = sync.clear();
      refreshStorage();
      addLog('Cleared ' + n + ' keys', 'warn');
    }
  };
  refreshStorage();
  
  // Peer count refresher
  setInterval(updatePeerCount, 2000);
  updatePeerCount();
}

function updatePeerCount(){
  var count = sync.peers();
  el('peer-count').textContent = count;
  el('peer-count').className = 'value ' + (count > 0 ? 'ok' : 'dim');
}

function sendMessage(){
  var text = el('msg-input').value.trim();
  if (!text) return;
  sync.broadcast('chat', text);
  addMessageToLog(text, sync.myId(), 'local');
  el('msg-input').value = '';
}

function addMessageToLog(text, peerId, type){
  var div = document.createElement('div');
  div.style.cssText = 'padding:6px 10px;margin-bottom:4px;background:var(--faint);border-radius:5px;font-size:13px';
  var badge = type === 'local' ? '<span class="ok">\[this tab\]</span>' : '<span class="accent">[from ' + esc((peerId||'').slice(0,8)) + ']</span>';
  div.innerHTML = '<span class="muted" style="font-size:11px">' + new Date().toLocaleTimeString() + ' ' + badge + '</span><div style="margin-top:2px">' + esc(text) + '</div>';
  var log = el('msg-log');
  log.insertBefore(div, log.firstChild);
  while (log.children.length > 30) log.removeChild(log.lastChild);
}

function applyTheme(color){
  el('theme-preview').style.background = color;
  el('theme-preview').textContent = 'Theme: ' + color;
}

function refreshStorage(){
  var keys = sync.keys();
  if (keys.length === 0) {
    el('storage-list').innerHTML = '<p class="muted">No data stored yet.</p>';
    return;
  }
  var html = '';
  keys.forEach(function(k){
    var v = sync.get(k);
    html += '<div class="metric" style="margin-bottom:6px"><div class="label">' + esc(k) + '</div><div style="font-family:var(--mono);font-size:12px;color:var(--ink);margin-top:4px;word-break:break-word">' + esc(JSON.stringify(v)) + '</div></div>';
  });
  el('storage-list').innerHTML = html;
}

function addLog(text, cls){
  console.log('\[sync\]', text);
}

function init(){
  // Show a "loading" placeholder while we wait for the plugin
  document.getElementById('page').innerHTML = '<p style="color:#7d8590;padding:40px 0;font:13px monospace">Loading plugin\u2026</p>';
  
  // Poll for the plugin to become available. The plugin is imported via
  // sync = '+String.fromCharCode(123)+'import:<slug>'+String.fromCharCode(125)+' in the lists editor.
  // Loading can take 2-5 seconds. We try every 300ms for up to 15s.
  var attempts = 0;
  var maxAttempts = 50;
  
  function attempt() {
    attempts++;
    
    // Has root.sync appeared yet?
    if (typeof root === 'undefined' || typeof root.sync !== 'function') {
      if (attempts < maxAttempts) {
        setTimeout(attempt, 300);
        return;
      }
      // Gave up
      showError('Plugin not imported. Add to your lists editor: <br><br><code style="background:var(--faint);padding:6px 10px;border-radius:4px;display:inline-block">sync = '+String.fromCharCode(123)+'import:cross-tab-sync-plugin'+String.fromCharCode(125)+'</code><br><br>Replace <code>cross-tab-sync-plugin</code> with the actual URL slug of your deployed plugin if different.');
      return;
    }
    
    // Invoke the plugin's $output to build & stash the API.
    // This is the missing piece \u2014 $output doesn't auto-run on import.
    try {
      var api = root.sync();
      // Prefer the window-stashed reference (set inside $output);
      // fall back to the direct return value if the stash didn't happen.
      sync = window.__sync || api;
    } catch (e) {
      showError('Plugin imported but failed to initialize: ' + (e && e.message ? e.message : String(e)));
      return;
    }
    
    if (!sync) {
      showError('Plugin returned no API.');
      return;
    }
    
    renderPage();
    
  }
  
  function showError(msg) {
    document.getElementById('page').innerHTML =
      '<header><h1 style="font-size:22px;font-weight:700;margin-bottom:12px">Plugin not ready</h1>' +
      '<p style="color:var(--muted);font-size:14px">' + msg + '</p>' +
      '<p style="color:var(--muted);font-size:13px;margin-top:18px">Or visit the <a href="https://perchance.org/cross-tab-sync-plugin" target="_blank">plugin docs page</a> for installation help.</p></header>';
  }
  
  attempt();
}
if(document.readyState==='loading'){document.addEventListener('DOMContentLoaded',function(){setTimeout(init,1500);});}
else{setTimeout(init,1500);}
})();

</script>
