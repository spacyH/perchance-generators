(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.comments) return window.weld.comments;

// -----------------------------------------------------------------
// weld.comments: per-item discussion threads.
//
// Perchance hosts a comments-plugin embed. The reusable trick is to
// give every item (image, post, scene...) its own channel by hashing a
// stable key, theme the embed to match your app, and present it either
// as an overlay modal or inline. weld.comments wraps all of that:
//
//   channel(key)            -> a stable channel id for that key
//   open({ key, title, ... })  themed overlay with the thread
//   inline(target, opts)       embed into an element
//   close()
//
// No app identity is baked in -- you pass your app slug and (optional)
// admin hash. Defaults are sensible and generic.
// -----------------------------------------------------------------

var EMBED_BASE = 'https://comments-plugin.perchance.org/embed/';
var STYLE_ID = 'weld-comments-style';
var overlay = null;

// Stable short hash (djb2 -> base36) for deriving a channel from any key.
function hash(text) {
  text = String(text == null ? '' : text);
  var h = 5381;
  for (var i = 0; i < text.length; i++) h = ((h << 5) + h + text.charCodeAt(i)) | 0;
  return Math.abs(h).toString(36);
}

// A channel id for an item. prefix groups channels under your app.
function channel(key, prefix) {
  prefix = prefix || defaultApp();
  return prefix + '-' + hash(key);
}

function defaultApp() {
  try { if (window.generatorName) return String(window.generatorName); } catch (e) {}
  return 'weld';
}

function resolveTheme(theme) {
  if (theme === 'light' || theme === 'dark') return theme;
  try { if (document.body && document.body.classList.contains('light-theme')) return 'light'; } catch (e) {}
  return 'dark';
}

function defaultConfig() {
  return {
    loadFonts: 'Open Sans, Inter',
    containerStyle: 'width:100%;height:100%;',
    inputAreaStyle: 'height:68px;',
    commentPlaceholderText: 'Add a comment...',
    submitButtonText: 'comment',
    allowBots: false
  };
}

// Build the embed src. app = generator slug the thread lives under.
function buildSrc(app, chan, config) {
  config = config || {};
  var merged = {};
  var d = defaultConfig();
  for (var k in d) merged[k] = d[k];
  for (var k2 in config) merged[k2] = config[k2];
  merged.channel = chan;
  var adminHash = merged.adminPasswordHash || '';
  var encoded = encodeURIComponent(JSON.stringify(merged));
  var src = EMBED_BASE + app + '+' + chan + '?';
  if (adminHash) src += 'adminPasswordHash=' + adminHash;
  return src + '#' + encoded;
}

// open(opts):
//   key       (required) anything stable -> derives the channel
//   title     header text
//   image     optional preview image url
//   app       generator slug the thread lives under (default generatorName)
//   theme     'auto' | 'light' | 'dark'
//   prefix    channel prefix (default app)
//   disabled  if true, show a "comments disabled" notice instead
//   config    passthrough to the comments embed (placeholder, submitButtonText,
//             allowBots, adminPasswordHash, ...)
function open(opts) {
  opts = opts || {};
  if (opts.disabled) { notifyDisabled(); return; }
  if (typeof document === 'undefined') return;
  var app = opts.app || defaultApp();
  var chan = channel(opts.key, opts.prefix || app);
  var cfg = {};
  var base = opts.config || {};
  for (var k in base) cfg[k] = base[k];
  cfg.forceColorScheme = resolveTheme(opts.theme);
  var src = buildSrc(app, chan, cfg);

  var ov = ensureOverlay();
  ov.querySelector('[data-title]').textContent = opts.title || 'Comments';
  var preview = ov.querySelector('[data-preview]');
  var img = ov.querySelector('[data-img]');
  if (opts.image) { preview.style.display = ''; img.src = opts.image; }
  else { preview.style.display = 'none'; }

  var frame = ov.querySelector('[data-frame]');
  var loading = ov.querySelector('[data-loading]');
  ov.style.display = 'flex';
  if (frame.getAttribute('data-src') !== src) {
    loading.style.display = 'flex'; loading.style.opacity = '1';
    frame.setAttribute('data-src', src);
    frame.src = src;
  }
  return chan;
}

function close() { if (overlay) overlay.style.display = 'none'; }

// inline(target, opts): embed the thread into an element instead of a modal.
function inline(target, opts) {
  opts = opts || {};
  var el = (typeof target === 'string') ? document.querySelector(target) : target;
  if (!el) return null;
  var app = opts.app || defaultApp();
  var chan = channel(opts.key, opts.prefix || app);
  var cfg = {};
  var base = opts.config || {};
  for (var k in base) cfg[k] = base[k];
  cfg.forceColorScheme = resolveTheme(opts.theme);
  var frame = document.createElement('iframe');
  frame.style.cssText = opts.style || 'width:100%;height:100%;border:none;background:transparent;';
  frame.setAttribute('allow', 'clipboard-read *; clipboard-write *');
  frame.src = buildSrc(app, chan, cfg);
  el.appendChild(frame);
  return frame;
}

  if (window.weld && window.weld.toast) { window.weld.toast.show('Comments are disabled for this item.', 'info'); return; }
  if (typeof document === 'undefined') return;
  var p = document.createElement('div');
  p.style.cssText = 'z-index:100000;position:fixed;left:50%;bottom:18px;transform:translateX(-50%);max-width:min(360px,calc(100vw - 28px));padding:10px 13px;border-radius:10px;border:1px solid var(--border-color,#444);background:var(--input-bg,#222);color:var(--text-color,#eee);box-shadow:0 8px 28px rgba(0,0,0,.35);font:13px/1.45 Open Sans,sans-serif;opacity:0;transition:opacity .18s ease;pointer-events:none;';
  p.textContent = 'Comments disabled for this item.';
  document.body.appendChild(p);
  requestAnimationFrame(function () { p.style.opacity = '1'; });
  setTimeout(function () { p.style.opacity = '0'; setTimeout(function () { p.remove(); }, 250); }, 2400);
}

// ---- secure datastore: comments-plugin as an append store, optionally
//      end-to-end encrypted (the pattern behind anonymous-inbox apps) ----
var attachedSecret = null;
function attach(deps) { if (deps && deps.secret) attachedSecret = deps.secret; return window.weld.comments; }
function resolveSecret() {
  var sp = attachedSecret || (window.weld && window.weld._secret) || window.secret;
  if (!sp || typeof sp.encrypt !== 'function') throw new Error('weld.comments: secret-plugin not attached -- call weld.comments.attach({ secret: secretPlugin })');
  return sp;
}

var CIPHER_RE = /^ENCRYPTED_[0-9]+_[\s\S]+_ENCRYPTED_END$/;
function isCiphertext(text) { return typeof text === 'string' && CIPHER_RE.test(text); }

function sha256Hex(text) {
  var bytes = new TextEncoder().encode(String(text == null ? '' : text));
  return Promise.resolve(crypto.subtle.digest('SHA-256', bytes)).then(function (buf) {
    var arr = new Uint8Array(buf), out = '';
    for (var i = 0; i < arr.length; i++) out += arr[i].toString(16).padStart(2, '0');
    return out;
  });
}

// Public-key crypto helpers (wrap secret-plugin). decrypt() is normalized
// through String() so a boxed-string return can never break === checks.
var cryptoApi = {
  generateKeyPair: function () { var k = resolveSecret().generateKeyPair(); return { publicKey: k.public || k.publicKey, privateKey: k.private || k.privateKey }; },
  encrypt: function (text, publicKey) { return String(resolveSecret().encrypt(String(text), publicKey)); },
  decrypt: function (ciphertext, privateKey) { return String(resolveSecret().decrypt(ciphertext, privateKey)); },
  keyPairMatches: function (publicKey, privateKey) { try { var sp = resolveSecret(), probe = 'weld:' + Date.now() + ':' + Math.random().toString(36).slice(2); return String(sp.decrypt(sp.encrypt(probe, publicKey), privateKey)) === probe; } catch (e) { return false; } },
  publicKeyHash: function (publicKey) { return sha256Hex(String(publicKey)); },
  isCiphertext: isCiphertext
};

// Build a comments-plugin channel string with the storage modifiers:
//   channelString(id)                          -> "id+ids:channel"   (inter-channel anonymity)
//   channelString(id, { u: 'name' })           -> "id+u:name,ids:channel"
//   channelString(id, { auth: hash })          -> "id+auth:hash,ids:channel"  (ownership via key)
//   channelString(id, { ids: false })          -> "id"
function channelString(id, opts) {
  opts = opts || {};
  var base = String(id == null ? '' : id), mods = [];
  if (opts.u) mods.push('u:' + String(opts.u).toLowerCase());
  else if (opts.auth) mods.push('auth:' + String(opts.auth));
  if (opts.ids !== false) mods.push('ids:channel');
  return mods.length ? base + '+' + mods.join(',') : base;
}

// store(opts): treat a comments-plugin controller as an append-only store.
//   opts.controller : the comments-plugin controller (submit(); optional
//                     setNicknameForNextComment(), getComments()/comments, onUpdate())
//   opts.keys       : { publicKey, privateKey } -> encrypt on post, decrypt on list
//   opts.anonymous  : post with no nickname (default true)
function store(opts) {
  opts = opts || {};
  var ctrl = opts.controller;
  if (!ctrl || typeof ctrl.submit !== 'function') throw new Error('weld.comments.store: a comments-plugin controller with submit() is required');
  var keys = opts.keys || null, anon = opts.anonymous !== false;
  function post(record) {
    var payload = (typeof record === 'string') ? record : JSON.stringify(record);
    if (keys) payload = cryptoApi.encrypt(payload, keys.publicKey);
    var pre = Promise.resolve();
    if (anon && typeof ctrl.setNicknameForNextComment === 'function') pre = Promise.resolve(ctrl.setNicknameForNextComment(''));
    return pre.then(function () { return ctrl.submit(payload); });
  }
  function rawComments() { if (typeof ctrl.getComments === 'function') return ctrl.getComments() || []; if (Array.isArray(ctrl.comments)) return ctrl.comments; return []; }
  function list(o) {
    o = o || {}; var blocked = o.blocked || {}, out = [];
    rawComments().forEach(function (c) {
      if (!c || typeof c.message !== 'string') return;
      var msg = c.message.trim(), uid = (c.user && typeof c.user.id === 'string') ? c.user.id : '';
      if (uid && blocked[uid]) return;
      var value = msg;
      if (keys) { if (!isCiphertext(msg)) return; try { value = cryptoApi.decrypt(msg, keys.privateKey); } catch (e) { return; } }
      if (o.parse !== false) { try { value = JSON.parse(value); } catch (e) {} }
      out.push({ value: value, userId: uid, time: Number((c.time || c.createdAt || 0)), raw: c });
    });
    if (o.sort !== false) out.sort(function (a, b) { return b.time - a.time; });
    return out;
  }
  function onNew(cb) { if (typeof ctrl.onComment === 'function') return ctrl.onComment(cb); if (typeof ctrl.onUpdate === 'function') return ctrl.onUpdate(function () { cb(list()); }); return function () {}; }
  return { post: post, list: list, onNew: onNew, channel: opts.channel || null };
}

window.weld.comments = {
  channel: channel,
  hash: hash,
  buildSrc: buildSrc,
  open: open,
  inline: inline,
  close: close,
  attach: attach,
  crypto: cryptoApi,
  channelString: channelString,
  isCiphertext: isCiphertext,
  store: store,
  version: '1.1.0'
};

return window.weld.comments;
})();
