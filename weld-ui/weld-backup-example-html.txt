// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

    class: 'w-select',
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,

}

'use strict';

if (!window.weld) window.weld = {};
if (window.weld.migrate) return;

// META store name -- tracks which data migrations have been applied
var META_STORE = '_weldMeta';
var META_KEY = 'appliedDataMigrations';

// Wrap an IDBRequest in a Promise
function req(idbRequest) {
  return new Promise(function(resolve, reject) {
    idbRequest.onsuccess = function(){ resolve(idbRequest.result); };
    idbRequest.onerror = function(){ reject(idbRequest.error); };
  });
}

// Wait for transaction completion
function txComplete(tx) {
  return new Promise(function(resolve, reject) {
    tx.oncomplete = function(){ resolve(); };
    tx.onerror = function(){ reject(tx.error); };
    tx.onabort = function(){ reject(tx.error || new Error('Transaction aborted')); };
  });
}

// Normalize an index spec - accepts string keyPath OR full object
function normalizeIndex(spec) {
  if (typeof spec === 'string') return { keyPath: spec, unique: false };
  return Object.assign({ unique: false }, spec);
}

// Apply schema upgrade INSIDE upgradeneeded handler
function applySchema(db, tx, schema, fromVer, toVer, log) {
  // Always ensure meta store exists from v1
  if (!db.objectStoreNames.contains(META_STORE)) {
    db.createObjectStore(META_STORE, { keyPath: 'key' });
    log('created meta store ' + META_STORE);
  }

  // Apply each version step
  for (var v = fromVer + 1; v <= toVer; v++) {
    var step = schema[v];
    if (!step) continue;
    log('applying schema v' + v);

    // Create new stores
    if (step.stores) {
      Object.keys(step.stores).forEach(function(name){
        if (db.objectStoreNames.contains(name)) {
          log('  store ' + name + ' already exists, skipping create');
          return;
        }
        var sp = step.stores[name];
        var opts = {};
        if (sp.keyPath) opts.keyPath = sp.keyPath;
        if (sp.autoIncrement) opts.autoIncrement = true;
        var store = db.createObjectStore(name, opts);
        log('  + store ' + name);
        if (sp.indexes) {
          Object.keys(sp.indexes).forEach(function(idxName){
            var idx = normalizeIndex(sp.indexes[idxName]);
            store.createIndex(idxName, idx.keyPath, { unique: idx.unique, multiEntry: !!idx.multiEntry });
            log('    + index ' + idxName);
          });
        }
      });
    }

    // Modify existing stores
    if (step.modify) {
      Object.keys(step.modify).forEach(function(name){
        if (!db.objectStoreNames.contains(name)) {
          log('  cannot modify ' + name + ' (does not exist)');
          return;
        }
        var store = tx.objectStore(name);
        var mods = step.modify[name];
        if (mods.dropIndexes) {
          mods.dropIndexes.forEach(function(idxName){
            if (store.indexNames.contains(idxName)) {
              store.deleteIndex(idxName);
              log('  - index ' + name + '.' + idxName);
            }
          });
        }
        if (mods.addIndexes) {
          Object.keys(mods.addIndexes).forEach(function(idxName){
            if (store.indexNames.contains(idxName)) return;
            var idx = normalizeIndex(mods.addIndexes[idxName]);
            store.createIndex(idxName, idx.keyPath, { unique: idx.unique, multiEntry: !!idx.multiEntry });
            log('  + index ' + name + '.' + idxName);
          });
        }
      });
    }

    // Drop stores
    if (step.drop) {
      step.drop.forEach(function(name){
        if (db.objectStoreNames.contains(name)) {
          db.deleteObjectStore(name);
          log('  - store ' + name);
        }
      });
    }
  }
}

// Open a database with declarative schema + data migrations
function open(name, config) {
  config = config || {};
  var schema = config.schema || {};
  var dataMigrations = config.dataMigrations || {};
  var onProgress = config.onProgress || function(){};

  // Highest schema version
  var versions = Object.keys(schema).map(Number).filter(function(n){ return !isNaN(n); });
  var targetVersion = versions.length ? Math.max.apply(null, versions) : 1;

  return new Promise(function(resolve, reject) {
    var req = indexedDB.open(name, targetVersion);

    req.onerror = function(){
      onProgress('error: ' + (req.error && req.error.message));
      reject(req.error);
    };

    req.onupgradeneeded = function(e){
      var db = req.result;
      var tx = e.target.transaction;
      var oldVer = e.oldVersion;
      var newVer = e.newVersion;
      onProgress('upgrading ' + name + ' from v' + oldVer + ' to v' + newVer);
      try {
        applySchema(db, tx, schema, oldVer, newVer, onProgress);
      } catch (err) {
        onProgress('schema error: ' + err.message);
        tx.abort();
        reject(err);
      }
    };

    req.onblocked = function(){
      onProgress('blocked: another connection has an older version open');
    };

    req.onsuccess = function(){
      var db = req.result;
      db.onversionchange = function(){
        db.close();
        onProgress('closed due to newer version opening');
      };

      // Run data migrations
      runDataMigrations(db, dataMigrations, onProgress).then(function(){
        resolve(wrapDb(db, name));
      }).catch(reject);
    };
  });
}

// Run any data migrations whose version hasn't been applied yet
function runDataMigrations(db, migrations, onProgress) {
  var versions = Object.keys(migrations).map(Number).sort(function(a, b){ return a - b; });
  if (!versions.length) return Promise.resolve();

  // Read applied set from meta
  var tx = db.transaction([META_STORE], 'readonly');
  var store = tx.objectStore(META_STORE);
  return req(store.get(META_KEY)).then(function(entry){
    var applied = (entry && entry.value) || [];
    var todo = versions.filter(function(v){ return applied.indexOf(v) === -1; });
    if (!todo.length) return;

    onProgress('running ' + todo.length + ' data migration(s): ' + todo.join(', '));
    var wrapped = wrapDb(db, '_internal');
    var chain = Promise.resolve();
    todo.forEach(function(v){
      chain = chain.then(function(){
        onProgress('  data migration v' + v + ' starting');
        return Promise.resolve(migrations[v](wrapped));
      }).then(function(){
        onProgress('  data migration v' + v + ' done');
        // Mark applied
        var tx2 = db.transaction([META_STORE], 'readwrite');
        var s2 = tx2.objectStore(META_STORE);
        return req(s2.get(META_KEY)).then(function(cur){
          var arr = (cur && cur.value) || [];
          if (arr.indexOf(v) === -1) arr.push(v);
          return req(s2.put({ key: META_KEY, value: arr }));
        }).then(function(){ return txComplete(tx2); });
      });
    });
    return chain;
  });
}

// Wrap a raw IDBDatabase in a friendly Promise-based API
function wrapDb(db, name) {
  function tx(stores, mode, callback) {
    if (!Array.isArray(stores)) stores = [stores];
    var t = db.transaction(stores, mode || 'readonly');
    var helper = {
      _tx: t,
      objectStore: function(s){ return t.objectStore(s); },
      put: function(s, v){ return req(t.objectStore(s).put(v)); },
      get: function(s, k){ return req(t.objectStore(s).get(k)); },
      delete: function(s, k){ return req(t.objectStore(s).delete(k)); },
      getAll: function(s){ return req(t.objectStore(s).getAll()); },
      clear: function(s){ return req(t.objectStore(s).clear()); },
      count: function(s){ return req(t.objectStore(s).count()); },
    };
    var result = Promise.resolve(callback(helper));
    return result.then(function(r){
      return txComplete(t).then(function(){ return r; });
    });
  }

  return {
    name: name,
    _db: db,
    version: db.version,
    storeNames: Array.from(db.objectStoreNames),

    put: function(store, value) {
      var t = db.transaction([store], 'readwrite');
      return req(t.objectStore(store).put(value));
    },
    get: function(store, key) {
      var t = db.transaction([store], 'readonly');
      return req(t.objectStore(store).get(key));
    },
    delete: function(store, key) {
      var t = db.transaction([store], 'readwrite');
      return req(t.objectStore(store).delete(key));
    },
    getAll: function(store, opts) {
      opts = opts || {};
      var t = db.transaction([store], 'readonly');
      var s = t.objectStore(store);
      if (opts.limit != null) return req(s.getAll(opts.range || null, opts.limit));
      return req(s.getAll(opts.range || null));
    },
    count: function(store) {
      var t = db.transaction([store], 'readonly');
      return req(t.objectStore(store).count());
    },
    clear: function(store) {
      var t = db.transaction([store], 'readwrite');
      return req(t.objectStore(store).clear());
    },
    query: function(store, indexName, value) {
      var t = db.transaction([store], 'readonly');
      var s = t.objectStore(store);
      var idx = s.index(indexName);
      return req(idx.getAll(value));
    },
    queryFirst: function(store, indexName, value) {
      var t = db.transaction([store], 'readonly');
      var s = t.objectStore(store);
      var idx = s.index(indexName);
      return req(idx.get(value));
    },
    queryRange: function(store, indexName, range) {
      var t = db.transaction([store], 'readonly');
      var s = t.objectStore(store);
      var idx = s.index(indexName);
      return req(idx.getAll(range));
    },
    keys: function(store) {
      var t = db.transaction([store], 'readonly');
      return req(t.objectStore(store).getAllKeys());
    },
    tx: tx,
    close: function() { db.close(); },

    // Bulk helpers
    bulkPut: function(store, items) {
      var t = db.transaction([store], 'readwrite');
      var s = t.objectStore(store);
      var promises = items.map(function(item){ return req(s.put(item)); });
      return Promise.all(promises).then(function(){ return txComplete(t); });
    },
    bulkDelete: function(store, keys) {
      var t = db.transaction([store], 'readwrite');
      var s = t.objectStore(store);
      var promises = keys.map(function(k){ return req(s.delete(k)); });
      return Promise.all(promises).then(function(){ return txComplete(t); });
    },

    // Iterator for huge datasets - avoids loading all into memory
    each: function(store, callback) {
      var t = db.transaction([store], 'readonly');
      var s = t.objectStore(store);
      return new Promise(function(resolve, reject) {
        var cur = s.openCursor();
        cur.onsuccess = function(){
          var c = cur.result;
          if (!c) { resolve(); return; }
          try {
            var r = callback(c.value, c.key);
            if (r === false) { resolve(); return; }
          } catch (e) { reject(e); return; }
          c.continue();
        };
        cur.onerror = function(){ reject(cur.error); };
      });
    },
  };
}

var FORMAT_VERSION = 1;     // bumped only if envelope format changes (not the data inside)
var SIGNATURE = 'weld-backup';

// Build an export envelope from arbitrary data
function exportData(opts) {
  opts = opts || {};
  var envelope = {
    signature: SIGNATURE,
    formatVersion: FORMAT_VERSION,
    dataVersion: opts.version != null ? opts.version : 1,
    generator: opts.generator || 'unknown',
    createdAt: new Date().toISOString(),
    metadata: opts.metadata || {},
    data: opts.data || {},
  };
  return envelope;
}

// Serialize envelope to a JSON Blob
function exportBlob(opts) {
  var env = exportData(opts);
  var json = JSON.stringify(env, null, 2);
  return new Blob([json], { type: 'application/json' });
}

// Trigger a browser download
function download(opts) {
  opts = opts || {};
  var blob = exportBlob(opts);
  var fname = opts.filename;
  if (!fname) {
    var stamp = new Date().toISOString().replace(/[:.]/g, '-').slice(0, 19);
    fname = (opts.generator || 'backup') + '-' + stamp + '.weld.json';
  }
  var url = URL.createObjectURL(blob);
  var a = document.createElement('a');
  a.href = url;
  a.download = fname;
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  setTimeout(function(){ URL.revokeObjectURL(url); }, 1000);
  return fname;
}

// Read a File/Blob/string into an envelope, validating signature
function readEnvelope(source) {
  return new Promise(function(resolve, reject) {
    if (typeof source === 'string') {
      try { resolve(JSON.parse(source)); }
      catch (e) { reject(new Error('Not valid JSON: ' + e.message)); }
      return;
    }
    if (source instanceof Blob || source instanceof File) {
      var reader = new FileReader();
      reader.onload = function(){
        try { resolve(JSON.parse(reader.result)); }
        catch (e) { reject(new Error('Not valid JSON: ' + e.message)); }
      };
      reader.onerror = function(){ reject(reader.error); };
      reader.readAsText(source);
      return;
    }
    if (source && typeof source === 'object') {
      resolve(source);
      return;
    }
    reject(new Error('Unsupported source type'));
  });
}

function validateEnvelope(env) {
  if (!env || typeof env !== 'object') {
    return { ok: false, error: 'Not an object' };
  }
  if (env.signature !== SIGNATURE) {
    return { ok: false, error: 'Missing or wrong signature (expected "' + SIGNATURE + '")' };
  }
  if (env.formatVersion == null) {
    return { ok: false, error: 'Missing formatVersion' };
  }
  if (env.formatVersion > FORMAT_VERSION) {
    return { ok: false, error: 'Envelope is newer (format v' + env.formatVersion + ') than this importer (v' + FORMAT_VERSION + ')' };
  }
  return {
    ok: true,
    formatVersion: env.formatVersion,
    dataVersion: env.dataVersion,
    generator: env.generator,
    createdAt: env.createdAt,
    metadata: env.metadata || {},
  };
}

function validate(source) {
  return readEnvelope(source).then(validateEnvelope).catch(function(e){
    return { ok: false, error: e.message };
  });
}

// Apply data migrations (provided by caller) to bring dataVersion up to current
function migrateData(env, migrations) {
  migrations = migrations || {};
  var versions = Object.keys(migrations).map(Number).sort(function(a, b){ return a - b; });
  var maxVer = versions.length ? Math.max.apply(null, versions) : env.dataVersion;
  var data = env.data;
  var fromVer = env.dataVersion;
  if (fromVer < maxVer) {
    for (var v = fromVer + 1; v <= maxVer; v++) {
      if (migrations[v]) {
        data = migrations[v](data) || data;
      }
    }
  }
  return { data: data, fromVersion: fromVer, toVersion: maxVer };
}

// Import an envelope - returns the data (optionally migrated)
function importData(source, opts) {
  opts = opts || {};
  return readEnvelope(source).then(function(env){
    var v = validateEnvelope(env);
    if (!v.ok) throw new Error(v.error);
    var migrated = migrateData(env, opts.migrations);
    return {
      data: migrated.data,
      generator: env.generator,
      fromVersion: migrated.fromVersion,
      toVersion: migrated.toVersion,
      metadata: env.metadata || {},
      createdAt: env.createdAt,
      importedAt: new Date().toISOString(),
    };
  });
}

// Restore an envelope into a weld.migrate DB
// mode: 'merge' (default) - put each item, overwriting same-key
//       'replace'          - clear store first, then put all
//       'append'           - only put items whose key doesn't exist
function toDb(source, db, opts) {
  opts = opts || {};
  var mode = opts.mode || 'merge';
  var onConflict = opts.onConflict; // (existing, incoming) => resolved | null to skip

  return importData(source, { migrations: opts.migrations }).then(function(result){
    var data = result.data;
    var stores = opts.stores || Object.keys(data);
    var summary = { imported: {}, skipped: {}, conflicts: {} };

    var chain = Promise.resolve();
    stores.forEach(function(s){
      if (!data[s]) return;
      var items = data[s];

      chain = chain.then(function(){
        if (mode === 'replace') {
          return db.clear(s).then(function(){
            return db.bulkPut(s, items);
          }).then(function(){
            summary.imported[s] = items.length;
          });
        }

        if (mode === 'append') {
          // Check each item's key
          var skipped = 0, imported = 0;
          var toPut = [];
          var lookups = items.map(function(item){
            // We need the keyPath of the store to know the key - use raw IDB
            var t = db._db.transaction([s], 'readonly');
            var store = t.objectStore(s);
            var keyPath = store.keyPath;
            var key = (typeof keyPath === 'string') ? item[keyPath] : null;
            return new Promise(function(resolve){
              if (key == null) { toPut.push(item); imported++; resolve(); return; }
              var r = store.get(key);
              r.onsuccess = function(){
                if (r.result == null) { toPut.push(item); imported++; }
                else { skipped++; }
                resolve();
              };
              r.onerror = function(){ resolve(); };
            });
          });
          return Promise.all(lookups).then(function(){
            summary.imported[s] = imported;
            summary.skipped[s] = skipped;
            if (toPut.length) return db.bulkPut(s, toPut);
          });
        }

        // merge mode
        if (!onConflict) {
          return db.bulkPut(s, items).then(function(){
            summary.imported[s] = items.length;
          });
        }

        // merge with conflict resolution
        var imported = 0, conflicts = 0;
        var resolved = [];
        var lookups = items.map(function(item){
          var t = db._db.transaction([s], 'readonly');
          var store = t.objectStore(s);
          var keyPath = store.keyPath;
          var key = (typeof keyPath === 'string') ? item[keyPath] : null;
          return new Promise(function(resolve){
            if (key == null) { resolved.push(item); imported++; resolve(); return; }
            var r = store.get(key);
            r.onsuccess = function(){
              var existing = r.result;
              if (existing == null) { resolved.push(item); imported++; }
              else {
                conflicts++;
                var merged = onConflict(existing, item, s);
                if (merged != null) resolved.push(merged);
              }
              resolve();
            };
            r.onerror = function(){ resolve(); };
          });
        });
        return Promise.all(lookups).then(function(){
          summary.imported[s] = imported;
          summary.conflicts[s] = conflicts;
          if (resolved.length) return db.bulkPut(s, resolved);
        });
      });
    });

    return chain.then(function(){
      summary.fromVersion = result.fromVersion;
      summary.toVersion = result.toVersion;
      summary.generator = result.generator;
      return summary;
    });
  });
}

// Count total records across all stores
function countTotal(data) {
  var n = 0;
  Object.keys(data).forEach(function(k){
    if (Array.isArray(data[k])) n += data[k].length;
  });
  return n;
}

var db = null;

async function init() {
  ui = window.weld.ui;
  if (!ui || !window.weld.migrate || !window.weld.backup) { setTimeout(init, 50); return; }

  // Set up a sample database
  db = await window.weld.migrate.open('weld-example-backup', {
    schema: {
      1: {
        stores: {
          characters: {
            keyPath: 'id',
            indexes: { byName: 'name' }
          }
        }
      }
    }
  });

  // Seed if empty
  var count = await db.count('characters');
  if (count === 0) {
    var seed = [
      { id: '1', name: 'Aria',     species: 'human',  level: 12 },
      { id: '2', name: 'Bex',      species: 'dwarf',  level:  8 },
      { id: '3', name: 'Casimir',  species: 'elf',    level: 15 },
    ];
    await db.bulkPut('characters', seed);
  }

  // SECTION 01 - Inspect current data
  await renderInspect();

  // SECTION 02 - Export
  buildExportSection();

  // SECTION 03 - Validate
  buildValidateSection();

  // SECTION 04 - Import (with optional migration)
  buildImportSection();

  buildMigrationDemo();

function buildMigrationDemo() {
  var sec = document.getElementById('sec-migrate-demo');

  // An older v1 backup with different field names
  var oldBackup = {
    signature: 'weld-backup',
    formatVersion: 1,
    dataVersion: 1,
    generator: 'old-example',
    createdAt: '2025-01-15T10:00:00.000Z',
    metadata: { note: 'Hand-crafted v1 backup from before the schema changed' },
    data: {
      characters: [
        { id: 'old1', first: 'Mira', last: 'Hollow', class: 'rogue' },
        { id: 'old2', first: 'Nyx',  last: 'Quill',  class: 'mage'  },
      ]
    }
  };

  sec.appendChild(ui.col([
    ui.el('p', { text: 'A pretend v1 backup with old field names (first/last/class). Importing without migrations gives you that old data shape. Importing with migrations transforms it to the new shape (name/species) on the way in.', style: { color: 'var(--muted)', fontSize: '14px', marginBottom: '10px' } }),
    ui.el('pre', { text: JSON.stringify(oldBackup, null, 2), style: { background: 'var(--faint)', maxHeight: '180px' } }),
    ui.row([
      ui.button({
        label: 'Import as-is',
        variant: 'ghost',
        onClick: async function() {
          var result = await window.weld.backup.importData(oldBackup);
          ui.modal({
            title: 'Imported without migrations',
            content: ui.el('pre', { text: JSON.stringify(result, null, 2), style: { background: 'var(--bg)', padding: '14px', borderRadius: '6px', maxHeight: '60vh', overflow: 'auto' } }),
          });
        }
      }),
      ui.button({
        label: 'Import with migrations',
        variant: 'primary',
        icon: 'refresh',
        onClick: async function() {
          var result = await window.weld.backup.importData(oldBackup, {
            migrations: {
              2: function(data) {
                data.characters.forEach(function(c) {
                  c.name = c.first + ' ' + c.last;
                  c.species = c.class === 'rogue' ? 'human' : (c.class === 'mage' ? 'elf' : 'human');
                  delete c.first;
                  delete c.last;
                  delete c.class;
                });
                return data;
              }
            }
          });
          ui.modal({
            title: 'Imported with v2 migration applied',
            content: ui.el('pre', { text: JSON.stringify(result, null, 2), style: { background: 'var(--bg)', padding: '14px', borderRadius: '6px', maxHeight: '60vh', overflow: 'auto' } }),
          });
        }
      }),
    ]),
  ]));
}
