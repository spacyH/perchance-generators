<script>
(function () {
  // Imported plugins are function DEFINITIONS -- trigger both so their $output runs.
  try { if (typeof root !== 'undefined' && typeof root.weldSkybridge === 'function') root.weldSkybridge(); } catch (e) {}
  try { if (typeof root !== 'undefined' && typeof root.weldAgent === 'function') root.weldAgent(); } catch (e) {}

  function $(id) { return document.getElementById(id); }
  var sb = (window.weld && window.weld.skybridge) || null;
  var AG = (window.weld && window.weld.agent) || null;

  var useOwnModel = !!(sb && sb.connected && typeof sb.has === 'function' && sb.has('ai'));

  function paint() {
    var c = $('conn'), t = $('conn-txt');
    if (sb && sb.connected) { c.className = 'pill on'; t.textContent = 'companion linked'; }
    else { c.className = 'pill off'; t.textContent = sb ? 'no companion (fallback)' : 'bridge not loaded'; }
    $('model').textContent = 'model: ' + (useOwnModel ? 'your own (via bridge)' : 'scripted demo');
    $('model').className = 'pill ' + (useOwnModel ? 'on' : 'off');
    var be = AG ? '(set on run)' : '\u2014';
    $('backend').textContent = 'memory: ' + be;
  }

  // dice like "2d6" or "d20" -> { total, rolls }
  function rollDice(spec) {
    var m = String(spec || 'd20').toLowerCase().replace(/\s+/g, '').match(/^(\d*)d(\d+)$/);
    if (!m) return { error: 'bad dice spec: ' + spec };
    var n = parseInt(m[1] || '1', 10), sides = parseInt(m[2], 10), rolls = [], total = 0;
    for (var i = 0; i < n; i++) { var r = 1 + Math.floor(Math.random() * sides); rolls.push(r); total += r; }
    return { total: total, rolls: rolls };
  }

  // A scripted demo reasoner: a fixed but plausible ReAct sequence, so the loop
  // is visible even with no own model. It reads the goal from the transcript.
  function demoReasoner() {
    var step = 0;
    return function (prompt /*, system */) {
      step++;
      var goalLine = (prompt.match(/GOAL:\s*(.*)/) || [])[1] || 'the goal';
      if (step === 1) return Promise.resolve(JSON.stringify({ thought: 'First, note the goal in memory so I remember it later.', tool: 'memory.write', args: { text: goalLine, tags: ['quest'] } }));
      if (step === 2) return Promise.resolve(JSON.stringify({ thought: 'Now roll 2d6 for initiative.', tool: 'roll', args: { dice: '2d6' } }));
      if (step === 3) return Promise.resolve(JSON.stringify({ thought: 'I noted the goal and rolled initiative; that is enough to plan.', tool: 'memory.search', args: { query: '' } }));
      return Promise.resolve(JSON.stringify({ thought: 'Done.', final: 'Encounter planned: I saved the goal to memory and rolled 2d6 for initiative. (Scripted demo reasoner \u2014 configure your own model in Weld Companion for real reasoning.)' }));
    };
  }

  function makeAgent(reason) {
    var ws = ($('ws').value || 'demo').trim();
    var opts = { workspace: ws, maxSteps: 8 };
    if (reason) opts.reason = reason;          // demo reasoner; omit => plugin uses sb.ai (your model)
    var a = AG.create(opts);
    a.registerTool({ name: 'roll', description: 'roll dice, e.g. 2d6 or d20', args: { dice: 'string like "2d6"' }, run: function (x) { return rollDice(x.dice); } });
    return a;
  }

  // Decide once per run: use the real model only if a quick probe actually returns
  // a completion (the companion advertises 'ai' even when no model is configured).
  async function pickReasoner() {
    if (sb && sb.connected && typeof sb.has === 'function' && sb.has('ai') && typeof sb.ai === 'function') {
      try { var r = await sb.ai('Reply with: ok', { system: 'Reply with the single word ok.' }); if (r && r.ok) { useOwnModel = true; return null; } } catch (e) {}
    }
    useOwnModel = false;
    return demoReasoner();
  }

  function clearStream() { $('stream').innerHTML = ''; }
  function addEv(kind, what, detail) {
    var li = document.createElement('li'); li.className = 'ev';
    var tag = document.createElement('span'); tag.className = 'tag ' + kind; tag.textContent = kind;
    var body = document.createElement('div'); body.className = 'body';
    var w = document.createElement('div'); w.className = 'what'; w.textContent = what;
    body.appendChild(w);
    if (detail != null && detail !== '') { var d = document.createElement('div'); d.className = 'detail'; d.textContent = (typeof detail === 'string') ? detail : JSON.stringify(detail, null, 2); body.appendChild(d); }
    li.appendChild(tag); li.appendChild(body); $('stream').appendChild(li);
  }
  function onStep(s) {
    dbgLog(s);
    if (s.type === 'start') addEv('start', 'Goal', s.goal);
    else if (s.type === 'thought') addEv('thought', 'thinking', s.thought);
    else if (s.type === 'action') addEv('action', s.tool, JSON.stringify(s.args));
    else if (s.type === 'observation') addEv('obs', 'observation', s.observation);
    else if (s.type === 'final') addEv('final', 'answer', s.answer);
    else if (s.type === 'maxsteps') addEv('maxsteps', 'stopped at step cap', '');
    else if (s.type === 'error') addEv('error', 'error', s.error);
    // 'prompt' and 'raw' go to the debug console only
  }

  // ---- debug console: a live, timestamped feed of everything the agent does ----
  var t0 = 0;
  function dbgLine(kind, text) {
    var el = document.createElement('span'); el.className = 'l';
    var ts = document.createElement('span'); ts.className = 'ts';
    var ms = t0 ? ('+' + String(Date.now() - t0) + 'ms ') : '';
    ts.textContent = ms;
    var k = document.createElement('span'); k.className = 'k-' + kind; k.textContent = kind.toUpperCase() + ' ';
    var b = document.createElement('span'); b.className = 'body'; b.textContent = text;
    el.appendChild(ts); el.appendChild(k); el.appendChild(b);
    $('dbg').appendChild(el);
    if ($('dbg-auto').checked) $('dbg').scrollTop = $('dbg').scrollHeight;
  }
  function dbgLog(s) {
    var showRaw = $('dbg-raw').checked;
    if (s.type === 'prompt') { if (showRaw) dbgLine('prompt', 'step ' + s.step + ' \u2192 model\n' + s.transcript + '\n--- system ---\n' + s.system); return; }
    if (s.type === 'raw') { if (showRaw) dbgLine('raw', 'model \u2192\n' + s.reply); return; }
    if (s.type === 'start') dbgLine('start', 'run started: ' + s.goal);
    else if (s.type === 'thought') dbgLine('thought', s.thought);
    else if (s.type === 'action') dbgLine('action', s.tool + ' ' + JSON.stringify(s.args));
    else if (s.type === 'observation') dbgLine('obs', JSON.stringify(s.observation));
    else if (s.type === 'final') dbgLine('final', s.answer);
    else if (s.type === 'maxsteps') dbgLine('maxsteps', 'hit step cap');
    else if (s.type === 'error') dbgLine('error', s.error);
  }

  var current = null;
  async function refreshMem() { if (!current) return; var m = await current.memory.list(); $('mem').textContent = m.length ? m.map(function (x) { return '\u2022 ' + x.text + (x.tags && x.tags.length ? '  #' + x.tags.join(' #') : ''); }).join('\n') : '(empty)'; }
  async function refreshScratch() { if (!current) return; var sp = await current.scratchpad.read(); $('scratch').textContent = sp.length ? sp.map(function (x) { return '\u2022 ' + x.note; }).join('\n') : '(empty)'; }

  async function run() {
    if (!AG) { addEv('error', 'weld.agent not loaded', 'Is \{import:weld-agent-plugin\} present, and was root.weldAgent() triggered?'); return; }
    $('run').disabled = true;
    clearStream();
    t0 = Date.now();
    var reason = await pickReasoner();
    paint();
    current = makeAgent(reason);
    $('backend').textContent = 'memory: ' + current.backend();
    try {
      await current.run($('goal').value, { onStep: onStep });
    } catch (e) {
      addEv('error', 'run failed', String(e && e.message || e));
    }
    await refreshMem(); await refreshScratch();
    $('run').disabled = false;
  }

  paint();
  if (sb && typeof sb.onConnect === 'function') sb.onConnect(function () { useOwnModel = !!(sb.connected && sb.has('ai')); paint(); });
  $('run').onclick = run;
  $('clear').onclick = function () { clearStream(); addEv('start', 'cleared', ''); };
  $('mem-refresh').onclick = refreshMem;
  $('scratch-refresh').onclick = refreshScratch;
  $('dbg-clear').onclick = function () { $('dbg').innerHTML = ''; };
})();
</script>
