<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.privacy) return window.weld.privacy;

// -----------------------------------------------------------------
// weld.privacy: make an app legible about where its data goes.
//
// Duck.ai's best idea is showing, per model, what privacy guarantee
// applies BEFORE you use it -- anonymized, zero-retention, no-training,
// and so on. weld.privacy ports the *pattern* (not the guarantees, which
// are platform-specific): a small claim vocabulary, an honest registry of
// how the suite's own backends behave, and a way to render a data-handling
// summary an app can show its users.
//
//   weld.privacy.claimsFor('kv')   returns an array of claim objects
//   weld.privacy.summary of the backends an app uses -> human-readable text
//   weld.privacy.register a backend with its claim kinds
//
// IMPORTANT HONESTY RULE: a claim describes data-handling POSTURE; it does
// NOT enforce anything. The labels here reflect what Perchance's backends
// actually do, to the best of documented behavior. An app must not attach
// a claim it can't stand behind -- an overstated label is worse than none.
// -----------------------------------------------------------------

// The claim vocabulary. Each kind has a stable id, a short title, a
// plain-language detail, and a coarse 'leaves-device' flag apps can sort on.
var KINDS = {
  'local-only': {
    title: 'Stays on this device',
    detail: 'Data is kept in this browser only (IndexedDB / localStorage) and is never sent anywhere. Clearing your browser data removes it.',
    leavesDevice: false
  },
  'anonymized': {
    title: 'Sent without identifying you',
    detail: 'Data is sent to a backend with identifying metadata stripped or proxied, so the request is not tied to your identity.',
    leavesDevice: true
  },
  'server-stored': {
    title: 'Stored on a server',
    detail: 'Data is stored on a remote server and persists there (often so it can sync across your devices). It is not only on this device.',
    leavesDevice: true
  },
  'public': {
    title: 'Publicly accessible',
    detail: 'Data is uploaded to a public URL that anyone with the link can access. Do not put anything sensitive here.',
    leavesDevice: true
  },
  'expiring': {
    title: 'Auto-deletes later',
    detail: 'Data is stored with an expiry and is removed automatically after a set time.',
    leavesDevice: true
  },
  'third-party': {
    title: 'Handled by a third party',
    detail: 'Data passes through a service outside your control, whose own data-handling policy applies.',
    leavesDevice: true
  },
  'transient': {
    title: 'Not stored at all',
    detail: 'Data is used for a single request and not retained after the response.',
    leavesDevice: true
  }
};

function kindIds() { return Object.keys(KINDS); }
function describeKind(id) {
  var k = Reflect.get(KINDS, id);
  return k ? { kind: id, title: k.title, detail: k.detail, leavesDevice: k.leavesDevice } : null;
}

// Honest, pre-registered claims for the suite's own backends. These reflect
// documented Perchance behavior; an app can override or extend them.
//   persist/history/backup -> local IndexedDB only          -> local-only
//   cache/state            -> in-memory or local            -> local-only
//   kv                     -> kv-plugin, server cross-device -> server-stored
//   upload                 -> public URL (+ optional expiry) -> public (+expiring)
//   stream/image           -> Perchance broker (proxied)     -> anonymized + third-party
//   fetch                  -> superFetch proxy (strips hdrs) -> anonymized + third-party
//   registry               -> reads Perchance public API     -> transient + third-party
//   report                 -> encrypted, expiring upload     -> public + expiring
var REGISTRY = {
  persist:  ['local-only'],
  history:  ['local-only'],
  backup:   ['local-only'],
  cache:    ['local-only'],
  state:    ['local-only'],
  memory:   ['local-only'],
  kv:       ['server-stored'],
  upload:   ['public', 'expiring'],
  report:   ['public', 'expiring'],
  stream:   ['anonymized', 'third-party'],
  image:    ['anonymized', 'third-party'],
  fetch:    ['anonymized', 'third-party'],
  registry: ['transient', 'third-party']
};

// register(name, kinds): declare (or override) the claims for a backend.
// kinds is an array of claim ids; unknown ids are dropped with no throw.
function register(name, kinds) {
  if (!name) return null;
  var list = (Array.isArray(kinds) ? kinds : Array.of(kinds)).filter(function (k) { return !!Reflect.get(KINDS, k); });
  Reflect.set(REGISTRY, String(name), list);
  return claimsFor(name);
}

// claimsFor(name): the resolved claim objects for a backend (or empty array).
function claimsFor(name) {
  var ids = Reflect.get(REGISTRY, String(name)) || new Array();
  return ids.map(describeKind).filter(Boolean);
}

// worst-case posture across a set of backends: does ANY of them leave the
// device? Useful for a single at-a-glance indicator.
function leavesDevice(names) {
  return toArray(names).some(function (n) {
    return claimsFor(n).some(function (c) { return c.leavesDevice; });
  });
}

function toArray(x) { return Array.isArray(x) ? x : (x == null ? new Array() : Array.of(x)); }

// summary(names): a human-readable data-handling digest for the backends an
// app uses. Apps can show this in a privacy panel or an info modal.
function summary(names) {
  var list = toArray(names);
  if (!list.length) return 'No backends declared.';
  var lines = new Array();
  lines.push(leavesDevice(list)
    ? 'Some data leaves this device:'
    : 'All data stays on this device.');
  list.forEach(function (n) {
    var claims = claimsFor(n);
    if (!claims.length) {
      lines.push('  ' + n + ': (no privacy claim declared)');
    } else {
      lines.push('  ' + n + ': ' + claims.map(function (c) { return c.title; }).join('; '));
    }
  });
  return lines.join('\n');
}

// report(names): a structured version of summary() for building UI.
function report(names) {
  var list = toArray(names);
  return {
    leavesDevice: leavesDevice(list),
    backends: list.map(function (n) { return { name: n, claims: claimsFor(n) }; })
  };
}

// Audit helper: which declared backends keep data only on-device?
function localOnly(names) {
  return toArray(names).filter(function (n) {
    var c = claimsFor(n);
    return c.length && c.every(function (x) { return !x.leavesDevice; });
  });
}

window.weld.privacy = {
  kinds: KINDS,
  kindIds: kindIds,
  describeKind: describeKind,
  register: register,
  claimsFor: claimsFor,
  leavesDevice: leavesDevice,
  localOnly: localOnly,
  summary: summary,
  report: report,
  version: '1.0.0'
};

return window.weld.privacy;
})();
</script>
<script>

  function boot(){ if(!window.weld||!window.weld.privacy){return setTimeout(boot,100);} init(); }
  function init(){
    var P=window.weld.privacy, $=function(id){return document.getElementById(id);};
    var BACKENDS=['persist','history','cache','kv','upload','stream','image','fetch','registry','report'];
    var chosen={persist:true, stream:true};
    var picks=$('picks');
    BACKENDS.forEach(function(b){
      var el=document.createElement('button');
      el.className='btn sm '+(chosen[b]?'':'ghost');
      el.textContent=b;
      el.onclick=function(){ chosen[b]=!chosen[b]; el.className='btn sm '+(chosen[b]?'':'ghost'); render(); };
      picks.appendChild(el);
    });
    function render(){
      var names=BACKENDS.filter(function(b){return chosen[b];});
      if(!names.length){ $('verdict').textContent='no backends selected'; $('out').textContent=''; return; }
      var r=P.report(names);
      $('verdict').textContent = r.leavesDevice ? 'some data leaves this device' : 'all data stays on this device';
      $('verdict').className = 'chip '+(r.leavesDevice?'':'k');
      $('out').textContent = P.summary(names);
    }
    render();
  }
  if(document.readyState==='loading')document.addEventListener('DOMContentLoaded',boot);else boot();

</script>
