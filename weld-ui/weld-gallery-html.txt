  <h2>Overview</h2>
<p>weld.gallery manages a persistent collection of images in IndexedDB: store blobs with metadata, sort and filter by date or favorites, evict old entries when full, render a responsive grid, and open images in a lightbox. The slideshow app's 200-line cache layer, extracted into a reusable plugin.</p>
<p>Each image is an IndexedDB Blob keyed by a unique ID. A separate manifest tracks metadata per image (prompt, seed, resolution, timestamp, favorited). The two are kept in sync through atomic store/remove operations.</p>
<p>Soft dependency: weld.ui (lightbox integration for fullscreen viewing).</p>
<h2>Quick start</h2>

<h2>Configuration</h2>
<ul>
<li><code>dbName</code>: IndexedDB database name. Default <code>'weld-gallery'</code>.</li>
<li><code>maxItems</code>: maximum images before eviction kicks in. Default 200.</li>
<li><code>sortMode</code>: initial sort. <code>'newest'</code> (default), <code>'oldest'</code>, or <code>'favorites'</code>.</li>
</ul>
<h2>Storing images</h2>
<div class="sig" data-codeid="sig3"></div>
<p>Store a Blob (JPEG, PNG, etc.) with optional metadata. Returns the generated ID. If the gallery is full, the oldest non-favorited entry is evicted first. Only evicts a favorite as a last resort (every slot favorited).</p>
<p>Metadata fields: <code>prompt</code>, <code>seed</code>, <code>resolution</code>, <code>metadata</code> (arbitrary extra data).</p>

<h2>Rendering</h2>
<div class="sig" data-codeid="sig15"></div>
<div class="sig" data-codeid="sig16"></div>
<p><code>render</code> sets up the container and loads the manifest from IndexedDB. <code>renderView</code> re-renders the grid from the current view (call after sort/filter changes). Images load lazily from IndexedDB into object URLs. Click any image to open the lightbox (if <code>weld.ui</code> is loaded).</p>
<p>The grid uses CSS class <code>w-gallery</code> with responsive <code>auto-fill</code> columns. Add the class to your container element, or let <code>render</code> set it up.</p>
<h2>Download</h2>
<div class="sig" data-codeid="sig17"></div>
<p>Trigger a browser download of the image Blob. Default filename is the image ID + <code>.jpg</code>.</p>
<h2>Events</h2>
<div class="sig" data-codeid="sig18"></div>
<p>Called after any mutation (store, remove, toggleFavorite, sort change). Receives the current manifest. Use it to re-render a custom UI or update counters.</p>
<h2>Eviction strategy</h2><div class="note">
Eviction removes the oldest non-favorited entry. If every entry is favorited, the absolute oldest is evicted (with a console warning). The new image is stored after eviction, so it never evicts itself. Favorited entries survive until the user explicitly removes them or every slot is favorited and space is needed.

<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.gallery) return; // already loaded

// -----------------------------------------------------------------
// weld.gallery: Image grid with IndexedDB blob storage.
// Manifest tracks metadata per image (prompt, seed, timestamp, fav).
// Integrates with weld.ui.lightbox for fullscreen viewing.
// Eviction: when full, drop oldest non-favorited first.
// -----------------------------------------------------------------

var DB_STORE = 'blobs';

function uid() {
  return 'img_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 6);
}

function createGallery(opts) {
  opts = opts || {};
  var dbName = opts.dbName || 'weld-gallery';
  var maxItems = opts.maxItems || 200;
  var manifest = [];
  var dbInstance = null;
  var dbPromise = null;
  var container = null;
  var lightboxInstance = null;
  var sortMode = opts.sortMode || 'newest';
  var favOnly = false;
  var changeListeners = [];

  // -------- IndexedDB for blobs --------
  function openDb() {
    if (dbInstance) return Promise.resolve(dbInstance);
    if (dbPromise) return dbPromise;
    dbPromise = new Promise(function(resolve, reject) {
      var req = indexedDB.open(dbName, 1);
      req.onupgradeneeded = function(e) {
        var db = e.target.result;
        if (!db.objectStoreNames.contains(DB_STORE)) {
          db.createObjectStore(DB_STORE);
        }
        if (!db.objectStoreNames.contains('meta')) {
          db.createObjectStore('meta');
        }
      };
      req.onsuccess = function(e) { dbInstance = e.target.result; resolve(dbInstance); };
      req.onerror = function(e) { dbPromise = null; reject(e.target.error); };
    });
    return dbPromise;
  }

  function idbOp(store, mode, fn) {
    return openDb().then(function(db) {
      return new Promise(function(resolve, reject) {
        var tx = db.transaction(store, mode);
        var s = tx.objectStore(store);
        var req = fn(s);
        if (req && req.onsuccess !== undefined) {
          req.onsuccess = function() { resolve(req.result); };
          req.onerror = function() { reject(req.error); };
        } else {
          tx.oncomplete = function() { resolve(); };
          tx.onerror = function() { reject(tx.error); };
        }
      });
    });
  }

  function saveManifest() {
    return idbOp('meta', 'readwrite', function(s) { return s.put(manifest, 'manifest'); });
  }

  function loadManifest() {
    return idbOp('meta', 'readonly', function(s) { return s.get('manifest'); }).then(function(m) {
      manifest = m || [];
    });
  }

  function emit() {
    for (var i = 0; i < changeListeners.length; i++) {
      try { changeListeners[i](manifest); } catch (e) {}
    }
  }

  // -------- Core operations --------
  function store(blob, meta) {
    if (!(blob instanceof Blob)) {
      return Promise.reject(new Error('gallery.store: expected a Blob'));
    }
    meta = meta || {};
    var id = uid();
    var entry = {
      id: id,
      prompt: meta.prompt || '',
      seed: meta.seed != null ? meta.seed : null,
      resolution: meta.resolution || null,
      createdAt: Date.now(),
      favorited: false,
      metadata: meta.metadata || null
    };

    return idbOp(DB_STORE, 'readwrite', function(s) { return s.put(blob, id); }).then(function() {
      // Eviction
      while (manifest.length >= maxItems) {
        var evictIdx = -1;
        for (var i = 0; i < manifest.length; i++) {
          if (!manifest[i].favorited) { evictIdx = i; break; }
        }
        if (evictIdx === -1) evictIdx = 0;
        var removed = manifest.splice(evictIdx, 1)[0];
        // Best-effort blob cleanup
        idbOp(DB_STORE, 'readwrite', function(s) { return s.delete(removed.id); }).catch(function() {});
      }
      manifest.push(entry);
      return saveManifest();
    }).then(function() {
      emit();
      return id;
    });
  }

  function remove(id) {
    manifest = manifest.filter(function(e) { return e.id !== id; });
    return Promise.all([
      idbOp(DB_STORE, 'readwrite', function(s) { return s.delete(id); }).catch(function() {}),
      saveManifest()
    ]).then(function() { emit(); return true; });
  }

  function removeMany(ids) {
    var idSet = {};
    for (var i = 0; i < ids.length; i++) idSet[ids[i]] = true;
    manifest = manifest.filter(function(e) { return !idSet[e.id]; });
    var chain = Promise.resolve();
    ids.forEach(function(id) {
      chain = chain.then(function() {
        return idbOp(DB_STORE, 'readwrite', function(s) { return s.delete(id); }).catch(function() {});
      });
    });
    return chain.then(saveManifest).then(function() { emit(); });
  }

  function clear() {
    var ids = manifest.map(function(e) { return e.id; });
    manifest = [];
    return removeMany(ids).then(saveManifest).then(function() { emit(); });
  }

  function toggleFavorite(id) {
    var entry = manifest.find(function(e) { return e.id === id; });
    if (!entry) return Promise.resolve(null);
    entry.favorited = !entry.favorited;
    return saveManifest().then(function() { emit(); return entry.favorited; });
  }

  function getBlob(id) {
    return idbOp(DB_STORE, 'readonly', function(s) { return s.get(id); });
  }

  function getBlobUrl(id) {
    return getBlob(id).then(function(blob) {
      return blob ? URL.createObjectURL(blob) : null;
    });
  }

  // -------- Sorted/filtered view --------
  function getView() {
    var view = manifest.slice();
    if (favOnly) view = view.filter(function(e) { return e.favorited; });
    switch (sortMode) {
      case 'oldest':
        view.sort(function(a, b) { return (a.createdAt || 0) - (b.createdAt || 0); });
        break;
      case 'favorites':
        view.sort(function(a, b) {
          if (a.favorited !== b.favorited) return a.favorited ? -1 : 1;
          return (b.createdAt || 0) - (a.createdAt || 0);
        });
        break;
      default:
        view.sort(function(a, b) { return (b.createdAt || 0) - (a.createdAt || 0); });
    }
    return view;
  }

  function setSort(mode) { sortMode = mode; emit(); }
  function setFavOnly(v) { favOnly = !!v; emit(); }

  // -------- Rendering --------
  function render(targetEl) {
    container = typeof targetEl === 'string' ? document.querySelector(targetEl) : targetEl;
    if (!container) return Promise.reject(new Error('gallery.render: container not found'));
    return loadManifest().then(function() { return renderView(); });
  }

  function renderView() {
    if (!container) return Promise.resolve();
    var view = getView();
    container.innerHTML = '';
    if (view.length === 0) {
      container.innerHTML = '<div style="color:var(--weld-muted,#7d8590);font:13px var(--weld-mono,monospace);padding:40px;text-align:center;">No images yet.</div>';
      return Promise.resolve();
    }
    var promises = [];
    view.forEach(function(entry) {
      var cell = document.createElement('div');
      cell.className = 'w-gallery-item' + (entry.favorited ? ' is-fav' : '');
      cell.dataset.id = entry.id;
      cell.innerHTML = '<div class="w-gallery-loading"></div>';
      container.appendChild(cell);
      var p = getBlobUrl(entry.id).then(function(url) {
        if (!url) { cell.innerHTML = '<div class="w-gallery-empty">missing</div>'; return; }
        var img = document.createElement('img');
        img.src = url;
        img.alt = entry.prompt || '';
        img.addEventListener('click', function() { openLightbox(entry.id); });
        cell.innerHTML = '';
        cell.appendChild(img);
        // Seed tag
        if (entry.seed != null) {
          var tag = document.createElement('div');
          tag.className = 'w-gallery-seed';
          tag.textContent = 'seed ' + entry.seed;
          cell.appendChild(tag);
        }
      });
      promises.push(p);
    });
    return Promise.all(promises);
  }

  function openLightbox(startId) {
    if (!window.weld || !window.weld.ui || !window.weld.ui.lightbox) return;
    if (!lightboxInstance) lightboxInstance = window.weld.ui.lightbox();
    var view = getView();
    var items = [];
    var startIdx = 0;
    view.forEach(function(entry, i) {
      items.push({ src: '', id: entry.id, meta: entry.prompt || '', alt: entry.prompt || '' });
      if (entry.id === startId) startIdx = i;
    });
    // Resolve blob URLs
    var chain = Promise.resolve();
    items.forEach(function(item, idx) {
      chain = chain.then(function() {
        return getBlobUrl(item.id).then(function(url) { items[idx].src = url || ''; });
      });
    });
    chain.then(function() {
      lightboxInstance.open(items, { index: startIdx });
    });
  }

  // -------- Download --------
  function downloadImage(id, filename) {
    return getBlob(id).then(function(blob) {
      if (!blob) return false;
      var url = URL.createObjectURL(blob);
      var a = document.createElement('a');
      a.href = url;
      a.download = filename || (id + '.jpg');
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      setTimeout(function() { URL.revokeObjectURL(url); }, 1000);
      return true;
    });
  }

  // -------- Init --------
  function init() {
    return loadManifest();
  }

  function onChange(fn) {
    changeListeners.push(fn);
    return function() {
      var idx = changeListeners.indexOf(fn);
      if (idx >= 0) changeListeners.splice(idx, 1);
    };
  }

  return {
    init: init,
    store: store,
    remove: remove,
    removeMany: removeMany,
    clear: clear,
    toggleFavorite: toggleFavorite,
    getBlob: getBlob,
    getBlobUrl: getBlobUrl,
    getView: getView,
    getManifest: function() { return manifest.slice(); },
    setSort: setSort,
    setFavOnly: setFavOnly,
    render: render,
    renderView: renderView,
    downloadImage: downloadImage,
    onChange: onChange,
    count: function() { return manifest.length; },
    maxItems: function() { return maxItems; }
  };
}
