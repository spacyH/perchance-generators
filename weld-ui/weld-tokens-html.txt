    <p class="lead">Accurate token counting via the DeepSeek-R1 tokenizer. The same tokenizer Perchance uses for AI calls, so counts match exactly.</p>
  </div>
  <h2>Overview</h2>
<p>weld.tokens loads the same DeepSeek-R1 tokenizer that Perchance's AI broker uses, then exposes precise counting and budget operations. Use it when you need exact token counts for context-window planning, prompt truncation, document chunking, or anywhere the rough <code>chars/4</code> estimate is too coarse.</p>
<p>The tokenizer downloads on first use (a few hundred KB via jsdelivr's transformers.js CDN). All counting methods are async; <code>countSync</code> is available once the model is loaded.</p>
<p>Pairs naturally with weld.lorebook: lorebook's default estimator is char/4 for speed, but you can swap in <code>weld.tokens.countSync</code> for accuracy after preload.</p>

<div class="sig" data-codeid="sig4"></div>
<div class="sig" data-codeid="sig5"></div>
<p>The tokenizer lazy-loads on first use of any async method. To avoid latency on the first <code>count</code> call, kick it off explicitly with <code>preload()</code> when the page loads.</p>
<p><code>isLoaded()</code> returns true once the model is in memory, after which <code>countSync</code> works.</p>
<h2>Counting</h2><h3>weld.tokens.count(text)</h3>
<div class="sig" data-codeid="sig6"></div>
<p>Returns the exact token count. Loads the tokenizer if not already loaded.</p>
<h3>weld.tokens.countSync(text)</h3>
<div class="sig" data-codeid="sig7"></div>
<p>Synchronous version. Returns <code>null</code> if the tokenizer hasn't loaded yet -- check <code>isLoaded()</code> first or use <code>count</code> for safety.</p>
<h3>weld.tokens.estimate(text)</h3>
<div class="sig" data-codeid="sig8"></div>
<p>Fast char/4 heuristic. No tokenizer needed. Use when speed matters and accuracy doesn't.</p>
<h3>weld.tokens.countMessages(messages)</h3>
<div class="sig" data-codeid="sig9"></div>
<p>Counts an array of messages with a per-message overhead (4 tokens) to approximate role tags and separators. Each message can be a string or an object with <code>content</code> or <code>text</code>.</p>
<h2>Truncation</h2><h3>weld.tokens.truncate(text, max)</h3>
<div class="sig" data-codeid="sig10"></div>
<p>Drops tokens from the end until the text fits within <code>max</code>. Use for trimming a system prompt or instruction block where the beginning matters most.</p>
<h3>weld.tokens.truncateStart(text, max)</h3>
<div class="sig" data-codeid="sig11"></div>
<p>Drops tokens from the start until the text fits. Use for chat histories where the most recent messages matter most.</p>
<pre data-codeid="code12"></pre>
<h3>weld.tokens.chunk(text, perChunk)</h3>
<div class="sig" data-codeid="sig13"></div>
<p>Splits long text into chunks of approximately <code>perChunk</code> tokens each. Useful for document QA pipelines that process content in segments.</p>

<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.tokens) return; // already loaded

// -----------------------------------------------------------------
// Accurate tokenizer (DeepSeek-R1-0528) via transformers.js.
// Same tokenizer Perchance uses for the AI broker, so counts are exact.
// -----------------------------------------------------------------
var tokenizerPromise = null;
var tokenizer = null;
var loadError = null;

function load() {
  if (tokenizerPromise) return tokenizerPromise;
  tokenizerPromise = (function() {
    return Promise.resolve().then(function() {
      return import('https://cdn.jsdelivr.net/npm/@huggingface/transformers@latest/+esm');
    }).then(function(mod) {
      return mod.AutoTokenizer.from_pretrained('deepseek-ai/DeepSeek-R1-0528');
    }).then(function(t) {
      tokenizer = t;
      return t;
    }).catch(function(e) {
      loadError = e;
      throw e;
    });
  })();
  return tokenizerPromise;
}

function preload() { return load(); }
function ready()   { return load(); }
function isLoaded(){ return tokenizer !== null; }
function getLoadError() { return loadError; }

// Cheap fallback (~4 chars per token)
function estimate(text) {
  return Math.ceil(String(text || '').length / 4);
}

// Exact count (async)
function count(text) {
  return load().then(function(t) {
    return t.encode(String(text)).length;
  });
}

// Sync count -- null if not loaded yet
function countSync(text) {
  if (!tokenizer) return null;
  return tokenizer.encode(String(text)).length;
}

function encode(text) {
  return load().then(function(t) { return t.encode(String(text)); });
}

function decode(ids) {
  return load().then(function(t) {
    return t.decode(ids, { skip_special_tokens: true });
  });
}

function budget(text, maxTokens) {
  return count(text).then(function(used) {
    return {
      used: used,
      max: maxTokens,
      remaining: maxTokens - used,
      fits: used <= maxTokens,
      percent: Math.round((used / maxTokens) * 100)
    };
  });
}

// Truncate preserving the START of the text
function truncate(text, maxTokens) {
  return load().then(function(t) {
    var ids = t.encode(String(text));
    if (ids.length <= maxTokens) return String(text);
    return t.decode(ids.slice(0, maxTokens), { skip_special_tokens: true });
  });
}

// Truncate preserving the END of the text (useful for chat histories)
function truncateStart(text, maxTokens) {
  return load().then(function(t) {
    var ids = t.encode(String(text));
    if (ids.length <= maxTokens) return String(text);
    return t.decode(ids.slice(-maxTokens), { skip_special_tokens: true });
  });
}

// Split text into chunks of ~N tokens each
function chunk(text, maxTokensPerChunk) {
  return load().then(function(t) {
    var ids = t.encode(String(text));
    var chunks = [];
    for (var i = 0; i < ids.length; i += maxTokensPerChunk) {
      chunks.push(t.decode(ids.slice(i, i + maxTokensPerChunk), { skip_special_tokens: true }));
    }
    return chunks;
  });
}

// Count tokens across an array of messages
function countMessages(messages) {
  var total = 0;
  function step(i) {
    if (i >= messages.length) return Promise.resolve(total);
    var msg = messages[i];
    var txt = typeof msg === 'string' ? msg : (msg.content || msg.text || '');
    return count(txt).then(function(n) {
      total += n + 4;
      return step(i + 1);
    });
  }
  return step(0);
}

function info() {
  return {
    model: 'deepseek-ai/DeepSeek-R1-0528',
    loaded: tokenizer !== null,
    error: loadError ? String(loadError.message || loadError) : null,
    vocabSize: tokenizer ? tokenizer.vocab_size : null
  };
}

window.weld.tokens = {
  preload: preload,
  ready: ready,
  isLoaded: isLoaded,
  loadError: getLoadError,
  estimate: estimate,
  count: count,
  countSync: countSync,
  encode: encode,
  decode: decode,
  budget: budget,
  truncate: truncate,
  truncateStart: truncateStart,
  chunk: chunk,
  countMessages: countMessages,
  info: info,
  version: '1.0.0'
};

})();

</script>
