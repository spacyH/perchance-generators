// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

// COMPONENTS
function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

function input(opts) {
  opts = opts || {};
  return el('input', {
    class: 'w-input',
    type: opts.type || 'text',
    placeholder: opts.placeholder,
    value: opts.value != null ? opts.value : '',
    oninput: opts.onInput ? function(e){ opts.onInput(e.target.value, e); } : null,
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
    readonly: opts.readonly ? 'readonly' : null,
  });
}

function select(opts) {
  opts = opts || {};
  var options = (opts.options || []).map(function(o){
    if (typeof o === 'string') o = { value: o, label: o };
    return el('option', {
      value: o.value,
      selected: o.value === opts.value ? 'selected' : null,
    }, o.label);
  });
  return el('select', {
    class: 'w-select',
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
  }, options);
}

function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = opts.title || opts.dismissable !== false
    ? el('div', { class: 'w-modal-hdr' }, [
        opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div'),
        opts.dismissable !== false ? closeBtn : null,
      ])
    : null;

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {
    if (e.key === 'Escape' && opts.dismissable !== false) close();
  }
  document.addEventListener('keydown', escHandler);

  document.body.appendChild(overlay);
  return { el: overlay, close: close, body: body };
}

<script>(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.branching) return; // already loaded

// -----------------------------------------------------------------
// Node id generator
// -----------------------------------------------------------------
function uid() {
  return 'n_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 8);
}

// -----------------------------------------------------------------
// Tree instance factory
// -----------------------------------------------------------------
function createTree(opts) {
  opts = opts || {};
  var storage = opts.storage;
  var storageKey = opts.storageKey || 'weld-branching';

  // Internal state
  var nodes = {};      // id -> node
  var rootIds = [];    // ordered list of root node ids
  var activeId = null; // currently active leaf

  function save() {
    if (!storage) return;
    try {
      storage.setItem(storageKey, JSON.stringify({
        nodes: nodes,
        rootIds: rootIds,
        activeId: activeId
      }));
    } catch (e) { /* swallow */ }
  }

  function load() {
    if (!storage) return false;
    try {
      var raw = storage.getItem(storageKey);
      if (!raw) return false;
      var parsed = JSON.parse(raw);
      if (parsed && typeof parsed === 'object') {
        nodes = parsed.nodes || {};
        rootIds = parsed.rootIds || [];
        activeId = parsed.activeId || null;
        return true;
      }
    } catch (e) { /* swallow */ }
    return false;
  }

  // Try loading initial state
  load();

  // -----------------------------------------------------------------
  // Core: append a node
  // -----------------------------------------------------------------
  function append(data, parentId) {
    if (data === undefined) data = null;
    if (parentId === undefined) {
      // Default parent: the currently active leaf, if any
      parentId = activeId;
    }
    var id = uid();
    var node = {
      id: id,
      parentId: parentId || null,
      childIds: [],
      data: data,
      createdAt: Date.now()
    };
    nodes[id] = node;

    if (parentId && nodes[parentId]) {
      nodes[parentId].childIds.push(id);
    } else {
      // New root
      rootIds.push(id);
    }
    activeId = id;
    save();
    return id;
  }

  // -----------------------------------------------------------------
  // Branch -- explicit alias for append with parent
  // -----------------------------------------------------------------
  function branch(fromId, data) {
    if (!nodes[fromId]) throw new Error('branching: unknown node ' + fromId);
    var parentId = nodes[fromId].parentId;
    var id = append(data, parentId);
    return id;
  }

  // -----------------------------------------------------------------
  // Get / list
  // -----------------------------------------------------------------
  function get(id) {
    var n = nodes[id];
    if (!n) return null;
    return {
      id: n.id,
      parentId: n.parentId,
      childIds: n.childIds.slice(),
      data: n.data,
      createdAt: n.createdAt
    };
  }

  function getAll() {
    var out = [];
    for (var id in nodes) out.push(get(id));
    return out;
  }

  function getRoots() {
    return rootIds.map(get);
  }

  // -----------------------------------------------------------------
  // Path: from root to a given node (or active leaf)
  // -----------------------------------------------------------------
  function path(toId) {
    var target = toId || activeId;
    if (!target || !nodes[target]) return [];
    var chain = [];
    var cursor = nodes[target];
    while (cursor) {
      chain.unshift(get(cursor.id));
      cursor = cursor.parentId ? nodes[cursor.parentId] : null;
    }
    return chain;
  }

  function activeBranch() {
    return path(activeId);
  }

  function setActive(id) {
    if (!nodes[id]) throw new Error('branching: unknown node ' + id);
    activeId = id;
    save();
    return id;
  }

  function getActiveId() {
    return activeId;
  }

  // -----------------------------------------------------------------
  // Siblings -- nodes with the same parent
  // -----------------------------------------------------------------
  function siblings(id) {
    var n = nodes[id];
    if (!n) return [];
    var parentChildren;
    if (n.parentId && nodes[n.parentId]) {
      parentChildren = nodes[n.parentId].childIds;
    } else {
      parentChildren = rootIds;
    }
    return parentChildren.map(get);
  }

  // -----------------------------------------------------------------
  // Switching to a sibling -- pick by index relative to current
  // -----------------------------------------------------------------
  function switchSibling(id, direction) {
    var sibs = siblings(id);
    if (sibs.length === 0) return null;
    var idx = sibs.findIndex(function(s) { return s.id === id; });
    if (idx === -1) return null;
    var newIdx;
    if (direction === 'next' || direction === 1) {
      newIdx = (idx + 1) % sibs.length;
    } else if (direction === 'prev' || direction === -1) {
      newIdx = (idx - 1 + sibs.length) % sibs.length;
    } else if (typeof direction === 'number') {
      newIdx = Math.max(0, Math.min(sibs.length - 1, direction));
    } else {
      return null;
    }
    return sibs[newIdx].id;
  }

  // -----------------------------------------------------------------
  // Switch branch by selecting a node; updates active to that node's deepest descendant
  // along its primary (first-child) path
  // -----------------------------------------------------------------
  function switchTo(id) {
    if (!nodes[id]) throw new Error('branching: unknown node ' + id);
    // Walk down following first child until leaf
    var cursor = nodes[id];
    while (cursor.childIds.length > 0) {
      cursor = nodes[cursor.childIds[0]];
    }
    activeId = cursor.id;
    save();
    return activeId;
  }

  // -----------------------------------------------------------------
  // Remove a node and all its descendants
  // -----------------------------------------------------------------
  function remove(id) {
    var n = nodes[id];
    if (!n) return false;
    // Collect descendants
    var toRemove = [];
    var stack = [id];
    while (stack.length) {
      var cur = stack.pop();
      toRemove.push(cur);
      var node = nodes[cur];
      if (node) {
        for (var i = 0; i < node.childIds.length; i++) stack.push(node.childIds[i]);
      }
    }
    // Detach from parent
    if (n.parentId && nodes[n.parentId]) {
      nodes[n.parentId].childIds = nodes[n.parentId].childIds.filter(function(c) { return c !== id; });
    } else {
      rootIds = rootIds.filter(function(r) { return r !== id; });
    }
    // Delete all
    for (var k = 0; k < toRemove.length; k++) delete nodes[toRemove[k]];
    // Reset active if needed
    if (toRemove.indexOf(activeId) !== -1) {
      activeId = (n.parentId && nodes[n.parentId]) ? n.parentId : (rootIds[0] || null);
    }
    save();
    return true;
  }

  function clear() {
    nodes = {};
    rootIds = [];
    activeId = null;
    save();
  }

  // -----------------------------------------------------------------
  // Serialize / deserialize
  // -----------------------------------------------------------------
  function serialize() {
    return JSON.parse(JSON.stringify({
      nodes: nodes,
      rootIds: rootIds,
      activeId: activeId
    }));
  }

  function deserialize(state) {
    if (!state || typeof state !== 'object') throw new Error('branching: invalid state');
    nodes = state.nodes ? JSON.parse(JSON.stringify(state.nodes)) : {};
    rootIds = state.rootIds ? state.rootIds.slice() : [];
    activeId = state.activeId || null;
    save();
  }

  function size() {
    var n = 0;
    for (var k in nodes) n++;
    return n;
  }

  function isLeaf(id) {
    var n = nodes[id];
    return n && n.childIds.length === 0;
  }

  return {
    append: append,
    branch: branch,
    get: get,
    getAll: getAll,
    getRoots: getRoots,
    path: path,
    activeBranch: activeBranch,
    setActive: setActive,
    getActiveId: getActiveId,
    siblings: siblings,
    switchSibling: switchSibling,
    switchTo: switchTo,
    remove: remove,
    clear: clear,
    serialize: serialize,
    deserialize: deserialize,
    size: size,
    isLeaf: isLeaf
  };
}

window.weld.branching = {
  create: createTree,
  version: '1.0.0'
};

})();
