// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

// COMPONENTS
function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

function tabs(opts) {
  opts = opts || {};
  var items = opts.items || [];
  var active = opts.active || (items[0] && items[0].id);
  var tabBar = el('div', { class: 'w-tabs' });
  var panel = el('div', { class: 'w-tab-panel' });

  function render(activeId) {
    tabBar.innerHTML = '';
    items.forEach(function(it){
      var btn = el('button', {
        class: 'w-tab' + (it.id === activeId ? ' w-tab-active' : ''),
        onclick: function(){ render(it.id); if (opts.onChange) opts.onChange(it.id); },
      }, it.label);
      tabBar.appendChild(btn);
    });
    panel.innerHTML = '';
    var found = items.find(function(it){ return it.id === activeId; });
    if (found && found.content) {
      panel.appendChild(typeof found.content === 'function' ? found.content() : found.content);
    }
  }
  render(active);

  var wrap = el('div', { class: 'w-tabs-wrap' }, [tabBar, panel]);
  wrap.setActive = function(id){ render(id); };
  return wrap;
}

function lightbox(opts) {
  opts = opts || {};
  lightboxInjectCss();
  var items = [];
  var currentIdx = -1;
  var overlay = document.createElement('div');
  overlay.className = 'w-lb';
  var img = document.createElement('img');
  img.className = 'w-lb-img';

  closeBtn.className = 'w-lb-close';
  closeBtn.textContent = 'CLOSE';
  var prevBtn = document.createElement('button');
  prevBtn.className = 'w-lb-nav w-lb-prev';
  prevBtn.innerHTML = '&#8249;';
  var nextBtn = document.createElement('button');
  nextBtn.className = 'w-lb-nav w-lb-next';
  nextBtn.innerHTML = '&#8250;';
  var metaEl = document.createElement('div');
  metaEl.className = 'w-lb-meta';
  overlay.appendChild(closeBtn);
  overlay.appendChild(prevBtn);
  overlay.appendChild(nextBtn);
  overlay.appendChild(img);
  overlay.appendChild(metaEl);
  document.body.appendChild(overlay);

  function updateNav() {
    prevBtn.style.display = items.length > 1 ? '' : 'none';
    nextBtn.style.display = items.length > 1 ? '' : 'none';
  }

  function show(idx) {
    if (idx < 0 || idx >= items.length) return;
    currentIdx = idx;
    var item = items[idx];
    img.src = typeof item === 'string' ? item : (item.src || item.url || '');
    img.alt = (item && item.alt) || '';
    metaEl.textContent = (item && item.meta) || '';
    metaEl.style.display = metaEl.textContent ? '' : 'none';
    updateNav();
  }

  function open(src, openOpts) {
    openOpts = openOpts || {};
    if (Array.isArray(src)) {
      items = src;
      show(openOpts.index || 0);
    } else {
      items = [src];
      show(0);
    }
    overlay.classList.add('open');
  }

  function close() {
    overlay.classList.remove('open');
    img.src = '';
    if (opts.onClose) opts.onClose();
  }

  function next() {
    if (items.length < 2) return;
    show((currentIdx + 1) % items.length);
  }

  function prev() {
    if (items.length < 2) return;
    show((currentIdx - 1 + items.length) % items.length);
  }

  overlay.addEventListener('click', function(e) {
    if (e.target === overlay) close();
  });
  closeBtn.addEventListener('click', close);
  prevBtn.addEventListener('click', function(e) { e.stopPropagation(); prev(); });
  nextBtn.addEventListener('click', function(e) { e.stopPropagation(); next(); });

  function keyHandler(e) {
    if (!overlay.classList.contains('open')) return;
    if (e.key === 'Escape') close();
    else if (e.key === 'ArrowRight' || e.key === 'ArrowDown') next();
    else if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') prev();
  }
  document.addEventListener('keydown', keyHandler);

  function destroy() {
    document.removeEventListener('keydown', keyHandler);
    if (overlay.parentNode) overlay.parentNode.removeChild(overlay);
  }

  return {
    open: open,
    close: close,
    next: next,
    prev: prev,
    destroy: destroy,
    el: overlay,
    isOpen: function() { return overlay.classList.contains('open'); }
  };
}

// CLIPBOARD — copy text with optional toast feedback
function clipboard(text, msg) {
  var p;
  if (navigator.clipboard && navigator.clipboard.writeText) {
    p = navigator.clipboard.writeText(String(text));
  } else {
    p = new Promise(function(resolve, reject) {
      try {
        var ta = document.createElement('textarea');
        ta.value = String(text);
        ta.style.cssText = 'position:fixed;left:-9999px;';
        document.body.appendChild(ta);
        ta.select();
        document.execCommand('copy');
        document.body.removeChild(ta);
        resolve();
      } catch (e) { reject(e); }
    });
  }
  return p.then(function() {
    if (msg !== false && window.weld && window.weld.toast) {
      window.weld.toast.show(msg || 'Copied to clipboard.', 'success');
    }
    return true;
  }).catch(function() {
    if (window.weld && window.weld.toast) {
      window.weld.toast.show('Copy failed.', 'error');
    }
    return false;
  });
}

<script>(function(){

if (!window.weld) window.weld = {};

// -----------------------------------------------------------------
// weld.gallery: Image grid with IndexedDB blob storage.
// Manifest tracks metadata per image (prompt, seed, timestamp, fav).
// Integrates with weld.ui.lightbox for fullscreen viewing.
// Eviction: when full, drop oldest non-favorited first.
// -----------------------------------------------------------------

var DB_STORE = 'blobs';

function uid() {
  return 'img_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 6);
}

function createGallery(opts) {
  opts = opts || {};
  var dbName = opts.dbName || 'weld-gallery';
  var maxItems = opts.maxItems || 200;
  var manifest = [];
  var dbInstance = null;
  var dbPromise = null;
  var container = null;
  var lightboxInstance = null;
  var sortMode = opts.sortMode || 'newest';
  var favOnly = false;
  var changeListeners = [];

  // -------- IndexedDB for blobs --------
  function openDb() {
    if (dbInstance) return Promise.resolve(dbInstance);
    if (dbPromise) return dbPromise;
    dbPromise = new Promise(function(resolve, reject) {
      var req = indexedDB.open(dbName, 1);
      req.onupgradeneeded = function(e) {
        var db = e.target.result;
        if (!db.objectStoreNames.contains(DB_STORE)) {
          db.createObjectStore(DB_STORE);
        }
        if (!db.objectStoreNames.contains('meta')) {
          db.createObjectStore('meta');
        }
      };
      req.onsuccess = function(e) { dbInstance = e.target.result; resolve(dbInstance); };
      req.onerror = function(e) { dbPromise = null; reject(e.target.error); };
    });
    return dbPromise;
  }

  function idbOp(store, mode, fn) {
    return openDb().then(function(db) {
      return new Promise(function(resolve, reject) {
        var tx = db.transaction(store, mode);
        var s = tx.objectStore(store);
        var req = fn(s);
        if (req && req.onsuccess !== undefined) {
          req.onsuccess = function() { resolve(req.result); };
          req.onerror = function() { reject(req.error); };
        } else {
          tx.oncomplete = function() { resolve(); };
          tx.onerror = function() { reject(tx.error); };
        }
      });
    });
  }

  function saveManifest() {
    return idbOp('meta', 'readwrite', function(s) { return s.put(manifest, 'manifest'); });
  }

  function loadManifest() {
    return idbOp('meta', 'readonly', function(s) { return s.get('manifest'); }).then(function(m) {
      manifest = m || [];
    });
  }

  function emit() {
    for (var i = 0; i < changeListeners.length; i++) {
      try { changeListeners[i](manifest); } catch (e) {}
    }
  }

  // -------- Core operations --------
  function store(blob, meta) {
    if (!(blob instanceof Blob)) {
      return Promise.reject(new Error('gallery.store: expected a Blob'));
    }
    meta = meta || {};
    var id = uid();
    var entry = {
      id: id,
      prompt: meta.prompt || '',
      seed: meta.seed != null ? meta.seed : null,
      resolution: meta.resolution || null,
      createdAt: Date.now(),
      favorited: false,
      metadata: meta.metadata || null
    };

    return idbOp(DB_STORE, 'readwrite', function(s) { return s.put(blob, id); }).then(function() {
      // Eviction
      while (manifest.length >= maxItems) {
        var evictIdx = -1;
        for (var i = 0; i < manifest.length; i++) {
          if (!manifest[i].favorited) { evictIdx = i; break; }
        }
        if (evictIdx === -1) evictIdx = 0;
        var removed = manifest.splice(evictIdx, 1)[0];
        // Best-effort blob cleanup
        idbOp(DB_STORE, 'readwrite', function(s) { return s.delete(removed.id); }).catch(function() {});
      }
      manifest.push(entry);
      return saveManifest();
    }).then(function() {
      emit();
      return id;
    });
  }

  function remove(id) {
    manifest = manifest.filter(function(e) { return e.id !== id; });
    return Promise.all([
      idbOp(DB_STORE, 'readwrite', function(s) { return s.delete(id); }).catch(function() {}),
      saveManifest()
    ]).then(function() { emit(); return true; });
  }

  function removeMany(ids) {
    var idSet = {};
    for (var i = 0; i < ids.length; i++) idSet[ids[i]] = true;
    manifest = manifest.filter(function(e) { return !idSet[e.id]; });
    var chain = Promise.resolve();
    ids.forEach(function(id) {
      chain = chain.then(function() {
        return idbOp(DB_STORE, 'readwrite', function(s) { return s.delete(id); }).catch(function() {});
      });
    });
    return chain.then(saveManifest).then(function() { emit(); });
  }

  function clear() {
    var ids = manifest.map(function(e) { return e.id; });
    manifest = [];
    return removeMany(ids).then(saveManifest).then(function() { emit(); });
  }

  function toggleFavorite(id) {
    var entry = manifest.find(function(e) { return e.id === id; });
    if (!entry) return Promise.resolve(null);
    entry.favorited = !entry.favorited;
    return saveManifest().then(function() { emit(); return entry.favorited; });
  }

  function getBlob(id) {
    return idbOp(DB_STORE, 'readonly', function(s) { return s.get(id); });
  }

  function getBlobUrl(id) {
    return getBlob(id).then(function(blob) {
      return blob ? URL.createObjectURL(blob) : null;
    });
  }

  // -------- Sorted/filtered view --------
  function getView() {
    var view = manifest.slice();
    if (favOnly) view = view.filter(function(e) { return e.favorited; });
    switch (sortMode) {
      case 'oldest':
        view.sort(function(a, b) { return (a.createdAt || 0) - (b.createdAt || 0); });
        break;
      case 'favorites':
        view.sort(function(a, b) {
          if (a.favorited !== b.favorited) return a.favorited ? -1 : 1;
          return (b.createdAt || 0) - (a.createdAt || 0);
        });
        break;
      default:
        view.sort(function(a, b) { return (b.createdAt || 0) - (a.createdAt || 0); });
    }
    return view;
  }

  function setSort(mode) { sortMode = mode; emit(); }
  function setFavOnly(v) { favOnly = !!v; emit(); }

  // -------- Rendering --------
  function render(targetEl) {
    container = typeof targetEl === 'string' ? document.querySelector(targetEl) : targetEl;
    if (!container) return Promise.reject(new Error('gallery.render: container not found'));
    return loadManifest().then(function() { return renderView(); });
  }

  function renderView() {
    if (!container) return Promise.resolve();
    var view = getView();
    container.innerHTML = '';
    if (view.length === 0) {
      container.innerHTML = '<div style="color:var(--weld-muted,#7d8590);font:13px var(--weld-mono,monospace);padding:40px;text-align:center;">No images yet.</div>';
      return Promise.resolve();
    }
    var promises = [];
    view.forEach(function(entry) {
      var cell = document.createElement('div');
      cell.className = 'w-gallery-item' + (entry.favorited ? ' is-fav' : '');
      cell.dataset.id = entry.id;
      cell.innerHTML = '<div class="w-gallery-loading"></div>';
      container.appendChild(cell);
      var p = getBlobUrl(entry.id).then(function(url) {
        if (!url) { cell.innerHTML = '<div class="w-gallery-empty">missing</div>'; return; }
        var img = document.createElement('img');
        img.src = url;
        img.alt = entry.prompt || '';
        img.addEventListener('click', function() { openLightbox(entry.id); });
        cell.innerHTML = '';
        cell.appendChild(img);
        // Seed tag
        if (entry.seed != null) {
          var tag = document.createElement('div');
          tag.className = 'w-gallery-seed';
          tag.textContent = 'seed ' + entry.seed;
          cell.appendChild(tag);
        }
      });
      promises.push(p);
    });
    return Promise.all(promises);
  }

  function openLightbox(startId) {
    if (!window.weld || !window.weld.ui || !window.weld.ui.lightbox) return;
    if (!lightboxInstance) lightboxInstance = window.weld.ui.lightbox();
    var view = getView();
    var items = [];
    var startIdx = 0;
    view.forEach(function(entry, i) {
      items.push({ src: '', id: entry.id, meta: entry.prompt || '', alt: entry.prompt || '' });
      if (entry.id === startId) startIdx = i;
    });
    // Resolve blob URLs
    var chain = Promise.resolve();
    items.forEach(function(item, idx) {
      chain = chain.then(function() {
        return getBlobUrl(item.id).then(function(url) { items[idx].src = url || ''; });
      });
    });
    chain.then(function() {
      lightboxInstance.open(items, { index: startIdx });
    });
  }

  // -------- Download --------
  function downloadImage(id, filename) {
    return getBlob(id).then(function(blob) {
      if (!blob) return false;
      var url = URL.createObjectURL(blob);
      var a = document.createElement('a');
      a.href = url;
      a.download = filename || (id + '.jpg');
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      setTimeout(function() { URL.revokeObjectURL(url); }, 1000);
      return true;
    });
  }

  // -------- Init --------
  function init() {
    return loadManifest();
  }

  function onChange(fn) {
    changeListeners.push(fn);
    return function() {
      var idx = changeListeners.indexOf(fn);
      if (idx >= 0) changeListeners.splice(idx, 1);
    };
  }

  return {
    init: init,
    store: store,
    remove: remove,
    removeMany: removeMany,
    clear: clear,
    toggleFavorite: toggleFavorite,
    getBlob: getBlob,
    getBlobUrl: getBlobUrl,
    getView: getView,
    getManifest: function() { return manifest.slice(); },
    setSort: setSort,
    setFavOnly: setFavOnly,
    render: render,
    renderView: renderView,
    downloadImage: downloadImage,
    onChange: onChange,
    count: function() { return manifest.length; },
    maxItems: function() { return maxItems; }
  };
}
