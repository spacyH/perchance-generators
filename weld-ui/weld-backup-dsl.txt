$meta
  title       = weld.backup
  description = Format-versioned export/import with conflict resolution and migration. Share links: compress + upload + URL. Consume shared data on load.
  tags        = weld, backup, export, import, share, plugin

$output() =>
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.backup) return window.weld.backup;

  var FORMAT_VERSION = 1;     // bumped only if envelope format changes (not the data inside)
  var SIGNATURE = 'weld-backup';

  // Build an export envelope from arbitrary data
  function exportData(opts) {
    opts = opts || {};
    var envelope = {
      signature: SIGNATURE,
      formatVersion: FORMAT_VERSION,
      dataVersion: opts.version != null ? opts.version : 1,
      generator: opts.generator || 'unknown',
      createdAt: new Date().toISOString(),
      metadata: opts.metadata || {},
      data: opts.data || {},
    };
    return envelope;
  }

  // Serialize envelope to a JSON Blob
  function exportBlob(opts) {
    var env = exportData(opts);
    var json = JSON.stringify(env, null, 2);
    return new Blob([json], { type: 'application/json' });
  }

  // Trigger a browser download
  function download(opts) {
    opts = opts || {};
    var blob = exportBlob(opts);
    var fname = opts.filename;
    if (!fname) {
      var stamp = new Date().toISOString().replace(/[:.]/g, '-').slice(0, 19);
      fname = (opts.generator || 'backup') + '-' + stamp + '.weld.json';
    }
    var url = URL.createObjectURL(blob);
    var a = document.createElement('a');
    a.href = url;
    a.download = fname;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    setTimeout(function(){ URL.revokeObjectURL(url); }, 1000);
    return fname;
  }

  // Read a File/Blob/string into an envelope, validating signature
  function readEnvelope(source) {
    return new Promise(function(resolve, reject) {
      if (typeof source === 'string') {
        try { resolve(JSON.parse(source)); }
        catch (e) { reject(new Error('Not valid JSON: ' + e.message)); }
        return;
      }
      if (source instanceof Blob || source instanceof File) {
        var reader = new FileReader();
        reader.onload = function(){
          try { resolve(JSON.parse(reader.result)); }
          catch (e) { reject(new Error('Not valid JSON: ' + e.message)); }
        };
        reader.onerror = function(){ reject(reader.error); };
        reader.readAsText(source);
        return;
      }
      if (source && typeof source === 'object') {
        resolve(source);
        return;
      }
      reject(new Error('Unsupported source type'));
    });
  }

  function validateEnvelope(env) {
    if (!env || typeof env !== 'object') {
      return { ok: false, error: 'Not an object' };
    }
    if (env.signature !== SIGNATURE) {
      return { ok: false, error: 'Missing or wrong signature (expected "' + SIGNATURE + '")' };
    }
    if (env.formatVersion == null) {
      return { ok: false, error: 'Missing formatVersion' };
    }
    if (env.formatVersion > FORMAT_VERSION) {
      return { ok: false, error: 'Envelope is newer (format v' + env.formatVersion + ') than this importer (v' + FORMAT_VERSION + ')' };
    }
    return {
      ok: true,
      formatVersion: env.formatVersion,
      dataVersion: env.dataVersion,
      generator: env.generator,
      createdAt: env.createdAt,
      metadata: env.metadata || {},
    };
  }

  function validate(source) {
    return readEnvelope(source).then(validateEnvelope).catch(function(e){
      return { ok: false, error: e.message };
    });
  }

  // Apply data migrations (provided by caller) to bring dataVersion up to current
  function migrateData(env, migrations) {
    migrations = migrations || {};
    var versions = Object.keys(migrations).map(Number).sort(function(a, b){ return a - b; });
    var maxVer = versions.length ? Math.max.apply(null, versions) : env.dataVersion;
    var data = env.data;
    var fromVer = env.dataVersion;
    if (fromVer < maxVer) {
      for (var v = fromVer + 1; v <= maxVer; v++) {
        if (migrations[v]) {
          data = migrations[v](data) || data;
        }
      }
    }
    return { data: data, fromVersion: fromVer, toVersion: maxVer };
  }

  // Import an envelope - returns the data (optionally migrated)
  function importData(source, opts) {
    opts = opts || {};
    return readEnvelope(source).then(function(env){
      var v = validateEnvelope(env);
      if (!v.ok) throw new Error(v.error);
      var migrated = migrateData(env, opts.migrations);
      return {
        data: migrated.data,
        generator: env.generator,
        fromVersion: migrated.fromVersion,
        toVersion: migrated.toVersion,
        metadata: env.metadata || {},
        createdAt: env.createdAt,
        importedAt: new Date().toISOString(),
      };
    });
  }

  // Show a hidden file picker, resolve with selected File
  function pickFile(accept) {
    return new Promise(function(resolve, reject) {
      var input = document.createElement('input');
      input.type = 'file';
      input.accept = accept || '.json,.weld.json,application/json';
      input.style.display = 'none';
      input.onchange = function(){
        var f = input.files && input.files[0];
        document.body.removeChild(input);
        if (f) resolve(f);
        else reject(new Error('No file selected'));
      };
      document.body.appendChild(input);
      input.click();
    });
  }

  // Dump an entire weld.migrate DB to an envelope
  function fromDb(db, opts) {
    opts = opts || {};
    var stores = opts.stores || db.storeNames.filter(function(s){ return s.indexOf('_weld') !== 0; });
    var data = {};
    var chain = Promise.resolve();
    stores.forEach(function(s){
      chain = chain.then(function(){
        return db.getAll(s).then(function(items){ data[s] = items; });
      });
    });
    return chain.then(function(){
      return exportData({
        version: opts.version != null ? opts.version : (db.version || 1),
        generator: opts.generator || db.name,
        metadata: Object.assign({ stores: stores, recordCount: countTotal(data) }, opts.metadata || {}),
        data: data,
      });
    });
  }

  // Convenience - dump and download in one call
  function downloadFromDb(db, opts) {
    return fromDb(db, opts).then(function(env){
      var blob = new Blob([JSON.stringify(env, null, 2)], { type: 'application/json' });
      var fname = (opts && opts.filename) || (env.generator + '-' + new Date().toISOString().replace(/[:.]/g, '-').slice(0, 19) + '.weld.json');
      var url = URL.createObjectURL(blob);
      var a = document.createElement('a');
      a.href = url; a.download = fname;
      document.body.appendChild(a); a.click(); document.body.removeChild(a);
      setTimeout(function(){ URL.revokeObjectURL(url); }, 1000);
      return { filename: fname, env: env };
    });
  }

  // Restore an envelope into a weld.migrate DB
  // mode: 'merge' (default) - put each item, overwriting same-key
  //       'replace'          - clear store first, then put all
  //       'append'           - only put items whose key doesn't exist
  function toDb(source, db, opts) {
    opts = opts || {};
    var mode = opts.mode || 'merge';
    var onConflict = opts.onConflict; // (existing, incoming) => resolved | null to skip

    return importData(source, { migrations: opts.migrations }).then(function(result){
      var data = result.data;
      var stores = opts.stores || Object.keys(data);
      var summary = { imported: {}, skipped: {}, conflicts: {} };

      var chain = Promise.resolve();
      stores.forEach(function(s){
        if (!data[s]) return;
        var items = data[s];

        chain = chain.then(function(){
          if (mode === 'replace') {
            return db.clear(s).then(function(){
              return db.bulkPut(s, items);
            }).then(function(){
              summary.imported[s] = items.length;
            });
          }

          if (mode === 'append') {
            // Check each item's key
            var skipped = 0, imported = 0;
            var toPut = [];
            var lookups = items.map(function(item){
              // We need the keyPath of the store to know the key - use raw IDB
              var t = db._db.transaction([s], 'readonly');
              var store = t.objectStore(s);
              var keyPath = store.keyPath;
              var key = (typeof keyPath === 'string') ? item[keyPath] : null;
              return new Promise(function(resolve){
                if (key == null) { toPut.push(item); imported++; resolve(); return; }
                var r = store.get(key);
                r.onsuccess = function(){
                  if (r.result == null) { toPut.push(item); imported++; }
                  else { skipped++; }
                  resolve();
                };
                r.onerror = function(){ resolve(); };
              });
            });
            return Promise.all(lookups).then(function(){
              summary.imported[s] = imported;
              summary.skipped[s] = skipped;
              if (toPut.length) return db.bulkPut(s, toPut);
            });
          }

          // merge mode
          if (!onConflict) {
            return db.bulkPut(s, items).then(function(){
              summary.imported[s] = items.length;
            });
          }

          // merge with conflict resolution
          var imported = 0, conflicts = 0;
          var resolved = [];
          var lookups = items.map(function(item){
            var t = db._db.transaction([s], 'readonly');
            var store = t.objectStore(s);
            var keyPath = store.keyPath;
            var key = (typeof keyPath === 'string') ? item[keyPath] : null;
            return new Promise(function(resolve){
              if (key == null) { resolved.push(item); imported++; resolve(); return; }
              var r = store.get(key);
              r.onsuccess = function(){
                var existing = r.result;
                if (existing == null) { resolved.push(item); imported++; }
                else {
                  conflicts++;
                  var merged = onConflict(existing, item, s);
                  if (merged != null) resolved.push(merged);
                }
                resolve();
              };
              r.onerror = function(){ resolve(); };
            });
          });
          return Promise.all(lookups).then(function(){
            summary.imported[s] = imported;
            summary.conflicts[s] = conflicts;
            if (resolved.length) return db.bulkPut(s, resolved);
          });
        });
      });

      return chain.then(function(){
        summary.fromVersion = result.fromVersion;
        summary.toVersion = result.toVersion;
        summary.generator = result.generator;
        return summary;
      });
    });
  }

  // Count total records across all stores
  function countTotal(data) {
    var n = 0;
    Object.keys(data).forEach(function(k){
      if (Array.isArray(data[k])) n += data[k].length;
    });
    return n;
  }

  // SHARE LINK — compress + upload + URL
  // uploader: the uploadPlugin function (from DSL import)
  // opts: { paramName?, generatorName? }
  function share(data, uploader, opts) {
    opts = opts || {};
    if (typeof uploader !== 'function') {
      return Promise.reject(new Error('backup.share: uploader function required (pass uploadPlugin)'));
    }
    var json = JSON.stringify(data);
    // Compress with gzip if available, otherwise send raw
    var blobPromise;
    if (typeof CompressionStream !== 'undefined') {
      blobPromise = new Response(
        new Blob([json]).stream().pipeThrough(new CompressionStream('gzip'))
      ).blob();
    } else {
      blobPromise = Promise.resolve(new Blob([json], { type: 'application/json' }));
    }
    return blobPromise.then(function(blob) {
      return Promise.resolve(uploader(blob));
    }).then(function(result) {
      if (!result || result.error) {
        throw new Error('upload failed: ' + (result && result.error || 'unknown'));
      }
      var url = String(result.url || result);
      // Extract filename from CDN URL
      var fileName = url.replace('https://user.uploads.dev/file/', '');
      var paramName = opts.paramName || 'data';
      var base = 'https://perchance.org/' + (opts.generatorName || window.generatorName || 'generator');
      return base + '?' + paramName + '=' + encodeURIComponent(fileName);
    });
  }

  // CONSUME SHARE — check URL for shared data, fetch + decompress
  // fetcher: a fetch-compatible function (superFetch or window.fetch)
  // opts: { paramName?, onFound? }
  function consumeShare(fetcher, opts) {
    opts = opts || {};
    var paramName = opts.paramName || 'data';
    var params;
    try { params = new URL(window.location.href).searchParams; }
    catch (e) { return Promise.resolve(null); }
    var raw = params.get(paramName);
    if (!raw) return Promise.resolve(null);

    // Extract just the filename (handle legacy formats with ~ separator)
    var fileName = raw.split('~').slice(-1)[0];
    if (!fileName) return Promise.resolve(null);

    var fetchFn = typeof fetcher === 'function' ? fetcher : window.fetch;
    var cdnUrl = 'https://user.uploads.dev/file/' + fileName;
    return Promise.resolve(fetchFn(cdnUrl, {
      signal: typeof AbortSignal !== 'undefined' && AbortSignal.timeout ? AbortSignal.timeout(15000) : undefined
    })).then(function(r) {
      if (r && typeof r.blob === 'function') return r.blob();
      // superFetch might return different shape
      if (r instanceof Blob) return r;
      return new Blob([typeof r === 'string' ? r : JSON.stringify(r)]);
    }).then(function(blob) {
      // Try to decompress
      if (typeof DecompressionStream !== 'undefined') {
        return new Response(
          blob.stream().pipeThrough(new DecompressionStream('gzip'))
        ).text().catch(function() {
          // Not gzipped — try raw
          return blob.text();
        });
      }
      return blob.text();
    }).then(function(text) {
      try { return JSON.parse(text); }
      catch (e) { throw new Error('backup.consumeShare: invalid JSON in share data'); }
    }).then(function(data) {
      if (typeof opts.onFound === 'function') opts.onFound(data);
      return data;
    });
  }

  // PUBLIC API
  // ---- upload-plugin helpers: unguessable-id blob store / link-identity
  //      (same injected-uploader style as share/consumeShare) ----
  var UPLOAD_PREFIX = 'https://user.uploads.dev/file/';
  function parseUploadId(url) {
    try {
      var base = (typeof location !== 'undefined' && location.href) ? location.href : 'https://x/';
      var path = new URL(String(url), base).pathname;
      var name = (path.split('/').filter(Boolean).pop() || '');
      var id = (name.split('.')[0] || '').toLowerCase().replace(/[^0-9a-f]/g, '');
      return (id.length === 32 && /^[0-9a-f]+$/.test(id)) ? id : '';
    } catch (e) { return ''; }
  }
  function safeUploadUrl(url) {
    try { var u = new URL(String(url)); if (u.protocol !== 'https:') return ''; if (u.hostname !== 'user.uploads.dev') return ''; if (u.pathname.indexOf('/file/') !== 0) return ''; return u.href; } catch (e) { return ''; }
  }
  function upload(uploader, blobOrString) {
    if (typeof uploader !== 'function') return Promise.reject(new Error('backup.upload: uploader function required (pass uploadPlugin)'));
    return Promise.resolve(uploader(blobOrString)).then(function (res) {
      if (!res || res.error || !res.url) throw new Error('backup.upload: upload failed: ' + ((res && res.error) || 'unknown'));
      return { url: res.url, id: parseUploadId(res.url) };
    });
  }
  function uploadJson(uploader, obj) { return upload(uploader, JSON.stringify(obj)); }
  function fetchJson(fetcher, url) {
    var f = fetcher || (typeof fetch !== 'undefined' ? fetch : null);
    if (typeof f !== 'function') return Promise.reject(new Error('backup.fetchJson: fetcher function required'));
    return Promise.resolve(f(url)).then(function (r) { return (r && typeof r.text === 'function') ? r.text() : r; }).then(function (t) { try { return JSON.parse(t); } catch (e) { throw new Error('backup.fetchJson: invalid JSON'); } });
  }

  window.weld.backup = {
    exportData: exportData,
    exportBlob: exportBlob,
    download: download,
    importData: importData,
    validate: validate,
    pickFile: pickFile,
    fromDb: fromDb,
    downloadFromDb: downloadFromDb,
    toDb: toDb,
    share: share,
    upload: upload,
    uploadJson: uploadJson,
    fetchJson: fetchJson,
    parseUploadId: parseUploadId,
    safeUploadUrl: safeUploadUrl,
    consumeShare: consumeShare,
    FORMAT_VERSION: FORMAT_VERSION,
    version: '1.1.0',
  };

  return window.weld.backup;
