$meta
  title       = weld.image
  description = Promise-based wrapper around text-to-image-plugin. Style presets, resolution helpers, batching with progress callbacks, variations, retry with exponential backoff.
  tags        = weld, image, t2i, generation, plugin

t2iSource     = {import:text-to-image-plugin}

$output() =>
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.image) { window.weld.image.setT2i(t2iSource); return window.weld.image; }

  // -----------------------------------------------------------------
  // weld.image: Promise-based wrapper around Perchance's text-to-image-plugin.
  // Adds batching, style presets, retry, resolution helpers.
  //
  // The actual t2i function comes from Perchance's text-to-image-plugin,
  // which must be imported in the DSL panel under any alias, e.g.:
  //   t2iSource = the import of text-to-image-plugin
  // The DSL form passes that function into weld.image.setT2i(...).
  // Standalone JS callers must call weld.image.setT2i(fn) themselves.
  // -----------------------------------------------------------------

  var t2iFn = null;

  // Resolution presets
  var RESOLUTIONS = {
    square:      '512x512',
    portrait:    '512x768',
    landscape:   '768x512',
    largeSquare: '768x768'
  };

  // Positive-only style presets (Perchance backend ignores negatives)
  var STYLES = {
    photo:      ', photorealistic, sharp focus, detailed, 8k, professional photography',
    cinematic:  ', cinematic shot, dramatic lighting, depth of field, film grain, atmospheric',
    anime:      ', anime style, cel shaded, vibrant, detailed line art',
    painting:   ', oil painting, classical art, brush strokes, painterly, masterpiece',
    digital:    ', digital art, concept art, trending on artstation, highly detailed',
    watercolor: ', watercolor painting, soft edges, paper texture, flowing colors',
    sketch:     ', pencil sketch, hand drawn, graphite, shading, detailed lines',
    lowpoly:    ', low poly 3d render, geometric, isometric, minimalist',
    pixelart:   ', pixel art, 16-bit style, retro game graphics, pixelated',
    cyberpunk:  ', cyberpunk, neon lights, futuristic, blade runner aesthetic, rain'
  };

  function setT2i(fn) {
    t2iFn = fn;
  }

  function getT2i() {
    if (!t2iFn) {
      throw new Error('weld.image: no text-to-image function set. Call weld.image.setT2i(fn) where fn is the imported text-to-image-plugin.');
    }
    return t2iFn;
  }

  function styleNames() { return Object.keys(STYLES); }
  function resolutionNames() { return Object.keys(RESOLUTIONS); }

  function withStyle(prompt, styleName) {
    var suffix = STYLES[styleName] || '';
    return String(prompt) + suffix;
  }

  function withResolution(opts, resName) {
    var res = RESOLUTIONS[resName];
    if (!res) return Object.assign({}, opts);
    var parts = res.split('x');
    return Object.assign({}, opts, { width: parseInt(parts[0], 10), height: parseInt(parts[1], 10) });
  }

  // --- truth-layer helpers (text-to-image-plugin's sharp edges) ---
  // Only four resolutions are accepted by the backend; anything else is
  // rejected. Validate up front with a clear error instead of a silent fail.
  var VALID_WH = { '512x512': 1, '512x768': 1, '768x512': 1, '768x768': 1 };
  function resolutionOf(opts) {
    if (!opts) return null;
    if (opts.resolution && VALID_WH[opts.resolution]) return opts.resolution;
    if (opts.width && opts.height) return opts.width + 'x' + opts.height;
    return null;
  }
  function validateResolution(opts) {
    var wh = resolutionOf(opts);
    if (wh && !VALID_WH[wh]) {
      throw new Error('weld.image: unsupported resolution "' + wh + '". Use one of 512x512, 512x768, 768x512, 768x768 (or a preset: ' + resolutionNames().join('/') + ').');
    }
  }
  function promptText(opts) { return String((opts && (opts.prompt != null ? opts.prompt : opts.text)) || '').trim(); }

  // The awaited t2i result is a BOXED String carrying canvas/dataUrl/inputs.
  // Comparing it with === fails and reading it as text needs coercion.
  // normalizeResult turns it into a plain, predictable object.
  function normalizeResult(data) {
    if (data == null) return { dataUrl: null, canvas: null, inputs: null, text: '' };
    var text = '';
    try { text = String(data); } catch (e) { text = ''; }
    return {
      dataUrl: (data.dataUrl != null) ? String(data.dataUrl) : (/^data:/.test(text) ? text : null),
      canvas: data.canvas || null,
      inputs: data.inputs || null,
      seed: (data.inputs && data.inputs.seed != null) ? data.inputs.seed : (data.seed != null ? data.seed : null),
      text: text,
      raw: data
    };
  }

  // Synchronous embed: returns the iframe HTML for direct insertion
  function embed(opts) {
    return String(getT2i()(opts));
  }

  // Async single generation: resolves with the result data
  function generate(opts) {
    return new Promise(function(resolve, reject) {
      // Guard the two silent failure modes up front.
      try {
        validateResolution(opts);
      } catch (e) { reject(e); return; }
      if (!promptText(opts)) {
        reject(new Error('weld.image: empty prompt. text-to-image hangs forever on an empty prompt, so it is rejected here.'));
        return;
      }
      var settled = false;
      var fn = getT2i();
      var enriched = Object.assign({}, opts, {
        onFinish: function(data) {
          if (settled) return;
          settled = true;
          resolve(normalizeResult(data));
        }
      });
      var html;
      try {
        html = String(fn(enriched));
      } catch (e) {
        reject(e);
        return;
      }

      try {
        if (opts && opts.into) {
          var target = typeof opts.into === 'string' ? document.querySelector(opts.into) : opts.into;
          if (target) target.innerHTML = html;
        }
      } catch (e) { /* ignore */ }

      var ms = (opts && opts.timeoutMs) || 120000;
      setTimeout(function() {
        if (!settled) {
          settled = true;
          reject(new Error('weld.image: generation timeout after ' + ms + 'ms'));
        }
      }, ms);
    });
  }

  // Generate N images sequentially with the same prompt (variation via seed)
  function batch(opts, count, onProgress) {
    return new Promise(function(resolve) {
      var results = [];
      var i = 0;
      function step() {
        if (i >= count) { resolve(results); return; }
        var cellOpts = Object.assign({}, opts);
        if (cellOpts.seed === undefined || cellOpts.seed === -1) {
          cellOpts.seed = Math.floor(Math.random() * 1000000);
        } else if (i > 0) {
          cellOpts.seed = opts.seed + i;
        }
        generate(cellOpts).then(function(data) {
          results.push(data);
          if (onProgress) onProgress(i + 1, count, data);
          i++;
          step();
        }).catch(function(e) {
          results.push({ error: e.message });
          if (onProgress) onProgress(i + 1, count, { error: e.message });
          i++;
          step();
        });
      }
      step();
    });
  }

  // Variations: same opts but different prompts
  function variations(prompts, sharedOpts, onProgress) {
    return new Promise(function(resolve) {
      var results = [];
      var i = 0;
      function step() {
        if (i >= prompts.length) { resolve(results); return; }
        var cellOpts = Object.assign({}, sharedOpts, { prompt: prompts[i] });
        generate(cellOpts).then(function(data) {
          results.push(data);
          if (onProgress) onProgress(i + 1, prompts.length, data);
          i++;
          step();
        }).catch(function(e) {
          results.push({ error: e.message, prompt: prompts[i] });
          if (onProgress) onProgress(i + 1, prompts.length, { error: e.message });
          i++;
          step();
        });
      }
      step();
    });
  }

  // Auto-retry on failure with exponential backoff
  function generateWithRetry(opts, maxRetries) {
    maxRetries = maxRetries || 3;
    return new Promise(function(resolve, reject) {
      var attempt = 0;
      var lastError = null;
      function tryOnce() {
        generate(opts).then(resolve).catch(function(e) {
          lastError = e;
          attempt++;
          if (attempt >= maxRetries) { reject(lastError); return; }
          var backoff = 1000 * Math.pow(2, attempt - 1);
          setTimeout(tryOnce, backoff);
        });
      }
      tryOnce();
    });
  }

  // regenerate(opts): reproduce or vary an image deterministically. With an
  // explicit opts.seed, the same prompt+seed reproduces the same image (the
  // "regenerate at full quality from a saved thumbnail" pattern). Pass
  // opts.vary:true to roll a fresh seed for an alternate take.
  function regenerate(opts) {
    opts = opts || {};
    var next = Object.assign({}, opts);
    if (opts.vary || next.seed == null || next.seed === -1) {
      next.seed = Math.floor(Math.random() * 1000000);
    }
    return generate(next).then(function (r) { r.seed = (next.seed != null ? next.seed : r.seed); return r; });
  }

  window.weld.image = {
    setT2i: setT2i,
    withStyle: withStyle,
    withResolution: withResolution,
    styleNames: styleNames,
    resolutionNames: resolutionNames,
    styles: STYLES,
    resolutions: RESOLUTIONS,
    validResolution: function (wh) { return !!VALID_WH[wh]; },
    normalizeResult: normalizeResult,
    embed: embed,
    generate: generate,
    regenerate: regenerate,
    batch: batch,
    variations: variations,
    generateWithRetry: generateWithRetry,
    version: '1.1.0'
  };

  window.weld.image.setT2i(t2iSource);

  return window.weld.image;
