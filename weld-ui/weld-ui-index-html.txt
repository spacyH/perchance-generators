  <div class="hero">
    <h1>weld</h1>
    <p class="tagline">A composable plugin suite for Perchance generators.</p>
    <p class="intro">Each weld plugin is a self-contained module that <strong>welds onto</strong> your generator: load only the ones you need, mix and match freely. Plugins share a common namespace (<code>window.weld.&lt;name&gt;</code>) and design system so the whole thing feels coherent without forcing tight coupling.</p>
    <p style="margin-top:18px;"><a href="https://perchance.org/weld-cookbook" style="display:inline-block;background:var(--accent);color:#fff;padding:9px 18px;border-radius:7px;font:600 13px var(--mono);">Read the Cookbook &rarr;</a> <a href="https://perchance.org/weld-platform" style="display:inline-block;background:var(--faint);color:var(--ink);padding:9px 18px;border-radius:7px;font:600 13px var(--mono);margin-left:8px;">Platform reference &rarr;</a><span style="display:block;margin-top:8px;font-size:12.5px;color:var(--muted);">New to Perchance plugins? Start with how to build one, how imports work, and the brace-interception rules.</span></p>
  </div>

  <div class="banner">
    <h3>HOW TO USE</h3>
    <pre data-codeid="install"></pre>
    <p>Drop the import lines into your generator's lists editor, then trigger each plugin once in your HTML panel (e.g. <code>root.weldUi()</code>) so its <code>$output</code> runs. Each plugin attaches itself to <code>window.weld</code>. Loading order does not matter; double-loading is a no-op.</p>
  </div>

  <div class="principles">
    <div class="principle"><h4>SOFT DEPENDENCIES</h4><p>Plugins use each other when available, work standalone when not.</p></div>
    <div class="principle"><h4>RAW JS</h4><p>All plugin code is plain readable JavaScript. Inspect anything in DevTools.</p></div>
    <div class="principle"><h4>DSL-SAFE</h4><p>Every plugin is audited for Perchance's parser quirks. Drop-in compatible with any generator.</p></div>

// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

// Returns { el, close } and is auto-attached to body
function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = null;
  if (opts.title || opts.dismissable !== false) {
    var titleEl = opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div');
    var dismissEl = opts.dismissable !== false ? closeBtn : null;
    header = el('div', { class: 'w-modal-hdr' }, [titleEl, dismissEl]);
  }

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {

var META_STORE = '_weldMeta';

// Wrap an IDBRequest in a Promise
function req(idbRequest) {
  return new Promise(function(resolve, reject) {
    idbRequest.onsuccess = function(){ resolve(idbRequest.result); };
    idbRequest.onerror = function(){ reject(idbRequest.error); };

// Wait for transaction completion
function txComplete(tx) {
  return new Promise(function(resolve, reject) {
    tx.oncomplete = function(){ resolve(); };
    tx.onerror = function(){ reject(tx.error); };
    tx.onabort = function(){ reject(tx.error || new Error('Transaction aborted')); };
  });
}

// Normalize an index spec - accepts string keyPath OR full object
function normalizeIndex(spec) {
  if (typeof spec === 'string') return { keyPath: spec, unique: false };
  return Object.assign({ unique: false }, spec);
}

// Apply schema upgrade INSIDE upgradeneeded handler
function applySchema(db, tx, schema, fromVer, toVer, log) {
  // Always ensure meta store exists from v1
  if (!db.objectStoreNames.contains(META_STORE)) {
    db.createObjectStore(META_STORE, { keyPath: 'key' });
    log('created meta store ' + META_STORE);
  }

  // Apply each version step
  for (var v = fromVer + 1; v <= toVer; v++) {
    var step = schema[v];
    if (!step) continue;
    log('applying schema v' + v);

    // Create new stores
    if (step.stores) {
      Object.keys(step.stores).forEach(function(name){
        if (db.objectStoreNames.contains(name)) {
          log('  store ' + name + ' already exists, skipping create');
          return;
        }
        var sp = step.stores[name];
        var opts = {};
        if (sp.keyPath) opts.keyPath = sp.keyPath;
        if (sp.autoIncrement) opts.autoIncrement = true;
        var store = db.createObjectStore(name, opts);
        log('  + store ' + name);
        if (sp.indexes) {
          Object.keys(sp.indexes).forEach(function(idxName){
            var idx = normalizeIndex(sp.indexes[idxName]);
            store.createIndex(idxName, idx.keyPath, { unique: idx.unique, multiEntry: !!idx.multiEntry });
            log('    + index ' + idxName);
          });
        }
      });
    }

    // Modify existing stores
    if (step.modify) {
      Object.keys(step.modify).forEach(function(name){
        if (!db.objectStoreNames.contains(name)) {
          log('  cannot modify ' + name + ' (does not exist)');
          return;

            if (store.indexNames.contains(idxName)) {
              store.deleteIndex(idxName);
              log('  - index ' + name + '.' + idxName);
            }
          });
        }

          Object.keys(mods.addIndexes).forEach(function(idxName){
            if (store.indexNames.contains(idxName)) return;

// Open a database with declarative schema + data migrations
function open(name, config) {
  config = config || {};
  var schema = config.schema || {};
  var dataMigrations = config.dataMigrations || {};
  var onProgress = config.onProgress || function(){};

  // Highest schema version
  var versions = Object.keys(schema).map(Number).filter(function(n){ return !isNaN(n); });
  var targetVersion = versions.length ? Math.max.apply(null, versions) : 1;

  return new Promise(function(resolve, reject) {
    var req = indexedDB.open(name, targetVersion);

    req.onerror = function(){
      onProgress('error: ' + (req.error && req.error.message));
      reject(req.error);
    };

    req.onupgradeneeded = function(e){
      var db = req.result;
      var tx = e.target.transaction;
      var oldVer = e.oldVersion;
      var newVer = e.newVersion;
      onProgress('upgrading ' + name + ' from v' + oldVer + ' to v' + newVer);
      try {
        applySchema(db, tx, schema, oldVer, newVer, onProgress);
      } catch (err) {
        onProgress('schema error: ' + err.message);
        tx.abort();
        reject(err);
      }
    };

    req.onblocked = function(){
      onProgress('blocked: another connection has an older version open');
    };

    req.onsuccess = function(){
      var db = req.result;
      db.onversionchange = function(){
        db.close();
        onProgress('closed due to newer version opening');
      };

      // Run data migrations
      runDataMigrations(db, dataMigrations, onProgress).then(function(){
        resolve(wrapDb(db, name));
      }).catch(reject);
    };
  });
}

// Run any data migrations whose version hasn't been applied yet
function runDataMigrations(db, migrations, onProgress) {
  var versions = Object.keys(migrations).map(Number).sort(function(a, b){ return a - b; });
  if (!versions.length) return Promise.resolve();

  // Read applied set from meta
  var tx = db.transaction([META_STORE], 'readonly');
  var store = tx.objectStore(META_STORE);
  return req(store.get(META_KEY)).then(function(entry){
    var applied = (entry && entry.value) || [];
    var todo = versions.filter(function(v){ return applied.indexOf(v) === -1; });
    if (!todo.length) return;

    onProgress('running ' + todo.length + ' data migration(s): ' + todo.join(', '));
    var wrapped = wrapDb(db, '_internal');
    var chain = Promise.resolve();
    todo.forEach(function(v){
      chain = chain.then(function(){
        onProgress('  data migration v' + v + ' starting');
        return Promise.resolve(migrations[v](wrapped));
      }).then(function(){
        onProgress('  data migration v' + v + ' done');
        // Mark applied
        var tx2 = db.transaction([META_STORE], 'readwrite');
        var s2 = tx2.objectStore(META_STORE);
        return req(s2.get(META_KEY)).then(function(cur){
          var arr = (cur && cur.value) || [];
          if (arr.indexOf(v) === -1) arr.push(v);
          return req(s2.put({ key: META_KEY, value: arr }));
        }).then(function(){ return txComplete(tx2); });
      });
    });
    return chain;
  });
}

// Wrap a raw IDBDatabase in a friendly Promise-based API
function wrapDb(db, name) {

    if (!Array.isArray(stores)) stores = [stores];
    var t = db.transaction(stores, mode || 'readonly');
    var helper = {
      _tx: t,
      objectStore: function(s){ return t.objectStore(s); },
      put: function(s, v){ return req(t.objectStore(s).put(v)); },
      get: function(s, k){ return req(t.objectStore(s).get(k)); },
      delete: function(s, k){ return req(t.objectStore(s).delete(k)); },
      getAll: function(s){ return req(t.objectStore(s).getAll()); },
      clear: function(s){ return req(t.objectStore(s).clear()); },
      count: function(s){ return req(t.objectStore(s).count()); },
    };
    var result = Promise.resolve(callback(helper));
    return result.then(function(r){
      return txComplete(t).then(function(){ return r; });
    });
  }

    name: name,
    _db: db,
    version: db.version,
    storeNames: Array.from(db.objectStoreNames),

    get: function(store, key) {

      return req(t.objectStore(store).get(key));

      return req(t.objectStore(store).delete(key));

      var t = db.transaction([store], 'readwrite');

    },

    },

      var t = db.transaction([store], 'readonly');
      var s = t.objectStore(store);
      var idx = s.index(indexName);
      return req(idx.getAll(range));

      return req(t.objectStore(store).getAllKeys());

    tx: tx,
    close: function() { db.close(); },

    // Bulk helpers
    bulkPut: function(store, items) {
      var t = db.transaction([store], 'readwrite');
      var s = t.objectStore(store);
      var promises = items.map(function(item){ return req(s.put(item)); });
      return Promise.all(promises).then(function(){ return txComplete(t); });
    },
    bulkDelete: function(store, keys) {
      var t = db.transaction([store], 'readwrite');
      var s = t.objectStore(store);
      var promises = keys.map(function(k){ return req(s.delete(k)); });
      return Promise.all(promises).then(function(){ return txComplete(t); });
    },

    each: function(store, callback) {

var FORMAT_VERSION = 1;     // bumped only if envelope format changes (not the data inside)
var SIGNATURE = 'weld-backup';

// Build an export envelope from arbitrary data
function exportData(opts) {
  opts = opts || {};
  var envelope = {
    signature: SIGNATURE,
    formatVersion: FORMAT_VERSION,
    dataVersion: opts.version != null ? opts.version : 1,
    generator: opts.generator || 'unknown',
    createdAt: new Date().toISOString(),
    metadata: opts.metadata || {},
    data: opts.data || {},
  };

}

// Serialize envelope to a JSON Blob
function exportBlob(opts) {
  var env = exportData(opts);
  var json = JSON.stringify(env, null, 2);
  return new Blob([json], { type: 'application/json' });
}

// Trigger a browser download
function download(opts) {
  opts = opts || {};
  var blob = exportBlob(opts);
  var fname = opts.filename;
  if (!fname) {
    var stamp = new Date().toISOString().replace(/[:.]/g, '-').slice(0, 19);

  a.href = url;

  setTimeout(function(){ URL.revokeObjectURL(url); }, 1000);
  return fname;
}

// Read a File/Blob/string into an envelope, validating signature
function readEnvelope(source) {
  return new Promise(function(resolve, reject) {
    if (typeof source === 'string') {
      try { resolve(JSON.parse(source)); }
      catch (e) { reject(new Error('Not valid JSON: ' + e.message)); }
      return;
    }
    if (source instanceof Blob || source instanceof File) {
      var reader = new FileReader();
      reader.onload = function(){
        try { resolve(JSON.parse(reader.result)); }
        catch (e) { reject(new Error('Not valid JSON: ' + e.message)); }
      };
      reader.onerror = function(){ reject(reader.error); };
      reader.readAsText(source);
      return;
    }
    if (source && typeof source === 'object') {
      resolve(source);
      return;
    }
    reject(new Error('Unsupported source type'));
  });
}

function validateEnvelope(env) {
  if (!env || typeof env !== 'object') {
    return { ok: false, error: 'Not an object' };
  }
  if (env.signature !== SIGNATURE) {
    return { ok: false, error: 'Missing or wrong signature (expected "' + SIGNATURE + '")' };
  }
  if (env.formatVersion == null) {
    return { ok: false, error: 'Missing formatVersion' };
  }
  if (env.formatVersion > FORMAT_VERSION) {
    return { ok: false, error: 'Envelope is newer (format v' + env.formatVersion + ') than this importer (v' + FORMAT_VERSION + ')' };
  }
  return {
    ok: true,
    formatVersion: env.formatVersion,
    dataVersion: env.dataVersion,
    generator: env.generator,
    createdAt: env.createdAt,
    metadata: env.metadata || {},
  };
}

function validate(source) {
  return readEnvelope(source).then(validateEnvelope).catch(function(e){
    return { ok: false, error: e.message };
  });
}

// Apply data migrations (provided by caller) to bring dataVersion up to current
function migrateData(env, migrations) {
  migrations = migrations || {};
  var versions = Object.keys(migrations).map(Number).sort(function(a, b){ return a - b; });
  var maxVer = versions.length ? Math.max.apply(null, versions) : env.dataVersion;
  var data = env.data;
  var fromVer = env.dataVersion;
  if (fromVer < maxVer) {
    for (var v = fromVer + 1; v <= maxVer; v++) {
      if (migrations[v]) {
        data = migrations[v](data) || data;
      }
    }
  }
  return { data: data, fromVersion: fromVer, toVersion: maxVer };
}

// Import an envelope - returns the data (optionally migrated)
function importData(source, opts) {
  opts = opts || {};
  return readEnvelope(source).then(function(env){
    var v = validateEnvelope(env);
    if (!v.ok) throw new Error(v.error);
    var migrated = migrateData(env, opts.migrations);
    return {
      data: migrated.data,
      generator: env.generator,
      fromVersion: migrated.fromVersion,
      toVersion: migrated.toVersion,
      metadata: env.metadata || {},
      createdAt: env.createdAt,
      importedAt: new Date().toISOString(),
    };
  });
}

  var stores = opts.stores || db.storeNames.filter(function(s){ return s.indexOf('_weld') !== 0; });
  var data = {};
  var chain = Promise.resolve();
  stores.forEach(function(s){
    chain = chain.then(function(){
      return db.getAll(s).then(function(items){ data[s] = items; });
    });
  });
  return chain.then(function(){
    return exportData({
      version: opts.version != null ? opts.version : (db.version || 1),
      generator: opts.generator || db.name,
      metadata: Object.assign({ stores: stores, recordCount: countTotal(data) }, opts.metadata || {}),
      data: data,
    });
  });
}

// Restore an envelope into a weld.migrate DB
// mode: 'merge' (default) - put each item, overwriting same-key
//       'replace'          - clear store first, then put all
//       'append'           - only put items whose key doesn't exist
function toDb(source, db, opts) {
  opts = opts || {};
  var mode = opts.mode || 'merge';
  var onConflict = opts.onConflict; // (existing, incoming) => resolved | null to skip

  return importData(source, { migrations: opts.migrations }).then(function(result){
    var data = result.data;
    var stores = opts.stores || Object.keys(data);
    var summary = { imported: {}, skipped: {}, conflicts: {} };

    var chain = Promise.resolve();
    stores.forEach(function(s){
      if (!data[s]) return;
      var items = data[s];

      chain = chain.then(function(){
        if (mode === 'replace') {
          return db.clear(s).then(function(){
            return db.bulkPut(s, items);
          }).then(function(){
            summary.imported[s] = items.length;
          });
        }

        if (mode === 'append') {
          // Check each item's key
          var skipped = 0, imported = 0;
          var toPut = [];
          var lookups = items.map(function(item){
            // We need the keyPath of the store to know the key - use raw IDB
            var t = db._db.transaction([s], 'readonly');
            var store = t.objectStore(s);
            var keyPath = store.keyPath;
            var key = (typeof keyPath === 'string') ? item[keyPath] : null;
            return new Promise(function(resolve){
              if (key == null) { toPut.push(item); imported++; resolve(); return; }
              var r = store.get(key);
              r.onsuccess = function(){
                if (r.result == null) { toPut.push(item); imported++; }
                else { skipped++; }
                resolve();
              };
              r.onerror = function(){ resolve(); };
            });
          });
          return Promise.all(lookups).then(function(){
            summary.imported[s] = imported;
            summary.skipped[s] = skipped;
            if (toPut.length) return db.bulkPut(s, toPut);
          });
        }

        // merge mode
        if (!onConflict) {
          return db.bulkPut(s, items).then(function(){
            summary.imported[s] = items.length;
          });
        }

        // merge with conflict resolution
        var imported = 0, conflicts = 0;
        var resolved = [];
        var lookups = items.map(function(item){
          var t = db._db.transaction([s], 'readonly');
          var store = t.objectStore(s);
          var keyPath = store.keyPath;
          var key = (typeof keyPath === 'string') ? item[keyPath] : null;
          return new Promise(function(resolve){
            if (key == null) { resolved.push(item); imported++; resolve(); return; }
            var r = store.get(key);
            r.onsuccess = function(){
              var existing = r.result;
              if (existing == null) { resolved.push(item); imported++; }
              else {
                conflicts++;
                var merged = onConflict(existing, item, s);
                if (merged != null) resolved.push(merged);
              }
              resolve();
            };
            r.onerror = function(){ resolve(); };
          });
        });
        return Promise.all(lookups).then(function(){
          summary.imported[s] = imported;
          summary.conflicts[s] = conflicts;
          if (resolved.length) return db.bulkPut(s, resolved);
        });
      });
    });

    return chain.then(function(){
      summary.fromVersion = result.fromVersion;
      summary.toVersion = result.toVersion;
      summary.generator = result.generator;
      return summary;
    });
  });
}

// Count total records across all stores
function countTotal(data) {
  var n = 0;
  Object.keys(data).forEach(function(k){
    if (Array.isArray(data[k])) n += data[k].length;
  });
  return n;
}

// SHARE LINK — compress + upload + URL
// uploader: the uploadPlugin function (from DSL import)
// opts: { paramName?, generatorName? }
function share(data, uploader, opts) {
  opts = opts || {};
  if (typeof uploader !== 'function') {
    return Promise.reject(new Error('backup.share: uploader function required (pass uploadPlugin)'));
  }
  var json = JSON.stringify(data);
  // Compress with gzip if available, otherwise send raw
  var blobPromise;
  if (typeof CompressionStream !== 'undefined') {
    blobPromise = new Response(
      new Blob([json]).stream().pipeThrough(new CompressionStream('gzip'))
    ).blob();
  } else {
    blobPromise = Promise.resolve(new Blob([json], { type: 'application/json' }));
  }
  return blobPromise.then(function(blob) {
    return Promise.resolve(uploader(blob));
  }).then(function(result) {
    if (!result || result.error) {
      throw new Error('upload failed: ' + (result && result.error || 'unknown'));
    }
    var url = String(result.url || result);
    // Extract filename from CDN URL
    var fileName = url.replace('https://user.uploads.dev/file/', '');
    var paramName = opts.paramName || 'data';
    var base = 'https://perchance.org/' + (opts.generatorName || window.generatorName || 'generator');
    return base + '?' + paramName + '=' + encodeURIComponent(fileName);
  });
}

// CONSUME SHARE — check URL for shared data, fetch + decompress
// fetcher: a fetch-compatible function (superFetch or window.fetch)
// opts: { paramName?, onFound? }
function consumeShare(fetcher, opts) {
  opts = opts || {};
  var paramName = opts.paramName || 'data';
  var params;
  try { params = new URL(window.location.href).searchParams; }
  catch (e) { return Promise.resolve(null); }
  var raw = params.get(paramName);
  if (!raw) return Promise.resolve(null);

  // Extract just the filename (handle legacy formats with ~ separator)
  var fileName = raw.split('~').slice(-1)[0];
  if (!fileName) return Promise.resolve(null);

  var fetchFn = typeof fetcher === 'function' ? fetcher : window.fetch;
  var cdnUrl = 'https://user.uploads.dev/file/' + fileName;
  return Promise.resolve(fetchFn(cdnUrl, {
    signal: typeof AbortSignal !== 'undefined' && AbortSignal.timeout ? AbortSignal.timeout(15000) : undefined
  })).then(function(r) {
    if (r && typeof r.blob === 'function') return r.blob();
    // superFetch might return different shape
    if (r instanceof Blob) return r;
    return new Blob([typeof r === 'string' ? r : JSON.stringify(r)]);
  }).then(function(blob) {
    // Try to decompress
    if (typeof DecompressionStream !== 'undefined') {
      return new Response(
        blob.stream().pipeThrough(new DecompressionStream('gzip'))
      ).text().catch(function() {
        // Not gzipped — try raw
        return blob.text();
      });
    }
    return blob.text();
  }).then(function(text) {
    try { return JSON.parse(text); }
    catch (e) { throw new Error('backup.consumeShare: invalid JSON in share data'); }
  }).then(function(data) {
    if (typeof opts.onFound === 'function') opts.onFound(data);
    return data;
  });
}

function createStack(opts) {

  var maxDepth = opts.maxDepth || 50;
  var getState = opts.getState;
  var setState = opts.setState;

  if (typeof getState !== 'function' || typeof setState !== 'function') {
    throw new Error('weld.undo.create requires opts.getState and opts.setState functions');

      if (i !== -1) listeners.splice(i, 1);

        }
      } else if (k === 'y') {
        if (canRedo()) { if (preventDefault) e.preventDefault(); redo(); }
      }
    }

    unbindKbd = function() {
      target.removeEventListener('keydown', handler);
      bound = false;
      unbindKbd = null;
    };
    return unbindKbd;

function coerce(s, value) {
  if (s.type === 'number') {
    var n = Number(value);
    if (isNaN(n)) throw new Error('not a number');
    if (s.min != null && n < s.min) n = s.min;
    if (s.max != null && n > s.max) n = s.max;
    return n;
  }
  if (s.type === 'boolean') return !!value;
  if (s.type === 'select') {
    var opts = s.options || [];
    var vals = opts.map(function(o) { return typeof o === 'object' ? o.value : o; });
    if (vals.indexOf(value) === -1) {
      throw new Error('value not in options');
    }
    return value;
  }
  // string, color, textarea
  return value == null ? '' : String(value);
}

  while (activeToasts.length >= config.max) {
    var oldest = activeToasts.shift();

  }

  return handle;

  handle.dismissed = true;

    if (handle.toast.parentNode) handle.toast.parentNode.removeChild(handle.toast);
    return;
  }
  handle.toast.classList.remove('shown');
  setTimeout(function() {
    if (handle.toast.parentNode) handle.toast.parentNode.removeChild(handle.toast);

// -----------------------------------------------------------------
// HTML escape -- the foundation of safety
// -----------------------------------------------------------------
function escapeHtml(s) {
  return String(s)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

// -----------------------------------------------------------------
// URL sanitization -- reject javascript:, vbscript:, data: (except images)
// -----------------------------------------------------------------
var SAFE_URL_RE = /^(?:https?:|mailto:|tel:|\/|#|\.\/|\.\.\/|[a-zA-Z0-9_\-]+\.)/;
var SAFE_IMG_URL_RE = /^(?:https?:|data:image\/(?:png|jpe?g|gif|webp|svg\+xml);|\/|\.\/|\.\.\/)/;

function safeUrl(url, forImg) {
  url = String(url).trim();
  var re = forImg ? SAFE_IMG_URL_RE : SAFE_URL_RE;
  return re.test(url) ? url : '#';
}

// -----------------------------------------------------------------
// Inline formatting: bold, italic, strikethrough -- applied to text
// where no HTML has been injected yet
// -----------------------------------------------------------------
function applyFormatting(text) {
  // Bold first (longer marker wins)
  text = text.replace(/\*\*([^*\n]+)\*\*/g, '<strong>$1</strong>');
  text = text.replace(/__([^_\n]+)__/g, '<strong>$1</strong>');
  // Italic (single marker)
  text = text.replace(/(^|[^*])\*([^*\n]+)\*(?!\*)/g, '$1<em>$2</em>');
  text = text.replace(/(^|[^_])_([^_\n]+)_(?!_)/g, '$1<em>$2</em>');
  // Strikethrough
  text = text.replace(/~~([^~\n]+)~~/g, '<del>$1</del>');
  return text;
}

// -----------------------------------------------------------------
// Inline parsing -- code, images, links, auto-links, then formatting
// -----------------------------------------------------------------
function parseInline(text) {
  var slots = [];
  function slot(html) {
    slots.push(html);
    return '\u0001S' + (slots.length - 1) + '\u0001';
  }

  // 1. Code spans -- extract first, content escaped, no further processing
  text = text.replace(/`([^`]+)`/g, function(_, code) {
    return slot('<code>' + escapeHtml(code) + '</code>');
  });

  // 2. Images:  exclamation + label-in-brackets + paren-url
  text = text.replace(/!\[([^\]]*)\]\(([^)\s]+)(?:\s+"([^"]*)")?\)/g, function(_, alt, url, title) {
    var t = title ? ' title="' + escapeHtml(title) + '"' : '';
    return slot('<img src="' + safeUrl(url, true) + '" alt="' + escapeHtml(alt) + '"' + t + '>');
  });

  // 3. Links: label-in-brackets followed by paren-url
  text = text.replace(/\[([^\]]+)\]\(([^)\s]+)(?:\s+"([^"]*)")?\)/g, function(_, label, url, title) {
    var t = title ? ' title="' + escapeHtml(title) + '"' : '';
    // Allow inline formatting inside link label
    var inner = applyFormatting(escapeHtml(label));
    return slot('<a href="' + safeUrl(url) + '"' + t + ' target="_blank" rel="noopener">' + inner + '</a>');
  });

  // 4. Auto-links: bare URLs (not already inside a slot)
  text = text.replace(/\b(https?:\/\/[^\s<]+[^\s<.,;:?!])/g, function(url) {
    return slot('<a href="' + safeUrl(url) + '" target="_blank" rel="noopener">' + escapeHtml(url) + '</a>');
  });

  // 5. Escape any remaining HTML in the plain text
  text = escapeHtml(text);

  // 6. Apply bold/italic/strikethrough on the now-safe text
  text = applyFormatting(text);

  // 7. Restore slot placeholders
  text = text.replace(/\u0001S(\d+)\u0001/g, function(_, idx) {
    return slots[parseInt(idx, 10)];
  });

  return text;
}

// -----------------------------------------------------------------
// Block-level parsing
// -----------------------------------------------------------------
function render(input, opts) {
  opts = opts || {};
  if (typeof input !== 'string') input = String(input == null ? '' : input);

  // Normalize line endings
  input = input.replace(/\r\n?/g, '\n');

  var lines = input.split('\n');
  var out = [];
  var i = 0;

  while (i < lines.length) {
    var line = lines[i];

    // Fenced code block: ``` or ~~~
    var fence = line.match(/^(```|~~~)\s*([a-zA-Z0-9_-]*)\s*$/);
    if (fence) {
      var marker = fence[1];
      var lang = fence[2] || '';
      i++;
      var code = [];
      while (i < lines.length && lines[i].indexOf(marker) !== 0) {
        code.push(lines[i]);
        i++;
      }
      i++;
      var langAttr = lang ? ' class="language-' + escapeHtml(lang) + '"' : '';
      out.push('<pre><code' + langAttr + '>' + escapeHtml(code.join('\n')) + '</code></pre>');
      continue;
    }

    // ATX header: # to ######
    var hdr = line.match(/^(#{1,6})\s+(.+)$/);
    if (hdr) {
      var n = hdr[1].length;
      out.push('<h' + n + '>' + parseInline(hdr[2]) + '</h' + n + '>');
      i++;
      continue;
    }

    // Horizontal rule
    if (/^\s*([-*_])\s*\1\s*\1[\s\1]*$/.test(line)) {
      out.push('<hr>');
      i++;
      continue;
    }

    // Blockquote
    if (line.charAt(0) === '>') {
      var quote = [];
      while (i < lines.length && lines[i].charAt(0) === '>') {
        quote.push(lines[i].replace(/^>\s?/, ''));
        i++;
      }
      out.push('<blockquote>' + render(quote.join('\n')) + '</blockquote>');
      continue;
    }

    // Unordered list
    if (/^\s*[-*+]\s+/.test(line)) {
      var items = [];
      while (i < lines.length && /^\s*[-*+]\s+/.test(lines[i])) {
        items.push(lines[i].replace(/^\s*[-*+]\s+/, ''));
        i++;
      }
      out.push('<ul>' + items.map(function(it) { return '<li>' + parseInline(it) + '</li>'; }).join('') + '</ul>');
      continue;
    }

    // Ordered list
    if (/^\s*\d+\.\s+/.test(line)) {
      var oitems = [];
      while (i < lines.length && /^\s*\d+\.\s+/.test(lines[i])) {
        oitems.push(lines[i].replace(/^\s*\d+\.\s+/, ''));
        i++;
      }
      out.push('<ol>' + oitems.map(function(it) { return '<li>' + parseInline(it) + '</li>'; }).join('') + '</ol>');
      continue;
    }

    // Empty line: skip
    if (line.trim() === '') {
      i++;
      continue;
    }

    // Paragraph: gather consecutive non-empty lines
    var para = [line];
    i++;
    while (i < lines.length && lines[i].trim() !== '' && !/^(#{1,6}\s|```|~~~|>|\s*[-*+]\s|\s*\d+\.\s)/.test(lines[i])) {
      para.push(lines[i]);
      i++;
    }
    out.push('<p>' + parseInline(para.join(' ')) + '</p>');
  }

  return out.join('\n');
}

  if (!el) throw new Error('weld.markdown.renderInto: element not found');
  el.innerHTML = render(input, opts);
  return el;
}

// -----------------------------------------------------------------
// Sanitize an arbitrary HTML string -- only allow safe tags/attrs
// -----------------------------------------------------------------
var ALLOWED_TAGS = {
  p: 1, br: 1, hr: 1,
  h1: 1, h2: 1, h3: 1, h4: 1, h5: 1, h6: 1,
  strong: 1, em: 1, del: 1, b: 1, i: 1, u: 1, s: 1,
  ul: 1, ol: 1, li: 1,
  blockquote: 1, pre: 1, code: 1,
  a: 1, img: 1,
  table: 1, thead: 1, tbody: 1, tr: 1, th: 1, td: 1,
  div: 1, span: 1
};
var ALLOWED_ATTRS = {
  a: { href: 1, title: 1, target: 1, rel: 1 },
  img: { src: 1, alt: 1, title: 1, width: 1, height: 1 },
  '*': { 'class': 1 }
};

function sanitize(html) {
  var tmp = document.createElement('template');
  tmp.innerHTML = String(html);
  walk(tmp.content);
  return tmp.innerHTML;
}

function walk(node) {
  var children = Array.prototype.slice.call(node.childNodes);
  for (var i = 0; i < children.length; i++) {
    var child = children[i];
    if (child.nodeType === 1) { // element
      var tag = child.tagName.toLowerCase();
      if (!ALLOWED_TAGS[tag]) {
        // Replace with its text content
        var text = document.createTextNode(child.textContent);
        child.parentNode.replaceChild(text, child);
        continue;
      }
      // Strip unsafe attributes
      var attrs = Array.prototype.slice.call(child.attributes);
      for (var j = 0; j < attrs.length; j++) {
        var attr = attrs[j];
        var tagAllowed = ALLOWED_ATTRS[tag] && ALLOWED_ATTRS[tag][attr.name];
        var globalAllowed = ALLOWED_ATTRS['*'][attr.name];
        if (!tagAllowed && !globalAllowed) {
          child.removeAttribute(attr.name);
          continue;
        }
        // URL attrs: sanitize
        if (attr.name === 'href') child.setAttribute('href', safeUrl(attr.value));
        if (attr.name === 'src') {
          var isImg = tag === 'img';
          child.setAttribute('src', safeUrl(attr.value, isImg));
        }
      }
      walk(child);
    }
  }
}

// Multi-channel chat spam guard.
// Detects copy-paste spam across channels using:
//   1. Normalised text fingerprint (lowercase, NFKD, emoji + punct stripped,
//      char-repeat dampened, whitespace collapsed)
//   2. Token-set Jaccard similarity for near-identical messages
// Cross-channel threshold + sliding time window controls ban trigger.
// -----------------------------------------------------------------

function fingerprint(text) {
  if (typeof text !== 'string') return '';
  return text
    .toLowerCase()
    .normalize('NFKD')
    .replace(/[\u0300-\u036f]/g, '')
    .replace(/[\p{Emoji_Presentation}\p{Extended_Pictographic}]/gu, '')
    .replace(/[^\p{L}\p{N}\s]/gu, ' ')
    .replace(/(.)\1{2,}/g, '$1$1')
    .replace(/\s+/g, ' ')
    .trim();
}

// Jaccard similarity on word-token sets, 0 (disjoint) to 1 (identical)
function jaccard(a, b) {
  if (a === b) return 1;
  if (!a || !b) return 0;
  var A = new Set(a.split(' '));
  var B = new Set(b.split(' '));
  var inter = 0;
  var itr = A.values();
  var step;
  while (!(step = itr.next()).done) {
    if (B.has(step.value)) inter++;
  }
  var union = A.size + B.size - inter;
  return union === 0 ? 0 : inter / union;
}

function similarEnough(a, b, threshold) {
  if (a === b) return true;
  return jaccard(a, b) >= threshold;
}

function createGuard(opts) {
  opts = opts || {};
  var DEFAULTS = {
    channelThreshold: 3,
    windowMs: 60000,
    similarityThreshold: 0.92,
    minMessageLength: 5,
    maxHistoryPerUser: 50,
    autoBan: true,
    autoDelete: true,
    banDurationMs: 0,
    whitelistUserIds: [],
    banFn: null,
    deleteFn: null,
    onSpamDetected: null,
    onBan: null
  };
  var merged = Object.assign({}, DEFAULTS, opts);

  var history = new Map();   // userId -> array of message records
  var banned  = new Map();   // userId -> ban record (reason / bannedAt / expiresAt)
  var stats   = { tracked: 0, banned: 0, totalReports: 0, totalBans: 0, totalDeletes: 0 };

  function pruneOld(userId, now) {
    var arr = history.get(userId);
    if (!arr) return;
    var cutoff = now - merged.windowMs;
    var i = 0;
    while (i < arr.length && arr[i].timestamp < cutoff) i++;
    if (i > 0) arr.splice(0, i);
    if (arr.length > merged.maxHistoryPerUser) {
      arr.splice(0, arr.length - merged.maxHistoryPerUser);
    }
  }

  function checkBanExpiry(userId, now) {
    var b = banned.get(userId);
    if (b && b.expiresAt && b.expiresAt <= now) {
      banned.delete(userId);
      return false;
    }
    return !!b;
  }

    var arr = history.get(userId) || [];
    var matchedByChannel = new Map();
    for (var i = 0; i < arr.length; i++) {
      var m = arr[i];
      if (similarEnough(m.fp, fp, merged.similarityThreshold)) {
        if (!matchedByChannel.has(m.channel)) matchedByChannel.set(m.channel, m);
      }
    }
    if (matchedByChannel.size >= merged.channelThreshold) {
      var matches = Array.from(matchedByChannel.values());
      return {
        fingerprint: fp,
        channels: Array.from(matchedByChannel.keys()),
        messages: matches,
        firstSeen: Math.min.apply(null, matches.map(function(m) { return m.timestamp; })),
        lastSeen: now
      };
    }
    return null;
  }

  function report(input) {
    return Promise.resolve().then(function() {
      stats.totalReports++;
      if (!input || !input.userId || !input.channel) {
        return { ok: false, reason: 'missing userId or channel' };
      }
      var channel = input.channel;
      var userId = input.userId;
      var messageId = input.messageId;
      var text = input.text;

      if (merged.whitelistUserIds.indexOf(userId) !== -1) {
        return { ok: true, whitelisted: true };
      }

      var now = input.timestamp || Date.now();
      if (checkBanExpiry(userId, now)) {
        return { ok: false, banned: true };
      }

      if (typeof text !== 'string' || text.length < merged.minMessageLength) {
        return { ok: true };
      }
      var fp = fingerprint(text);
      if (!fp) return { ok: true };

      pruneOld(userId, now);
      if (!history.has(userId)) history.set(userId, []);
      var arr = history.get(userId);
      arr.push({ channel: channel, fp: fp, text: text, messageId: messageId, timestamp: now });
      stats.tracked = history.size;

      var spam = classify(userId, fp, now);
      if (!spam) return { ok: true };

      // Include the triggering message itself
      var alreadyHas = false;
      for (var i = 0; i < spam.messages.length; i++) {
        if (spam.messages[i].channel === channel) { alreadyHas = true; break; }
      }
      if (!alreadyHas) {
        spam.channels.push(channel);
        spam.messages.push({ channel: channel, fp: fp, text: text, messageId: messageId, timestamp: now });
      }

      var verdict = Promise.resolve(true);
      if (typeof merged.onSpamDetected === 'function') {
        verdict = Promise.resolve(merged.onSpamDetected(spam));
      }

      return verdict.then(function(v) {
        if (v === false) return { ok: true, vetoed: true, spam: spam };

        var actions = Promise.resolve();
        if (merged.autoBan) {
          var expiresAt = merged.banDurationMs > 0 ? now + merged.banDurationMs : null;
          banned.set(userId, { reason: 'cross-channel-spam', bannedAt: now, expiresAt: expiresAt });
          stats.banned++;
          stats.totalBans++;
          if (typeof merged.banFn === 'function') {
            actions = actions.then(function() {
              return Promise.resolve(merged.banFn(userId, { reason: 'cross-channel-spam', spam: spam }))
                .catch(function(e) { console.error('[weld.spamguard] banFn threw:', e); });
            });
          }
          if (typeof merged.onBan === 'function') {
            actions = actions.then(function() {
              return Promise.resolve(merged.onBan(userId, { reason: 'cross-channel-spam', spam: spam }))
                .catch(function() {});
            });
          }
        }

        if (merged.autoDelete && typeof merged.deleteFn === 'function') {
          spam.messages.forEach(function(m) {
            actions = actions.then(function() {
              return Promise.resolve(merged.deleteFn(m.messageId, m.channel))
                .then(function() { stats.totalDeletes++; })
                .catch(function(e) { console.error('[weld.spamguard] deleteFn threw:', e, m); });
            });
          });
        }

        return actions.then(function() {
          history.delete(userId);
          return { ok: false, banned: true, spam: spam };
        });
      });
    });
  }

  function ban(userId, reason) {
    banned.set(userId, { reason: reason || 'manual', bannedAt: Date.now(), expiresAt: null });
    stats.banned++;
    stats.totalBans++;
  }

  function unban(userId) {
    if (banned.delete(userId)) stats.banned = Math.max(0, stats.banned - 1);
  }

  function isBanned(userId) {
    return checkBanExpiry(userId, Date.now());
  }

  function bannedUsers() {
    var now = Date.now();
    return Array.from(banned.keys()).filter(function(id) { return checkBanExpiry(id, now); });
  }

  function getHistory(userId) {
    return (history.get(userId) || []).slice();
  }

  function flush() {
    history.clear();
    stats.tracked = 0;
  }

  function getStats() {
    return Object.assign({}, stats, {
      tracked: history.size,
      banned: bannedUsers().length
    });
  }

  return {
    report: report,
    ban: ban,
    unban: unban,
    isBanned: isBanned,
    bannedUsers: bannedUsers,
    history: getHistory,
    flush: flush,
    options: merged,
    stats: getStats,
    fingerprint: fingerprint,
    jaccard: jaccard
  };
}

function batch(opts, count, onProgress) {
  return new Promise(function(resolve) {
    var results = [];
    var i = 0;
    function step() {
      if (i >= count) { resolve(results); return; }
      var cellOpts = Object.assign({}, opts);
      if (cellOpts.seed === undefined || cellOpts.seed === -1) {
        cellOpts.seed = Math.floor(Math.random() * 1000000);

        i++;
        step();

        results.push({ error: e.message });
        if (onProgress) onProgress(i + 1, count, { error: e.message });
        i++;
        step();
      });
    }
    step();
  });

// -----------------------------------------------------------------
// weld.prompt: Structured prompt assembly with token-aware packing.
// Builds the instruction string for aiTextPlugin from named sections
// (system, character card, lorebook, reminder) plus serialized chat
// history, fitting everything into a token budget.
//
// The wire format matches the Perchance AI character chat convention:
//   Name: Message content (double-bracketed)
// separated by double newlines, with standard stop sequences.
// -----------------------------------------------------------------

var DEFAULT_SEPARATOR = '\n\n';
var DEFAULT_STOP_SEQUENCES = ['\n\n' + '[' + '[', '\n' + '[' + '['];
var HIDDEN_AI_START = '<!--hidden-from-ai-start-->';
var HIDDEN_AI_END   = '<!--hidden-from-ai-end-->';

function estimateTokens(text) {
  if (!text) return 0;
  // Use weld.tokens.countSync if available (exact); otherwise chars/4
  if (window.weld && window.weld.tokens && window.weld.tokens.isLoaded && window.weld.tokens.isLoaded()) {
    var n = window.weld.tokens.countSync(text);
    if (n !== null) return n;
  }
  return Math.ceil(String(text).length / 4);
}

function substituteVars(text, vars) {
  if (!text || !vars) return text || '';
  var result = String(text);
  for (var key in vars) {
    // Replace double-brace vars: user/char are standard
    // Build the target pattern without literal double-brace in source
    var target = '\x7b\x7b' + key + '\x7d\x7d';
    var idx = result.indexOf(target);
    while (idx !== -1) {
      result = result.slice(0, idx) + String(vars[key]) + result.slice(idx + target.length);
      idx = result.indexOf(target, idx + String(vars[key]).length);
    }
  }
  return result;
}

function stripHiddenFromAi(text) {
  if (!text) return '';
  var s = String(text);
  var startIdx = s.indexOf(HIDDEN_AI_START);
  while (startIdx !== -1) {
    var endIdx = s.indexOf(HIDDEN_AI_END, startIdx);
    if (endIdx === -1) {
      s = s.slice(0, startIdx);
      break;
    }
    s = s.slice(0, startIdx) + s.slice(endIdx + HIDDEN_AI_END.length);
    startIdx = s.indexOf(HIDDEN_AI_START);
  }
  return s;
}

// Serialize a single message to wire format
function serializeMessage(msg, vars) {
  var name = msg.name || msg.role || 'Unknown';
  var content = msg.content || msg.message || msg.text || '';
  content = stripHiddenFromAi(content);
  content = substituteVars(content, vars);
  return '[' + '[' + name + ']' + ']: ' + content.trim();
}

// Serialize an array of messages to the Perchance wire format
function serializeHistory(messages, vars) {
  if (!messages || messages.length === 0) return '';
  var lines = [];
  for (var i = 0; i < messages.length; i++) {
    var msg = messages[i];
    // Skip AI-hidden messages
    if (msg.hiddenFrom && msg.hiddenFrom.indexOf('ai') !== -1) continue;
    lines.push(serializeMessage(msg, vars));
  }
  return lines.join(DEFAULT_SEPARATOR);
}

// Builder factory

function createBuilder(opts) {
  opts = opts || {};
  var budget = opts.budget || 4000;
  var counter = opts.counter || estimateTokens;
  var separator = opts.separator || DEFAULT_SEPARATOR;
  var vars = {};
  var _r = opts.roles || {};
  if (_r.user) vars.user = _r.user;
  if (_r.assistant) vars.char = _r.assistant;
  if (opts.vars) {
    for (var k in opts.vars) vars[k] = opts.vars[k];
  }
  var assistantName = _r.assistant || 'Assistant';

  // Sections: each has name / text / priority / position
  // position: 'start' (before history) or 'end' (after history)
  var sections = [];
  var historyMessages = [];

  function count(text) {
    if (typeof counter === 'function') return counter(text);
    return estimateTokens(text);
  }

  function addSection(name, text, sectionOpts) {
    sectionOpts = sectionOpts || {};
    sections.push({
      name: name,
      text: substituteVars(String(text || ''), vars),
      priority: typeof sectionOpts.priority === 'number' ? sectionOpts.priority : 50,
      position: sectionOpts.position || 'start'
    });
  }

  // -------- Convenience methods --------
  function system(text, priority) {
    addSection('system', text, { priority: priority || 100, position: 'start' });
  }

  function character(text, priority) {
    addSection('character', text, { priority: priority || 90, position: 'start' });
  }

  function lorebook(text, priority) {
    addSection('lorebook', text, { priority: priority || 70, position: 'start' });
  }

  function reminder(text, priority) {
    addSection('reminder', text, { priority: priority || 95, position: 'end' });
  }

  function history(messages) {
    historyMessages = (messages || []).slice();
  }

  function user(text) {
    var name = vars.user || 'User';
    historyMessages.push({ name: name, content: text, role: 'user' });
  }

  // -------- Main packing logic --------
  function pack() {
    // 1. Sort sections: start-positioned first (by priority desc), then end-positioned
    var startSections = [];
    var endSections = [];
    for (var i = 0; i < sections.length; i++) {
      var s = sections[i];
      if (s.position === 'end') endSections.push(s);
      else startSections.push(s);
    }
    startSections.sort(function(a, b) { return b.priority - a.priority; });
    endSections.sort(function(a, b) { return b.priority - a.priority; });

    // 2. Serialize history
    var historyText = serializeHistory(historyMessages, vars);

    // 3. Build the instruction by fitting sections + history into the budget.
    //    Strategy: start with everything, then drop lowest-priority sections,
    //    then truncate history from the oldest end.

    // Start with all sections included
    var includedStart = startSections.slice();
    var includedEnd = endSections.slice();
    var currentHistory = historyText;

    function assemble() {
      var parts = [];
      for (var j = 0; j < includedStart.length; j++) parts.push(includedStart[j].text);
      if (currentHistory) parts.push(currentHistory);
      for (var k = 0; k < includedEnd.length; k++) parts.push(includedEnd[k].text);
      return parts.filter(Boolean).join(separator);
    }

    var instruction = assemble();
    var tokens = count(instruction);

    // Drop lowest-priority sections until we fit (or run out of droppable sections)
    if (tokens > budget) {
      // Collect all sections sorted by priority ascending (lowest first to drop)
      var allSorted = startSections.concat(endSections).slice();
      allSorted.sort(function(a, b) { return a.priority - b.priority; });

      for (var d = 0; d < allSorted.length && tokens > budget; d++) {
        var toDrop = allSorted[d];
        includedStart = includedStart.filter(function(s) { return s !== toDrop; });
        includedEnd = includedEnd.filter(function(s) { return s !== toDrop; });
        instruction = assemble();
        tokens = count(instruction);
      }
    }

    // If still over budget, truncate history from the oldest end
    if (tokens > budget && historyMessages.length > 0) {
      var trimmedMessages = historyMessages.slice();
      while (tokens > budget && trimmedMessages.length > 1) {
        trimmedMessages.shift();
        currentHistory = serializeHistory(trimmedMessages, vars);
        instruction = assemble();
        tokens = count(instruction);
      }
    }

    // Build the startWith for continuation
    var startWith = '[' + '[' + assistantName + ']' + ']:';

    return {
      instruction: instruction,
      startWith: startWith,
      stopSequences: opts.stopSequences || DEFAULT_STOP_SEQUENCES,
      hideStartWith: true,
      tokens: tokens,
      budget: budget,
      fits: tokens <= budget,
      sectionsIncluded: includedStart.concat(includedEnd).map(function(s) { return s.name; }),
      historyLength: historyMessages.length
    };
  }

  // -------- Reset --------
  function reset() {
    sections = [];
    historyMessages = [];
  }

  return {
    section: addSection,
    system: system,
    character: character,
    lorebook: lorebook,
    reminder: reminder,
    history: history,
    user: user,
    pack: pack,
    reset: reset
  };
}

  serializeMessage: serializeMessage,
  substituteVars: substituteVars,
  stripHiddenFromAi: stripHiddenFromAi,
  estimateTokens: estimateTokens,
  version: '1.0.0'
};

return window.weld.prompt;

var STORE_NAME = 'records';
var DB_VERSION = 1;

// Cache of opened databases: dbName -> Promise<IDBDatabase>
var dbCache = {};

function openDatabase(dbName) {
  if (dbCache[dbName]) return dbCache[dbName];
  dbCache[dbName] = new Promise(function(resolve, reject) {
    var request = indexedDB.open(dbName, DB_VERSION);
    request.onupgradeneeded = function(e) {
      var db = e.target.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        var store = db.createObjectStore(STORE_NAME, { keyPath: ['_col', '_id'] });
        store.createIndex('by_col', '_col', { unique: false });
      }

    request.onsuccess = function(e) { resolve(e.target.result); };
    request.onerror = function(e) {
      delete dbCache[dbName];
      reject(new Error('weld.persist: failed to open "' + dbName + '": ' + (e.target.error || 'unknown')));
    };
  });
  return dbCache[dbName];
}

function uid() {
  return Date.now().toString(36) + '_' + Math.random().toString(36).slice(2, 8);

// -----------------------------------------------------------------
// Collection handle

function makeCollection(dbName, colName) {
  function getStore(mode) {
    return openDatabase(dbName).then(function(db) {
      var tx = db.transaction(STORE_NAME, mode);
      return tx.objectStore(STORE_NAME);
    });
  }

  function wrapRequest(req) {
    return new Promise(function(resolve, reject) {
      req.onsuccess = function() { resolve(req.result); };
      req.onerror = function() { reject(req.error); };
    });
  }

  // Put: upsert a record. If record has no .id, one is generated.
  function put(record) {
    if (!record || typeof record !== 'object') {
      return Promise.reject(new Error('persist.put: record must be an object'));
    }
    var rec = Object.assign({}, record);
    if (rec.id == null) rec.id = uid();
    rec._col = colName;
    rec._id = String(rec.id);
    return getStore('readwrite').then(function(store) {
      return wrapRequest(store.put(rec));
    }).then(function() {
      return rec.id;
    });
  }

  // Put many records in a single transaction
  function putMany(records) {
    if (!Array.isArray(records)) return Promise.reject(new Error('persist.putMany: expected array'));
    return openDatabase(dbName).then(function(db) {
      return new Promise(function(resolve, reject) {
        var tx = db.transaction(STORE_NAME, 'readwrite');
        var store = tx.objectStore(STORE_NAME);
        var ids = [];
        for (var i = 0; i < records.length; i++) {
          var rec = Object.assign({}, records[i]);
          if (rec.id == null) rec.id = uid();
          rec._col = colName;
          rec._id = String(rec.id);
          ids.push(rec.id);
          store.put(rec);
        }
        tx.oncomplete = function() { resolve(ids); };
        tx.onerror = function() { reject(tx.error); };
      });
    });
  }

  // Get a single record by id
  function get(id) {
    return getStore('readonly').then(function(store) {
      return wrapRequest(store.get([colName, String(id)]));
    }).then(function(rec) {
      if (!rec) return null;
      var out = Object.assign({}, rec);
      delete out._col;
      delete out._id;
      return out;
    });
  }

  // Get all records in this collection
  function getAll() {
    return getStore('readonly').then(function(store) {
      var index = store.index('by_col');
      return wrapRequest(index.getAll(colName));
    }).then(function(recs) {
      return (recs || []).map(function(rec) {
        var out = Object.assign({}, rec);
        delete out._col;
        delete out._id;
        return out;
      });
    });
  }

  // Query: get all records matching a filter function
  function query(filterFn) {
    return getAll().then(function(all) {
      if (typeof filterFn !== 'function') return all;
      return all.filter(filterFn);
    });
  }

  // Delete a single record by id
  function del(id) {
    return getStore('readwrite').then(function(store) {
      return wrapRequest(store.delete([colName, String(id)]));
    }).then(function() { return true; });
  }

  // Delete many records by id array
  function delMany(ids) {
    if (!Array.isArray(ids)) return Promise.reject(new Error('persist.delMany: expected array'));
    return openDatabase(dbName).then(function(db) {
      return new Promise(function(resolve, reject) {
        var tx = db.transaction(STORE_NAME, 'readwrite');
        var store = tx.objectStore(STORE_NAME);
        for (var i = 0; i < ids.length; i++) {
          store.delete([colName, String(ids[i])]);
        }
        tx.oncomplete = function() { resolve(ids.length); };
        tx.onerror = function() { reject(tx.error); };
      });
    });
  }

  // Clear all records in this collection
  function clear() {
    return getAll().then(function(all) {
      var ids = all.map(function(r) { return r.id; });
      return delMany(ids);
    });
  }

  // Count records in this collection
  function count() {
    return getStore('readonly').then(function(store) {
      var index = store.index('by_col');
      return wrapRequest(index.count(colName));
    });
  }

  // Update: merge a patch into an existing record
  function update(id, patch) {
    return get(id).then(function(existing) {
      if (!existing) return Promise.reject(new Error('persist.update: record not found: ' + id));
      var merged = Object.assign({}, existing, patch, { id: id });
      return put(merged);
    });
  }

  return {
    put: put,
    putMany: putMany,
    get: get,
    getAll: getAll,
    query: query,
    delete: del,
    deleteMany: delMany,
    clear: clear,
    count: count,
    update: update
  };
}

function createHistory(opts) {
  opts = opts || {};
  var dbName = opts.dbName || 'weld-history';
  var autoSave = opts.autoSave !== false;
  var autoSaveDebounce = opts.autoSaveDebounceMs || 2000;

  // Require weld.persist
  if (!window.weld || !window.weld.persist) {
    throw new Error('weld.history requires weld.persist to be loaded first.');
  }

  var db = window.weld.persist.open(dbName);
  var sessions = db.collection('sessions');
  var trees = db.collection('trees');

  var activeSessionId = null;
  var activeTree = null;        // weld.branching instance if available
  var activeMessages = null;    // fallback: plain array of messages
  var saveTimer = null;
  var changeListeners = [];

  function hasBranching() {
    return !!(window.weld && window.weld.branching);
  }

  function emit(event, data) {
    for (var i = 0; i < changeListeners.length; i++) {
      try { changeListeners[i](event, data); } catch (e) { /* swallow */ }
    }
  }

    if (!autoSave || !activeSessionId) return;

    meta = meta || {};
    var id = meta.id || uid();
    var now = Date.now();
    var session = {
      id: id,
      title: meta.title || 'New conversation',
      characterId: meta.characterId || null,
      characterName: meta.characterName || null,

      updatedAt: now,
      messageCount: 0,
      metadata: meta.metadata || null

      return saveTree();

      emit('create', { sessionId: id });
      return id;
    });

// weld.queue: Async task pipeline.
// Configurable concurrency (1-N), progress callbacks, retry with
// backoff, cancel, pause/resume. The slideshow image-gen queue
// pattern, generalized.
// -----------------------------------------------------------------

function createQueue(opts) {
  opts = opts || {};
  var maxConcurrent = Math.max(1, opts.concurrency || 1);
  var retryCount = opts.retries || 0;
  var retryDelay = opts.retryDelay || 1000;
  var retryBackoff = opts.retryBackoff || 2;

  var pending = [];
  var active = [];
  var completed = [];
  var failed = [];
  var paused = false;
  var listeners = { progress: [], done: [], error: [], drain: [] };

  function emit(event, data) {
    var fns = listeners[event] || [];
    for (var i = 0; i < fns.length; i++) {
      try { fns[i](data); } catch (e) { /* swallow */ }
    }
  }

  function emitProgress() {
    emit('progress', {
      pending: pending.length,
      active: active.length,
      completed: completed.length,
      failed: failed.length,
      total: pending.length + active.length + completed.length + failed.length
    });
  }

  function runNext() {
    if (paused) return;
    while (active.length < maxConcurrent && pending.length > 0) {
      var task = pending.shift();
      active.push(task);
      emitProgress();
      executeTask(task);
    }
  }

  function executeTask(task) {
    var attempt = 0;
    function tryOnce() {
      var result;
      try {
        result = Promise.resolve(task.fn(task.data, task));
      } catch (e) {
        result = Promise.reject(e);
      }
      result.then(function(value) {
        var idx = active.indexOf(task);
        if (idx >= 0) active.splice(idx, 1);
        task.result = value;
        task.status = 'done';
        completed.push(task);
        emit('done', { task: task, value: value });
        emitProgress();
        if (pending.length === 0 && active.length === 0) emit('drain', stats());
        else runNext();
      }).catch(function(err) {
        attempt++;
        if (attempt <= retryCount && !task.cancelled) {
          var delay = retryDelay * Math.pow(retryBackoff, attempt - 1);
          setTimeout(tryOnce, delay);
        } else {
          var idx = active.indexOf(task);
          if (idx >= 0) active.splice(idx, 1);
          task.error = err;
          task.status = 'failed';
          failed.push(task);
          emit('error', { task: task, error: err });
          emitProgress();
          if (pending.length === 0 && active.length === 0) emit('drain', stats());
          else runNext();
        }
      });
    }
    task.status = 'active';
    tryOnce();
  }

  // -------- PUBLIC --------
  function add(fn, data) {
    var task = {
      id: 't_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 6),
      fn: fn,
      data: data || null,
      status: 'pending',
      result: null,
      error: null,
      cancelled: false,
      addedAt: Date.now()
    };
    pending.push(task);
    emitProgress();
    runNext();
    return task;
  }

  function addMany(fn, dataArray) {
    var tasks = [];
    for (var i = 0; i < dataArray.length; i++) {
      tasks.push(add(fn, dataArray[i]));
    }
    return tasks;
  }

  function cancel(taskOrId) {
    var id = typeof taskOrId === 'string' ? taskOrId : taskOrId.id;
    // Remove from pending
    for (var i = pending.length - 1; i >= 0; i--) {
      if (pending[i].id === id) {
        pending[i].status = 'cancelled';
        pending[i].cancelled = true;
        pending.splice(i, 1);
        emitProgress();
        return true;
      }
    }
    // Mark active as cancelled (will reject on next attempt)
    for (var j = 0; j < active.length; j++) {
      if (active[j].id === id) {
        active[j].cancelled = true;
        return true;
      }
    }
    return false;
  }

  function cancelAll() {
    while (pending.length) {
      var t = pending.pop();
      t.status = 'cancelled';
      t.cancelled = true;
    }
    for (var i = 0; i < active.length; i++) active[i].cancelled = true;
    emitProgress();
  }

  function pause() {
    paused = true;
  }

  function resume() {
    paused = false;
    runNext();
  }

  function clear() {
    cancelAll();
    active = [];
    pending = [];
    completed = [];
    failed = [];
    emitProgress();
  }

  function stats() {
    return {
      pending: pending.length,
      active: active.length,
      completed: completed.length,
      failed: failed.length,
      total: pending.length + active.length + completed.length + failed.length,
      paused: paused,
      concurrency: maxConcurrent
    };
  }

  function setConcurrency(n) {
    maxConcurrent = Math.max(1, n);
    if (!paused) runNext();
  }

  function on(event, fn) {
    if (!listeners[event]) listeners[event] = [];
    listeners[event].push(fn);
    return function() {
      var arr = listeners[event];
      var idx = arr.indexOf(fn);
      if (idx >= 0) arr.splice(idx, 1);
    };
  }

  // Wait for all currently pending + active tasks to finish
  function drain() {
    if (pending.length === 0 && active.length === 0) {
      return Promise.resolve(stats());
    }
    return new Promise(function(resolve) {
      var unsub = on('drain', function(s) {
        unsub();
        resolve(s);
      });
    });
  }

  return {
    add: add,
    addMany: addMany,
    cancel: cancel,
    cancelAll: cancelAll,
    pause: pause,
    resume: resume,
    clear: clear,
    stats: stats,
    setConcurrency: setConcurrency,
    on: on,
    drain: drain,
    isPaused: function() { return paused; }
  };
}

    assertOpts = assertOpts || {};
    return Promise.resolve(promise).then(function(val) {
      return record(group, name, 'PASS', 'should resolve', 'resolved: ' + stringify(val), assertOpts.note);
    }).catch(function(err) {
      return record(group, name, 'FAIL', 'should resolve', 'rejected: ' + (err && err.message || err), assertOpts.note);
    });
  }

  function assertRejects(group, name, promise, assertOpts) {
    assertOpts = assertOpts || {};
    return Promise.resolve(promise).then(function(val) {
      return record(group, name, 'FAIL', 'should reject', 'resolved: ' + stringify(val), assertOpts.note);
    }).catch(function(err) {
      return record(group, name, 'PASS', 'should reject', 'rejected: ' + (err && err.message || err), assertOpts.note);
    });
  }

  function skip(group, name, reason) {
    return record(group, name, 'SKIP', null, null, reason || 'skipped');
  }

  function info(group, name, message) {
    return record(group, name, 'INFO', null, null, message);
  }

  // -------- Timing --------
  function start() {
    startTime = Date.now();
    results = [];
    counts = { PASS: 0, FAIL: 0, DRIFT: 0, SKIP: 0, INFO: 0 };
    groups = {};
    if (container) container.innerHTML = '';
    emit('start', {});
  }

  function finish() {
    var elapsed = startTime ? Date.now() - startTime : 0;
    var summary = {
      counts: Object.assign({}, counts),
      total: results.length,
      elapsed: elapsed,
      passed: counts.FAIL === 0 && counts.DRIFT === 0,
      groups: Object.keys(groups)
    };
    emit('finish', summary);
    return summary;
  }

  // -------- Progress --------
  function progress(current, total, label) {
    emit('progress', { current: current, total: total, label: label || '' });
  }

  // -------- Export --------
  function exportResults() {
    return {
      results: results.slice(),
      counts: Object.assign({}, counts),
      groups: Object.keys(groups),
      elapsed: startTime ? Date.now() - startTime : 0,
      exportedAt: Date.now()
    };
  }

  function exportText() {
    var lines = [];
    lines.push('Test results: ' + new Date().toISOString());
    lines.push('');
    for (var g in groups) {
      lines.push('== ' + g + ' ==');
      for (var i = 0; i < groups[g].length; i++) {
        var r = groups[g][i];
        var line = '  ' + r.status + '  ' + r.name;
        if (r.note) line += '  -- ' + r.note;
        if (r.status === 'FAIL' || r.status === 'DRIFT') {
          line += '  (expected: ' + stringify(r.expected) + ', got: ' + stringify(r.actual) + ')';
        }
        lines.push(line);
      }
      lines.push('');
    }
    lines.push('PASS: ' + counts.PASS + '  FAIL: ' + counts.FAIL +
      '  DRIFT: ' + counts.DRIFT + '  SKIP: ' + counts.SKIP);
    return lines.join('\n');
  }

  // -------- Rendering --------
  function renderResult(entry) {

    var row = document.createElement('div');

    var statusColor = { PASS: '#3fb950', FAIL: '#f85149', DRIFT: '#d29922', SKIP: '#7d8590', INFO: '#4493f8' };
    var detail = '';
    if (entry.status === 'PASS') detail = '<span style="color:#3fb950;">' + escapeHtml(stringify(entry.actual)) + '</span>';
    else if (entry.status === 'SKIP' || entry.status === 'INFO') detail = '<span style="color:#7d8590;">' + escapeHtml(entry.note || '') + '</span>';
    else detail = '<span style="color:#7d8590;">expected: ' + escapeHtml(stringify(entry.expected)) + '</span> <span style="color:#f85149;">got: ' + escapeHtml(stringify(entry.actual)) + '</span>';
    if (entry.note && entry.status !== 'SKIP' && entry.status !== 'INFO') {

  function escapeHtml(s) {
    return String(s).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
  }

  // -------- Helpers --------
  function sleep(ms) { return new Promise(function(r) { setTimeout(r, ms); }); }

  function timeout(promise, ms, label) {
    return new Promise(function(resolve, reject) {
      var done = false;
      var t = setTimeout(function() { if (!done) { done = true; reject(new Error('TIMEOUT' + (label ? ' ' + label : ''))); } }, ms);
      Promise.resolve(promise).then(
        function(v) { if (!done) { done = true; clearTimeout(t); resolve(v); } },
        function(e) { if (!done) { done = true; clearTimeout(t); reject(e); } }
      );
    });
  }

  function on(fn) {
    listeners.push(fn);
    return function() {
      var idx = listeners.indexOf(fn);
      if (idx >= 0) listeners.splice(idx, 1);
    };
  }

    counts: function() { return Object.assign({}, counts); },
    results: function() { return results.slice(); },
    groupNames: function() { return Object.keys(groups); },
    on: on,
    // Helpers
    sleep: sleep,
    timeout: timeout
  };
}

    if (!demoDb) return showToast('Database not ready', 'error');
    window.weld.backup.pickFile().then(function(file){
      return window.weld.backup.toDb(file, demoDb, { mode: 'merge' });
    }).then(function(){
      return demoDb.get('settings', 'user');
    }).then(function(record){
      if (record && record.value) {
        Object.assign(settings, record.value);
        window.weld.ui.setTheme({
          accent: settings.accent,
          radius: settings.radius + 'px',
          radiusSm: Math.max(0, settings.radius - 2) + 'px',
          radiusLg: (settings.radius + 4) + 'px',
        });
      }
      showToast('Settings imported -- rebuilding panel', 'success');
      refreshLiveState();
      var section = document.getElementById('demo-section');
      section.innerHTML = '';
      demoBuilt = false;
      buildDemo();
    }).catch(function(e){
      if (e.message !== 'No file selected') showToast('Import failed: ' + e.message, 'error');
    });
  }

  function wipeSettings(){
    var ui = window.weld.ui;
    ui.modal({
      title: 'Wipe all demo data?',
      content: 'This deletes the weld-index-demo database and resets all settings. The page will reload.',
      size: 'sm',
      actions: [
        { label: 'Cancel' },
        { label: 'Wipe and reload', variant: 'danger', onClick: function(){
          if (demoDb) demoDb.close();
          window.weld.migrate.deleteDatabase('weld-index-demo').then(function(){ location.reload(); });
        }}
      ]
    });
  }

  function openSettingsModal(){
    var ui = window.weld.ui;
    var sizeSel = ui.select({
      options: [
        { value: 'comfortable', label: 'Comfortable' },
        { value: 'compact', label: 'Compact' },
        { value: 'spacious', label: 'Spacious' },
      ], value: 'comfortable'
    });
    var notifSel = ui.checkbox({ label: 'Show notifications', checked: true });
    var bioInput = ui.textarea({ placeholder: 'Tell us about yourself...', rows: 3 });
    var content = ui.col([
      ui.label('Density'), sizeSel,
      ui.label('Notifications'), notifSel,
      ui.label('Bio'), bioInput,
    ], 12);
    ui.modal({
      title: 'Demo settings',
      content: content,
      size: 'md',
      actions: [
        { label: 'Cancel' },
        { label: 'Save', variant: 'primary', onClick: function(){
          showToast('Settings saved (demo only -- not persisted)', 'success');
        }}
      ]
    });
  }

  function showConfirmModal(){
    var ui = window.weld.ui;
    ui.modal({
      title: 'Are you sure?',
      content: 'This is a demo of a destructive confirmation. Nothing will actually be deleted.',
      size: 'sm',
      actions: [
        { label: 'Cancel' },
        { label: 'Yes, do it', variant: 'danger', onClick: function(){
          showToast('Confirmed (no-op demo)', 'success');
        }}
      ]
    });
  }

  function showToast(msg, kind){
    var ui = window.weld.ui;
    var area = document.getElementById('toast-area');
    if (!area) return;
    var iconName = kind === 'success' ? 'check' : (kind === 'error' ? 'error' : 'info');
    var icon = ui.icon(iconName, { size: 16 });
    var toast = document.createElement('div');
    toast.className = 'toast ' + (kind || 'info');
    var iw = document.createElement('span'); iw.className = 'ico'; iw.appendChild(icon);
    var tx = document.createElement('span'); tx.textContent = msg;
    toast.appendChild(iw); toast.appendChild(tx);
    area.appendChild(toast);
    setTimeout(function(){
      toast.style.animation = 'toast-out 240ms ease-in forwards';
      setTimeout(function(){ if (toast.parentNode) toast.parentNode.removeChild(toast); }, 240);
    }, 3500);
  }

  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', detectLoop);
  else detectLoop();
})();
