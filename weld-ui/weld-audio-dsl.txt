$meta
  title       = weld.audio
  description = Background music, looping ambience, and event-triggered sound effects for story and visual-novel apps. Built on the Web Audio API with no dependencies: named layers that crossfade, a pooled SFX player that won't stutter on rapid fire, a master mixer with per-channel volume, and a one-gesture unlock for browser autoplay policy. Stash on window.weld.audio.
  tags        = weld, audio, sound, music, ambience, sfx, plugin

$output() =>
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.audio) return window.weld.audio;

  // -----------------------------------------------------------------
  // weld.audio: sound for story apps, without the Web Audio boilerplate.
  //
  // Three things every illustrated-chat / VN wants and none of the base
  // plugins provide: looping background MUSIC that crossfades when the mood
  // changes, looping AMBIENCE that layers under it (rain, a tavern, wind),
  // and one-shot SFX fired on events (a door, a sword, a notification).
  // weld.audio gives each its own channel on a shared mixer:
  //
  //   await weld.audio.unlock();                 // call once from a click
  //   weld.audio.music.play('https://.../theme.mp3', { fade: 1.5 });
  //   weld.audio.ambience.play('https://.../rain.mp3', { volume: 0.5 });
  //   weld.audio.sfx.play('https://.../door.mp3'); // pooled, no stutter
  //   weld.audio.setChannelVolume('music', 0.4);
  //
  // Browsers block audio until a user gesture; unlock() resumes the context
  // and is safe to call repeatedly. Everything degrades to a quiet no-op if
  // Web Audio is unavailable (so it never throws in a headless sandbox).
  // -----------------------------------------------------------------

  var AC = (typeof AudioContext !== 'undefined') ? AudioContext
         : (typeof webkitAudioContext !== 'undefined') ? webkitAudioContext : null;

  var ctx = null;
  var master = null;
  var channels = {};            // name -> { gain, volume }
  var buffers = {};             // url -> AudioBuffer (decoded, cached)
  var loops = {};               // channel -> { src, gain, url }
  var unlocked = false;

  function supported() { return !!AC; }

  function ensureCtx() {
    if (!AC) return null;
    if (!ctx) {
      try {
        ctx = new AC();
        master = ctx.createGain();
        master.gain.value = 1;
        master.connect(ctx.destination);
        // standard channels
        ['music', 'ambience', 'sfx'].forEach(function (name) { ensureChannel(name); });
      } catch (e) { ctx = null; }
    }
    return ctx;
  }

  function ensureChannel(name) {
    ensureCtx();
    if (!ctx) return null;
    if (!channels[name]) {
      var g = ctx.createGain();
      g.gain.value = 1;
      g.connect(master);
      channels[name] = { gain: g, volume: 1 };
    }
    return channels[name];
  }

  // Resume the audio context from a user gesture. Returns a Promise<bool>.
  function unlock() {
    if (!ensureCtx()) return Promise.resolve(false);
    var p = (ctx.state === 'suspended' && ctx.resume) ? ctx.resume() : Promise.resolve();
    return Promise.resolve(p).then(function () { unlocked = (ctx.state === 'running'); return unlocked; }, function () { return false; });
  }

  // Load + decode a URL into an AudioBuffer (cached). Uses fetch/superFetch.
  function load(url) {
    url = String(url || '');
    if (!url) return Promise.reject(new Error('weld.audio: empty url'));
    if (!ensureCtx()) return Promise.reject(new Error('weld.audio: Web Audio not available'));
    if (buffers[url]) return Promise.resolve(buffers[url]);
    var fetcher = (typeof fetch === 'function') ? fetch : null;
    if (!fetcher) return Promise.reject(new Error('weld.audio: no fetch to load audio'));
    return fetcher(url).then(function (r) { return r.arrayBuffer(); }).then(function (ab) {
      return new Promise(function (resolve, reject) {
        // decodeAudioData has both callback and promise forms; support both.
        var ret = ctx.decodeAudioData(ab, function (buf) { buffers[url] = buf; resolve(buf); }, reject);
        if (ret && typeof ret.then === 'function') ret.then(function (buf) { buffers[url] = buf; resolve(buf); }, reject);
      });
    });
  }

  function rampGain(gainNode, to, seconds) {
    if (!ctx) return;
    var now = ctx.currentTime;
    try {
      gainNode.gain.cancelScheduledValues(now);
      gainNode.gain.setValueAtTime(gainNode.gain.value, now);
      gainNode.gain.linearRampToValueAtTime(to, now + Math.max(0.001, seconds || 0));
    } catch (e) { gainNode.gain.value = to; }
  }

  // Start (or crossfade to) a looping source on a channel.
  function playLoop(channelName, url, opts) {
    opts = opts || {};
    if (!ensureChannel(channelName)) return Promise.resolve(null);
    var fade = opts.fade != null ? opts.fade : 0.8;
    var vol = opts.volume != null ? opts.volume : 1;
    return load(url).then(function (buf) {
      var ch = channels[channelName];
      // fade out the previous loop on this channel
      var prev = loops[channelName];
      if (prev) {
        rampGain(prev.gain, 0, fade);
        (function (p) { setTimeout(function () { try { p.src.stop(); } catch (e) {} }, (fade * 1000) + 60); })(prev);
      }
      var src = ctx.createBufferSource();
      src.buffer = buf; src.loop = true;
      var g = ctx.createGain();
      g.gain.value = prev ? 0 : vol;      // start silent if crossfading in
      src.connect(g); g.connect(ch.gain);
      try { src.start(0); } catch (e) {}
      if (prev) rampGain(g, vol, fade);
      loops[channelName] = { src: src, gain: g, url: url };
      return loops[channelName];
    });
  }

  function stopLoop(channelName, fade) {
    var prev = loops[channelName];
    if (!prev) return;
    fade = fade != null ? fade : 0.6;
    rampGain(prev.gain, 0, fade);
    setTimeout(function () { try { prev.src.stop(); } catch (e) {} }, (fade * 1000) + 60);
    loops[channelName] = null;
  }

  // One-shot SFX: each call spins up its own short-lived source, so rapid
  // fire never cuts itself off (the "pool" is implicit — sources are cheap
  // and self-cleaning via onended).
  function playSfx(url, opts) {
    opts = opts || {};
    if (!ensureChannel('sfx')) return Promise.resolve(false);
    var vol = opts.volume != null ? opts.volume : 1;
    return load(url).then(function (buf) {
      var src = ctx.createBufferSource();
      src.buffer = buf;
      var g = ctx.createGain(); g.gain.value = vol;
      src.connect(g); g.connect(channels.sfx.gain);
      if (opts.rate) { try { src.playbackRate.value = opts.rate; } catch (e) {} }
      src.onended = function () { try { src.disconnect(); g.disconnect(); } catch (e) {} };
      try { src.start(0); } catch (e) {}
      return true;
    }, function () { return false; });
  }

  function setChannelVolume(name, v) {
    var ch = ensureChannel(name);
    if (!ch) return;
    ch.volume = v;
    rampGain(ch.gain, v, 0.15);
  }
  function setMasterVolume(v) { if (ensureCtx()) rampGain(master, v, 0.15); }
  function masterVolume() { return master ? master.gain.value : 0; }

  // Preload a set of URLs (e.g. a scene's SFX) so first play is instant.
  function preload(urls) {
    if (!Array.isArray(urls)) urls = [urls];
    return Promise.all(urls.map(function (u) { return load(u).then(function () { return true; }, function () { return false; }); }));
  }

  function stopAll(fade) {
    Object.keys(loops).forEach(function (c) { stopLoop(c, fade); });
  }

  // Per-channel facade so the API reads naturally (music.play / .stop).
  function channelFacade(name) {
    return {
      play: function (url, opts) { return playLoop(name, url, opts); },
      stop: function (fade) { return stopLoop(name, fade); },
      setVolume: function (v) { return setChannelVolume(name, v); },
      current: function () { return loops[name] ? loops[name].url : null; }
    };
  }

  window.weld.audio = {
    unlock: unlock,
    supported: supported,
    load: load,
    preload: preload,
    music: channelFacade('music'),
    ambience: channelFacade('ambience'),
    sfx: { play: playSfx, setVolume: function (v) { return setChannelVolume('sfx', v); } },
    channel: function (name) { ensureChannel(name); return channelFacade(name); },
    setChannelVolume: setChannelVolume,
    setMasterVolume: setMasterVolume,
    masterVolume: masterVolume,
    stopAll: stopAll,
    isUnlocked: function () { return unlocked; },
    version: '1.0.0'
  };

  return window.weld.audio;
