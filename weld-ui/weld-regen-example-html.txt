<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.regen) {
  window.weld.regen.attach({ stream: window.weld.stream, branching: window.weld.branching });
  return window.weld.regen;
}

// -----------------------------------------------------------------
// weld.regen: the "swipe" feature every chat app wants, built as the
// connective tissue between two plugins that already exist.
//
//   weld.branching  gives you the conversation tree (nodes, siblings).
//   weld.stream     runs the model.
//
// weld.regen ties them together: re-running the current assistant reply
// appends the new take as a SIBLING of the old one (a branch from the
// same parent), so you keep every alternative and can swipe between
// them -- exactly the character.ai style swipe deck.
//
// You supply how to build the instruction (a build() callback or a
// literal instruction); weld.regen owns the tree bookkeeping.
// -----------------------------------------------------------------

var deps = { stream: null, branching: null };
function attach(d) {
  if (d && d.stream) deps.stream = d.stream;
  if (d && d.branching) deps.branching = d.branching;
}
function stream() { return deps.stream || (window.weld && window.weld.stream) || null; }
function branching() { return deps.branching || (window.weld && window.weld.branching) || null; }

function resolveInstruction(opts) {
  if (typeof opts.build === 'function') return Promise.resolve(opts.build());
  if (opts.instruction != null) return Promise.resolve(opts.instruction);
  return Promise.resolve(null);
}

// Run one generation through weld.stream, forwarding stream callbacks.
function generate(instruction, opts) {
  var s = stream();
  if (!s) return Promise.resolve({ text: '', error: 'weld.regen: weld.stream is not loaded' });
  var runOpts = { instruction: instruction };
  ['startWith', 'hideStartWith', 'stopSequences', 'chat', 'onChunk', 'onToken'].forEach(function (k) {
    if (opts[k] != null) runOpts[k] = opts[k];
  });
  return s.run(runOpts);
}

// -----------------------------------------------------------------
// regenerate(opts) -> Promise<{ node, result, error }>
//
// opts:
//   tree         (required) a weld.branching tree
//   nodeId       the assistant node to make a sibling of
//                (default: the active leaf, tree.getActiveId())
//   build()/instruction   how to produce the prompt (see above)
//   data(result) optional: map a stream result -> node data
//                (default: { role:'assistant', content: result.text, meta })
//   setActive    (bool) switch the tree to the new node (default true)
//   ...plus any weld.stream run options (onChunk, chat, startWith, ...)
// -----------------------------------------------------------------
function regenerate(opts) {
  opts = opts || {};
  var tree = opts.tree;
  if (!tree || typeof tree.append !== 'function') {
    return Promise.resolve({ node: null, result: null, error: 'weld.regen: opts.tree must be a weld.branching tree' });
  }
  var targetId = opts.nodeId || tree.getActiveId();

  return resolveInstruction(opts).then(function (instruction) {
    if (instruction == null || String(instruction).trim() === '') {
      return { node: null, result: null, error: 'weld.regen: no instruction (pass build() or instruction)' };
    }
    return generate(instruction, opts).then(function (result) {
      if (result.error) return { node: null, result: result, error: result.error };

      var data = (typeof opts.data === 'function')
        ? opts.data(result)
        : { role: 'assistant', content: result.text, meta: { stopReason: result.stopReason, truncated: result.truncated, regenerated: true } };

      var node;
      var existing = targetId ? tree.get(targetId) : null;
      if (existing && existing.parentId) {
        // Sibling: branch from the same parent as the target node.
        node = tree.branch(targetId, data);
      } else if (existing) {
        // Target is a root: add another root-level sibling via append(null parent).
        node = tree.append(data, null);
      } else {
        // No target yet: just append at the active position.
        node = tree.append(data);
      }
      if (opts.setActive !== false && node && node.id) tree.setActive(node.id);
      return { node: node, result: result, error: null };
    });
  });
}

// Swipe among the siblings of a node. dir: +1 / -1 (default +1).
// Returns the node now active, or null.
function swipe(tree, nodeId, dir) {
  if (!tree) return null;
  var id = nodeId || tree.getActiveId();
  if (!id) return null;
  if (typeof tree.switchSibling === 'function') {
    var moved = tree.switchSibling(id, (dir == null ? 1 : dir));
    if (moved && moved.id && tree.setActive) tree.setActive(moved.id);
    return moved || null;
  }
  return null;
}

// The list of sibling takes for a node, with a 1-based position.
// Returns { index, count, items:[{ id, data, active }] }.
function variants(tree, nodeId) {
  var empty = { index: 0, count: 0, items: [] };
  if (!tree || typeof tree.siblings !== 'function') return empty;
  var id = nodeId || tree.getActiveId();
  if (!id) return empty;
  var sibs = tree.siblings(id) || [];
  var activeId = tree.getActiveId ? tree.getActiveId() : id;
  var items = sibs.map(function (n) {
    return { id: n.id, data: n.data, active: n.id === activeId };
  });
  var index = 0;
  for (var i = 0; i < items.length; i++) if (items[i].active) { index = i + 1; break; }
  return { index: index, count: items.length, items: items };
}

// Edit a node's text in place (e.g. the user fixes the reply).
function edit(tree, nodeId, newContent) {
  if (!tree) return null;
  var n = tree.get(nodeId);
  if (!n) return null;
  if (n.data && typeof n.data === 'object') n.data.content = newContent;
  else n.data = { content: newContent };
  return n;
}

window.weld.regen = {
  attach: attach,
  regenerate: regenerate,
  swipe: swipe,
  variants: variants,
  edit: edit,
  version: '1.0.0'
};

window.weld.regen.attach({ stream: window.weld.stream, branching: window.weld.branching });
return window.weld.regen;
})();

<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.branching) return window.weld.branching;

// -----------------------------------------------------------------
// Node id generator
// -----------------------------------------------------------------
function uid() {
  return 'n_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 8);
}

// -----------------------------------------------------------------
// Tree instance factory
// -----------------------------------------------------------------
function createTree(opts) {
  opts = opts || {};
  var storage = opts.storage;
  var storageKey = opts.storageKey || 'weld-branching';

  // Internal state
  var nodes = {};      // id -> node
  var rootIds = [];    // ordered list of root node ids
  var activeId = null; // currently active leaf

  function save() {
    if (!storage) return;
    try {
      storage.setItem(storageKey, JSON.stringify({
        nodes: nodes,
        rootIds: rootIds,
        activeId: activeId
      }));
    } catch (e) { /* swallow */ }
  }

  function load() {
    if (!storage) return false;
    try {
      var raw = storage.getItem(storageKey);
      if (!raw) return false;
      var parsed = JSON.parse(raw);
      if (parsed && typeof parsed === 'object') {
        nodes = parsed.nodes || {};
        rootIds = parsed.rootIds || [];
        activeId = parsed.activeId || null;
        return true;
      }
    } catch (e) { /* swallow */ }
    return false;
  }

  // Try loading initial state
  load();

  // -----------------------------------------------------------------
  // Core: append a node
  // -----------------------------------------------------------------
  function append(data, parentId) {
    if (data === undefined) data = null;
    if (parentId === undefined) {
      // Default parent: the currently active leaf, if any
      parentId = activeId;
    }
    var id = uid();
    var node = {
      id: id,
      parentId: parentId || null,
      childIds: [],
      data: data,
      createdAt: Date.now()
    };
    nodes[id] = node;

    if (parentId && nodes[parentId]) {
      nodes[parentId].childIds.push(id);
    } else {
      // New root
      rootIds.push(id);
    }
    activeId = id;
    save();
    return id;
  }

  // -----------------------------------------------------------------
  // Branch -- explicit alias for append with parent
  // -----------------------------------------------------------------
  function branch(fromId, data) {
    if (!nodes[fromId]) throw new Error('branching: unknown node ' + fromId);
    var parentId = nodes[fromId].parentId;
    var id = append(data, parentId);
    return id;
  }

  // -----------------------------------------------------------------
  // Get / list
  // -----------------------------------------------------------------
  function get(id) {
    var n = nodes[id];
    if (!n) return null;
    return {
      id: n.id,
      parentId: n.parentId,
      childIds: n.childIds.slice(),
      data: n.data,
      createdAt: n.createdAt
    };
  }

  function getAll() {
    var out = [];
    for (var id in nodes) out.push(get(id));
    return out;
  }

  function getRoots() {
    return rootIds.map(get);
  }

  // -----------------------------------------------------------------
  // Path: from root to a given node (or active leaf)
  // -----------------------------------------------------------------
  function path(toId) {
    var target = toId || activeId;
    if (!target || !nodes[target]) return [];
    var chain = [];
    var cursor = nodes[target];
    while (cursor) {
      chain.unshift(get(cursor.id));
      cursor = cursor.parentId ? nodes[cursor.parentId] : null;
    }
    return chain;
  }

  function activeBranch() {
    return path(activeId);
  }

  function setActive(id) {
    if (!nodes[id]) throw new Error('branching: unknown node ' + id);
    activeId = id;
    save();
    return id;
  }

  function getActiveId() {
    return activeId;
  }

  // -----------------------------------------------------------------
  // Siblings -- nodes with the same parent
  // -----------------------------------------------------------------
  function siblings(id) {
    var n = nodes[id];
    if (!n) return [];
    var parentChildren;
    if (n.parentId && nodes[n.parentId]) {
      parentChildren = nodes[n.parentId].childIds;
    } else {
      parentChildren = rootIds;
    }
    return parentChildren.map(get);
  }

  // -----------------------------------------------------------------
  // Switching to a sibling -- pick by index relative to current
  // -----------------------------------------------------------------
  function switchSibling(id, direction) {
    var sibs = siblings(id);
    if (sibs.length === 0) return null;
    var idx = sibs.findIndex(function(s) { return s.id === id; });
    if (idx === -1) return null;
    var newIdx;
    if (direction === 'next' || direction === 1) {
      newIdx = (idx + 1) % sibs.length;
    } else if (direction === 'prev' || direction === -1) {
      newIdx = (idx - 1 + sibs.length) % sibs.length;
    } else if (typeof direction === 'number') {
      newIdx = Math.max(0, Math.min(sibs.length - 1, direction));
    } else {
      return null;
    }
    return sibs[newIdx].id;
  }

  // -----------------------------------------------------------------
  // Switch branch by selecting a node; updates active to that node's deepest descendant
  // along its primary (first-child) path
  // -----------------------------------------------------------------
  function switchTo(id) {
    if (!nodes[id]) throw new Error('branching: unknown node ' + id);
    // Walk down following first child until leaf
    var cursor = nodes[id];
    while (cursor.childIds.length > 0) {
      cursor = nodes[cursor.childIds[0]];
    }
    activeId = cursor.id;
    save();
    return activeId;
  }

  // -----------------------------------------------------------------
  // Remove a node and all its descendants
  // -----------------------------------------------------------------
  function remove(id) {
    var n = nodes[id];
    if (!n) return false;
    // Collect descendants
    var toRemove = [];
    var stack = [id];
    while (stack.length) {
      var cur = stack.pop();
      toRemove.push(cur);
      var node = nodes[cur];
      if (node) {
        for (var i = 0; i < node.childIds.length; i++) stack.push(node.childIds[i]);
      }
    }
    // Detach from parent
    if (n.parentId && nodes[n.parentId]) {
      nodes[n.parentId].childIds = nodes[n.parentId].childIds.filter(function(c) { return c !== id; });
    } else {
      rootIds = rootIds.filter(function(r) { return r !== id; });
    }
    // Delete all
    for (var k = 0; k < toRemove.length; k++) delete nodes[toRemove[k]];
    // Reset active if needed
    if (toRemove.indexOf(activeId) !== -1) {
      activeId = (n.parentId && nodes[n.parentId]) ? n.parentId : (rootIds[0] || null);
    }
    save();
    return true;
  }

  function clear() {
    nodes = {};
    rootIds = [];
    activeId = null;
    save();
  }

  // -----------------------------------------------------------------
  // Serialize / deserialize
  // -----------------------------------------------------------------
  function serialize() {
    return JSON.parse(JSON.stringify({
      nodes: nodes,
      rootIds: rootIds,
      activeId: activeId
    }));
  }

  function deserialize(state) {
    if (!state || typeof state !== 'object') throw new Error('branching: invalid state');
    nodes = state.nodes ? JSON.parse(JSON.stringify(state.nodes)) : {};
    rootIds = state.rootIds ? state.rootIds.slice() : [];
    activeId = state.activeId || null;
    save();
  }

  function size() {
    var n = 0;
    for (var k in nodes) n++;
    return n;
  }

  function isLeaf(id) {
    var n = nodes[id];
    return n && n.childIds.length === 0;
  }

  return {
    append: append,
    branch: branch,
    get: get,
    getAll: getAll,
    getRoots: getRoots,
    path: path,
    activeBranch: activeBranch,
    setActive: setActive,
    getActiveId: getActiveId,
    siblings: siblings,
    switchSibling: switchSibling,
    switchTo: switchTo,
    remove: remove,
    clear: clear,
    serialize: serialize,
    deserialize: deserialize,
    size: size,
    isLeaf: isLeaf
  };
}

window.weld.branching = {
  create: createTree,
  version: '1.0.0'
};

return window.weld.branching;
})();
</script>
