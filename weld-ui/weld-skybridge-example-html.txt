<script>
(function () {
  // An imported plugin's $output is a function DEFINITION -- it does not run on
  // import; something must CALL it. The plugin builds its API as a side effect
  // (window.weld.skybridge) and is idempotent, so we trigger it once here.
  // Without this call the plugin never runs and window.weld.skybridge is absent.
  try {
    if (typeof root !== 'undefined' && typeof root.weldSkybridge === 'function') root.weldSkybridge();
  } catch (e) { try { if (window.console) console.log('\[skybridge\] could not trigger root.weldSkybridge():', e && e.message); } catch (e2) {} }
  var sb = (window.weld && window.weld.skybridge) || null;
  function $(id){ return document.getElementById(id); }
  function esc(s){ return String(s); }

  // ---- diagnostics panel ------------------------------------------------
  function diagSnapshot(){
    if (sb && typeof sb.diagnostics === 'function') return sb.diagnostics();
    var beacon = null; try { beacon = window.__weldSkybridge || null; } catch (e) {}
    if (sb && (sb.build || sb.version)) {
      return { error: 'plugin present (build ' + (sb.build || '?') + ', v' + (sb.version || '?') + ') but THIS build has no diagnostics() \u2014 it is older than the current source. Re-save weld-skybridge-plugin.' };
    }
    if (beacon && beacon.build) {
      return { error: 'deployment beacon found (build ' + beacon.build + ') but window.weld.skybridge was not built \u2014 the plugin threw AFTER setting the beacon. Open the generator console for the error.' };
    }
    return { error: 'window.weld.skybridge and the beacon are both absent \u2014 the plugin $output never ran. An imported plugin is NOT auto-invoked; this page calls root.weldSkybridge() to trigger it. Check that {import:weld-skybridge-plugin} is in the generator and that root.weldSkybridge is a function.' };
  }
  function fmtTrace(tr){
    if (!tr || !tr.length) return '   (no handshake events yet)';
    var t0 = tr[0].t;
    return tr.map(function(e){
      var dt = '+' + String(e.t - t0).padStart(5,' ') + 'ms';
      var line = '   ' + dt + '  ' + e.dir.toUpperCase().padEnd(5,' ') + ' ' + (e.type||'');
      if (e.origin) line += '  <- ' + e.origin;
      if (e.note)   line += '  (' + e.note + ')';
      return line;
    }).join('\n');
  }
  function renderDiag(){
    var d = diagSnapshot();
    var lines = [];
    if (d.error) { lines.push('PLUGIN: ' + d.error); }
    else {
      lines.push('PLUGIN build : ' + d.build + '   (v' + d.version + ')');
      lines.push('connected    : ' + d.connected + '   protocol ' + d.protocol + (d.anchorVersion ? '   anchor v' + d.anchorVersion : ''));
      lines.push('framed       : ' + d.framed + '   origin ' + d.origin + '   peers ' + d.peers);
      lines.push('capabilities : ' + (d.capabilities.length ? d.capabilities.join(', ') : '(none)') + '   storage backend: ' + d.storageBackend);
      lines.push('');
      lines.push('handshake trace (plugin side):');
      lines.push(fmtTrace(d.trace));
    }
    var el = $('diag-out');
    if (el) el.textContent = lines.join('\n');
    return d;
  }
  function copyReport(){
    var d = renderDiag();
    var report = 'weld.skybridge diagnostics report\n'
      + 'page: ' + location.href + '\n'
      + 'when: ' + new Date().toISOString() + '\n'
      + '----- plugin diagnostics -----\n'
      + JSON.stringify(d, null, 2) + '\n'
      + '----- note -----\n'
      + 'Also copy the browser console lines tagged [skybridge] and [WeldCompanion].\n'
      + 'In the top-frame console run: weldCompanion.skybridgeDiagnostics()  (companion side).\n';
    var done = function(ok){ var c = $('diag-copied'); if (c) c.textContent = ok ? 'copied \u2713' : 'copy failed -- select the box manually'; };
    try {
      if (navigator.clipboard && navigator.clipboard.writeText) navigator.clipboard.writeText(report).then(function(){done(true);}, function(){done(false);});
      else done(false);
    } catch (e) { done(false); }
  }

  // ---- tiny test runner -------------------------------------------------
  var tests = [];           // { name, run() -> Promise<{status, detail}> }
  var nodes = {};           // name -> { badge, detail }
  function def(name, run){ tests.push({ name: name, run: run }); }
  function mkRow(name){
    var li = document.createElement('li'); li.className = 'test';
    var b = document.createElement('span'); b.className = 'badge run'; b.textContent = '...';
    var body = document.createElement('div'); body.className = 'body';
    var nm = document.createElement('div'); nm.className = 'name'; nm.textContent = name;
    var dt = document.createElement('div'); dt.className = 'detail'; dt.textContent = 'queued';
    body.appendChild(nm); body.appendChild(dt);
    li.appendChild(b); li.appendChild(body);
    $('tests').appendChild(li);
    nodes[name] = { badge: b, detail: dt };
  }
  function setRow(name, status, detail){
    var n = nodes[name]; if (!n) return;
    n.badge.className = 'badge ' + status;
    n.badge.textContent = status;
    n.detail.textContent = detail || '';
  }
  function paintConn(){
    var c = $('conn'), t = $('conn-txt');
    if (sb && sb.connected) { c.className = 'pill on'; t.textContent = 'companion linked'; }
    else { c.className = 'pill off'; t.textContent = sb ? 'no companion (fallback)' : 'plugin not loaded'; }
    $('proto').textContent = 'protocol ' + (sb && sb.protocol ? sb.protocol : '\u2014');
    $('proto').className = 'pill ' + (sb && sb.connected ? 'on' : 'off');
    var be = sb ? sb.storage.backend() : '\u2014';
    $('backend').textContent = 'storage: ' + be;
    $('backend').className = 'pill ' + (be === 'companion' ? 'on' : 'off');
  }

  // ---- the probes -------------------------------------------------------
  // PASS = behaved correctly. SKIP = correct-but-not-applicable (e.g. no companion).
  function P(detail){ return { status: 'pass', detail: detail }; }
  function F(detail){ return { status: 'fail', detail: detail }; }
  function S(detail){ return { status: 'skip', detail: detail }; }

  def('plugin present', function(){
    return Promise.resolve(sb && typeof sb.request === 'function'
      ? P('window.weld.skybridge is installed (v' + sb.version + ')')
      : F('window.weld.skybridge missing \u2014 is the plugin imported?'));
  });

  def('advertises protocol range', function(){
    var r = sb.protocolRange;
    return Promise.resolve(r && r.min === 1 && r.max >= r.min
      ? P('this end speaks protocol ' + r.min + '..' + r.max)
      : F('unexpected protocolRange: ' + JSON.stringify(r)));
  });

  def('handshake', function(){
    // give the async hello/here a moment if it hasn't settled yet
    function check(){ return sb.connected; }
    if (check()) return Promise.resolve(P('companion answered \u2014 anchor v' + (sb.anchorVersion || '?')));
    return new Promise(function(resolve){
      var done = false;
      var off = sb.onConnect(function(){ if (!done){ done = true; resolve(P('companion answered \u2014 anchor v' + (sb.anchorVersion || '?'))); } });
      setTimeout(function(){
        if (done) return;
        // no companion is a valid state; this is a SKIP, not a FAIL.
        // (waited past the companion's announce schedule + our hello retries)
        resolve(S('no companion linked \u2014 running in fallback mode (this is fine)'));
      }, 3500);
    });
  });

  def('negotiated protocol', function(){
    if (!sb.connected) return Promise.resolve(S('skipped \u2014 not connected'));
    return Promise.resolve(sb.protocol >= 1
      ? P('agreed on protocol ' + sb.protocol)
      : F('connected but protocol is ' + sb.protocol));
  });

  def('capabilities advertised', function(){
    if (!sb.connected) return Promise.resolve(S('skipped \u2014 not connected'));
    var caps = ['storage','ai'].filter(function(c){ return sb.has(c); });
    return Promise.resolve(caps.length
      ? P('companion offers: ' + caps.join(', '))
      : F('connected but advertises no capabilities'));
  });

  def('storage backend resolves', function(){
    var be = sb.storage.backend();
    var ok = be === 'companion' || be === 'persist' || be === 'memory';
    if (!ok) return Promise.resolve(F('unknown backend: ' + be));
    return Promise.resolve(P('routing to "' + be + '"' + (be === 'companion' ? ' (cross-generator)' : ' (this browser only)')));
  });

  def('storage round-trip', function(){
    var key = 'selftest:rt', val = { n: Date.now(), tag: 'skybridge-live' };
    return sb.storage.set(key, val).then(function(ok){
      if (!ok) return F('set() returned false (companion may have denied permission)');
      return sb.storage.get(key).then(function(got){
        if (got && got.n === val.n && got.tag === val.tag) return P('set then get returned the same record via "' + sb.storage.backend() + '"');
        return F('get() did not return what set() wrote: ' + JSON.stringify(got));
      });
    }).catch(function(e){ return F('threw: ' + (e && e.message)); });
  });

  def('storage list sees the key', function(){
    return sb.storage.list('selftest:').then(function(keys){
      var hit = keys && keys.indexOf && keys.indexOf('selftest:rt') !== -1;
      // some fallbacks return bare keys; accept any array that includes our suffix
      var loose = keys && keys.join && keys.join(',').indexOf('selftest:rt') !== -1;
      return (hit || loose) ? P('list("selftest:") includes the probe key')
                            : F('probe key not found in list: ' + JSON.stringify(keys));
    }).catch(function(e){ return F('threw: ' + (e && e.message)); });
  });

  def('ai routes to your model', function(){
    if (!sb.connected) return Promise.resolve(S('skipped \u2014 AI needs a linked companion'));
    if (!sb.has('ai'))  return Promise.resolve(S('skipped \u2014 companion does not advertise "ai"'));
    return sb.ai('Name a cosy riverside tavern. Reply with just the name.', { system: 'You name fantasy taverns. Reply with only the name.' }).then(function(r){
      if (r.ok) return P('completion received (key never crossed the bridge): \u201c' + String(r.value).slice(0, 80) + '\u201d');
      if (r.reason === 'no-own-model') return S('skipped \u2014 no personal model set in the companion (AI Helper tab)');
      if (r.reason === 'denied')       return S('skipped \u2014 you declined the permission prompt');
      return F('ai() failed: ' + r.reason);
    }).catch(function(e){ return F('threw: ' + (e && e.message)); });
  });

  def('fallback contract (when offline)', function(){
    if (sb.connected) return Promise.resolve(S('skipped \u2014 companion is linked, fallback path not exercised'));
    var okHas = sb.has('storage') === false && sb.has('ai') === false;
    return sb.ai('x').then(function(r){
      var okAi = r.ok === false && (r.reason === 'disconnected' || r.reason === 'unsupported');
      return (okHas && okAi)
        ? P('offline: has()=false and ai() returns {ok:false, reason:"' + r.reason + '"} \u2014 degrades cleanly')
        : F('offline contract off: has()=' + sb.has('storage') + '/' + sb.has('ai') + ', ai=' + JSON.stringify(r));
    });
  });

  // ---- run loop ---------------------------------------------------------
  function runAll(){
    $('tests').innerHTML = '';
    nodes = {};
    tests.forEach(function(t){ mkRow(t.name); });
    var pass = 0, fail = 0, skip = 0, total = tests.length;
    $('s-total').textContent = total; $('s-pass').textContent = '0';
    $('s-fail').style.display = 'none'; $('s-skip').style.display = 'none';
    paintConn();

    var i = 0;
    function next(){
      if (i >= tests.length){
        $('s-pass').textContent = pass;
        if (fail){ var f = $('s-fail'); f.style.display = ''; f.textContent = fail + ' failed'; }
        if (skip){ var s = $('s-skip'); s.style.display = ''; s.textContent = skip + ' skipped'; }
        paintConn();
        return;
      }
      var t = tests[i++];
      Promise.resolve().then(t.run).then(function(res){
        res = res || F('no result');
        setRow(t.name, res.status, res.detail);
        if (res.status === 'pass') pass++; else if (res.status === 'fail') fail++; else skip++;
        $('s-pass').textContent = pass;
        paintConn();
        next();
      }, function(err){
        setRow(t.name, 'fail', 'threw: ' + (err && err.message));
        fail++; next();
      });
    }
    next();
  }

  if (!sb) {
    paintConn();
    $('tests').innerHTML = '<li class="test"><span class="badge fail">fail</span><div class="body"><div class="name">plugin present</div><div class="detail">window.weld.skybridge is missing \u2014 import {import:weld-skybridge-plugin} in this generator.</div></div></li>';
  } else {
    runAll();
    $('rerun').onclick = runAll;
  }

  // ---- diagnostics wiring ------------------------------------------------
  renderDiag();
  $('diag-refresh').onclick = renderDiag;
  $('diag-copy').onclick = copyReport;
  if (sb && typeof sb.onConnect === 'function') sb.onConnect(renderDiag);
  // the trace fills in over the first few seconds; refresh a few times
  [600, 1500, 3500, 6000].forEach(function (t) { setTimeout(renderDiag, t); });

  // ---- manual playground ------------------------------------------------
  function show(el, v){ el.textContent = (typeof v === 'string') ? v : JSON.stringify(v, null, 2); }
  if (sb) {
    $('save').onclick = function(){ sb.storage.set('demo:' + $('slot').value, { text: 'saved at ' + new Date().toLocaleTimeString(), at: Date.now() }).then(function(ok){ show($('store-out'), ok ? ('saved \u2713 via ' + sb.storage.backend()) : 'save failed/denied'); }); };
    $('load').onclick = function(){ sb.storage.get('demo:' + $('slot').value).then(function(v){ show($('store-out'), v ? v : '(nothing in that slot)'); }); };
    $('list').onclick = function(){ sb.storage.list('demo:').then(function(k){ show($('store-out'), k && k.length ? k : '(no slots yet)'); }); };
    $('ai-go').onclick = function(){ show($('ai-out'), 'thinking\u2026'); sb.ai($('ai-in').value, { system: 'You name fantasy taverns. Reply with just the name.' }).then(function(r){ show($('ai-out'), r.ok ? r.value : ('unavailable: ' + r.reason)); }); };
  }
})();
</script>
