// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

function button(opts) {

  var classes = ['w-btn'];

  if (opts.size === 'sm') classes.push('w-btn-sm');

    class: classes.join(' '),

}

function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = opts.title || opts.dismissable !== false
    ? el('div', { class: 'w-modal-hdr' }, [
        opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div'),
        opts.dismissable !== false ? closeBtn : null,
      ])
    : null;

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {
    if (e.key === 'Escape' && opts.dismissable !== false) close();
  }
  document.addEventListener('keydown', escHandler);

  document.body.appendChild(overlay);
  return { el: overlay, close: close, body: body };
}

<script>(function(){
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.quota) return;

  var STORAGE_KEY = 'weld-quota-state';
  var watchers = [];
  var lastEstimate = null;

  // -----------------------------------------------------------------
  // STORAGE ESTIMATION
  // -----------------------------------------------------------------
  function estimate() {
    if (!navigator.storage || typeof navigator.storage.estimate !== 'function') {
      var unsupported = {
        used: 0, available: 0, total: 0, percentUsed: 0,
        supported: false, usageDetails: null
      };
      lastEstimate = unsupported;
      return Promise.resolve(unsupported);
    }
    return navigator.storage.estimate().then(function(est) {
      var used = est.usage || 0;
      var total = est.quota || 0;
      var available = Math.max(0, total - used);
      var percentUsed = total > 0 ? (used / total) * 100 : 0;
      var result = {
        used: used,
        available: available,
        total: total,
        percentUsed: percentUsed,
        supported: true,
        usageDetails: est.usageDetails || null
      };
      lastEstimate = result;
      checkWatchers(result);
      return result;
    });
  }

  function formatBytes(bytes) {
    if (bytes == null || isNaN(bytes)) return '?';
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
    if (bytes < 1024 * 1024 * 1024) return (bytes / 1024 / 1024).toFixed(1) + ' MB';
    return (bytes / 1024 / 1024 / 1024).toFixed(2) + ' GB';
  }

  // -----------------------------------------------------------------
  // STORAGE WATCHERS
  // -----------------------------------------------------------------
  function watch(opts) {
    opts = opts || {};
    var w = {
      threshold: opts.threshold != null ? opts.threshold : 80,
      onCross: typeof opts.onCross === 'function' ? opts.onCross : function(){},
      every: opts.every || 60000,
      crossed: false,
      interval: null
    };
    watchers.push(w);
    w.interval = setInterval(estimate, w.every);
    estimate(); // initial check
    return function stop() {
      clearInterval(w.interval);
      var i = watchers.indexOf(w);
      if (i !== -1) watchers.splice(i, 1);
    };
  }

  function checkWatchers(est) {
    for (var i = 0; i < watchers.length; i++) {
      var w = watchers[i];
      var over = est.percentUsed >= w.threshold;
      if (over !== w.crossed) {
        w.crossed = over;
        try { w.onCross(est, over); } catch (e) { /* swallow */ }
      }
    }
  }

  // -----------------------------------------------------------------
  // RATE LIMITERS - sliding window, persisted
  // -----------------------------------------------------------------
  var rateConfigs = {};

  function loadRateState() {
    try {
      var raw = localStorage.getItem(STORAGE_KEY);
      if (!raw) return {};
      return JSON.parse(raw) || {};
    } catch (e) { return {}; }
  }

  function saveRateState(state) {
    try { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); }
    catch (e) { /* quota or disabled */ }
  }

  function pruneOld(timestamps, windowMs) {
    var cutoff = Date.now() - windowMs;
    var out = [];
    for (var i = 0; i < timestamps.length; i++) {
      if (timestamps[i] >= cutoff) out.push(timestamps[i]);
    }
    return out;
  }

  function makeStatus(name, timestamps, cfg, allowed) {
    var remaining = Math.max(0, cfg.maxPerWindow - timestamps.length);
    var resetsIn = timestamps.length > 0
      ? Math.max(0, (timestamps[0] + cfg.windowMs) - Date.now())
      : 0;
    return {
      name: name,
      allowed: allowed,
      remaining: remaining,
      used: timestamps.length,
      total: cfg.maxPerWindow,
      windowMs: cfg.windowMs,
      resetsIn: resetsIn
    };
  }

  function createRate(name, opts) {
    if (!name || typeof name !== 'string') throw new Error('rate name required');
    opts = opts || {};
    rateConfigs[name] = {
      maxPerWindow: opts.maxPerWindow || 60,
      windowMs: opts.windowMs || 60000
    };
    return getRate(name);
  }

  function getRate(name) {
    return {
      name: name,
      check: function() { return checkRate(name); },
      consume: function() { return consumeRate(name); },
      reset: function() { return resetRate(name); },
      history: function() { return rateHistory(name); }
    };
  }

  function checkRate(name) {
    var cfg = rateConfigs[name];
    if (!cfg) throw new Error('unknown rate: ' + name);
    var state = loadRateState();
    var timestamps = pruneOld(state[name] || [], cfg.windowMs);
    return makeStatus(name, timestamps, cfg, timestamps.length < cfg.maxPerWindow);
  }

  function consumeRate(name) {
    var cfg = rateConfigs[name];
    if (!cfg) throw new Error('unknown rate: ' + name);
    var state = loadRateState();
    var timestamps = pruneOld(state[name] || [], cfg.windowMs);
    if (timestamps.length >= cfg.maxPerWindow) {
      return makeStatus(name, timestamps, cfg, false);
    }
    timestamps.push(Date.now());
    state[name] = timestamps;
    saveRateState(state);
    return makeStatus(name, timestamps, cfg, true);
  }

  function resetRate(name) {
    var state = loadRateState();
    delete state[name];
    saveRateState(state);
    return true;
  }

  function rateHistory(name) {
    var cfg = rateConfigs[name];
    if (!cfg) return [];
    var state = loadRateState();
    return pruneOld(state[name] || [], cfg.windowMs);
  }

  function listRates() { return Object.keys(rateConfigs); }

  // -----------------------------------------------------------------
  // API
  // -----------------------------------------------------------------
  window.weld.quota = {
    // Storage
    estimate: estimate,
    lastEstimate: function() { return lastEstimate; },
    watch: watch,
    formatBytes: formatBytes,

    // Rate limiting
    rate: {
      create: createRate,
      get: getRate,
      check: checkRate,
      consume: consumeRate,
      reset: resetRate,
      history: rateHistory,
      list: listRates
    },

    version: '1.0.0'
  };
})();

(function(){
  function start() {

  function buildDemo() {
    var ui = window.weld.ui;

    // -------- STORAGE CARD --------
    var storageCard = document.getElementById('storage-card');
    var storageRow = storageCard.querySelector('.row');

    }));
    storageRow.appendChild(ui.button({
      label: 'Fill 1MB to IndexedDB', icon: 'plus', size: 'sm',
      onClick: function() { fillIdb(1).then(refreshStorage); }
    }));
    storageRow.appendChild(ui.button({
      label: 'Fill 5MB', icon: 'plus', size: 'sm',
      onClick: function() { fillIdb(5).then(refreshStorage); }
    }));
    storageRow.appendChild(ui.button({
      label: 'Wipe demo data', variant: 'danger', size: 'sm', icon: 'trash',
      onClick: function() { wipeIdb().then(refreshStorage); }
    }));

    function refreshStorage() {
      quota.estimate().then(function(e) {
        var fill = document.getElementById('storage-fill');
        var label = document.getElementById('storage-label');
        var details = document.getElementById('storage-details');

        if (!e.supported) {
          fill.style.width = '0%';
          label.textContent = 'storage estimate not supported';
          details.textContent = '';
          return;
        }
        var pct = Math.min(100, e.percentUsed);
        fill.style.width = pct.toFixed(1) + '%';
        label.textContent = pct.toFixed(2) + '% used';
        details.textContent = quota.formatBytes(e.used) + ' of ' + quota.formatBytes(e.total) +
                              ' (' + quota.formatBytes(e.available) + ' available)';
      });
    }

    function fillIdb(mb) {
      return new Promise(function(resolve) {
        var req = indexedDB.open('weld-quota-demo', 1);
        req.onupgradeneeded = function(e){
          var db = e.target.result;
          if (!db.objectStoreNames.contains('blobs')) db.createObjectStore('blobs', {autoIncrement:true});
        };
        req.onsuccess = function(){
          var db = req.result;
          var tx = db.transaction('blobs', 'readwrite');
          var store = tx.objectStore('blobs');
          // Generate ~1MB blob
          var chunk = new Uint8Array(1024 * 1024 * mb);
          for (var i = 0; i < chunk.length; i++) chunk[i] = i & 0xff;
          store.add(chunk);
          tx.oncomplete = function(){ db.close(); resolve(); };
          tx.onerror = function(){ db.close(); resolve(); };
        };
        req.onerror = function(){ resolve(); };
      });
    }

    function wipeIdb() {
      return new Promise(function(resolve) {
        var req = indexedDB.deleteDatabase('weld-quota-demo');
        req.onsuccess = req.onerror = function(){ resolve(); };
      });
    }

    // -------- WATCHER CARD --------
    var watcherCard = document.getElementById('watcher-card');
    var watcherRow = watcherCard.querySelector('.row');
    var threshold = 0.5;  // 0.5% for easy testing
    var thresholdLabel = document.getElementById('threshold-label');
    var stopFn = null;

    watcherRow.appendChild(ui.button({
      label: 'Start watching at 0.5%', variant: 'primary', size: 'sm', icon: 'eye',
      onClick: function(){

                       isOver ? 'warn' : '');
          }
        });
      }
    }));
    watcherRow.appendChild(ui.button({
      label: 'Stop watcher', size: 'sm', variant: 'ghost',
      onClick: function(){
        if (stopFn) { stopFn(); stopFn = null; logWatcher('Watcher stopped'); }
        else logWatcher('No watcher running');
      }
    }));

    function logWatcher(msg, cls) {
      var log = document.getElementById('watcher-log');
      if (log.querySelector('.empty')) log.innerHTML = '';
      var entry = document.createElement('div');
      entry.className = 'entry' + (cls ? ' ' + cls : '');
      entry.textContent = '[' + new Date().toLocaleTimeString() + '] ' + msg;
      log.insertBefore(entry, log.firstChild);
    }

    // -------- RATE LIMITER CARD --------
    var rateCard = document.getElementById('rate-card');
    var rateRow = rateCard.querySelector('.row');

    var rateLimit = quota.rate.create('demo-api', { maxPerWindow: 10, windowMs: 30000 });

    rateRow.appendChild(ui.button({
      label: 'Consume (call API)', variant: 'primary', size: 'sm', icon: 'send',
      onClick: function(){
        var s = rateLimit.consume();
        renderRate();
        if (!s.allowed) {
          logRate('BLOCKED -- ' + Math.ceil(s.resetsIn/1000) + 's until reset', 'danger');
        } else {
          logRate('Consumed ' + s.used + '/' + s.total + ' (' + s.remaining + ' remaining)');
        }
      }
    }));
    rateRow.appendChild(ui.button({
      label: 'Check (no consume)', size: 'sm', icon: 'info',
      onClick: function(){
        var s = rateLimit.check();
        logRate('Check: ' + s.used + '/' + s.total + ' used, ' +
                Math.ceil(s.resetsIn/1000) + 's until reset');
      }
    }));
    rateRow.appendChild(ui.button({
      label: 'Reset rate', size: 'sm', variant: 'ghost', icon: 'refresh',
      onClick: function(){ rateLimit.reset(); renderRate(); logRate('Rate state cleared'); }
    }));

      meter.innerHTML = '';
