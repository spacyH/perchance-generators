// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

// COMPONENTS
function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

function input(opts) {
  opts = opts || {};
  return el('input', {
    class: 'w-input',
    type: opts.type || 'text',
    placeholder: opts.placeholder,
    value: opts.value != null ? opts.value : '',
    oninput: opts.onInput ? function(e){ opts.onInput(e.target.value, e); } : null,
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
    readonly: opts.readonly ? 'readonly' : null,
  });
}

function select(opts) {
  opts = opts || {};
  var options = (opts.options || []).map(function(o){
    if (typeof o === 'string') o = { value: o, label: o };
    return el('option', {
      value: o.value,
      selected: o.value === opts.value ? 'selected' : null,
    }, o.label);
  });
  return el('select', {
    class: 'w-select',
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
  }, options);
}

function tabs(opts) {
  opts = opts || {};
  var items = opts.items || [];
  var active = opts.active || (items[0] && items[0].id);
  var tabBar = el('div', { class: 'w-tabs' });
  var panel = el('div', { class: 'w-tab-panel' });

  function render(activeId) {
    tabBar.innerHTML = '';
    items.forEach(function(it){
      var btn = el('button', {
        class: 'w-tab' + (it.id === activeId ? ' w-tab-active' : ''),
        onclick: function(){ render(it.id); if (opts.onChange) opts.onChange(it.id); },
      }, it.label);
      tabBar.appendChild(btn);
    });
    panel.innerHTML = '';
    var found = items.find(function(it){ return it.id === activeId; });
    if (found && found.content) {
      panel.appendChild(typeof found.content === 'function' ? found.content() : found.content);
    }
  }
  render(active);

  var wrap = el('div', { class: 'w-tabs-wrap' }, [tabBar, panel]);
  wrap.setActive = function(id){ render(id); };
  return wrap;
}

// Returns { el, close } and is auto-attached to body
function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = opts.title || opts.dismissable !== false
    ? el('div', { class: 'w-modal-hdr' }, [
        opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div'),
        opts.dismissable !== false ? closeBtn : null,
      ])
    : null;

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {
    if (e.key === 'Escape' && opts.dismissable !== false) close();
  }
  document.addEventListener('keydown', escHandler);

  document.body.appendChild(overlay);
  return { el: overlay, close: close, body: body };
}

// Toast host management
// -----------------------------------------------------------------
var config = {
  position: 'top-right',
  duration: 4000,
  max: 5
};
var hosts = {};
var activeToasts = [];

function getHost(position) {
  if (hosts[position]) return hosts[position];
  var h = document.createElement('div');
  h.className = 'weld-toast-host ' + position;
  document.body.appendChild(h);
  hosts[position] = h;
  return h;
}

function configure(opts) {
  if (!opts) return;
  if (opts.position) config.position = opts.position;
  if (opts.duration != null) config.duration = opts.duration;
  if (opts.max != null) config.max = opts.max;
}

// -----------------------------------------------------------------
// Toast notification
// -----------------------------------------------------------------
var ICONS = {
  info:    'i',
  success: '\u2713',
  warn:    '!',
  error:   'x'
};

function show(message, opts) {
  injectCss();
  opts = opts || {};
  var type = opts.type || 'info';
  var duration = opts.duration != null ? opts.duration : config.duration;
  var position = opts.position || config.position;
  var dismissible = opts.dismissible !== false;

  var host = getHost(position);

  var toast = document.createElement('div');
  toast.className = 'weld-toast ' + type;

  var icon = document.createElement('div');
  icon.className = 'weld-toast-icon';
  icon.textContent = ICONS[type] || ICONS.info;
  toast.appendChild(icon);

  var msg = document.createElement('div');
  msg.className = 'weld-toast-msg';
  msg.textContent = String(message);
  toast.appendChild(msg);

  if (dismissible) {
    var close = document.createElement('button');
    close.className = 'weld-toast-close';
    close.textContent = '\u00d7';
    close.setAttribute('aria-label', 'close');
    close.addEventListener('click', function() { dismiss(handle); });
    toast.appendChild(close);
  }

  // Enforce max
  while (activeToasts.length >= config.max) {
    var oldest = activeToasts.shift();
    if (oldest) dismiss(oldest, true);
  }

  host.appendChild(toast);
  // Trigger CSS transition
  setTimeout(function() { toast.classList.add('shown'); }, 10);

  var timer = null;
  var handle = { toast: toast, host: host, dismissed: false, _t: null };
  activeToasts.push(handle);

  if (duration > 0) {
    handle._t = setTimeout(function() { dismiss(handle); }, duration);
  }

  return handle;
}

function dismiss(handle, immediate) {
  if (!handle || handle.dismissed) return;
  handle.dismissed = true;
  if (handle._t) clearTimeout(handle._t);
  var idx = activeToasts.indexOf(handle);
  if (idx !== -1) activeToasts.splice(idx, 1);

  if (immediate) {
    if (handle.toast.parentNode) handle.toast.parentNode.removeChild(handle.toast);
    return;
  }
  handle.toast.classList.remove('shown');
  setTimeout(function() {
    if (handle.toast.parentNode) handle.toast.parentNode.removeChild(handle.toast);
  }, 240);
}

function clearAll() {
  while (activeToasts.length) {
    dismiss(activeToasts[0], true);
  }
}

  }, 200);
}

function confirmDialog(opts) {
  opts = opts || {};
  var title = opts.title || 'Confirm';
  var message = opts.message || 'Are you sure?';
  var okLabel = opts.okLabel || 'OK';
  var cancelLabel = opts.cancelLabel || 'Cancel';
  var danger = opts.danger === true;

  return new Promise(function(resolve) {
    var bg = buildBackdrop();
    var modal = document.createElement('div');
    modal.className = 'weld-toast-modal';

    var h = document.createElement('h4');
    h.textContent = title;
    modal.appendChild(h);

    var p = document.createElement('p');
    p.textContent = message;
    modal.appendChild(p);

    var actions = document.createElement('div');
    actions.className = 'weld-toast-actions';

    var cancel = document.createElement('button');
    cancel.className = 'weld-toast-btn ghost';
    cancel.textContent = cancelLabel;
    cancel.addEventListener('click', function() { teardown(bg); resolve(false); });
    actions.appendChild(cancel);

    var ok = document.createElement('button');
    ok.className = 'weld-toast-btn ' + (danger ? 'danger' : 'primary');
    ok.textContent = okLabel;
    ok.addEventListener('click', function() { teardown(bg); resolve(true); });
    actions.appendChild(ok);

    modal.appendChild(actions);
    bg.appendChild(modal);
    setTimeout(function() { ok.focus(); }, 100);
  });
}

function promptDialog(opts) {
  opts = opts || {};
  var title = opts.title || 'Prompt';
  var message = opts.message || '';
  var placeholder = opts.placeholder || '';
  var defaultValue = opts.defaultValue || '';
  var multiline = opts.multiline === true;
  var okLabel = opts.okLabel || 'OK';
  var cancelLabel = opts.cancelLabel || 'Cancel';

  return new Promise(function(resolve) {
    var bg = buildBackdrop();
    var modal = document.createElement('div');
    modal.className = 'weld-toast-modal';

    var h = document.createElement('h4');
    h.textContent = title;
    modal.appendChild(h);

    if (message) {
      var p = document.createElement('p');
      p.textContent = message;
      modal.appendChild(p);
    }

    var input;
    if (multiline) {
      input = document.createElement('textarea');
    } else {
      input = document.createElement('input');
      input.type = 'text';
    }
    input.placeholder = placeholder;
    input.value = defaultValue;
    modal.appendChild(input);

    var actions = document.createElement('div');
    actions.className = 'weld-toast-actions';

    var cancel = document.createElement('button');
    cancel.className = 'weld-toast-btn ghost';
    cancel.textContent = cancelLabel;
    cancel.addEventListener('click', function() { teardown(bg); resolve(null); });
    actions.appendChild(cancel);

    var ok = document.createElement('button');
    ok.className = 'weld-toast-btn primary';
    ok.textContent = okLabel;
    ok.addEventListener('click', function() { teardown(bg); resolve(input.value); });
    actions.appendChild(ok);

    modal.appendChild(actions);
    bg.appendChild(modal);
    setTimeout(function() { input.focus(); input.select(); }, 100);

    if (!multiline) {
      input.addEventListener('keydown', function(e) {
        if (e.key === 'Enter') { teardown(bg); resolve(input.value); }
        if (e.key === 'Escape') { teardown(bg); resolve(null); }
      });
    }
  });
}
