      <span class="ver">v1.0.0</span>
    </div>
    <p class="lead">Storage estimation and API rate limiting. Watch usage, throttle calls, prevent failures before they happen.</p>
  </div>
  <h2>Overview</h2>
<p>weld.quota tracks two kinds of limit your generator can hit: <strong>storage quota</strong> (how much of the browser's IndexedDB allowance you've used) and <strong>API rate limits</strong> (how many calls you've made to a particular endpoint in a sliding window).</p>
<p>Both are common pain points. Storage usually fails silently then catastrophically; rate limits get hit during quick demos and erode user trust. This plugin gives you cheap programmatic access to both, plus a simple watcher pattern for warning users before things break.</p>

<h2>Storage estimation</h2><h3>weld.quota.estimate()</h3>
<div class="sig" data-codeid="sig4"></div>
<p>Wraps <code>navigator.storage.estimate()</code> and returns a normalized status object:</p>
<pre data-codeid="code5"></pre>

<p>The first call also caches the result -- <code>weld.quota.lastEstimate()</code> returns the most recent without doing another async lookup.</p>
<h3>weld.quota.watch(\{threshold, onCross, every\})</h3>
<div class="sig" data-codeid="sig6"></div>
<p>Poll storage every <code>every</code> ms (default 60000) and fire <code>onCross(estimate, isOver)</code> when crossing the threshold percentage. Fires once on the way up and again on the way down -- not repeatedly while above. Returns a function to stop watching.</p>
<pre data-codeid="code7"></pre>
<h3>weld.quota.formatBytes(n)</h3>
<div class="sig" data-codeid="sig8"></div>
<p>Convenience formatter. Picks the unit (B / KB / MB / GB) and renders 1 decimal place.</p>
<pre data-codeid="code9"></pre>
<h2>Rate limiters</h2><h3>weld.quota.rate.create(name, \{maxPerWindow, windowMs\})</h3>
<div class="sig" data-codeid="sig10"></div>
<p>Create a named rate limiter. <code>maxPerWindow</code> defaults to 60, <code>windowMs</code> to 60000. State is persisted in <code>localStorage</code> so it survives page reloads.</p>
<pre data-codeid="code11"></pre>
<p>Returns a limiter object with these methods:</p>
<h3>limit.check()</h3>
<div class="sig" data-codeid="sig12"></div>
<p>See how many calls are available without consuming. Same shape as <code>consume()</code>:</p>
<pre data-codeid="code13"></pre>
<h3>limit.consume()</h3>
<div class="sig" data-codeid="sig14"></div>
<p>Record a call. Returns the same status. If <code>allowed</code> is false, the call was not recorded -- you should not perform the action.</p>
<pre data-codeid="code15"></pre>
<h3>limit.reset() / limit.history()</h3>
<p><code>reset()</code> clears the consumption history for this limit. <code>history()</code> returns the array of timestamps in the current sliding window (useful for debugging or charting).</p>

<div class="sig" data-codeid="sig17"></div>
<p>Names of every rate limiter currently configured. Useful for diagnostics.</p>

<h2>Persistence</h2><div class="note">
Rate limit state is stored in <code>localStorage</code> under the key <code>weld-quota-state</code>. It survives page reloads but is per-origin -- a user's rate state on perchance.org applies across all generators on that domain. If you want per-generator state, prefix your rate names: <code>weld.quota.rate.create('mygen:openai', ...)</code>.

<script>
(function(){
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.quota) return;

  var STORAGE_KEY = 'weld-quota-state';
  var watchers = [];
  var lastEstimate = null;

  // -----------------------------------------------------------------
  // STORAGE ESTIMATION
  // -----------------------------------------------------------------
  function estimate() {
    if (!navigator.storage || typeof navigator.storage.estimate !== 'function') {
      var unsupported = {
        used: 0, available: 0, total: 0, percentUsed: 0,
        supported: false, usageDetails: null
      };
      lastEstimate = unsupported;
      return Promise.resolve(unsupported);
    }
    return navigator.storage.estimate().then(function(est) {
      var used = est.usage || 0;
      var total = est.quota || 0;
      var available = Math.max(0, total - used);
      var percentUsed = total > 0 ? (used / total) * 100 : 0;
      var result = {
        used: used,
        available: available,
        total: total,
        percentUsed: percentUsed,
        supported: true,
        usageDetails: est.usageDetails || null
      };
      lastEstimate = result;
      checkWatchers(result);
      return result;
    });
  }

  function formatBytes(bytes) {
    if (bytes == null || isNaN(bytes)) return '?';
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
    if (bytes < 1024 * 1024 * 1024) return (bytes / 1024 / 1024).toFixed(1) + ' MB';
    return (bytes / 1024 / 1024 / 1024).toFixed(2) + ' GB';
  }

  // -----------------------------------------------------------------
  // STORAGE WATCHERS
  // -----------------------------------------------------------------
  function watch(opts) {
    opts = opts || {};
    var w = {
      threshold: opts.threshold != null ? opts.threshold : 80,
      onCross: typeof opts.onCross === 'function' ? opts.onCross : function(){},
      every: opts.every || 60000,
      crossed: false,
      interval: null
    };
    watchers.push(w);
    w.interval = setInterval(estimate, w.every);
    estimate(); // initial check
    return function stop() {
      clearInterval(w.interval);
      var i = watchers.indexOf(w);
      if (i !== -1) watchers.splice(i, 1);
    };
  }

  function checkWatchers(est) {
    for (var i = 0; i < watchers.length; i++) {
      var w = watchers[i];
      var over = est.percentUsed >= w.threshold;
      if (over !== w.crossed) {
        w.crossed = over;
        try { w.onCross(est, over); } catch (e) { /* swallow */ }
      }
    }
  }

  // -----------------------------------------------------------------
  // RATE LIMITERS - sliding window, persisted
  // -----------------------------------------------------------------
  var rateConfigs = {};

  function loadRateState() {
    try {
      var raw = localStorage.getItem(STORAGE_KEY);
      if (!raw) return {};
      return JSON.parse(raw) || {};
    } catch (e) { return {}; }
  }

  function saveRateState(state) {
    try { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); }
    catch (e) { /* quota or disabled */ }
  }

  function pruneOld(timestamps, windowMs) {
    var cutoff = Date.now() - windowMs;
    var out = [];
    for (var i = 0; i < timestamps.length; i++) {
      if (timestamps[i] >= cutoff) out.push(timestamps[i]);
    }
    return out;
  }

  function makeStatus(name, timestamps, cfg, allowed) {
    var remaining = Math.max(0, cfg.maxPerWindow - timestamps.length);
    var resetsIn = timestamps.length > 0
      ? Math.max(0, (timestamps[0] + cfg.windowMs) - Date.now())
      : 0;
    return {
      name: name,
      allowed: allowed,
      remaining: remaining,
      used: timestamps.length,
      total: cfg.maxPerWindow,
      windowMs: cfg.windowMs,
      resetsIn: resetsIn
    };
  }

  function createRate(name, opts) {
    if (!name || typeof name !== 'string') throw new Error('rate name required');
    opts = opts || {};
    rateConfigs[name] = {
      maxPerWindow: opts.maxPerWindow || 60,
      windowMs: opts.windowMs || 60000
    };
    return getRate(name);
  }

  function getRate(name) {
    return {
      name: name,
      check: function() { return checkRate(name); },
      consume: function() { return consumeRate(name); },
      reset: function() { return resetRate(name); },
      history: function() { return rateHistory(name); }
    };
  }

  function checkRate(name) {
    var cfg = rateConfigs[name];
    if (!cfg) throw new Error('unknown rate: ' + name);
    var state = loadRateState();
    var timestamps = pruneOld(state[name] || [], cfg.windowMs);
    return makeStatus(name, timestamps, cfg, timestamps.length < cfg.maxPerWindow);
  }

  function consumeRate(name) {
    var cfg = rateConfigs[name];
    if (!cfg) throw new Error('unknown rate: ' + name);
    var state = loadRateState();
    var timestamps = pruneOld(state[name] || [], cfg.windowMs);
    if (timestamps.length >= cfg.maxPerWindow) {
      return makeStatus(name, timestamps, cfg, false);
    }
    timestamps.push(Date.now());
    state[name] = timestamps;
    saveRateState(state);
    return makeStatus(name, timestamps, cfg, true);
  }

  function resetRate(name) {
    var state = loadRateState();
    delete state[name];
    saveRateState(state);
    return true;
  }

  function rateHistory(name) {
    var cfg = rateConfigs[name];
    if (!cfg) return [];
    var state = loadRateState();
    return pruneOld(state[name] || [], cfg.windowMs);
  }

  function listRates() { return Object.keys(rateConfigs); }

  // -----------------------------------------------------------------
  // API
  // -----------------------------------------------------------------
  window.weld.quota = {
    // Storage
    estimate: estimate,
    lastEstimate: function() { return lastEstimate; },
    watch: watch,
    formatBytes: formatBytes,

    // Rate limiting
    rate: {
      create: createRate,
      get: getRate,
      check: checkRate,
      consume: consumeRate,
      reset: resetRate,
      history: rateHistory,
      list: listRates
    },

    version: '1.0.0'
  };
})();

</script>
