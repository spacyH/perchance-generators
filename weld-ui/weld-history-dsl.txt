$meta
  title       = weld.history
  description = Chat session manager with auto-save. Each session wraps a branching tree plus metadata. Save, load, list, rename, delete, search. Backed by weld.persist, uses weld.branching for tree ops.
  tags        = weld, history, session, chat, save, plugin

$output() =>
  'use strict';

  if (!window.weld) window.weld = {};
  if (window.weld.history) return window.weld.history;

  // -----------------------------------------------------------------
  // weld.history: Chat session manager.
  // Each session = metadata (title, character, timestamps, message count)
  //              + a serialized weld.branching tree.
  // Backed by weld.persist for storage. Uses weld.branching for in-memory
  // tree operations.
  //
  // Soft dependencies:
  //   weld.persist  - required for persistent storage (fatal without it)
  //   weld.branching - optional; if not loaded, sessions store raw message
  //                    arrays instead of branching trees
  // -----------------------------------------------------------------

  function uid() {
    return 's_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 8);
  }

  function createHistory(opts) {
    opts = opts || {};
    var dbName = opts.dbName || 'weld-history';
    var autoSave = opts.autoSave !== false;
    var autoSaveDebounce = opts.autoSaveDebounceMs || 2000;

    // Require weld.persist
    if (!window.weld || !window.weld.persist) {
      throw new Error('weld.history requires weld.persist to be loaded first.');
    }

    var db = window.weld.persist.open(dbName);
    var sessions = db.collection('sessions');
    var trees = db.collection('trees');

    var activeSessionId = null;
    var activeTree = null;        // weld.branching instance if available
    var activeMessages = null;    // fallback: plain array of messages
    var saveTimer = null;
    var changeListeners = new Array();

    function hasBranching() {
      return !!(window.weld && window.weld.branching);
    }

    function emit(event, data) {
      for (var i = 0; i < changeListeners.length; i++) {
        try { Reflect.get(changeListeners, i)(event, data); } catch (e) { /* swallow */ }
      }
    }

    function scheduleSave() {
      if (!autoSave || !activeSessionId) return;
      if (saveTimer) clearTimeout(saveTimer);
      saveTimer = setTimeout(function() {
        saveTimer = null;
        save().catch(function(e) {
          console.error(String.fromCharCode(91)+'weld.history'+String.fromCharCode(93)+' auto-save failed:', e);
        });
      }, autoSaveDebounce);
    }

    // -------- Create a new session --------
    function create(meta) {
      meta = meta || {};
      var id = meta.id || uid();
      var now = Date.now();
      var session = {
        id: id,
        title: meta.title || 'New conversation',
        characterId: meta.characterId || null,
        characterName: meta.characterName || null,
        createdAt: now,
        updatedAt: now,
        messageCount: 0,
        pinned: meta.pinned === true,
        summary: meta.summary || '',
        metadata: meta.metadata || null
      };

      // Initialize tree or message array
      if (hasBranching()) {
        activeTree = window.weld.branching.create();
        activeMessages = null;
      } else {
        activeTree = null;
        activeMessages = new Array();
      }
      activeSessionId = id;

      return sessions.put(session).then(function() {
        return saveTree();
      }).then(function() {
        emit('create', { sessionId: id });
        return id;
      });
    }

    // -------- Load a session --------
    function load(sessionId) {
      activeSessionId = null;
      activeTree = null;
      activeMessages = null;

      return sessions.get(sessionId).then(function(session) {
        if (!session) throw new Error('weld.history: session not found: ' + sessionId);
        return trees.get(sessionId).then(function(treeData) {
          activeSessionId = sessionId;
          if (treeData && treeData.tree && hasBranching()) {
            activeTree = window.weld.branching.create();
            activeTree.deserialize(treeData.tree);
            activeMessages = null;
          } else if (treeData && treeData.messages) {
            activeTree = null;
            activeMessages = treeData.messages;
          } else {
            // Empty session
            if (hasBranching()) {
              activeTree = window.weld.branching.create();
              activeMessages = null;
            } else {
              activeTree = null;
              activeMessages = new Array();
            }
          }
          // Update view time
          return sessions.update(sessionId, { updatedAt: Date.now() });
        });
      }).then(function() {
        emit('load', { sessionId: sessionId });
        return sessionId;
      });
    }

    // -------- Save the active session --------
    function save() {
      if (!activeSessionId) return Promise.reject(new Error('weld.history: no active session'));
      var treeData = { id: activeSessionId };
      if (activeTree) {
        treeData.tree = activeTree.serialize();
        treeData.messages = null;
      } else {
        treeData.tree = null;
        treeData.messages = activeMessages ? activeMessages.slice() : new Array();
      }
      var msgCount = activeTree ? activeTree.size() : (activeMessages ? activeMessages.length : 0);
      return Promise.all([
        trees.put(treeData),
        sessions.update(activeSessionId, {
          updatedAt: Date.now(),
          messageCount: msgCount
        })
      ]).then(function() {
        emit('save', { sessionId: activeSessionId });
      });
    }

    function saveTree() {
      var treeData = { id: activeSessionId };
      if (activeTree) {
        treeData.tree = activeTree.serialize();
        treeData.messages = null;
      } else {
        treeData.tree = null;
        treeData.messages = activeMessages ? activeMessages.slice() : new Array();
      }
      return trees.put(treeData);
    }

    // -------- Append a message to the active session --------
    function append(message) {
      if (!activeSessionId) throw new Error('weld.history: no active session');
      if (activeTree) {
        var id = activeTree.append(message);
        scheduleSave();
        emit('append', { sessionId: activeSessionId, nodeId: id });
        return id;
      } else {
        if (!message.id) message.id = uid();
        activeMessages.push(message);
        scheduleSave();
        emit('append', { sessionId: activeSessionId, messageId: message.id });
        return message.id;
      }
    }

    // -------- Get the active conversation --------
    function getMessages() {
      if (!activeSessionId) return new Array();
      if (activeTree) {
        return activeTree.activeBranch().map(function(n) { return n.data; });
      }
      return activeMessages ? activeMessages.slice() : new Array();
    }

    function getTree() {
      return activeTree;
    }

    // -------- List sessions --------
    function list(listOpts) {
      listOpts = listOpts || {};
      return sessions.getAll().then(function(all) {
        // Filter
        if (listOpts.characterId) {
          all = all.filter(function(s) { return s.characterId === listOpts.characterId; });
        }
        if (listOpts.search) {
          var q = String(listOpts.search).toLowerCase();
          all = all.filter(function(s) {
            var t = s.title ? s.title.toLowerCase() : '';
            var cn = s.characterName ? s.characterName.toLowerCase() : '';
            var sm = s.summary ? s.summary.toLowerCase() : '';
            return t.indexOf(q) !== -1 || cn.indexOf(q) !== -1 || sm.indexOf(q) !== -1;
          });
        }
        if (listOpts.pinnedOnly) {
          all = all.filter(function(s) { return s.pinned === true; });
        }
        // Sort: pinned sessions first, then by updatedAt descending. Pass
        // pinnedFirst:false to sort purely by recency.
        var pinFirst = listOpts.pinnedFirst !== false;
        all.sort(function(a, b) {
          if (pinFirst) {
            var pa = a.pinned ? 1 : 0, pb = b.pinned ? 1 : 0;
            if (pa !== pb) return pb - pa;
          }
          return (b.updatedAt || 0) - (a.updatedAt || 0);
        });
        // Limit
        if (listOpts.limit && listOpts.limit > 0) {
          all = all.slice(0, listOpts.limit);
        }
        return all;
      });
    }

    // -------- Session metadata operations --------
    function rename(sessionId, title) {
      return sessions.update(sessionId, { title: String(title), updatedAt: Date.now() }).then(function() {
        emit('rename', { sessionId: sessionId, title: title });
      });
    }

    // Pin / unpin a session. Pinning is not "activity", so it does NOT bump
    // updatedAt -- a pinned old chat should keep its real last-touched time,
    // it just sorts to the top via list()'s pinnedFirst.
    function setPinned(sessionId, pinned) {
      var id = sessionId || activeSessionId;
      return sessions.update(id, { pinned: pinned === true }).then(function() {
        emit('pin', { sessionId: id, pinned: pinned === true });
        return pinned === true;
      });
    }
    function pin(sessionId) { return setPinned(sessionId, true); }
    function unpin(sessionId) { return setPinned(sessionId, false); }
    function togglePin(sessionId) {
      var id = sessionId || activeSessionId;
      return sessions.get(id).then(function(s) { return setPinned(id, !(s && s.pinned)); });
    }

    // Set (or clear) a session's summary -- a short recap an app can show in
    // the session list or generate from the conversation. Not activity, so it
    // doesn't bump updatedAt either.
    function setSummary(sessionId, summary) {
      var id = sessionId || activeSessionId;
      return sessions.update(id, { summary: String(summary == null ? '' : summary) }).then(function() {
        emit('summary', { sessionId: id, summary: String(summary == null ? '' : summary) });
      });
    }

    function getMeta(sessionId) {
      return sessions.get(sessionId || activeSessionId);
    }

    function updateMeta(sessionId, patch) {
      return sessions.update(sessionId, Object.assign({}, patch, { updatedAt: Date.now() }));
    }

    // -------- Delete --------
    function del(sessionId) {
      if (sessionId === activeSessionId) {
        activeSessionId = null;
        activeTree = null;
        activeMessages = null;
      }
      return Promise.all([
        sessions.delete(sessionId),
        trees.delete(sessionId)
      ]).then(function() {
        emit('delete', { sessionId: sessionId });
        return true;
      });
    }

    function clear() {
      activeSessionId = null;
      activeTree = null;
      activeMessages = null;
      return Promise.all([
        sessions.clear(),
        trees.clear()
      ]).then(function() {
        emit('clear', {});
      });
    }

    // -------- Listeners --------
    function onChange(fn) {
      changeListeners.push(fn);
      return function() {
        var idx = changeListeners.indexOf(fn);
        if (idx >= 0) changeListeners.splice(idx, 1);
      };
    }

    function getActiveId() {
      return activeSessionId;
    }

    function isActive() {
      return activeSessionId !== null;
    }

    return {
      create: create,
      load: load,
      save: save,
      append: append,
      getMessages: getMessages,
      getTree: getTree,
      list: list,
      rename: rename,
      pin: pin,
      unpin: unpin,
      togglePin: togglePin,
      setSummary: setSummary,
      getMeta: getMeta,
      updateMeta: updateMeta,
      delete: del,
      clear: clear,
      onChange: onChange,
      getActiveId: getActiveId,
      isActive: isActive
    };
  }

  window.weld.history = {
    create: createHistory,
    version: '1.1.0'
  };

  return window.weld.history;
