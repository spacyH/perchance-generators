// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

// COMPONENTS
function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

function input(opts) {
  opts = opts || {};
  return el('input', {
    class: 'w-input',
    type: opts.type || 'text',
    placeholder: opts.placeholder,
    value: opts.value != null ? opts.value : '',
    oninput: opts.onInput ? function(e){ opts.onInput(e.target.value, e); } : null,
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
    readonly: opts.readonly ? 'readonly' : null,
  });
}

function select(opts) {
  opts = opts || {};
  var options = (opts.options || []).map(function(o){
    if (typeof o === 'string') o = { value: o, label: o };
    return el('option', {
      value: o.value,
      selected: o.value === opts.value ? 'selected' : null,
    }, o.label);
  });
  return el('select', {
    class: 'w-select',
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
  }, options);
}

function tabs(opts) {
  opts = opts || {};
  var items = opts.items || [];
  var active = opts.active || (items[0] && items[0].id);
  var tabBar = el('div', { class: 'w-tabs' });
  var panel = el('div', { class: 'w-tab-panel' });

  function render(activeId) {
    tabBar.innerHTML = '';
    items.forEach(function(it){
      var btn = el('button', {
        class: 'w-tab' + (it.id === activeId ? ' w-tab-active' : ''),
        onclick: function(){ render(it.id); if (opts.onChange) opts.onChange(it.id); },
      }, it.label);
      tabBar.appendChild(btn);
    });
    panel.innerHTML = '';
    var found = items.find(function(it){ return it.id === activeId; });
    if (found && found.content) {
      panel.appendChild(typeof found.content === 'function' ? found.content() : found.content);
    }
  }
  render(active);

  var wrap = el('div', { class: 'w-tabs-wrap' }, [tabBar, panel]);
  wrap.setActive = function(id){ render(id); };

// Returns { el, close } and is auto-attached to body
function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = null;
  if (opts.title || opts.dismissable !== false) {
    var titleEl = opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div');
    var dismissEl = opts.dismissable !== false ? closeBtn : null;
    header = el('div', { class: 'w-modal-hdr' }, [titleEl, dismissEl]);
  }

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {
    if (e.key === 'Escape' && opts.dismissable !== false) close();
  }
  document.addEventListener('keydown', escHandler);

  document.body.appendChild(overlay);
  return { el: overlay, close: close, body: body };
}

function lightbox(opts) {
  opts = opts || {};
  lightboxInjectCss();
  var items = [];
  var currentIdx = -1;
  var overlay = document.createElement('div');
  overlay.className = 'w-lb';
  var img = document.createElement('img');
  img.className = 'w-lb-img';
  var closeBtn = document.createElement('button');
  closeBtn.className = 'w-lb-close';
  closeBtn.textContent = 'CLOSE';
  var prevBtn = document.createElement('button');
  prevBtn.className = 'w-lb-nav w-lb-prev';
  prevBtn.innerHTML = '&#8249;';
  var nextBtn = document.createElement('button');
  nextBtn.className = 'w-lb-nav w-lb-next';
  nextBtn.innerHTML = '&#8250;';
  var metaEl = document.createElement('div');
  metaEl.className = 'w-lb-meta';
  overlay.appendChild(closeBtn);
  overlay.appendChild(prevBtn);
  overlay.appendChild(nextBtn);
  overlay.appendChild(img);
  overlay.appendChild(metaEl);
  document.body.appendChild(overlay);

  function updateNav() {
    prevBtn.style.display = items.length > 1 ? '' : 'none';
    nextBtn.style.display = items.length > 1 ? '' : 'none';
  }

  function show(idx) {
    if (idx < 0 || idx >= items.length) return;
    currentIdx = idx;
    var item = items[idx];
    img.src = typeof item === 'string' ? item : (item.src || item.url || '');
    img.alt = (item && item.alt) || '';
    metaEl.textContent = (item && item.meta) || '';
    metaEl.style.display = metaEl.textContent ? '' : 'none';
    updateNav();
  }

  function open(src, openOpts) {
    openOpts = openOpts || {};
    if (Array.isArray(src)) {
      items = src;
      show(openOpts.index || 0);
    } else {
      items = [src];
      show(0);
    }
    overlay.classList.add('open');
  }

  function close() {
    overlay.classList.remove('open');
    img.src = '';
    if (opts.onClose) opts.onClose();
  }

  function next() {
    if (items.length < 2) return;
    show((currentIdx + 1) % items.length);
  }

  function prev() {
    if (items.length < 2) return;
    show((currentIdx - 1 + items.length) % items.length);
  }

  overlay.addEventListener('click', function(e) {
    if (e.target === overlay) close();
  });
  closeBtn.addEventListener('click', close);
  prevBtn.addEventListener('click', function(e) { e.stopPropagation(); prev(); });
  nextBtn.addEventListener('click', function(e) { e.stopPropagation(); next(); });

  function keyHandler(e) {
    if (!overlay.classList.contains('open')) return;
    if (e.key === 'Escape') close();
    else if (e.key === 'ArrowRight' || e.key === 'ArrowDown') next();
    else if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') prev();
  }
  document.addEventListener('keydown', keyHandler);

  function destroy() {
    document.removeEventListener('keydown', keyHandler);
    if (overlay.parentNode) overlay.parentNode.removeChild(overlay);
  }

  return {
    open: open,
    close: close,
    next: next,
    prev: prev,
    destroy: destroy,
    el: overlay,
    isOpen: function() { return overlay.classList.contains('open'); }
  };
}

// CLIPBOARD — copy text with optional toast feedback
function clipboard(text, msg) {
  var p;
  if (navigator.clipboard && navigator.clipboard.writeText) {
    p = navigator.clipboard.writeText(String(text));
  } else {
    p = new Promise(function(resolve, reject) {
      try {
        var ta = document.createElement('textarea');
        ta.value = String(text);
        ta.style.cssText = 'position:fixed;left:-9999px;';
        document.body.appendChild(ta);
        ta.select();
        document.execCommand('copy');
        document.body.removeChild(ta);
        resolve();
      } catch (e) { reject(e); }
    });
  }
  return p.then(function() {
    if (msg !== false && window.weld && window.weld.toast) {
      window.weld.toast.show(msg || 'Copied to clipboard.', 'success');
    }
    return true;
  }).catch(function() {
    if (window.weld && window.weld.toast) {
      window.weld.toast.show('Copy failed.', 'error');
    }
    return false;
  });
}

if (!window.weld) window.weld = {};

// -----------------------------------------------------------------
// weld.testrig: Assertion framework for Perchance plugin testing.
// PASS / FAIL / DRIFT / SKIP status levels, test groups with headers,
// live stats, progress bar, export. The platform monitor pattern,
// generalized.
// -----------------------------------------------------------------

function createTestRig(opts) {
  opts = opts || {};
  var results = [];
  var counts = { PASS: 0, FAIL: 0, DRIFT: 0, SKIP: 0, INFO: 0 };
  var groups = {};
  var startTime = null;
  var listeners = [];
  var container = typeof opts.container === 'string'
    ? document.querySelector(opts.container) : opts.container;

  function emit(event, data) {
    for (var i = 0; i < listeners.length; i++) {
      try { listeners[i](event, data); } catch (e) {}
    }
  }

  function stringify(v) {
    if (v === undefined) return 'undefined';
    if (v === null) return 'null';
    if (typeof v === 'string') return v.length > 80 ? JSON.stringify(v.slice(0, 80)) + '...' : JSON.stringify(v);
    try {
      var s = JSON.stringify(v);
      return s.length > 100 ? s.slice(0, 100) + '...' : s;
    } catch (e) { return String(v); }
  }

  function record(group, name, status, expected, actual, note) {
    var entry = {
      group: group,
      name: name,
      status: status,
      expected: expected,
      actual: actual,
      note: note || null,
      timestamp: Date.now()
    };
    results.push(entry);
    counts[status] = (counts[status] || 0) + 1;
    if (!groups[group]) groups[group] = [];
    groups[group].push(entry);
    emit('result', entry);
    if (container) renderResult(entry);
    return status === 'PASS';
  }

  // -------- Assertions --------
  function assertEqual(group, name, expected, actual, assertOpts) {
    assertOpts = assertOpts || {};
    var cmp = assertOpts.cmp
      ? assertOpts.cmp(expected, actual)
      : stringify(expected) === stringify(actual);
    var status = cmp ? 'PASS' : (assertOpts.soft ? 'DRIFT' : 'FAIL');
    return record(group, name, status, expected, actual, assertOpts.note);
  }

  function assertTrue(group, name, actual, assertOpts) {
    return assertEqual(group, name, true, !!actual, assertOpts);
  }

  function assertFalse(group, name, actual, assertOpts) {
    return assertEqual(group, name, false, !!actual, assertOpts);
  }

  function assertThrows(group, name, fn, assertOpts) {
    assertOpts = assertOpts || {};
    var threw = false;
    var error = null;
    try { fn(); } catch (e) { threw = true; error = e; }
    var status = threw ? 'PASS' : 'FAIL';
    return record(group, name, status, 'should throw', threw ? 'threw: ' + (error && error.message || error) : 'did not throw', assertOpts.note);
  }

  // Async version: resolves the promise and checks the result
  function assertResolves(group, name, promise, assertOpts) {
    assertOpts = assertOpts || {};
    return Promise.resolve(promise).then(function(val) {
      return record(group, name, 'PASS', 'should resolve', 'resolved: ' + stringify(val), assertOpts.note);
    }).catch(function(err) {
      return record(group, name, 'FAIL', 'should resolve', 'rejected: ' + (err && err.message || err), assertOpts.note);
    });
  }

  function assertRejects(group, name, promise, assertOpts) {
    assertOpts = assertOpts || {};
    return Promise.resolve(promise).then(function(val) {
      return record(group, name, 'FAIL', 'should reject', 'resolved: ' + stringify(val), assertOpts.note);
    }).catch(function(err) {
      return record(group, name, 'PASS', 'should reject', 'rejected: ' + (err && err.message || err), assertOpts.note);
    });
  }

  function skip(group, name, reason) {
    return record(group, name, 'SKIP', null, null, reason || 'skipped');
  }

  function info(group, name, message) {
    return record(group, name, 'INFO', null, null, message);
  }

  // -------- Timing --------
  function start() {
    startTime = Date.now();
    results = [];
    counts = { PASS: 0, FAIL: 0, DRIFT: 0, SKIP: 0, INFO: 0 };
    groups = {};
    if (container) container.innerHTML = '';
    emit('start', {});
  }

  function finish() {
    var elapsed = startTime ? Date.now() - startTime : 0;
    var summary = {
      counts: Object.assign({}, counts),
      total: results.length,
      elapsed: elapsed,
      passed: counts.FAIL === 0 && counts.DRIFT === 0,
      groups: Object.keys(groups)
    };
    emit('finish', summary);
    return summary;
  }

  // -------- Progress --------
  function progress(current, total, label) {
    emit('progress', { current: current, total: total, label: label || '' });
  }

  // -------- Export --------
  function exportResults() {
    return {
      results: results.slice(),
      counts: Object.assign({}, counts),
      groups: Object.keys(groups),
      elapsed: startTime ? Date.now() - startTime : 0,
      exportedAt: Date.now()
    };
  }

  function exportText() {
    var lines = [];
    lines.push('Test results: ' + new Date().toISOString());
    lines.push('');
    for (var g in groups) {
      lines.push('== ' + g + ' ==');
      for (var i = 0; i < groups[g].length; i++) {
        var r = groups[g][i];
        var line = '  ' + r.status + '  ' + r.name;
        if (r.note) line += '  -- ' + r.note;
        if (r.status === 'FAIL' || r.status === 'DRIFT') {
          line += '  (expected: ' + stringify(r.expected) + ', got: ' + stringify(r.actual) + ')';
        }
        lines.push(line);
      }
      lines.push('');
    }
    lines.push('PASS: ' + counts.PASS + '  FAIL: ' + counts.FAIL +
      '  DRIFT: ' + counts.DRIFT + '  SKIP: ' + counts.SKIP);
    return lines.join('\n');
  }

  // -------- Rendering --------
  function renderResult(entry) {
    if (!container) return;
    // Find or create group section
    var groupEl = container.querySelector('[data-group="' + entry.group + '"]');
    if (!groupEl) {
      groupEl = document.createElement('div');
      groupEl.dataset.group = entry.group;
      groupEl.innerHTML =
        '<div style="font:600 11px var(--weld-mono,monospace);color:var(--weld-muted,#7d8590);' +
        'text-transform:uppercase;letter-spacing:1.5px;padding:10px 0 6px;' +
        'border-bottom:1px solid var(--weld-border,rgba(255,255,255,0.10));margin-top:12px;">' +
        escapeHtml(entry.group) + '</div>';
      container.appendChild(groupEl);
    }
    var row = document.createElement('div');
    row.style.cssText = 'display:grid;grid-template-columns:60px 240px 1fr;gap:8px;' +
      'padding:4px 0;border-bottom:1px dotted var(--weld-border,rgba(255,255,255,0.05));' +
      'align-items:baseline;font:12px var(--weld-mono,monospace);';
    var statusColor = { PASS: '#3fb950', FAIL: '#f85149', DRIFT: '#d29922', SKIP: '#7d8590', INFO: '#4493f8' };
    var detail = '';
    if (entry.status === 'PASS') detail = '<span style="color:#3fb950;">' + escapeHtml(stringify(entry.actual)) + '</span>';
    else if (entry.status === 'SKIP' || entry.status === 'INFO') detail = '<span style="color:#7d8590;">' + escapeHtml(entry.note || '') + '</span>';
    else detail = '<span style="color:#7d8590;">expected: ' + escapeHtml(stringify(entry.expected)) + '</span> <span style="color:#f85149;">got: ' + escapeHtml(stringify(entry.actual)) + '</span>';
    if (entry.note && entry.status !== 'SKIP' && entry.status !== 'INFO') {
      detail += ' <span style="color:#7d8590;">' + escapeHtml(entry.note) + '</span>';
    }
    row.innerHTML =
      '<span style="font-weight:700;color:' + (statusColor[entry.status] || '#7d8590') + ';">' + entry.status + '</span>' +
      '<span style="color:var(--weld-ink,#e6edf3);">' + escapeHtml(entry.name) + '</span>' +
      '<span>' + detail + '</span>';
    groupEl.appendChild(row);
  }

  function escapeHtml(s) {
    return String(s).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
  }

  // -------- Helpers --------
  function sleep(ms) { return new Promise(function(r) { setTimeout(r, ms); }); }

  function timeout(promise, ms, label) {
    return new Promise(function(resolve, reject) {
      var done = false;
      var t = setTimeout(function() { if (!done) { done = true; reject(new Error('TIMEOUT' + (label ? ' ' + label : ''))); } }, ms);
      Promise.resolve(promise).then(
        function(v) { if (!done) { done = true; clearTimeout(t); resolve(v); } },
        function(e) { if (!done) { done = true; clearTimeout(t); reject(e); } }
      );
    });
  }

  function on(fn) {
    listeners.push(fn);
    return function() {
      var idx = listeners.indexOf(fn);
      if (idx >= 0) listeners.splice(idx, 1);
    };
  }

  return {
    // Assertions
    assertEqual: assertEqual,
    assertTrue: assertTrue,
    assertFalse: assertFalse,
    assertThrows: assertThrows,
    assertResolves: assertResolves,
    assertRejects: assertRejects,
    skip: skip,
    info: info,
    // Lifecycle
    start: start,
    finish: finish,
    progress: progress,
    // Export
    exportResults: exportResults,
    exportText: exportText,
    // State
    counts: function() { return Object.assign({}, counts); },
    results: function() { return results.slice(); },
    groupNames: function() { return Object.keys(groups); },
    on: on,
    // Helpers
    sleep: sleep,
    timeout: timeout
  };
}

window.weld.testrig = {
  create: createTestRig,
  version: '1.0.0'
};

})();
