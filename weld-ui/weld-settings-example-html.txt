// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

// COMPONENTS
function button(opts) {
  opts = opts || {};
  var classes = ['w-btn'];
  if (opts.variant === 'primary') classes.push('w-btn-primary');
  else if (opts.variant === 'danger') classes.push('w-btn-danger');
  else if (opts.variant === 'ghost') classes.push('w-btn-ghost');
  if (opts.size === 'sm') classes.push('w-btn-sm');
  else if (opts.size === 'lg') classes.push('w-btn-lg');
  var children = [];
  if (opts.icon) children.push(icon(opts.icon, { size: opts.size === 'sm' ? 14 : 16 }));
  if (opts.label != null) children.push(opts.label);
  return el('button', {
    class: classes.join(' '),
    type: opts.type || 'button',
    disabled: opts.disabled ? 'disabled' : null,
    onclick: opts.onClick,
  }, children);
}

function input(opts) {
  opts = opts || {};
  return el('input', {
    class: 'w-input',
    type: opts.type || 'text',
    placeholder: opts.placeholder,
    value: opts.value != null ? opts.value : '',
    oninput: opts.onInput ? function(e){ opts.onInput(e.target.value, e); } : null,
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
    readonly: opts.readonly ? 'readonly' : null,
  });
}

function select(opts) {
  opts = opts || {};
  var options = (opts.options || []).map(function(o){
    if (typeof o === 'string') o = { value: o, label: o };
    return el('option', {
      value: o.value,
      selected: o.value === opts.value ? 'selected' : null,
    }, o.label);
  });
  return el('select', {
    class: 'w-select',
    onchange: opts.onChange ? function(e){ opts.onChange(e.target.value, e); } : null,
    disabled: opts.disabled ? 'disabled' : null,
  }, options);
}

    open: opts.open ? 'open' : null,
  }, [summary, body]);
}

function tabs(opts) {
  opts = opts || {};
  var items = opts.items || [];
  var active = opts.active || (items[0] && items[0].id);
  var tabBar = el('div', { class: 'w-tabs' });
  var panel = el('div', { class: 'w-tab-panel' });

  function render(activeId) {
    tabBar.innerHTML = '';
    items.forEach(function(it){
      var btn = el('button', {
        class: 'w-tab' + (it.id === activeId ? ' w-tab-active' : ''),
        onclick: function(){ render(it.id); if (opts.onChange) opts.onChange(it.id); },
      }, it.label);
      tabBar.appendChild(btn);
    });
    panel.innerHTML = '';
    var found = items.find(function(it){ return it.id === activeId; });
    if (found && found.content) {
      panel.appendChild(typeof found.content === 'function' ? found.content() : found.content);
    }
  }
  render(active);

  var wrap = el('div', { class: 'w-tabs-wrap' }, [tabBar, panel]);
  wrap.setActive = function(id){ render(id); };
  return wrap;
}

// Returns { el, close } and is auto-attached to body
function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = opts.title || opts.dismissable !== false
    ? el('div', { class: 'w-modal-hdr' }, [
        opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div'),
        opts.dismissable !== false ? closeBtn : null,
      ])
    : null;

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {
    if (e.key === 'Escape' && opts.dismissable !== false) close();
  }
  document.addEventListener('keydown', escHandler);

  document.body.appendChild(overlay);
  return { el: overlay, close: close, body: body };
}

// -----------------------------------------------------------------
// Validation and coercion per type
// -----------------------------------------------------------------
function coerce(s, value) {
  if (s.type === 'number') {
    var n = Number(value);
    if (isNaN(n)) throw new Error('not a number');
    if (s.min != null && n < s.min) n = s.min;
    if (s.max != null && n > s.max) n = s.max;
    return n;
  }
  if (s.type === 'boolean') return !!value;
  if (s.type === 'select') {
    var opts = s.options || [];
    var vals = opts.map(function(o) { return typeof o === 'object' ? o.value : o; });
    if (vals.indexOf(value) === -1) {
      throw new Error('value not in options');
    }
    return value;
  }
  // string, color, textarea
  return value == null ? '' : String(value);
}

// -----------------------------------------------------------------
// Schema instance factory
// -----------------------------------------------------------------
function createSettings(opts) {
  opts = opts || {};
  var schema = opts.schema || {};
  var storageKey = opts.storageKey || 'weld-settings';
  var storage = opts.storage;
  if (storage === undefined) {
    storage = (typeof localStorage !== 'undefined') ? localStorage : null;
  }

  var values = {};
  var listeners = {};
  listeners['*'] = [];

  // Seed defaults
  for (var key in schema) {
    var def = schema[key]['default'];
    values[key] = def !== undefined ? def : null;
  }

  // Load persisted
  if (storage) {
    try {
      var raw = storage.getItem(storageKey);
      if (raw) {
        var loaded = JSON.parse(raw);
        for (var k in loaded) {
          if (schema[k]) values[k] = loaded[k];
        }
      }
    } catch (e) { /* swallow */ }
  }

  function save() {
    if (!storage) return;
    try { storage.setItem(storageKey, JSON.stringify(values)); }
    catch (e) { /* swallow */ }
  }

  function get(key) {
    return values[key];
  }

  function getAll() {
    var out = {};
    for (var k in values) out[k] = values[k];
    return out;
  }

  function set(key, value) {
    if (!schema[key]) throw new Error('unknown setting: ' + key);
    value = coerce(schema[key], value);
    var old = values[key];
    if (old === value) return false;
    values[key] = value;
    save();
    fireListeners(key, value, old);
    return true;
  }

  function fireListeners(key, value, old) {
    var L = listeners[key];
    if (L) {
      for (var i = 0; i < L.length; i++) {
        try { L[i](value, old); } catch (e) { /* swallow */ }
      }
    }
    var Lall = listeners['*'];
    for (var j = 0; j < Lall.length; j++) {
      try { Lall[j](key, value, old); } catch (e) { /* swallow */ }
    }
  }

  function reset(key) {
    if (key) {
      if (schema[key]) set(key, schema[key]['default']);
    } else {
      for (var k in schema) {
        var d = schema[k]['default'];
        if (d !== undefined) set(k, d);
      }
    }
  }

  function subscribe(key, fn) {
    if (typeof fn !== 'function') throw new Error('subscribe requires a function');
    if (!listeners[key]) listeners[key] = [];
    listeners[key].push(fn);
    return function() {
      var idx = listeners[key].indexOf(fn);
      if (idx !== -1) listeners[key].splice(idx, 1);
    };
  }

  function buildField(key) {
    var s = schema[key];
    if (!s) return null;

    var row = document.createElement('div');
    row.className = 'weld-settings-row';

    var label = document.createElement('label');
    label.textContent = s.label || key;
    row.appendChild(label);

    var control;
    if (s.type === 'boolean') {
      control = document.createElement('input');
      control.type = 'checkbox';
      control.checked = !!values[key];
      control.addEventListener('change', function() { set(key, control.checked); });
    } else if (s.type === 'select') {
      control = document.createElement('select');
      var opts = s.options || [];
      for (var i = 0; i < opts.length; i++) {
        var opt = opts[i];
        var val = typeof opt === 'object' ? opt.value : opt;
        var lbl = typeof opt === 'object' ? opt.label : opt;
        var o = document.createElement('option');
        o.value = val;
        o.textContent = lbl;
        if (values[key] === val) o.selected = true;
        control.appendChild(o);
      }
      control.addEventListener('change', function() { set(key, control.value); });
    } else if (s.type === 'number') {
      control = document.createElement('input');
      control.type = 'number';
      if (s.min != null) control.min = s.min;
      if (s.max != null) control.max = s.max;
      if (s.step != null) control.step = s.step;
      control.value = values[key];
      control.addEventListener('change', function() {
        var v = parseFloat(control.value);
        if (!isNaN(v)) set(key, v);
      });
    } else if (s.type === 'color') {
      control = document.createElement('input');
      control.type = 'color';
      control.value = values[key] || '#000000';
      control.addEventListener('change', function() { set(key, control.value); });
    } else if (s.type === 'textarea') {
      control = document.createElement('textarea');
      control.rows = s.rows || 3;
      control.value = values[key] || '';
      control.addEventListener('change', function() { set(key, control.value); });
    } else {
      control = document.createElement('input');
      control.type = s.secret ? 'password' : 'text';
      control.value = values[key] || '';
      control.addEventListener('change', function() { set(key, control.value); });
    }

    row.appendChild(control);

    if (s.help) {
      var helpEl = document.createElement('div');
      helpEl.className = 'weld-settings-help';
      helpEl.textContent = s.help;
      row.appendChild(helpEl);
    }

    return row;
  }

  function buildPanel(panelOpts) {
    injectCss();
    panelOpts = panelOpts || {};
    var panel = document.createElement('div');
    panel.className = 'weld-settings-panel';

    if (panelOpts.title) {
      var h = document.createElement('h3');
      h.textContent = panelOpts.title;
      panel.appendChild(h);
    }

    var sections = panelOpts.sections;
    var rendered = {};

    if (sections && sections.length) {
      for (var i = 0; i < sections.length; i++) {
        var section = sections[i];
        var sec = document.createElement('div');
        sec.className = 'weld-settings-section';
        if (section.title) {
          var t = document.createElement('h4');
          t.textContent = section.title;
          sec.appendChild(t);
        }
        var keys = section.keys || [];
        for (var j = 0; j < keys.length; j++) {
          var field = buildField(keys[j]);
          if (field) {
            sec.appendChild(field);
            rendered[keys[j]] = true;
          }
        }
        panel.appendChild(sec);
      }
    }

    // Render any keys not yet in a section
    for (var key in schema) {
      if (rendered[key]) continue;
      var f = buildField(key);
      if (f) panel.appendChild(f);
    }

    return panel;
  }

  return {
    get: get,
    set: set,
    getAll: getAll,
    reset: reset,
    subscribe: subscribe,
    buildPanel: buildPanel,
    buildField: buildField,
    schema: schema
  };
}

window.weld.settings = {
  create: createSettings,
  version: '1.0.0'
};
