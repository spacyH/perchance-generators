<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.embed) return window.weld.embed;

// -----------------------------------------------------------------
// weld.embed: turn text into vectors, in the browser, with no server.
//
// Two serious community chat apps independently reached for the same
// thing -- transformers.js feature-extraction in a Web Worker, cached
// in IndexedDB -- to get semantic memory and lore retrieval. This packs
// that pattern into one plugin so weld.memory and weld.lorebook can
// rank by meaning instead of keyword overlap.
//
//   init(opts)          boot the worker + model (lazy; rank/embed call it)
//   embed(text)         -> Promise<vector | null>
//   embedBatch(texts)   -> Promise<(vector|null)[]>  (cached + batched)
//   similarity(a, b)    cosine (vectors are L2-normalized, so a dot)
//   rank(query, items, opts) -> Promise<[{ item, index, score }]> top-k
//   useEmbedder(fn)     swap in your own async embedder (or for testing)
//   clearCache/cacheSize/info/isReady
//
// Robustness: the real model loads from a CDN inside the Perchance
// frame. If that fails (offline, blocked, no WASM), weld.embed falls
// back to a deterministic lexical hash embedding -- lower quality, but
// rank() still returns sensible keyword-overlap results instead of
// throwing. info().backend tells you which engine is live.
// -----------------------------------------------------------------

var DEFAULTS = {
  model: 'Xenova/all-MiniLM-L6-v2', // small, 384-dim, widely cached
  fallbackDim: 256,
  cdn: 'https://cdn.jsdelivr.net/npm/@xenova/transformers@2.17.2',
  timeoutMs: 12000,
  dbName: 'weld-embed',
  store: 'vectors'
};

var cfg = {};
for (var k in DEFAULTS) cfg[k] = DEFAULTS[k];

var worker = null;
var reqId = 0;
var pending = {};        // id -> { resolve, reject, timer }
var ready = false;
var booting = null;      // Promise while init is in flight
var backend = 'none';    // 'worker' | 'fallback' | 'custom'
var customEmbedder = null;
var modelDim = 0;
var db = null;
var mem = {};            // in-memory vector cache (key -> vector)

function cacheKey(text) { return (backend === 'fallback' ? 'lex' : cfg.model) + '|' + text; }

// ---- IndexedDB cache (best-effort; never fatal) ----
function openDb() {
  return new Promise(function (resolve) {
    if (db) { resolve(db); return; }
    if (typeof indexedDB === 'undefined') { resolve(null); return; }
    var req = indexedDB.open(cfg.dbName, 1);
    req.onupgradeneeded = function (e) {
      var d = e.target.result;
      if (!d.objectStoreNames.contains(cfg.store)) d.createObjectStore(cfg.store, { keyPath: 'key' });
    };
    req.onsuccess = function (e) { db = e.target.result; resolve(db); };
    req.onerror = function () { resolve(null); };
  });
}

function cacheGetMany(keys) {
  return openDb().then(function (d) {
    if (!d) return {};
    return new Promise(function (resolve) {
      var out = {}, left = keys.length;
      if (!left) { resolve(out); return; }
      var tx = d.transaction([cfg.store], 'readonly');
      var store = tx.objectStore(cfg.store);
      keys.forEach(function (key) {
        var r = store.get(key);
        r.onsuccess = function () { if (r.result) out[key] = r.result.vector; if (--left === 0) resolve(out); };
        r.onerror = function () { if (--left === 0) resolve(out); };
      });
    });
  });
}

function cachePut(key, vector) {
  mem[key] = vector;
  openDb().then(function (d) {
    if (!d) return;
    try {
      var tx = d.transaction([cfg.store], 'readwrite');
      tx.objectStore(cfg.store).put({ key: key, vector: vector });
    } catch (e) { /* ignore */ }
  });
}

// ---- the worker source (built so the Perchance scanner sees no traps) ----
// No doubled square or curly brackets, no import-directive token; object literals keep spaces.
function workerSource() {
  var L = [];
  L.push("var pipe = null;");
  L.push("var CDN = self.__cdn;");
  L.push("function reply(id, ok, payload){ self.postMessage(Object.assign({ id: id, ok: ok }, payload)); }");
  L.push("self.onmessage = async function(e){");
  L.push("  var msg = e.data || {};");
  L.push("  try {");
  L.push("    if (msg.type === 'init') {");
  L.push("      if (!pipe) {");
  L.push("        importScripts(msg.cdn || CDN);");
  L.push("        var T = self.transformers || self.Transformers;");
  L.push("        if (T && T.env) { T.env.allowLocalModels = false; }");
  L.push("        pipe = await T.pipeline('feature-extraction', msg.model);");
  L.push("      }");
  L.push("      reply(msg.id, true, { dim: 0 });");
  L.push("      return;");
  L.push("    }");
  L.push("    if (msg.type === 'embed') {");
  L.push("      if (!pipe) { reply(msg.id, false, { error: 'model not initialized' }); return; }");
  L.push("      var out = await pipe(msg.texts, { pooling: 'mean', normalize: true });");
  L.push("      var data = out.data;");
  L.push("      var dims = out.dims || [msg.texts.length, data.length / msg.texts.length];");
  L.push("      var dim = dims[dims.length - 1];");
  L.push("      var vecs = [];");
  L.push("      for (var i = 0; i < msg.texts.length; i++) {");
  L.push("        var start = i * dim, v = new Array(dim);");
  L.push("        for (var j = 0; j < dim; j++) { v[j] = data[start + j]; }");
  L.push("        vecs.push(v);");
  L.push("      }");
  L.push("      reply(msg.id, true, { vectors: vecs, dim: dim });");
  L.push("      return;");
  L.push("    }");
  L.push("  } catch (err) { reply(msg.id, false, { error: String(err && err.message || err) }); }");
  L.push("};");
  return L.join('\n');
}

function startWorker() {
  if (typeof Worker === 'undefined' || typeof Blob === 'undefined') return null;
  // Prepend the CDN as a worker global so importScripts can default to it.
  var src = "self.__cdn=" + JSON.stringify(cfg.cdn) + ";\n" + workerSource();
  var url = URL.createObjectURL(new Blob([src], { type: 'text/javascript' }));
  var w = new Worker(url);
  w.onmessage = function (e) {
    var m = e.data || {};
    var p = pending[m.id];
    if (!p) return;
    delete pending[m.id];
    if (p.timer) clearTimeout(p.timer);
    if (m.ok) p.resolve(m);
    else p.reject(new Error(m.error || 'worker error'));
  };
  w.onerror = function () { /* surfaced per-request via timeout */ };
  return w;
}

function call(type, payload) {
  return new Promise(function (resolve, reject) {
    var id = ++reqId;
    pending[id] = {
      resolve: resolve, reject: reject,
      timer: setTimeout(function () {
        if (pending[id]) { delete pending[id]; reject(new Error('weld.embed: worker timeout')); }
      }, cfg.timeoutMs)
    };
    var msg = { id: id, type: type };
    for (var key in payload) msg[key] = payload[key];
    worker.postMessage(msg);
  });
}

// ---- deterministic lexical fallback (no model, no network) ----
var STOP = { the:1,a:1,an:1,and:1,or:1,but:1,is:1,are:1,was:1,were:1,to:1,of:1,
  in:1,on:1,at:1,for:1,with:1,it:1,this:1,that:1,as:1,by:1 };
function tokens(text) {
  return String(text || '').toLowerCase().replace(/[^a-z0-9\u00c0-\u024f' ]+/g, ' ')
    .split(/\s+/).filter(function (w) { return w.length > 1 && !STOP[w]; });
}
function hashStr(s) {
  var h = 2166136261 >>> 0;
  for (var i = 0; i < s.length; i++) { h ^= s.charCodeAt(i); h = Math.imul(h, 16777619); }
  return h >>> 0;
}
function lexEmbed(text) {
  var dim = cfg.fallbackDim, v = new Array(dim);
  for (var i = 0; i < dim; i++) v[i] = 0;
  var toks = tokens(text);
  for (var t = 0; t < toks.length; t++) {
    var tok = toks[t];
    // two hashed projections per token soften collisions
    v[hashStr(tok) % dim] += 1;
    v[hashStr('#' + tok) % dim] += 0.5;
  }
  // L2 normalize so cosine == dot
  var norm = 0; for (var n = 0; n < dim; n++) norm += v[n] * v[n];
  norm = Math.sqrt(norm) || 1;
  for (var m2 = 0; m2 < dim; m2++) v[m2] = v[m2] / norm;
  return v;
}

// ---- init / engine selection ----
function init(opts) {
  if (opts) for (var key in opts) if (opts[key] != null) cfg[key] = opts[key];
  if (ready) return Promise.resolve(info());
  if (booting) return booting;

  if (customEmbedder) { backend = 'custom'; ready = true; return Promise.resolve(info()); }

  booting = new Promise(function (resolve) {
    worker = startWorker();
    if (!worker) { backend = 'fallback'; ready = true; resolve(info()); return; }
    call('init', { model: cfg.model, cdn: cfg.cdn }).then(function () {
      backend = 'worker'; ready = true; resolve(info());
    }, function () {
      // Worker present but model failed to load -> lexical fallback.
      try { worker.terminate(); } catch (e) {}
      worker = null; backend = 'fallback'; ready = true; resolve(info());
    });
  });
  return booting;
}

// ---- embedding (cache-aware, batched) ----
function embedBatch(texts) {
  if (!Array.isArray(texts) || !texts.length) return Promise.resolve([]);
  return init().then(function () {
    var keys = texts.map(cacheKey);
    // in-memory first
    var result = new Array(texts.length);
    var missIdx = [], missTexts = [], missKeys = [];
    for (var i = 0; i < texts.length; i++) {
      if (mem[keys[i]]) result[i] = mem[keys[i]];
      else { missIdx.push(i); missTexts.push(texts[i]); missKeys.push(keys[i]); }
    }
    if (!missIdx.length) return result;

    return cacheGetMany(missKeys).then(function (fromDb) {
      var stillIdx = [], stillTexts = [], stillKeys = [];
      for (var j = 0; j < missIdx.length; j++) {
        var key = missKeys[j];
        if (fromDb[key]) { result[missIdx[j]] = fromDb[key]; mem[key] = fromDb[key]; }
        else { stillIdx.push(missIdx[j]); stillTexts.push(missTexts[j]); stillKeys.push(key); }
      }
      if (!stillIdx.length) return result;

      // compute the remainder
      var compute;
      if (backend === 'custom') {
        compute = Promise.resolve(customEmbedder(stillTexts)).then(function (vs) { return { vectors: vs }; });
      } else if (backend === 'worker') {
        compute = call('embed', { texts: stillTexts }).then(function (m) {
          if (m.dim) modelDim = m.dim;
          return { vectors: m.vectors };
        }, function () {
          // mid-session worker failure: fall back lexically for this batch
          backend = 'fallback';
          return { vectors: stillTexts.map(lexEmbed) };
        });
      } else {
        compute = Promise.resolve({ vectors: stillTexts.map(lexEmbed) });
      }

      return compute.then(function (r) {
        var vs = r.vectors || [];
        for (var z = 0; z < stillIdx.length; z++) {
          var vec = vs[z] || null;
          result[stillIdx[z]] = vec;
          if (vec) cachePut(stillKeys[z], vec);
        }
        return result;
      });
    });
  });
}

function embed(text) { return embedBatch([text]).then(function (a) { return a[0]; }); }

// ---- math ----
function similarity(a, b) {
  if (!a || !b || a.length !== b.length) return 0;
  var s = 0; for (var i = 0; i < a.length; i++) s += a[i] * b[i];
  return s; // vectors are normalized at the source -> cosine
}

// rank(query, items, opts) -> top-k by similarity.
// items: string[] | object[]; opts: { topK, minScore, key }
function rank(query, items, opts) {
  opts = opts || {};
  var key = opts.key || 'text';
  var topK = opts.topK || 5;
  var minScore = opts.minScore != null ? opts.minScore : 0;
  if (!items || !items.length) return Promise.resolve([]);
  var texts = items.map(function (it) {
    return (typeof it === 'string') ? it : String((it && it[key]) != null ? it[key] : '');
  });
  return embedBatch([query].concat(texts)).then(function (vecs) {
    var qv = vecs[0];
    if (!qv) return [];
    var scored = [];
    for (var i = 0; i < items.length; i++) {
      var v = vecs[i + 1];
      if (!v) continue;
      var score = similarity(qv, v);
      if (score >= minScore) scored.push({ item: items[i], index: i, score: score });
    }
    scored.sort(function (a, b) { return b.score - a.score; });
    return scored.slice(0, topK);
  });
}

function useEmbedder(fn) {
  customEmbedder = (typeof fn === 'function') ? fn : null;
  if (customEmbedder) { backend = 'custom'; ready = true; }
}

function clearCache() {
  mem = {};
  return openDb().then(function (d) {
    if (!d) return false;
    return new Promise(function (resolve) {
      try {
        var tx = d.transaction([cfg.store], 'readwrite');
        var r = tx.objectStore(cfg.store).clear();
        r.onsuccess = function () { resolve(true); };
        r.onerror = function () { resolve(false); };
      } catch (e) { resolve(false); }
    });
  });
}

function cacheSize() { return Object.keys(mem).length; }
function isReady() { return ready; }
function info() { return { model: cfg.model, backend: backend, ready: ready, dim: modelDim || cfg.fallbackDim }; }

window.weld.embed = {
  init: init,
  embed: embed,
  embedBatch: embedBatch,
  similarity: similarity,
  rank: rank,
  search: rank,
  useEmbedder: useEmbedder,
  clearCache: clearCache,
  cacheSize: cacheSize,
  isReady: isReady,
  info: info,
  version: '1.0.0'
};

return window.weld.embed;
})();

<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.cache) return window.weld.cache;

// -----------------------------------------------------------------
// weld.cache: don't pay for the same answer twice. Exact caching is the
// floor (LRU-bounded, TTL-expiring). The interesting part is SEMANTIC
// reuse: with weld.embed attached, a new prompt that means the same as a
// recent one ("what's the capital of France?" vs "France's capital?")
// can return the cached answer instead of calling the model again. The
// serial broker in weld.stream makes every saved call meaningfully
// faster, and quota goes further.
//
//   var c = weld.cache.create({ max: 200, ttlMs: 3600000 });
//   var ask = c.wrap(myGenerate);          // memoized by arguments
//   c.attach({ embed: weld.embed });
//   await c.memoize(prompt, () => generate(prompt), { threshold: 0.93 });
// -----------------------------------------------------------------

function getEmbed(deps) { return (deps && deps.embed) || (window.weld && window.weld.embed) || null; }

function stableKey(x) {
  if (typeof x === 'string') return x;
  try { return JSON.stringify(x, Object.keys(x || {}).sort()); }
  catch (e) { return String(x); }
}

function create(createOpts) {
  createOpts = createOpts || {};
  var max = createOpts.max || 200;
  var ttlMs = createOpts.ttlMs || 0;          // 0 = no expiry
  var deps = { embed: createOpts.embed || null };
  var defaultThreshold = createOpts.threshold != null ? createOpts.threshold : 0.93;

  var store = {};        // key -> { value, at, vec? }
  var order = [];        // LRU: most-recent at the end
  var hits = 0, misses = 0, semanticHits = 0;

  function touch(key) {
    var i = order.indexOf(key); if (i !== -1) order.splice(i, 1);
    order.push(key);
  }
  function evictIfNeeded() {
    while (order.length > max) { var k = order.shift(); delete store[k]; }
  }
  function fresh(entry) { return !ttlMs || (Date.now() - entry.at) < ttlMs; }

  function get(key) {
    key = stableKey(key);
    var e = store[key];
    if (e && fresh(e)) { hits++; touch(key); return e.value; }
    if (e) { delete store[key]; var i = order.indexOf(key); if (i !== -1) order.splice(i, 1); }
    misses++;
    return undefined;
  }
  function has(key) { var e = store[stableKey(key)]; return !!(e && fresh(e)); }

  function set(key, value, opts) {
    key = stableKey(key);
    var entry = { value: value, at: Date.now() };
    if (opts && opts.vec) entry.vec = opts.vec;
    store[key] = entry; touch(key); evictIfNeeded();
    return value;
  }
  function del(key) { key = stableKey(key); delete store[key]; var i = order.indexOf(key); if (i !== -1) order.splice(i, 1); }
  function clear() { store = {}; order = []; }

  // wrap(fn): memoize an async (or sync) function by its arguments.
  function wrap(fn, wrapOpts) {
    wrapOpts = wrapOpts || {};
    var keyFn = wrapOpts.key || function () { return stableKey([].slice.call(arguments)); };
    return function () {
      var args = [].slice.call(arguments);
      var key = keyFn.apply(null, args);
      var cached = get(key);
      if (cached !== undefined) return Promise.resolve(cached);
      return Promise.resolve(fn.apply(this, args)).then(function (result) {
        set(key, result);
        return result;
      });
    };
  }

  // memoize(text, producer, opts): exact-first, then optional semantic.
  // If an entry's text is within opts.threshold cosine of `text`, reuse
  // it; otherwise run producer(), store with its vector, and return.
  // opts: { threshold, semantic (default true if embed present) }
  function memoize(text, producer, opts) {
    opts = opts || {};
    var key = String(text);
    var exact = get(key);
    if (exact !== undefined) return Promise.resolve(exact);

    var embed = getEmbed(deps);
    var wantSemantic = (opts.semantic !== false) && embed;
    var threshold = opts.threshold != null ? opts.threshold : defaultThreshold;

    if (!wantSemantic) {
      return Promise.resolve(producer()).then(function (v) { set(key, v); return v; });
    }
    // gather entries that carry a vector
    var keys = order.filter(function (k) { return store[k] && store[k].vec && fresh(store[k]); });
    var lookup = wantSemanticMatch(embed, text, keys, threshold);
    return lookup.then(function (matchKey) {
      if (matchKey != null) { semanticHits++; touch(matchKey); return store[matchKey].value; }
      return embed.embed(text).then(function (vec) {
        return Promise.resolve(producer()).then(function (v) {
          set(key, v, { vec: vec });
          return v;
        });
      });
    });
  }

  function wantSemanticMatch(embed, text, keys, threshold) {
    if (!keys.length) return Promise.resolve(null);
    var items = keys.map(function (k) { return { k: k, vec: store[k].vec }; });
    // embed the query once, compare to stored vectors directly
    return embed.embed(text).then(function (qv) {
      if (!qv) return null;
      var best = null, bestScore = -1;
      for (var i = 0; i < items.length; i++) {
        var s = embed.similarity(qv, items[i].vec);
        if (s > bestScore) { bestScore = s; best = items[i].k; }
      }
      return bestScore >= threshold ? best : null;
    }, function () { return null; });
  }

  function attach(d) { if (d && d.embed) deps.embed = d.embed; }
  function stats() { return { size: order.length, hits: hits, misses: misses, semanticHits: semanticHits, max: max }; }

  return {
    get: get, set: set, has: has, delete: del, clear: clear,
    wrap: wrap, memoize: memoize, attach: attach, stats: stats,
    size: function () { return order.length; }
  };
}

window.weld.cache = {
  create: create,
  version: '1.0.0'
};

return window.weld.cache;
})();
