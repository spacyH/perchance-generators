      <h1>weld.director</h1>
      <span class="ver">v1.0.0</span>
  </div>
  <p class="lead">Multi-pass orchestration: run the reply in the foreground and resolve immediately, then run state reconcile, fact extraction, and illustration in the background. Dependency-ordered and failure-isolated.</p>
</div>
<h2>Install</h2>
<p>Import once; stashes on <code>window.weld.director</code>. No dependencies &mdash; it is pure scheduling. The passes you define do the work and may use any other plugin.</p>
<pre data-codeid="install"></pre>

<h2>The shape of a turn</h2>
<p>Define named passes, mark the slow non-essential ones <code>background</code>, and declare ordering with <code>after</code>. <code>run()</code> resolves <code>foreground</code> as soon as the essential passes finish, then runs the background passes.</p>
<pre data-codeid="shape"></pre>

<h2>define(name, fn, opts)</h2>
<div class="sig" data-codeid="sig_define"></div>
<p>The pass function receives <code>(ctx, input)</code>. <code>ctx</code> accumulates each pass's return value by name (<code>ctx.reply</code>, <code>ctx.audit</code>...) and carries <code>ctx.errors</code>. Options: <code>background</code> (run after the foreground resolves) and <code>after</code> (array of pass names this one depends on).</p>

<h2>run(input)</h2>
<div class="sig" data-codeid="sig_run"></div>
<p>Returns <code>\{ foreground, background, ctx \}</code>. Independent passes run concurrently; dependent passes wait for their <code>after</code> list. Await <code>foreground</code> to render the reply, and optionally <code>background</code> to know when the rest finished.</p>
<pre data-codeid="run_ex"></pre>

<h2>Failure isolation</h2>
<p>A pass that throws or rejects is caught and recorded in <code>ctx.errors[name]</code> (and reported via <code>onError</code>). Passes that depended on it are skipped with a clear reason; everything independent still runs. One flaky illustration call never sinks the reply.</p>
<pre data-codeid="fail_ex"></pre>
<div class="note">A dependency cycle can't hang the run &mdash; the scheduler detects the deadlock and settles the stuck passes as errors. Hooks: <code>onPass(name, result, ctx)</code> and <code>onError(name, error, ctx)</code>.</div>

<script>
(function(){
'use strict';

if (!window.weld) window.weld = {};
if (window.weld.director) return window.weld.director;

// -----------------------------------------------------------------
// weld.director: the conductor every serious chat app hand-rolls. A turn
// is rarely one call -- it is "write the reply (fast, streamed), THEN in
// the background reconcile state, extract memory facts, and illustrate
// the scene." Done naively the background work blocks the reply, or a
// failing illustrate pass takes down the whole turn.
//
// weld.director formalizes it:
//   var dir = weld.director.create();
//   dir.define('reply',      replyFn);                       // foreground
//   dir.define('audit',      auditFn,      { background:true, after:['reply'] });
//   dir.define('illustrate', illustrateFn, { background:true, after:['reply'] });
//   var run = dir.run(input);
//   var ctx = await run.foreground;   // resolves when reply is done
//   // ... render ctx.reply immediately ...
//   await run.background;             // optional: when audit+illustrate finish
//
// Each pass fn receives (ctx, input); ctx carries prior passes' results
// by name (ctx.reply, ctx.audit, ...) plus ctx.errors. Passes run in
// dependency order, in parallel where independent. A thrown/rejected
// pass is caught, recorded in ctx.errors[name], and reported via onError;
// dependents that needed it are skipped, everything else proceeds.
// -----------------------------------------------------------------

function create(createOpts) {
  createOpts = createOpts || {};
  var passes = {};   // name -> { fn, background, after }
  var order = [];    // insertion order for stable scheduling
  var hooks = { onPass: createOpts.onPass || null, onError: createOpts.onError || null };

  function define(name, fn, opts) {
    opts = opts || {};
    if (typeof fn !== 'function') throw new Error('weld.director: pass "' + name + '" needs a function');
    if (!passes[name]) order.push(name);
    passes[name] = { fn: fn, background: !!opts.background, after: opts.after || [] };
    return api;
  }

  function remove(name) {
    delete passes[name];
    var i = order.indexOf(name); if (i !== -1) order.splice(i, 1);
    return api;
  }

  // Topologically schedule a set of pass-names, then run respecting
  // dependencies and running independents concurrently. Mutates ctx.
  function runGroup(names, ctx, input) {
    var pending = {};
    names.forEach(function (n) { pending[n] = true; });
    var done = {};      // name -> true once settled (ok or skipped/failed)
    var running = {};

    function depsSatisfied(name) {
      var after = passes[name].after || [];
      for (var i = 0; i < after.length; i++) {
        var dep = after[i];
        // only wait on deps that are part of this run at all
        if ((pending[dep] || running[dep]) && !done[dep]) return false;
      }
      return true;
    }

    function depFailed(name) {
      var after = passes[name].after || [];
      for (var i = 0; i < after.length; i++) {
        if (ctx.errors[after[i]]) return after[i];
      }
      return null;
    }

    return new Promise(function (resolve) {
      function tick() {
        var launched = false;
        names.forEach(function (name) {
          if (done[name] || running[name]) return;
          if (!depsSatisfied(name)) return;

          // a required upstream pass failed -> skip this one
          var failedDep = depFailed(name);
          if (failedDep) {
            ctx.errors[name] = 'skipped: dependency "' + failedDep + '" failed';
            done[name] = true; delete pending[name];
            if (hooks.onError) try { hooks.onError(name, ctx.errors[name], ctx); } catch (e) {}
            launched = true; // state changed; re-evaluate
            return;
          }

          running[name] = true; delete pending[name]; launched = true;
          Promise.resolve().then(function () { return passes[name].fn(ctx, input); })
            .then(function (result) {
              ctx[name] = result;
              if (hooks.onPass) try { hooks.onPass(name, result, ctx); } catch (e) {}
            }, function (err) {
              ctx.errors[name] = String((err && err.message) || err);
              if (hooks.onError) try { hooks.onError(name, ctx.errors[name], ctx); } catch (e) {}
            })
            .then(function () {
              running[name] = false; done[name] = true;
              if (allDone()) resolve(ctx); else tick();
            });
        });
        if (allDone()) resolve(ctx);
        else if (!launched && !anyRunning()) {
          // deadlock guard (e.g. a dependency cycle): settle the rest as errors
          names.forEach(function (n) {
            if (!done[n]) { ctx.errors[n] = 'unresolved dependencies (cycle?)'; done[n] = true; }
          });
          resolve(ctx);
        }
      }
      function allDone() { return names.every(function (n) { return done[n]; }); }
      function anyRunning() { return names.some(function (n) { return running[n]; }); }
      if (!names.length) { resolve(ctx); return; }
      tick();
    });
  }

  // run(input, opts) -> { foreground, background, ctx }.
  // foreground resolves with ctx once all non-background passes finish;
  // background resolves once background passes finish (started after fg).
  function run(input, runOpts) {
    runOpts = runOpts || {};
    var ctx = { input: input, errors: {} };
    var fgNames = order.filter(function (n) { return !passes[n].background; });
    var bgNames = order.filter(function (n) { return passes[n].background; });

    var bgResolve;
    var background = new Promise(function (res) { bgResolve = res; });

    var foreground = runGroup(fgNames, ctx, input).then(function () {
      // kick background work without blocking the foreground resolution
      if (bgNames.length) {
        runGroup(bgNames, ctx, input).then(function () { bgResolve(ctx); });
      } else {
        bgResolve(ctx);
      }
      return ctx;
    });

    return { foreground: foreground, background: background, ctx: ctx };
  }

  function onPass(fn) { hooks.onPass = fn; return api; }
  function onError(fn) { hooks.onError = fn; return api; }
  function list() { return order.slice(); }

  var api = {
    define: define,
    remove: remove,
    run: run,
    onPass: onPass,
    onError: onError,
    list: list
  };
  return api;
}

window.weld.director = {
  create: create,
  version: '1.0.0'
};

return window.weld.director;
})();
</script>
