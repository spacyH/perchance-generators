// el(tag, attrs, children) - children is string or element or array
function el(tag, attrs, children) {
  var node = document.createElement(tag);
  if (attrs) {
    Object.keys(attrs).forEach(function(k){
      var v = attrs[k];
      if (v == null) return;
      if (k === 'class' || k === 'className') node.className = v;
      else if (k === 'style' && typeof v === 'object') Object.assign(node.style, v);
      else if (k === 'dataset' && typeof v === 'object') Object.assign(node.dataset, v);
      else if (k.indexOf('on') === 0 && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
      else if (k === 'html') node.innerHTML = v;
      else if (k === 'text') node.textContent = v;
      else node.setAttribute(k, v);
    });
  }
  if (children != null) {
    if (!Array.isArray(children)) children = [children];
    children.forEach(function(c){
      if (c == null) return;
      if (typeof c === 'string' || typeof c === 'number') node.appendChild(document.createTextNode(c));
      else if (c.nodeType) node.appendChild(c);
    });
  }
  return node;
}

function modal(opts) {
  opts = opts || {};
  var sizeClass = 'w-modal-' + (opts.size || 'md');
  var closeBtn = el('button', {
    class: 'w-modal-close',
    onclick: function(){ close(); },
    'aria-label': 'Close',
  }, [icon('x', { size: 18 })]);

  var header = opts.title || opts.dismissable !== false
    ? el('div', { class: 'w-modal-hdr' }, [
        opts.title ? el('h3', { class: 'w-modal-title', text: opts.title }) : el('div'),
        opts.dismissable !== false ? closeBtn : null,
      ])
    : null;

  var bodyContent = opts.content;
  if (typeof bodyContent === 'string') bodyContent = el('div', { html: bodyContent });
  var body = el('div', { class: 'w-modal-body' }, bodyContent);

  var actions = opts.actions ? el('div', { class: 'w-modal-actions' },
    opts.actions.map(function(a){
      return button({
        label: a.label,
        variant: a.variant || (a.primary ? 'primary' : null),
        onClick: function(){
          if (a.onClick) {
            var res = a.onClick();
            if (res !== false) close();
          } else { close(); }
        },
      });
    })
  ) : null;

  var modalEl = el('div', { class: 'w-modal ' + sizeClass }, [header, body, actions]);

  var overlay = el('div', {
    class: 'w-modal-overlay',
    onclick: function(e){ if (e.target === overlay && opts.dismissable !== false) close(); },
  }, [modalEl]);

  var closed = false;
  function close() {
    if (closed) return;
    closed = true;
    overlay.remove();
    document.removeEventListener('keydown', escHandler);
    if (opts.onClose) opts.onClose();
  }
  function escHandler(e) {
    if (e.key === 'Escape' && opts.dismissable !== false) close();
  }
  document.addEventListener('keydown', escHandler);

  document.body.appendChild(overlay);
  return { el: overlay, close: close, body: body };
}

// -----------------------------------------------------------------
// Accurate tokenizer (DeepSeek-R1-0528) via transformers.js.
// Same tokenizer Perchance uses for the AI broker, so counts are exact.
// -----------------------------------------------------------------
var tokenizerPromise = null;
var tokenizer = null;
var loadError = null;

function load() {
  if (tokenizerPromise) return tokenizerPromise;
  tokenizerPromise = (function() {
    return Promise.resolve().then(function() {
      return import('https://cdn.jsdelivr.net/npm/@huggingface/transformers@latest/+esm');
    }).then(function(mod) {
      return mod.AutoTokenizer.from_pretrained('deepseek-ai/DeepSeek-R1-0528');
    }).then(function(t) {
      tokenizer = t;
      return t;
    }).catch(function(e) {
      loadError = e;
      throw e;
    });
  })();
  return tokenizerPromise;
}

function preload() { return load(); }
function ready()   { return load(); }
function isLoaded(){ return tokenizer !== null; }
function getLoadError() { return loadError; }

// Cheap fallback (~4 chars per token)
function estimate(text) {
  return Math.ceil(String(text || '').length / 4);
}

// Exact count (async)
function count(text) {
  return load().then(function(t) {
    return t.encode(String(text)).length;
  });
}

// Sync count -- null if not loaded yet
function countSync(text) {
  if (!tokenizer) return null;
  return tokenizer.encode(String(text)).length;
}

function encode(text) {
  return load().then(function(t) { return t.encode(String(text)); });
}

function decode(ids) {
  return load().then(function(t) {
    return t.decode(ids, { skip_special_tokens: true });
  });
}

function budget(text, maxTokens) {
  return count(text).then(function(used) {
    return {
      used: used,
      max: maxTokens,
      remaining: maxTokens - used,
      fits: used <= maxTokens,
      percent: Math.round((used / maxTokens) * 100)
    };
  });
}

// Truncate preserving the START of the text
function truncate(text, maxTokens) {
  return load().then(function(t) {
    var ids = t.encode(String(text));
    if (ids.length <= maxTokens) return String(text);
    return t.decode(ids.slice(0, maxTokens), { skip_special_tokens: true });
  });
}

// Truncate preserving the END of the text (useful for chat histories)
function truncateStart(text, maxTokens) {
  return load().then(function(t) {
    var ids = t.encode(String(text));
    if (ids.length <= maxTokens) return String(text);
    return t.decode(ids.slice(-maxTokens), { skip_special_tokens: true });
  });
}

// Split text into chunks of ~N tokens each
function chunk(text, maxTokensPerChunk) {
  return load().then(function(t) {
    var ids = t.encode(String(text));
    var chunks = [];
    for (var i = 0; i < ids.length; i += maxTokensPerChunk) {
      chunks.push(t.decode(ids.slice(i, i + maxTokensPerChunk), { skip_special_tokens: true }));
    }
    return chunks;
  });
}

// Count tokens across an array of messages
function countMessages(messages) {
  var total = 0;
  function step(i) {
    if (i >= messages.length) return Promise.resolve(total);
    var msg = messages[i];
    var txt = typeof msg === 'string' ? msg : (msg.content || msg.text || '');
    return count(txt).then(function(n) {
      total += n + 4;
      return step(i + 1);
    });
  }
  return step(0);
}

function info() {
  return {
    model: 'deepseek-ai/DeepSeek-R1-0528',
    loaded: tokenizer !== null,
    error: loadError ? String(loadError.message || loadError) : null,
    vocabSize: tokenizer ? tokenizer.vocab_size : null
  };
}

window.weld.tokens = {
  preload: preload,
  ready: ready,
  isLoaded: isLoaded,
  loadError: getLoadError,
  estimate: estimate,
  count: count,
  countSync: countSync,
  encode: encode,
  decode: decode,
  budget: budget,
  truncate: truncate,
  truncateStart: truncateStart,
  chunk: chunk,
  countMessages: countMessages,
  info: info,
  version: '1.0.0'
};

})();
