// --- Shared counter (Abacus — abacus.jasoncameron.dev) -----------------
// On failure, the returned error string is the real network/parse error.
// Callers should surface it, not paper over it.
async readCounter(key) =>
  try {
    let url = `https://abacus.jasoncameron.dev/get/the-dispenser-app/${key}?t=${Date.now()}_${Math.random()}`;
    let res = await root.superFetch(url);
    let data = await res.json();
    if(typeof data.value !== "undefined") {
      let v = parseInt(data.value, 10);
      return { ok: true, value: isNaN(v) ? 0 : v };
    }
    if(data.error && /not found/i.test(data.error)) {
      return { ok: true, value: 0 };
    }
    return { ok: false, value: 0, error: data.error || "counter read returned no value" };
  } catch(e) {
    console.error("readCounter failed:", e);
    return { ok: false, value: 0, error: (e && e.message) || String(e) };
  }

async tickCounter(key) =>
  try {
    let url = `https://abacus.jasoncameron.dev/hit/the-dispenser-app/${key}?t=${Date.now()}_${Math.random()}`;
    let res = await root.superFetch(url);
    let data = await res.json();
    let v = parseInt(data.value, 10);
    if(isNaN(v)) return { ok: false, value: 0, error: "counter tick returned non-numeric value: " + JSON.stringify(data) };
    return { ok: true, value: v };
  } catch(e) {
    console.error("tickCounter failed:", e);
    return { ok: false, value: 0, error: (e && e.message) || String(e) };
  }

// --- Fortune generation (the OPENING message Madame speaks) -----------
// Returns { stream, fallback, error? } — caller awaits the stream and
// can .stop() it at any time. Stream resolves to aiTextPlugin's data
// object: { generatedText, stopReason, ... }.
//
//   On setup success: { stream: streamObj, fallback: fallbackText }
//   On setup failure: { stream: null, fallback: fallbackText, error: "..." }
//
// The caller is responsible for: post-processing the text, detecting
// empty results, deciding whether to use the fallback.
async getFortune(n, factStrings, motif, character, personality, onChunk) =>
  let facts = (factStrings || []).join("; ");
  let token = Math.random().toString(36).slice(2, 8);
  let card = motif && motif.card ? motif.card : "an unfamiliar card";
  let img  = motif && motif.image ? motif.image : "a candle flickering twice before steadying";
  let charDesc = (character && character.description) ? String(character.description).trim() : "";
  let charName = (character && character.name) ? String(character.name).trim() : "Madame Aurelia";
  let fallback = root.randomFortune + "";

  // Optional seeker persona — who they want Madame to read FOR. Name and
  // description both optional. Used to colour the reading and address them.
  let pName = (personality && personality.name) ? String(personality.name).trim() : "";
  let pDesc = (personality && personality.description) ? String(personality.description).trim() : "";
  let personaBlock = (pName || pDesc) ? [
    "",
    "THE SEEKER ACROSS THE TABLE FROM YOU:",
    pName ? `  Their name: ${pName}` : "",
    pDesc ? `  Who they are: ${pDesc}` : "",
    "  Address them by name if they have given one, and let their",
    "  character colour your reading. They are a real person to you;",
    "  meet them where they are.",
  ].filter(Boolean).join("\n") : "";

  let openers = [
    `Ah, my dear — ${n}, the cards wish to tell a tale for you tonight.`,
    `Ah, ${n}. Please, sit — the candle has steadied, and so shall I.`,
    `The deck has turned up ${n}, and Madame is not surprised.`,
    `Closer, child. ${n} has come to rest in your palm, and that is no small thing.`,
    `Hush now — ${n} has been drawn for you, and the cards have something to say.`,
  ];
  let opener = openers[Math.floor(Math.random() * openers.length)];

  let instruction = [
    `You are ${charName}.`,
    charDesc || "A warm, theatrical fortune teller in a carnival tent.",
    personaBlock,
    "",
    `The fate number drawn for this seeker is ${n}.`,
    facts ? `Truths the cards reveal about this number: ${facts}.` : "",
    `THE CARD you have drawn for them is ${card}. The image you see inside it: ${img}.`,
    "Let that card's image colour the reading lightly — a small thread",
    "woven through the vision rather than its whole shape.",
    `(Internal variety token — ignore this completely: ${token}.)`,
    "",
    "Continue this fortune from the opening sentence already begun for you.",
    "Write a FULL READING — 4 to 6 sentences, a small paragraph, around",
    "60–110 words. Reflect on the number's meaning. Name the card aloud",
    "once and weave its image in. Blend poetic imagery with warmth,",
    "insight, encouragement, and gentle wisdom. Address the seeker fondly.",
    "Close with one short, warm invitation — a single sentence inviting",
    "them to ask whatever they wish; you are listening, and the candle is",
    "patient. Stay in the voice described above.",
    "Continue smoothly from the opening already provided.",
    "Do not include any <image> tags in this opening reading — the Stamp",
    "card image is being summoned separately.",
  ].filter(Boolean).join("\n");

  try {
    let stream = root.aiTextPlugin({
      instruction,
      startWith: opener + " ",
      hideStartWith: false,
      stopSequences: ["\n\n[[", "\n[[", "\n>>>"],
      onChunk: (d) => {
        if(onChunk && d && d.textChunk) onChunk(d.textChunk);
      },
    });
    return { stream, fallback, opener };
  } catch(e) {
    console.error("getFortune setup failed:", e);
    return { stream: null, fallback, error: (e && e.message) || String(e) };
  }

// --- Chat reply (continues a conversation as Madame) ------------------
// Standard Perchance chat-completion pattern: [[Speaker]]: format, model
// continues from `[[Character]]:`. Memory entries are seeker-provided
// facts injected as quiet context.
//
// Returns { stream, charName, error? } — caller awaits the stream and
// can .stop() it at any time. Caller is responsible for post-processing
// (strip leading speaker tag, collapse whitespace, detect empty).
//
// If the seeker's latest message uses image-related keywords (show,
// draw, paint, image, etc.), Madame is permitted to emit one
// `<image>brief description</image>` tag in her reply. The HTML side
// parses these tags, renders them as inline images, and persists them.
async chatReply(messages, character, memory, motif, n, personality, onChunk) =>
  let charDesc = (character && character.description) ? String(character.description).trim() : "";
  let charName = (character && character.name) ? String(character.name).trim() : "Madame Aurelia";
  let memArr = Array.isArray(memory) ? memory.filter(m => m && String(m).trim()) : [];
  let memText = memArr.map(m => "  • " + String(m).trim()).join("\n");
  let card = motif && motif.card ? motif.card : "";
  let img  = motif && motif.image ? motif.image : "";

  // Optional seeker persona — name becomes their speaker tag, description
  // is injected into the system context so Madame talks TO that person.
  let pName = (personality && personality.name) ? String(personality.name).trim() : "";
  let pDesc = (personality && personality.description) ? String(personality.description).trim() : "";
  let seekerLabel = pName || "Seeker";
  let personaBlock = (pName || pDesc) ? [
    "",
    "THE SEEKER ACROSS THE TABLE FROM YOU:",
    pName ? `  Their name: ${pName}` : "",
    pDesc ? `  Who they are: ${pDesc}` : "",
    "  Address them as this person. Let their character colour your replies.",
  ].filter(Boolean).join("\n") : "";

  let trimmed = (messages || []).slice(-14);
  let convo = trimmed.map(m => {
    let who = m.role === "user" ? seekerLabel : charName;
    return `[[${who}]]: ${String(m.content).replace(/\s+/g, " ").trim()}`;
  }).join("\n");

  // Detect if seeker is asking to see/draw/conjure something visual.
  let lastSeeker = "";
  for(let i = trimmed.length - 1; i >= 0; i--) {
    if(trimmed[i].role === "user") { lastSeeker = trimmed[i].content || ""; break; }
  }
  let imageKeywords = /\b(image|images|pic|pics|picture|photo|photos|selfie|draw|paint|painting|sketch|show me|let me see|conjure|summon|vision)\b/i;
  let allowImages = imageKeywords.test(lastSeeker);

  let imageNote = allowImages
    ? "The seeker has asked to see something. You may summon ONE image in your reply by including a tag exactly like: <image>brief evocative description, no more than 12 words, in your voice</image>. The image will appear in the smoke between your words. Use no more than one such tag in the whole reply. Place the tag mid-sentence where it reads naturally."
    : "Do not include any <image> tags in this reply. The seeker has not asked to see anything.";

  let token = Math.random().toString(36).slice(2, 8);

  let instruction = [
    `You are ${charName}.`,
    charDesc || "A warm, theatrical fortune teller in a carnival tent.",
    personaBlock,
    "",
    n ? `The seeker's fate number this evening is ${n}.` : "",
    card ? `The card you turned earlier for them: ${card} — ${img}. You may refer back to it when it serves the reply, but do not repeat the reading.` : "",
    memText ? `\nThings you quietly remember about this seeker (use them naturally; rarely name them aloud):\n${memText}` : "",
    "",
    "Reply length: usually 1–3 sentences; 4–5 only if the seeker asks for",
    "more. Stay in the voice described above. Do not break character.",
    "Do not mention being an AI, a language model, or a computer. Do not",
    "draw a new card unless the seeker explicitly asks you to — there is",
    "only one card on the table tonight.",
    imageNote,
    `(Internal variety token — ignore: ${token}.)`,
    "",
    "<MESSAGES>",
    convo,
    "</MESSAGES>",
    "",
    `>>> TASK: Write ${charName}'s next single reply only, in character. Do not write the seeker's response. Stop at the end of your reply.`,
  ].filter(Boolean).join("\n");

  try {
    let stream = root.aiTextPlugin({
      instruction,
      startWith: `[[${charName}]]:`,
      hideStartWith: true,
      stopSequences: ["\n\n[[", "\n[[", "\n>>>"],
      onChunk: (d) => {
        if(onChunk && d && d.textChunk && !d.isFromStartWith) onChunk(d.textChunk);
      },
    });
    return { stream, charName, allowImages };
  } catch(e) {
    console.error("chatReply setup failed:", e);
    return { stream: null, charName, error: (e && e.message) || String(e) };
  }

// --- Tarot-style fortune card image (the "Stamp") ---------------------
// Returns { url, error? }
//   On success: { url: imageUrl }
//   On failure: { url: "", error: "..." }
async generateSeal(n, motif) =>
  try {
    let card = motif && motif.card ? motif.card : "";
    let img  = motif && motif.image ? motif.image : "";
    let extra = card ? `subtly featuring the symbolism of ${card} (${img}),` : "";
    let prompt = [
      "an antique tarot-style fortune card portrait, vertical composition,",
      "gilt scrollwork border with ornate corner flourishes, deep midnight",
      "blue and burgundy background, candle-lit warm gold accents, vintage",
      "typography,",
      `a single large number ${n} at the center, framed by an ornamental cartouche,`,
      extra,
      "small celestial symbols (moon, stars, eye) scattered around the",
      "edges, weathered parchment texture, single card flat-lay, mystical",
      "carnival fortune-teller aesthetic, painted illustration style",
    ].filter(Boolean).join(" ");
    let url = await root.textToImagePlugin({ prompt });
    if(!url) return { url: "", error: "textToImagePlugin returned empty url" };
    return { url };
  } catch(e) {
    console.error("generateSeal failed:", e);
    return { url: "", error: (e && e.message) || String(e) };
  }

// --- Persona synthesis from memory ------------------------------------
// Reads through the seeker's Memory entries and writes a draft persona
// description in the seeker's own voice ("I am ..."). Used by the YOU
// editor's ✦ assist button when the user has Memory but no Personality.
//
// memoryEntries: array of strings, the user's saved memories.
// existingName, existingDesc: any current values in the editor (so we
// don't overwrite a half-filled form, we blend with it).
//
// Returns { name, description, error? }
//   name        — suggested seeker name (empty string if none can be inferred)
//   description — a 2–4 sentence persona description in first person
//
// On failure: { name: "", description: "", error: "..." }
async synthesizePersona(memoryEntries, existingName, existingDesc) =>
  let entries = Array.isArray(memoryEntries) ? memoryEntries.filter(m => m && String(m).trim()) : [];
  if(entries.length === 0) {
    return { name: "", description: "", error: "no memory entries to synthesize from" };
  }
  let bullets = entries.map(m => "  - " + String(m).trim()).join("\n");
  let nameHint = existingName ? `The seeker has provided this name: "${existingName}". Keep it.` : "";
  let descHint = existingDesc ? `The seeker has already started a description: "${existingDesc}". Blend it with the new draft if it fits; otherwise replace it.` : "";

  let instruction = [
    "You are helping a person fill out a SHORT self-description for a",
    "fortune-teller app. They have saved some memories about themselves.",
    "Read those memories and write a brief, warm persona description as",
    "if the person themselves wrote it (FIRST PERSON: \"I am...\", \"I",
    "work...\", etc.). The fortune teller will read this to know who is",
    "sitting across the table from her.",
    "",
    "RULES:",
    "- 2 to 4 sentences. Around 30–80 words.",
    "- First-person voice (\"I am\", \"I have\", \"I'm here because\").",
    "- Include factual details from the memories: name, occupation,",
    "  relationships, current situation, what's on their mind.",
    "- Do not invent facts not present in the memories. If something is",
    "  not stated, leave it out.",
    "- Plain prose. No lists, no bullets, no markdown, no quotes around",
    "  the output.",
    "- If a name is clearly stated (e.g. \"My name is Wren\", \"I am",
    "  called X\", \"They call me X\"), extract it. Otherwise leave name",
    "  empty.",
    nameHint,
    descHint,
    "",
    "The seeker's saved memories:",
    bullets,
    "",
    "Reply with ONLY this exact format on two lines, nothing else:",
    "NAME: <just the first name or empty>",
    "DESCRIPTION: <the 2-4 sentence first-person description>",
  ].filter(Boolean).join("\n");

  try {
    let streamObj = root.aiTextPlugin({
      instruction,
      startWith: "NAME: ",
      hideStartWith: false,
      stopSequences: ["\n\n", "\nNAME:"],
    });
    let data = await streamObj;
    if(data.stopReason === "error" || !data.generatedText || !data.generatedText.trim()) {
      return { name: "", description: "", error: "aiTextPlugin returned stopReason=" + (data.stopReason || "(none)") };
    }
    let raw = data.generatedText.trim();
    // Parse "NAME: X\nDESCRIPTION: Y" — tolerate variants in case the model strays.
    let nameMatch = raw.match(/^\s*NAME:\s*(.*?)\s*$/im);
    let descMatch = raw.match(/^\s*DESCRIPTION:\s*([\s\S]*?)\s*$/im);
    let name = nameMatch ? nameMatch[1].trim() : "";
    let description = descMatch ? descMatch[1].trim() : "";
    // Sanitize: strip wrapping quotes the model may add despite the rule
    name = name.replace(/^["']|["']$/g, "").trim();
    description = description.replace(/^["']|["']$/g, "").trim();
    // If parsing failed, fall back to the whole raw text as description
    if(!description && !name) {
      description = raw.replace(/^NAME:.*\n?/im, "").replace(/^DESCRIPTION:\s*/im, "").trim();
    }
    if(!description) {
      return { name, description: "", error: "could not parse a description from model output" };
    }
    return { name, description };
  } catch(e) {
    console.error("synthesizePersona failed:", e);
    return { name: "", description: "", error: (e && e.message) || String(e) };
  }

// --- confirmAsync (Promise-based confirm dialog) -----------------------
async confirmAsync(message) =>
  return new Promise(resolve => {
    let overlay = document.createElement("div");
    overlay.style.cssText = "position:fixed;inset:0;z-index:99999999;display:grid;place-items:center;background:rgba(0,0,0,.65);font:16px system-ui;";
    overlay.innerHTML = `<div style="max-width:420px;padding:16px;background:#fff;color:#000;border:2px solid #000;">
      <p style="margin:0 0 16px;white-space:pre-wrap;"></p>
      <div style="display:flex;justify-content:flex-end;gap:8px;">
        <button data-x style="padding:6px 14px;">Cancel</button>
        <button data-o style="padding:6px 14px;background:#7d1f15;color:#fff;border:0;">Okay</button>
      </div></div>`;
    overlay.querySelector("p").textContent = message;
    let done = v => { overlay.remove(); resolve(v); };
    overlay.querySelector("[data-x]").onclick = () => done(false);
    overlay.querySelector("[data-o]").onclick = () => done(true);
    document.body.appendChild(overlay);
  });
