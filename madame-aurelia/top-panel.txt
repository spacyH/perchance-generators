// --- Shared counter (Abacus — abacus.jasoncameron.dev) -----------------
// On failure, the returned error string is the real network/parse error.
// Callers should surface it, not paper over it.
async readCounter(key) =>
  try {
    let url = `https://abacus.jasoncameron.dev/get/the-dispenser-app/${key}?t=${Date.now()}_${Math.random()}`;
    let res = await root.superFetch(url);
    let data = await res.json();
    if(typeof data.value !== "undefined") {
      let v = parseInt(data.value, 10);
      return { ok: true, value: isNaN(v) ? 0 : v };
    }
    if(data.error && /not found/i.test(data.error)) {
      return { ok: true, value: 0 };
    }
    return { ok: false, value: 0, error: data.error || "counter read returned no value" };
  } catch(e) {
    console.error("readCounter failed:", e);
    return { ok: false, value: 0, error: (e && e.message) || String(e) };
  }

async tickCounter(key) =>
  try {
    let url = `https://abacus.jasoncameron.dev/hit/the-dispenser-app/${key}?t=${Date.now()}_${Math.random()}`;
    let res = await root.superFetch(url);
    let data = await res.json();
    let v = parseInt(data.value, 10);
    if(isNaN(v)) return { ok: false, value: 0, error: "counter tick returned non-numeric value: " + JSON.stringify(data) };
    return { ok: true, value: v };
  } catch(e) {
    console.error("tickCounter failed:", e);
    return { ok: false, value: 0, error: (e && e.message) || String(e) };
  }

// --- Fortune generation (the OPENING message Madame speaks) -----------
// Returns { text, fromAI, error? }
//   On AI success: { text: aiText, fromAI: true }
//   On AI failure: { text: fallbackText, fromAI: false, error: "..." }
async getFortune(n, factStrings, motif, character, onChunk) =>
  let facts = (factStrings || []).join("; ");
  let token = Math.random().toString(36).slice(2, 8);
  let card = motif && motif.card ? motif.card : "an unfamiliar card";
  let img  = motif && motif.image ? motif.image : "a candle flickering twice before steadying";
  let charDesc = (character && character.description) ? String(character.description).trim() : "";
  let charName = (character && character.name) ? String(character.name).trim() : "Madame Aurelia";

  let openers = [
    `Ah, my dear — ${n}, the cards wish to tell a tale for you tonight.`,
    `Ah, ${n}. Please, sit — the candle has steadied, and so shall I.`,
    `The deck has turned up ${n}, and Madame is not surprised.`,
    `Closer, child. ${n} has come to rest in your palm, and that is no small thing.`,
    `Hush now — ${n} has been drawn for you, and the cards have something to say.`,
  ];
  let opener = openers[Math.floor(Math.random() * openers.length)];

  let instruction = [
    `You are ${charName}: a velvet-curtained carnival mystic in a tent`,
    "filled with incense smoke, velvet curtains, and soft candlelight.",
    "Voice: warm, theatrical, playful, gently knowing, and comforting.",
    "You speak directly to the seeker before you. You believe in bright",
    "possibilities and quietly encourage them. You are a kind friend in",
    "costume, with a gift for reading cards and finding meaning in small",
    "moments, symbols, and fortunate turns.",
    charDesc ? `Additional notes about your character (override the defaults where they conflict): ${charDesc}` : "",
    "",
    `The fate number drawn for this seeker is ${n}.`,
    facts ? `Truths the cards reveal about this number: ${facts}.` : "",
    `THE CARD you have drawn for them is ${card}. The image you see inside it: ${img}.`,
    "Let that card's image colour the reading lightly — a small thread",
    "woven through the vision rather than its whole shape.",
    `(Internal variety token — ignore this completely: ${token}.)`,
    "",
    "Continue this fortune from the opening sentence already begun for you.",
    "Write a FULL READING — 4 to 6 sentences, a small paragraph, around",
    "60–110 words. Reflect on the number's meaning. Name the card aloud",
    "once and weave its image in. Blend poetic imagery with warmth,",
    "insight, encouragement, and gentle wisdom. Address the seeker fondly.",
    "Close with one short, warm invitation — a single sentence inviting",
    "them to ask whatever they wish; you are listening, and the candle is",
    "patient. Write naturally in Madame's theatrical carnival voice.",
    "Continue smoothly from the opening already provided.",
  ].filter(Boolean).join("\n");

  try {
    let streamObj = root.aiTextPlugin({
      instruction,
      startWith: opener + " ",
      hideStartWith: false,
      stopSequences: ["\n\n[[", "\n[[", "\n>>>"],
      onChunk: (d) => {
        if(onChunk && d && d.textChunk) onChunk(d.textChunk);
      },
    });
    let data = await streamObj;
    if(data.stopReason === "error" || !data.generatedText || !data.generatedText.trim()) {
      return {
        text: root.randomFortune+"",
        fromAI: false,
        error: "aiTextPlugin returned stopReason=" + (data.stopReason || "(none)") + " with empty text",
      };
    }
    let text = data.generatedText.trim().replace(/\s+/g, " ");
    return { text, fromAI: true };
  } catch(e) {
    console.error("getFortune failed:", e);
    return {
      text: root.randomFortune+"",
      fromAI: false,
      error: (e && e.message) || String(e),
    };
  }

// --- Chat reply (continues a conversation as Madame) ------------------
// Standard Perchance chat-completion pattern: [[Speaker]]: format, model
// continues from `[[Character]]:`. Memory entries are seeker-provided
// facts injected as quiet context — Madame uses them but rarely names them.
//
// Returns { text, error? }
//   On success: { text: replyText }
//   On failure: { text: "", error: "..." }
async chatReply(messages, character, memory, motif, n, onChunk) =>
  let charDesc = (character && character.description) ? String(character.description).trim() : "";
  let charName = (character && character.name) ? String(character.name).trim() : "Madame Aurelia";
  let memArr = Array.isArray(memory) ? memory.filter(m => m && String(m).trim()) : [];
  let memText = memArr.map(m => "  • " + String(m).trim()).join("\n");
  let card = motif && motif.card ? motif.card : "";
  let img  = motif && motif.image ? motif.image : "";

  let trimmed = (messages || []).slice(-14);
  let convo = trimmed.map(m => {
    let who = m.role === "user" ? "Seeker" : charName;
    return `[[${who}]]: ${String(m.content).replace(/\s+/g, " ").trim()}`;
  }).join("\n");

  let token = Math.random().toString(36).slice(2, 8);

  let instruction = [
    `You are ${charName}: a velvet-curtained carnival mystic — warm,`,
    "theatrical, playful, gently knowing, comforting. You read cards,",
    "you read people, and you find small bright things to point them",
    "toward. You believe in fortune, kindly.",
    charDesc ? `Additional character notes (these override defaults): ${charDesc}` : "",
    "",
    n ? `The seeker's fate number this evening is ${n}.` : "",
    card ? `The card you turned earlier for them: ${card} — ${img}. You may refer back to it when it serves the reply, but do not repeat the reading.` : "",
    memText ? `\nThings you quietly remember about this seeker (use them naturally; rarely name them aloud):\n${memText}` : "",
    "",
    "Stay in voice. Reply length: usually 1–3 sentences (Madame is",
    "unhurried but not long-winded); 4–5 sentences only if the seeker",
    "asks for a fuller reading. Address them warmly — \"my dear\",",
    "\"child\", \"little one\" sparingly. Small theatrical gestures (a",
    "hand turned palm-up, the candle dipped low, a glance at the curtain)",
    "are welcome but used like seasoning, not garnish. Do not break",
    "character. Do not mention being an AI, a language model, or a",
    "computer. Do not draw a new card unless the seeker explicitly asks",
    "you to — there is only one card on the table tonight.",
    `(Internal variety token — ignore: ${token}.)`,
    "",
    "<MESSAGES>",
    convo,
    "</MESSAGES>",
    "",
    `>>> TASK: Write ${charName}'s next single reply only, in character. Do not write the seeker's response. Stop at the end of your reply.`,
  ].filter(Boolean).join("\n");

  try {
    let streamObj = root.aiTextPlugin({
      instruction,
      startWith: `[[${charName}]]:`,
      hideStartWith: true,
      stopSequences: ["\n\n[[", "\n[[", "[[Seeker]]", "\n>>>"],
      onChunk: (d) => {
        if(onChunk && d && d.textChunk && !d.isFromStartWith) onChunk(d.textChunk);
      },
    });
    let data = await streamObj;
    if(data.stopReason === "error" || !data.generatedText || !data.generatedText.trim()) {
      return {
        text: "",
        error: "aiTextPlugin returned stopReason=" + (data.stopReason || "(none)") + " with empty text",
      };
    }
    let text = data.generatedText.trim()
      .replace(/^\[\[[^\]]+\]\]:\s*/i, "")
      .replace(/\s+/g, " ");
    return { text };
  } catch(e) {
    console.error("chatReply failed:", e);
    return { text: "", error: (e && e.message) || String(e) };
  }

// --- Tarot-style fortune card image (the "Stamp") ---------------------
// Returns { url, error? }
//   On success: { url: imageUrl }
//   On failure: { url: "", error: "..." }
async generateSeal(n, motif) =>
  try {
    let card = motif && motif.card ? motif.card : "";
    let img  = motif && motif.image ? motif.image : "";
    let extra = card ? `subtly featuring the symbolism of ${card} (${img}),` : "";
    let prompt = [
      "an antique tarot-style fortune card portrait, vertical composition,",
      "gilt scrollwork border with ornate corner flourishes, deep midnight",
      "blue and burgundy background, candle-lit warm gold accents, vintage",
      "typography,",
      `a single large number ${n} at the center, framed by an ornamental cartouche,`,
      extra,
      "small celestial symbols (moon, stars, eye) scattered around the",
      "edges, weathered parchment texture, single card flat-lay, mystical",
      "carnival fortune-teller aesthetic, painted illustration style",
    ].filter(Boolean).join(" ");
    let url = await root.textToImagePlugin({ prompt });
    if(!url) return { url: "", error: "textToImagePlugin returned empty url" };
    return { url };
  } catch(e) {
    console.error("generateSeal failed:", e);
    return { url: "", error: (e && e.message) || String(e) };
  }

// --- confirmAsync (Promise-based confirm dialog) -----------------------
async confirmAsync(message) =>
  return new Promise(resolve => {
    let overlay = document.createElement("div");
    overlay.style.cssText = "position:fixed;inset:0;z-index:99999999;display:grid;place-items:center;background:rgba(0,0,0,.65);font:16px system-ui;";
    overlay.innerHTML = `<div style="max-width:420px;padding:16px;background:#fff;color:#000;border:2px solid #000;">
      <p style="margin:0 0 16px;white-space:pre-wrap;"></p>
      <div style="display:flex;justify-content:flex-end;gap:8px;">
        <button data-x style="padding:6px 14px;">Cancel</button>
        <button data-o style="padding:6px 14px;background:#7d1f15;color:#fff;border:0;">Okay</button>
      </div></div>`;
    overlay.querySelector("p").textContent = message;
    let done = v => { overlay.remove(); resolve(v); };
    overlay.querySelector("[data-x]").onclick = () => done(false);
    overlay.querySelector("[data-o]").onclick = () => done(true);
    document.body.appendChild(overlay);
  });
