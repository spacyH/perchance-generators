<script>
(function () {
  var MSG = 'marinara.bridge';

  function $(id) { return document.getElementById(id); }
  function log(line) {
    var el = $('log');
    var t = new Date().toISOString().slice(11, 19);
    el.textContent = (el.textContent === '—' ? '' : el.textContent + '\n') + '[' + t + '] ' + line;
  }
  function setPill(id, on, text) {
    var el = $(id);
    el.className = 'pill ' + (on ? 'on' : 'off');
    el.querySelector('span:last-child').textContent = text;
  }

  function grabPlugin(name) {
    try { if (typeof root !== 'undefined' && root[name] !== undefined) return root[name]; } catch (e) {}
    try { if (window[name] !== undefined) return window[name]; } catch (e2) {}
    return null;
  }

  var refState = { url: '', preview: '' };

  function updateRefPreview() {
    var box = $('ref-preview');
    box.innerHTML = '';
    var src = refState.preview || refState.url;
    if (!src) {
      box.innerHTML = '<span class="ref-meta">No reference image</span>';
      return;
    }
    var img = new Image();
    img.src = src;
    img.alt = 'reference';
    var meta = document.createElement('div');
    meta.className = 'ref-meta';
    var blur = Number($('ref-blur').value) || 0.35;
    meta.textContent = refState.url
      ? 'URL: ' + refState.url.replace(/^(.{48}).+/, '$1…') + '\nblur: ' + blur
      : 'preview only';
    box.appendChild(img);
    box.appendChild(meta);
  }

  function setRefUrl(url, preview) {
    refState.url = url || '';
    refState.preview = preview || url || '';
    if (url) $('ref-url').value = url;
    updateRefPreview();
  }

  function clearRef() {
    refState = { url: '', preview: '' };
    $('ref-file').value = '';
    $('ref-url').value = '';
    updateRefPreview();
    log('reference image cleared');
  }

  async function uploadRefFile(file) {
    var uploadFn = grabPlugin('uploadPlugin');
    if (!uploadFn) throw new Error('upload-plugin not available');
    log('uploading reference…');
    var result = await uploadFn(file);
    if (result && result.error) throw new Error(String(result.error));
    return String(result.url);
  }

  async function onRefFileChange(ev) {
    var file = ev.target.files && ev.target.files[0];
    if (!file) return;
    try {
      var localPreview = URL.createObjectURL(file);
      refState.preview = localPreview;
      updateRefPreview();
      var url = await uploadRefFile(file);
      setRefUrl(url, url);
      log('reference uploaded: ' + url);
    } catch (e) {
      log('reference upload failed: ' + (e && e.message));
      clearRef();
    }
  }

  // Imported plugins do not auto-run $output — trigger explicitly.
  try {
    if (typeof root !== 'undefined' && typeof root.marinaraBridge === 'function') root.marinaraBridge();
  } catch (e) { log('root.marinaraBridge() failed: ' + (e && e.message)); }

  var bridge = (window.marinara && window.marinara.bridge) || null;

  function refreshStatus() {
    var framed = false;
    try { framed = window.parent !== window; } catch (e) { framed = true; }
    var opted = bridge && bridge.isOptedIn ? bridge.isOptedIn() : false;
    var active = bridge && bridge.isActive ? bridge.isActive() : false;
    setPill('pill-bridge', active, 'bridge: ' + (active ? 'listening' : 'idle'));
    setPill('pill-frame', framed, 'frame: ' + (framed ? 'embedded' : 'top'));
    setPill('pill-opt', opted, 'opt-in: ' + (opted ? 'yes' : 'add ?marinara=1'));
    if (active) log('bridge ready — waiting for Marinara requests');
    else if (!opted) log('not in bridge mode — append ?marinara=1 for Marinara embed');
    else if (!framed) log('open inside Marinara iframe to activate bridge');
  }

  function readOpts() {
    var opts = {
      prompt: $('prompt').value.trim(),
      resolution: $('resolution').value,
      guidanceScale: Number($('guidance').value) || 7,
      seed: Number($('seed').value)
    };
    var refUrl = (refState.url || $('ref-url').value || '').trim();
    if (refUrl) {
      opts.referenceImage = {
        url: refUrl,
        blur: Number($('ref-blur').value)
      };
    }
    return opts;
  }

  function showImage(dataUrl) {
    var box = $('preview');
    box.innerHTML = '';
    if (!dataUrl) { box.textContent = 'No image returned'; return; }
    var img = new Image();
    img.src = String(dataUrl);
    img.alt = 'generated';
    box.appendChild(img);
  }

  async function generateLocal() {
    if (!bridge || typeof bridge.generate !== 'function') {
      log('bridge.generate unavailable');
      return;
    }
    var opts = readOpts();
    if (!opts.prompt) { log('prompt required'); return; }
    $('go').disabled = true;
    log('local generate: ' + opts.resolution + (opts.referenceImage ? ' + ref' : ''));
    try {
      var r = await bridge.generate(opts);
      if (!r.ok) { log('failed: ' + (r.reason || 'unknown')); return; }
      showImage(r.value && r.value.dataUrl);
      log('done (' + (r.value && r.value.inputs && r.value.inputs.resolution || opts.resolution) + ')');
    } catch (e) {
      log('error: ' + (e && e.message));
    } finally {
      $('go').disabled = false;
    }
  }

  function embedUrl() {
    var base = 'https://perchance.org/marinara-t2i-host';
    try {
      if (typeof window.generatorName === 'string') base = 'https://perchance.org/' + window.generatorName;
    } catch (e) {}
    return base + '?marinara=1';
  }

  $('go').addEventListener('click', generateLocal);
  $('copy-url').addEventListener('click', function () {
    var url = embedUrl();
    if (navigator.clipboard && navigator.clipboard.writeText) {
      navigator.clipboard.writeText(url).then(function () { log('copied embed URL'); });
    } else {
      prompt('Copy Marinara embed URL:', url);
    }
  });
  $('ref-file').addEventListener('change', onRefFileChange);
  $('ref-clear').addEventListener('click', clearRef);
  $('ref-url').addEventListener('change', function () {
    var url = $('ref-url').value.trim();
    if (url) setRefUrl(url, url);
    else clearRef();
  });
  $('ref-blur').addEventListener('input', function () {
    $('ref-blur-val').textContent = (Number($('ref-blur').value).toFixed(2)) + ' — lower = stronger likeness';
    updateRefPreview();
  });

  $('proto').textContent = [
    'Channel: ' + MSG,
    'Child → parent: { channel, type: "ready", proto, build, generatorName, capabilities: ["image"] }',
    'Parent → child: { channel, type: "request", cap: "image", nonce, payload: { prompt, resolution, guidanceScale, seed, referenceImage? } }',
    '  referenceImage: { url: string, blur: number }  // blur 0–1; url = upload CDN or data: URL',
    'Child → parent: { channel, type: "reply", nonce, result: { ok, value: { dataUrl, inputs } } | { ok:false, reason } }',
    '',
    'Embed URL: ' + embedUrl(),
    'Queue: one image job at a time (Perchance t2i broker is serial).'
  ].join('\n');

  refreshStatus();
  setInterval(refreshStatus, 2000);
})();
</script>
