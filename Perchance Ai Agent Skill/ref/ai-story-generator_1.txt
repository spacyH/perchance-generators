normalizeThemeSettings(settings, opts) =>
  if(!opts) opts = {};
  if(!settings || typeof settings !== "object") return opts.allowNull ? null : {...getThemeDefaults()};
  let s = {...getThemeDefaults(), ...settings};
  if(s.italicColor === undefined || s.italicColor === "") {
    let legacy = s.italicStyle || "";
    let colorMatch = legacy.match(/color\s*:\s*([^;]+)/i);
    if(colorMatch) s.italicColor = colorMatch[1].trim();
  }
  if(s.quoteColor === undefined || s.quoteColor === "") {
    let legacy = s.quoteStyle || "";
    let colorMatch = legacy.match(/color\s*:\s*([^;]+)/i);
    if(colorMatch) s.quoteColor = colorMatch[1].trim();
  }
  if(s.italicStyle) {
    s.italicItalic = /font-style\s*:\s*italic/i.test(s.italicStyle || "");
  }
  if(s.quoteStyle) {
    s.quoteItalic = /font-style\s*:\s*italic/i.test(s.quoteStyle || "");
  }
  if(!Array.isArray(s.advancedRules)) s.advancedRules = [];
  return s;

getParagraphStopIndex(text, limit) =>
  if(limit < 1) return -1;
  let leading = text.match(/^\n+/)?.[0] || "";
  let t = text.slice(leading.length);
  if(!t) return -1;
  let regex = /\n{2,}/g;
  let count = 0;
  let match;
  while((match = regex.exec(t))) {
    count++;
    if(count >= limit) return leading.length + match.index + match[0].length;
  }
  return -1;

formatRelativeTime(ts) =>
  if(!ts) return "";
  let diff = Date.now() - ts;
  if(diff < 1000*10) return "just now";
  if(diff < 1000*60) return `${Math.round(diff/1000)}s ago`;
  if(diff < 1000*60*60) return `${Math.round(diff/(1000*60))}m ago`;
  if(diff < 1000*60*60*24) return `${Math.round(diff/(1000*60*60))}h ago`;
  if(diff < 1000*60*60*24*14) return `${Math.round(diff/(1000*60*60*24))}d ago`;
  return new Date(ts).toString().split(" ").slice(1, 4).join(" ");

normalizeFolderPath(path) =>
  path = (path || "").toString().trim().replace(/\\+/g, "/");
  path = path.replace(/\/+/g, "/").replace(/^\/+|\/+$/g, "");
  return path;

getFolderSegments(path) =>
  return normalizeFolderPath(path).split("/").filter(Boolean);

getParentFolderPath(path) =>
  let segs = getFolderSegments(path);
  return segs.slice(0, -1).join("/");

isFolderPrefix(path, prefix) =>
  if(!prefix) return true;
  if(!path) return false;
  if(path === prefix) return true;
  return path.startsWith(prefix + "/");

getChildFolderPaths(meta, folderPath) =>
  let base = normalizeFolderPath(folderPath || "");
  let prefix = base ? base + "/" : "";
  let set = new Set();
  for(let m of meta) {
    let fp = normalizeFolderPath(m.folderPath || "");
    if(!fp) continue;
    if(prefix && !fp.startsWith(prefix)) continue;
    let remainder = prefix ? fp.slice(prefix.length) : fp;
    let seg = remainder.split("/")[0];
    if(seg) {
      let childPath = base ? (base + "/" + seg) : seg;
      set.add(childPath);
    }
  }
  return [...set].sort((a,b) => a.localeCompare(b));

getStoriesAtFolder(meta, folderPath) =>
  folderPath = normalizeFolderPath(folderPath || "");
  return meta.filter(m => normalizeFolderPath(m.folderPath || "") === folderPath);

sortStoriesForDisplay(metaArr) =>
  return (metaArr || []).slice(0).sort((a,b) => {
    let ap = a.pinned ? 1 : 0;
    let bp = b.pinned ? 1 : 0;
    if(ap !== bp) return bp - ap;
    return (b.lastEditTime||0) - (a.lastEditTime||0);
  });

countMatches(haystack, needle) =>
  if(!needle) return 0;
  let count = 0;
  let idx = 0;
  while(true) {
    idx = haystack.indexOf(needle, idx);
    if(idx === -1) break;
    count++;
    idx += needle.length || 1;
  }
  return count;

async computeStoriesSearchResults(query) =>
  ensureStoryManager();
  let q = (query || "").toString().trim().toLowerCase();
  if(!q) { window.storyManager.searchResults = null; renderStoriesList(); return; }
  let token = ++window.storyManager.searchToken;
  let meta = window.storyManager.meta || [];
  let results = [];
  for(let entry of meta) {
    let data = await kv.stories.get(entry.id);
    data = normalizeStoryData(data);
    let title = (entry.title || "Untitled story").toLowerCase();
    let overview = (data.storyOverview || "").toLowerCase();
    let story = (data.storySoFar || "").toLowerCase();
    let score = 0;
    score += 10 * countMatches(title, q);
    score += countMatches(overview, q);
    score += countMatches(story, q);
    if(score > 0) {
      results.push({ entry, score });
    }
  }
  if(token !== window.storyManager.searchToken) return;
  results.sort((a,b) => (b.score - a.score) || ((b.entry.lastEditTime||0) - (a.entry.lastEditTime||0)));
  window.storyManager.searchResults = results;
  renderStoriesList();

async saveCurrentStoryToKvNow() =>
  ensureStoryManager();
  if(window.isDeletingAllData) return;
  if(!window.storyManager.isReady) return;
  let id = window.storyManager.currentId;
  if(id === null || id === undefined) return;

  window.storyManager.currentData = data;

  // update meta timestamps in-memory
  let entry = getMetaEntryById(id);
  if(entry) entry.lastEditTime = Date.now();

  let metaSnapshot = (window.storyManager.meta || []).map(m => ({...m}));
  let activeId = id;

  window.storyManager.saveChain = window.storyManager.saveChain.then(async () => {
    await kv.stories.set(id, data);
    await kv.storiesMeta.set(STORIES_META_KEY, metaSnapshot);

  }).catch(console.error);

  return window.storyManager.saveChain;

async checkForExternalStoryUpdate() =>
  try {
    ensureStoryManager();
    if(!window.storyManager.isReady || window.storyManager.isLoading) return;
    if(window.externalUpdatePromptOpen) return;
    if(window.currentlyGenerating) return;
    let id = window.storyManager.currentId;
    if(id === null || id === undefined) return;
    let metaArr = await kv.storiesMeta.get(STORIES_META_KEY) || [];
    let remoteEntry = metaArr.find(m => m && m.id == id);
    let localEntry = getMetaEntryById(id);
    if(!remoteEntry || !localEntry) return;
    let remoteTime = remoteEntry.lastEditTime || 0;
    let localTime = localEntry.lastEditTime || 0;
    if(remoteTime <= localTime) return;
    if(window.lastExternalUpdateSeenByStoryId[id] === remoteTime) return;
    window.lastExternalUpdateSeenByStoryId[id] = remoteTime;
    if(typeof window.forceAutoBackupNow === "function") await window.forceAutoBackupNow();
    window.externalUpdatePromptOpen = true;
    let confirmed = await confirmAsync(
      "You've worked on a newer version of this story in another browser tab.\n\nAn auto-backup of your current version has been saved.\n\nLoad the newer version now? (Cancel to keep this version)"
    );
    window.externalUpdatePromptOpen = false;
    if(confirmed) {
      let data = await kv.stories.get(id);
      if(data && window.storyManager.currentId === id) {
        loadChatDataIntoInputAreas(data);
        localEntry.lastEditTime = remoteTime;
        requestSaveCurrentStory({immediate:true});
      }
    } else {
      requestSaveCurrentStory({immediate:true});
    }
  } catch(e) {
    console.error(e);
  }

async createNewStory(opts) =>
  if(!opts) opts = {};
  ensureStoryManager();
  if(window.currentlyGenerating && window.lastGenerationStreamObj) {
    let ok = await confirmAsync("A paragraph is currently generating. Stop it and create a new story?", {});
    if(!ok) return null;
    try { window.userClickedStop = true; window.lastGenerationStreamObj.stop(); } catch(e) {}
  }

  // flush current saves before switching
  await saveCurrentStoryToKvNow();
  await flushAllStorySaves();

  let id = generateNewStoryId();
  let now = Date.now();
  let data = normalizeStoryData(opts.data || getBlankStoryData());

  let title = (opts.title || "").trim();
  if(!title) title = getSuggestedTitleFromStoryData(data);
  if(!title) title = "Untitled story";

  let folderPath = normalizeFolderPath(opts.folderPath || "");
  if(!folderPath && !(window.storyManager.searchQuery || "").trim()) {
    folderPath = normalizeFolderPath(window.storyManager.currentFolderPath || "");
  }
  window.storyManager.meta.unshift({
    id,
    title,
    createdTime: now,
    lastEditTime: now,
    pinned: !!opts.pinned,
    folderPath,
    autoNameAttempted: false,
  });

  // persist new story + meta
  await kv.stories.set(id, data);
  await kv.storiesMeta.set(STORIES_META_KEY, window.storyManager.meta);
  await kv.storiesMeta.set(ACTIVE_STORY_ID_KEY, id);

  // switch to it
  await switchToStory(id, {skipSaveCurrent:true, closePanel:true});
  return id;

async downloadFolder(folderPath) =>
  ensureStoryManager();
  folderPath = normalizeFolderPath(folderPath);
  let meta = (window.storyManager.meta || []).filter(m => isFolderPrefix(normalizeFolderPath(m.folderPath || ""), folderPath));
  if(meta.length === 0) {
    alert("No stories found in that folder.");
    return;
  }
  await saveCurrentStoryToKvNow();
  await flushAllStorySaves();
  let zip = await ensureZipJs();
  let writer = new zip.ZipWriter(new zip.BlobWriter("application/zip"));
  let usedFolderNames = new Set();
  let rootName = folderPath ? folderPath.split("/").slice(-1)[0] : "folder";
  rootName = rootName.replace(/[\\/:*?"<>|]/g, "_") || "folder";
  for(let entry of meta) {
    let data = await kv.stories.get(entry.id);
    data = normalizeStoryData(data);
    let fullPath = normalizeFolderPath(entry.folderPath || "");
    let relPath = folderPath ? fullPath.slice(folderPath.length).replace(/^\/+/, "") : fullPath;
    let safeSegments = relPath ? relPath.split("/").map(s => s.replace(/[\\/:*?"<>|]/g, "_")) : [];
    let safeTitle = (entry.title || "Untitled story").replace(/[\\/:*?"<>|]/g, "_").slice(0, 80) || "Untitled story";
    let baseFolder = `${safeTitle} (${entry.id})`;
    let storyFolder = [rootName, ...safeSegments, baseFolder].join("/");
    if(usedFolderNames.has(storyFolder)) storyFolder += "-" + Math.random().toString(36).slice(2,6);
    usedFolderNames.add(storyFolder);
    let storyWithoutSummaries = (data.storySoFar || "").split(/\n{2,}/).map(p => p.trim()).filter(p => !p.startsWith("SUMMARY^")).join("\n\n");
    let metaJson = {
      meta: entry,
      storyData: data,
    };
    await writer.add(`${storyFolder}/overview.txt`, new zip.TextReader(data.storyOverview || ""));
    await writer.add(`${storyFolder}/story.txt`, new zip.TextReader(data.storySoFar || ""));
    await writer.add(`${storyFolder}/story_without_summaries.txt`, new zip.TextReader(storyWithoutSummaries));
    await writer.add(`${storyFolder}/meta.json`, new zip.TextReader(JSON.stringify(metaJson, null, 2)));
  }
  let blob = await writer.close();
  let filename = `${rootName}-stories.zip`;
  downloadBlob(blob, filename);

  // close panel (esp on mobile)
  if(opts.closePanel !== false) closeStoriesPanel();

  // scroll to top (nice when switching stories)
  try { document.scrollingElement.scrollTop = 0; } catch(e) {}

  window.storyManager.isLoading = false;

async initStoriesSystem() =>
  ensureStoryManager();
  window.storyManager.isLoading = true;
  window.storyManager.isReady = false;

  let meta = await kv.storiesMeta.get(STORIES_META_KEY);
  if(!Array.isArray(meta)) meta = [];

    return {

      createdTime: m.createdTime || Date.now(),

      pinned: !!m.pinned,
      folderPath: normalizeFolderPath(m.folderPath || ""),
      autoNameAttempted: !!m.autoNameAttempted,
    };
  });

  if(!Array.isArray(folderMeta)) folderMeta = [];
  folderMeta = folderMeta.filter(f => f && f.path).map(f => ({path: normalizeFolderPath(f.path || ""), emoji: (f.emoji || "")}));

  let appBackgrounds = await kv.storiesMeta.get(APP_CUSTOM_BACKGROUNDS_KEY);

  window.appCustomBackgrounds = normalizeCustomMediaList(appBackgrounds);
  let appNotebook = await kv.storiesMeta.get(APP_NOTEBOOK_KEY);
  window.appNotebookData = normalizeNotebookData(appNotebook);

  // Legacy localStorage -> kv upgrade
  if(legacyStoryDataExistsInLocalStorage()) {
    let legacyData = getLegacyStoryDataFromLocalStorage();
    clearLegacyStoryDataFromLocalStorage();

    let legacyId = 0;
    if(meta.some(m => m.id === legacyId)) legacyId = generateNewStoryId();

    let now = Date.now();
    let title = "Untitled story";
    // if it's effectively empty, still keep but label
    if((legacyData.storyOverview + legacyData.storySoFar + legacyData.whatHappensNext).trim() === "") {
      title = "Untitled story";
    }

    meta.unshift({id:legacyId, title, createdTime:now, lastEditTime:now, pinned:false, folderPath:"", autoNameAttempted:false});
    await kv.stories.set(legacyId, normalizeStoryData(legacyData));

    // If old autoBackups were keyed by 0 and legacyId changed, migrate them too.
    if(legacyId !== 0) {
      try {
        let backups = await kv.autoBackups.get(0);
        if(Array.isArray(backups) && backups.length > 0) {
          await kv.autoBackups.set(legacyId, backups);
          await kv.autoBackups.delete(0);
        }
      } catch(e) { console.error(e); }
    }

    await kv.storiesMeta.set(STORIES_META_KEY, meta);
    await kv.storiesMeta.set(ACTIVE_STORY_ID_KEY, legacyId);
  }

  // If no stories exist yet, create a first one
  if(meta.length === 0) {
    window.storyManager.hadNoStoriesOnLoad = true;
    let id = generateNewStoryId();
    let now = Date.now();
    meta.unshift({id, title:"Untitled story", createdTime:now, lastEditTime:now, pinned:false, folderPath:"", autoNameAttempted:false});
    await kv.stories.set(id, getBlankStoryData());
    await kv.storiesMeta.set(STORIES_META_KEY, meta);
    await kv.storiesMeta.set(ACTIVE_STORY_ID_KEY, id);
  }

  window.storyManager.meta = meta;

  // Choose active story
  let activeId = await kv.storiesMeta.get(ACTIVE_STORY_ID_KEY);
  if(activeId === undefined || activeId === null || !meta.some(m => m.id === activeId)) {
    activeId = sortMetaByLastEdit(meta)[0]?.id ?? meta[0].id;
    await kv.storiesMeta.set(ACTIVE_STORY_ID_KEY, activeId);
  }

  window.storyManager.currentId = activeId;
  window.currentStoryId = activeId;

  let data = await kv.stories.get(activeId);
  if(!data) {
    data = getBlankStoryData();
    await kv.stories.set(activeId, data);
  }
  data = normalizeStoryData(data);
  window.storyManager.currentData = data;

  // Load into UI
  window.loadingStoryInProgress = true;
  loadChatDataIntoInputAreas(data);
  window.loadingStoryInProgress = false;

  updateCurrentStoryTitleDisplay();
  renderStoriesList();
  updateStoryHeaderVisibility();

  window.storyManager.isReady = true;
  window.storyManager.isLoading = false;




getParagraphEndRegex() => return /[.。．！!？?ー":*»’”—–。]$/;

getNumParagraphsToPutInStartWith() =>
  // The <writing_instructions> are always put at the *end* of the instruction to the AI. We want to bias towards "injecting" those writing instructions at the point in the story *where the user actually wrote them*, so we move (up to) a few paragraphs into the startWith to do that.
  let marker = window.whatShouldHappenNextInsertionPointText;
  if(!marker || typeof marker !== "string") return 0;
  let storyText = (storySoFarEl?.value || "");
  if(!storyText) return 0;
  let index = storyText.lastIndexOf(marker);
  if(index < 0) return 0;
  let after = storyText.slice(index + marker.length);
  let nextDoubleNewline = after.indexOf("\n\n");
  if(nextDoubleNewline < 0) return 0;
  let remainder = after.slice(nextDoubleNewline + 2);
  if(!remainder) return 0;
  let count = remainder.split(/\n{2,}/).map(p => p.trim()).filter(Boolean).length;
  return Math.min(3, count);
getStorySoFarForInstruction() =>
  let paras = (window.preprocessedStorySoFarText || "").trim().split(/\n{2,}/).map(p => p.trim()).filter(Boolean);
  const storyEndsCleanly = getParagraphEndRegex().test((window.preprocessedStorySoFarText || "").trim());
  // If story doesn't end cleanly, always exclude last paragraph (it goes in startWith)
  // Otherwise, exclude based on numParagraphsToPutInStartWith setting
  const numToExclude = storyEndsCleanly ? getNumParagraphsToPutInStartWith() : 1;
  if(numToExclude === 0) return paras.join("\n\n").trim();
  return paras.slice(0, Math.max(0, paras.length - numToExclude)).join("\n\n").trim();
getStartWithText() =>
  let paras = (storySoFarEl.value || "").trim().split(/\n{2,}/).map(p => p.trim()).filter(Boolean);
  if(window.continueMode === "inline") return paras.slice(-1).join("\n\n").trim();
  const storyEndsCleanly = getParagraphEndRegex().test(storySoFarEl.value.trim());
  if(!storyEndsCleanly) return paras.slice(-1).join("\n\n").trim(); // If story doesn't end cleanly, always include last paragraph so AI can continue mid-sentence
  let numToPutInStartWith = getNumParagraphsToPutInStartWith();
  if(numToPutInStartWith === 0) return "\n\n";
  let startWithText = paras.slice(-numToPutInStartWith).join("\n\n").trim();
  return startWithText + "\n\n";


async continueStory(opts) =>

  window.currentlyGenerating = true;
  window.storySoFarAutoScrollEnabled = (storySoFarEl.scrollHeight - storySoFarEl.scrollTop - storySoFarEl.offsetHeight) < 65;

  continueTextBtn.disabled = true;
  storySoFarEl.spellcheck = false;
  
  // important for summary stuff, since we need "normalized" paragraphs so that summary replacements/injections work properly:
  antiAntiLayoutJank(() => { storySoFarEl.value = storySoFarEl.value.split(/\n{2,}/).map(p => p.trim()).join("\n\n"); if(window.storySoFarAutoScrollEnabled) storySoFarEl.scrollTop = storySoFarEl.scrollHeight; });

    antiAntiLayoutJank(() => storySoFarEl.value = storySoFarEl.value.trim() + "\n\n"); // then we add a couple of new lines to the end, ready for the next paragraph that's about to be generated
  }
  if(opts.continueInline) {
    antiAntiLayoutJank(() => storySoFarEl.value = storySoFarEl.value.trim());
    window.continueMode = "inline";
  } else {
    window.continueMode = "normal";
  }
  
  window.preprocessedStorySoFarText = storySoFarEl.value;
  try {
    // get a version of the message feed with hierarchical summaries swapped in:
    let messagesWithSummaryReplacements = getMessagesWithSummaryReplacements(storySoFarEl.value);

    window.preprocessedStorySoFarText = messagesWithSummaryReplacements.join("\n\n").trim();

    console.error("Falling back to using *all* messages because there was an error while trying to compute messagesWithSummaryReplacements:", e);

    window.preprocessedStorySoFarText = storySoFarEl.value;
  }
  
  window.storyTextBeforeLastGeneration = storySoFarEl.value;
  window.lastGenerationStreamObj = ai(storyWritingPrompt); // we put it into a 'global' variable so that we can use it in the 'onclick' of the stop button to stop the text generation
  
  loaderEl.innerHTML = window.lastGenerationStreamObj.loadingIndicatorHtml; 
  
  let data = await window.lastGenerationStreamObj;
  
  loaderEl.innerHTML = "";
  
  window.paragraphLimitOverrideForGeneration = null;
  
  window.currentlyGenerating = false;
  
  if(localStorage.generateCount === undefined || isNaN(Number(localStorage.generateCount))) localStorage.generateCount = "0";
  localStorage.generateCount = Number(localStorage.generateCount) + 1;
  if(!localStorage.firstGenerateTime) localStorage.firstGenerateTime = Date.now();
  try { renderStoriesFooterNotices(); } catch(e) {}
  updateLastParagraphButtonsDisplayIfNeeded();
  
  if(data.stopReason !== "error" && !window.userClickedStop) {
    enableRatingButtons();
  }
  
  continueTextBtn.disabled = false;
  continueTextBtn.style.visibility = "visible";
  storySoFarEl.spellcheck = true;
  
  let generateCount = Number(localStorage.generateCount);
  if(generateCount > 20 && !localStorage.haveUsedTabToContinueMessage) {
    let isTouchScreen = false;
    try { isTouchScreen = window.matchMedia("(pointer: coarse)").matches; } catch(e) { console.error(e); }
    if(window.innerWidth > window.innerHeight && !isTouchScreen) {
      continueTextBtnTabLabel.style.display = "";
    }
  }
  if(generateCount > 5) {

async compressBlobWithGzip(blob) =>
  const cs = new CompressionStream('gzip');
  const compressedStream = blob.stream().pipeThrough(cs);
  let outputBlob = await new Response(compressedStream).blob();
  return new Blob([outputBlob], { type: "application/gzip" }); // <-- to add the correct mime type

async loadDataFromUrlHash() => 
  let success = false;
  if(!window.DecompressionStream) {
    alert("Story share links use a browser feature that's only available in modern browsers. Please upgrade your browser to the latest version to allow for loading data from story share links.");
    return {success, error:"browser_compat"};
  }
  let proceed = await confirmAsync("You're about to load a shared story. If you already have stories saved in this browser, you can access them via the 📚 stories button.\n\nContinue loading this shared story?");
  if(!proceed) {
    try { window.location.hash = ""; } catch(e) {}
    return {success:false, error:"loading_cancelled_by_user"};
  }
  
  let loadingModal = document.createElement('div');
  loadingModal.innerHTML = `<div style="position: fixed; top: 0; left: 0; width: 100%; height: 100%; background-color: rgba(0, 0, 0, 0.5); z-index: 9999; display: flex; justify-content: center; align-items: center;">
    <div style="padding: 20px; background-color: var(--box-color); border-radius: 8px;">
      ⏳ loading story data...
    </div>
  </div>`;
  loadingModal = loadingModal.firstElementChild;
  document.body.append(loadingModal);
  
  try {
    let hashText = window.location.hash.slice(1);
    if(!hashText.startsWith("data=")) {
      throw new Error("Invalid share URL.");
    }
    let fileUrl = hashText.replace(/^data=/, "");
    if(/^uup1(:|%3A)/.test(fileUrl)) {
      fileUrl = fileUrl.replace(/^uup1(:|%3A)/, "https://user.uploads.dev/file/");
    }
    let fetchOptions = {};
    if(window.AbortSignal && AbortSignal.timeout) fetchOptions.signal = AbortSignal.timeout(10000);
    let blob = await fetch(fileUrl, fetchOptions).then(res => res.ok ? res.blob() : null).catch(console.error);
    if(!blob) {
      loadingModal.remove();
      await confirmAsync(`It seems you've tried to load a share URL, but the file specified by the URL does not exist. If you believe it should exist, you can ask for help on the community forum, or check if the file has been quarantined:\n\nperchance.org/quarantined-files`, {hideCancel:true});
      return {success:false, error:"loading_cancelled_by_user"};
    }
    let text;
    if(fileUrl.endsWith(".gz")) {
      let decompressedBlob = await decompressBlobWithGzip(blob);
      text = await decompressedBlob.text();
    } else {
      text = await blob.text();
    }
    let data = JSON.parse(text);
    if(data.format === "perchance-ai-story-v1") {
      try {
        // Import as new story (safer than overwriting current story in multi-story world)
        let title = extractAndStripTitleFromImportedStoryData(data, "Imported story");
        await createNewStory({title, data:normalizeStoryData(data)});
      } catch(e) {
        alert("Failed to import story into your browser storage. Please report this issue using the feedback button, including the story length (word count) and your device/browser.");
        console.error(e);
      }
      success = true;
    } else {
      alert("Unknown story data format.");
    }
  } catch(e) {
    alert(`Failed to load story data: ${e.message}`);
    console.error(e);
  }
  
  loadingModal.remove();
  return {success};
  
async decompressBlobWithGzip(blob) =>

  const decompressedStream = blob.stream().pipeThrough(ds);
  return await new Response(decompressedStream).blob();

async exportAllStories(opts) =>
  if(!opts) opts = {};
  let bundle = await getAllStoriesBundle();
  let filenameBase = `perchance-stories-${new Date().toISOString().slice(0,10)}`;
  let blob;
  if(opts.format === "folders") {
    let zip = await ensureZipJs();
    let writer = new zip.ZipWriter(new zip.BlobWriter("application/zip"));
    let usedFolderNames = new Set();
    for(let entry of bundle.storiesMeta) {
      let data = bundle.stories[entry.id];
      let folderPath = normalizeFolderPath(entry.folderPath || "");
      let safeSegments = folderPath ? folderPath.split("/").map(s => s.replace(/[\\/:*?"<>|]/g, "_")) : [];
      let safeTitle = (entry.title || "Untitled story").replace(/[\\/:*?"<>|]/g, "_").slice(0, 80) || "Untitled story";
      let baseFolder = `${safeTitle} (${entry.id})`;
      let storyFolder = [...safeSegments, baseFolder].join("/");
      if(usedFolderNames.has(storyFolder)) storyFolder += "-" + Math.random().toString(36).slice(2,6);
      usedFolderNames.add(storyFolder);
      let storyWithoutSummaries = (data.storySoFar || "").split(/\n{2,}/).map(p => p.trim()).filter(p => !p.startsWith("SUMMARY^")).join("\n\n");
      let metaJson = {
        meta: entry,
        storyData: data,
      };
      await writer.add(`${storyFolder}/overview.txt`, new zip.TextReader(data.storyOverview || ""));
      await writer.add(`${storyFolder}/story.txt`, new zip.TextReader(data.storySoFar || ""));
      await writer.add(`${storyFolder}/story_without_summaries.txt`, new zip.TextReader(storyWithoutSummaries));
      await writer.add(`${storyFolder}/meta.json`, new zip.TextReader(JSON.stringify(metaJson, null, 2)));
    }
    blob = await writer.close();
    filenameBase += ".stories.zip";
  } else {
    let CBOR = await ensureCBOR();
    if(!CBOR) { alert("CBOR library failed to load."); return; }
    let cborBytes = CBOR.encode(bundle);
    let cborBlob = new Blob([cborBytes], { type: 'application/cbor' });
    let compressedCborBlob = await new Response(cborBlob.stream().pipeThrough(new CompressionStream('gzip'))).blob();
    blob = new Blob([compressedCborBlob], { type: "application/gzip" });
    filenameBase += ".cbor.gz";
  }
  
  if(opts.encrypt) {
    if(!opts.password) {
      alert("Please enter an encryption password.");
      return;
    }
    let zip = await ensureZipJs();
    let writer = new zip.ZipWriter(new zip.BlobWriter("application/zip"), { password: opts.password });
    await writer.add(filenameBase, new zip.BlobReader(blob));
    let encryptedBlob = await writer.close();
    blob = encryptedBlob;
    filenameBase = filenameBase.replace(/\.[^.]+$/, "") + ".encrypted.zip";
  }
  return { blob, filename: filenameBase };

async importStoriesFromBlob(blob, opts) =>
  if(!opts) opts={};
  if(blob.name?.toLowerCase().endsWith(".json") || blob.type === "application/json" || blob.type === "text/json" || blob.type === "text/plain") {
    try {
      let text = await blob.text();
      let data = JSON.parse(text);
      if(data && data.format === "perchance-ai-story-v1") {
        let title = extractAndStripTitleFromImportedStoryData(data, "Imported story");
        return createNewStory({title, data:normalizeStoryData(data)});
      }
      if(data && data.format === "perchance-ai-stories-bundle-v1") {
        let bundles = Object.keys(data.stories || {}).map(id => ({meta: (data.storiesMeta || []).find(m => m.id == id) || {}, storyData: data.stories[id]}));
        if(Array.isArray(data.folderMeta)) {
          mergeFolderMeta(data.folderMeta);
        }
        return importStoryBundles(bundles);
      }
      if(data && data.storyData) {
        return importStoryBundles([{meta: data.meta || {}, storyData: data.storyData}]);
      }
    } catch(e) {
      console.error(e);
    }
  }
  if(blob.name?.toLowerCase().endsWith(".zip") || blob.type === "application/zip") {
    let zip = await ensureZipJs();
    let entries = null;
    try {
      let reader = new zip.ZipReader(new zip.BlobReader(blob));
      entries = await reader.getEntries();
      let needsPassword = entries.some(e => e.encrypted);
      await reader.close();
      if(needsPassword) {
        let password = prompt("This zip is encrypted. Enter password:");
        if(password === null) return;
        reader = new zip.ZipReader(new zip.BlobReader(blob), { password });
        entries = await reader.getEntries();
        await reader.close();
      }
    } catch(e) {
      let password = prompt("This zip may be encrypted. Enter password:");
      if(password === null) return;
      let reader = new zip.ZipReader(new zip.BlobReader(blob), { password });
      entries = await reader.getEntries();
      await reader.close();
    }
    if(entries && entries.length === 1 && entries[0].filename.toLowerCase().endsWith(".zip")) {
      let innerBlob = await entries[0].getData(new zip.BlobWriter());
      return importStoriesFromBlob(innerBlob);
    }
    if(entries) {
      return importStoriesFromZipEntries(entries);
    }
  }
  return importStoriesFromNativeBlob(blob);

async importStoriesFromZipEntries(entries) =>
  let zip = await ensureZipJs();
  let metaEntries = entries.filter(e => e.filename.toLowerCase().endsWith("meta.json"));
  let cborEntry = entries.find(e => /\.cbor(\.gz)?$/i.test(e.filename));
  if(cborEntry) {
    let cborBlob = await cborEntry.getData(new zip.BlobWriter());
    return importStoriesFromNativeBlob(cborBlob);
  }
  if(metaEntries.length > 0) {
    let storyBundles = [];
    for(let metaEntry of metaEntries) {
      let metaBlob = await metaEntry.getData(new zip.BlobWriter());
      let metaText = await metaBlob.text();
      let parsed = JSON.parse(metaText);
      let storyData = parsed.storyData || parsed;
      storyBundles.push({meta: parsed.meta || {}, storyData});
    }
    return importStoryBundles(storyBundles);
  }
  alert("No recognizable story data found in that zip.");

async importStoriesFromNativeBlob(blob) =>
  let CBOR = await ensureCBOR();
  if(!CBOR) { alert("CBOR library failed to load."); return; }
  let bytes = new Uint8Array(await blob.arrayBuffer());
  let isGzip = blob.name?.toLowerCase().endsWith(".gz") || blob.type === "application/gzip";
  if(bytes.length >= 2 && bytes[0] === 0x1f && bytes[1] === 0x8b) isGzip = true;
  if(isGzip) {
    const ds = new DecompressionStream("gzip");
    const decompressedStream = new Blob([bytes]).stream().pipeThrough(ds);
    bytes = new Uint8Array(await new Response(decompressedStream).arrayBuffer());
  }
  let data = CBOR.decode(bytes);
  if(data && data.format === "perchance-ai-stories-bundle-v1") {
    let bundles = Object.keys(data.stories || {}).map(id => ({meta: (data.storiesMeta || []).find(m => m.id == id) || {}, storyData: data.stories[id]}));
    if(Array.isArray(data.folderMeta)) {
      mergeFolderMeta(data.folderMeta);
    }
    return importStoryBundles(bundles);
  }
  if(data && data.format === "perchance-ai-story-v1") {
    let title = extractAndStripTitleFromImportedStoryData(data, "Imported story");
    return createNewStory({title, data:normalizeStoryData(data)});
  }
  alert("Unknown import format.");

mergeFolderMeta(folderMeta) =>
  ensureStoryManager();
  let existing = window.storyManager.folderMeta || [];
  let map = new Map(existing.map(f => [normalizeFolderPath(f.path || ""), f]));
  for(let f of folderMeta) {
    let path = normalizeFolderPath(f.path || "");
    if(!path) continue;
    map.set(path, {path, emoji: f.emoji || ""});
  }
  window.storyManager.folderMeta = [...map.values()];
  requestFolderMetaSaveDebounced({debounceMs:200});

async importStoryBundles(bundles) =>
  ensureStoryManager();

  let confirmed = await confirmAsync(`This will import ${bundles.length} story${bundles.length===1?"":"ies"} into your stories list. Continue?`);
  if(!confirmed) return;
  for(let b of bundles) {
    let data = normalizeStoryData(b.storyData || {});
    let title = (b.meta?.title || "Imported story");
    let folderPath = normalizeFolderPath(b.meta?.folderPath || "");
    let pinned = !!b.meta?.pinned;
    await createNewStory({title, data, folderPath, pinned});
  }
  renderStoriesList();

setupNotepadResizer() =>
  if(window.notepadResizerSetup) return;
  if(typeof notepadResizeHandle === "undefined" || typeof notepadPanel === "undefined") return;
  window.notepadResizerSetup = true;
  notepadResizeHandle.addEventListener("pointerdown", e => {
    e.preventDefault();
    let startX = e.clientX;
    let startWidth = notepadPanel.getBoundingClientRect().width;
    notepadResizeHandle.setPointerCapture(e.pointerId);
    function onMove(ev) {
      let delta = startX - ev.clientX;
      let nextWidth = Math.max(220, startWidth + delta);
      nextWidth = Math.min(nextWidth, Math.max(260, window.innerWidth - 160));
      document.documentElement.style.setProperty("--notepad-width", `${Math.round(nextWidth)}px`);
    }
    function onUp(ev) {
      notepadResizeHandle.releasePointerCapture(ev.pointerId);
      window.removeEventListener("pointermove", onMove);
      window.removeEventListener("pointerup", onUp);
      let width = parseInt(getComputedStyle(document.documentElement).getPropertyValue("--notepad-width")) || notepadPanel.getBoundingClientRect().width;
      localStorage.notepadWidth = Math.round(width);
    }
    window.addEventListener("pointermove", onMove);
    window.addEventListener("pointerup", onUp);
  });

    let originalPlaceholder = urlInputEl.placeholder;
    buttonEl.textContent = "⏳";

    } finally {
      buttonEl.textContent = originalText;
      buttonEl.disabled = false;
      urlInputEl.placeholder = originalPlaceholder;
    }
  };
  input.click();

normalizeColorScheme(scheme) =>
  return (scheme === "dark") ? "dark" : "light";

  let scheme = document.documentElement.style.colorScheme;
  if(scheme !== "dark" && scheme !== "light") {
    scheme = (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) ? "dark" : "light";
  }
  return normalizeColorScheme(scheme);

  scheme = normalizeColorScheme(scheme);
  localStorage.forceColorScheme = scheme;
  setColorScheme(scheme);
  let systemColorScheme = window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches ? "dark" : "light";
  if(systemColorScheme === scheme) {

  }

loadAppThemeSettingsByScheme() =>
  let raw = localStorage.themeSettingsByScheme || "";
  let settings = getThemeDefaultsByScheme();
  if(raw) {
    try { settings = normalizeThemeSettingsByScheme(JSON.parse(raw), {allowNull:false}); } catch(e) { console.error(e); }
  }
  for(let scheme of ["light","dark"]) {
    let entry = settings[scheme];
    if(entry?.storyFontFamily) entry.storyFontFamily = parseFontFamilyInput(entry.storyFontFamily);
  }
  window.themeSettingsByScheme = settings;
  return settings;

parseFontFamilyInput(raw) =>
  let val = (raw || "").toString().trim();
  if(!val) return "";
  return val.replace(/^https?:\/\//i, "");

extractFontNameFromInput(raw) =>
  let val = (raw || "").toString().trim();
  if(!val) return "";
  val = val.replace(/^https?:\/\//i, "");
  let specimenMatch = val.match(/fonts\.google\.com\/specimen\/([^?#/]+)/i);
  if(specimenMatch && specimenMatch[1]) {
    let name = specimenMatch[1].replace(/\+/g, " ");
    try { name = decodeURIComponent(name); } catch(e) {}
    return name.trim();
  }
  let familyMatch = val.match(/[?&]family=([^&#]+)/i);
  if(familyMatch && familyMatch[1]) {
    let name = familyMatch[1].split(":")[0].replace(/\+/g, " ");
    try { name = decodeURIComponent(name); } catch(e) {}
    return name.trim();
  }
  return val;

buildThemeAdvancedTextRules(rules) =>
  let output = [];
  if(!Array.isArray(rules)) return output;
  for(let r of rules) {
    let pattern = (r?.pattern || "").toString().trim();
    let css = (r?.css || "").toString().trim();
    if(!pattern || !css) continue;
    let regex = null;
    if(/^\/.+\/[gimsuy]*$/.test(pattern)) {
      let lastSlash = pattern.lastIndexOf("/");
      let body = pattern.slice(1, lastSlash);
      let flags = pattern.slice(lastSlash+1) || "g";
      if(!flags.includes("g")) flags += "g";
      try { regex = new RegExp(body, flags); } catch(e) { regex = null; }
    } else {
      let escaped = pattern.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
      try { regex = new RegExp(escaped, "g"); } catch(e) { regex = null; }
    }
    if(!regex) continue;
    output.push({match: regex, style: css});
  }
  return output;

async generateThemeRuleWithAi(rowEl, btnEl) =>
  if(!rowEl) return;
  let originalBtnText = btnEl?.textContent;
  if(btnEl) { btnEl.textContent = "⏳"; btnEl.disabled = true; }
  let description = prompt("Describe the rule you want, e.g. \"make the word Alice appear bold\".");
  if(!description) { if(btnEl) { btnEl.textContent = originalBtnText || "⏳"; btnEl.disabled = false; } return; }
  try {
    let result = await ai({
      instruction: [
        "Convert the user's description into a JSON object with keys:",
        `"pattern": a literal text or a /regex/flags string (include 'g' flag),`,
        `"css": CSS text styling rules like "color:#d33; font-weight:bold;"`,
        "Return ONLY JSON with those two keys and no extra text.",
        "",
        `User description: ${description}`,
      ].join("\n"),
      stopSequences: ["\n\n"],
    });
    if(result.stopReason === "error") {
      alert("Failed to generate rule.");
      return;
    }
    let text = (result.generatedText || "").trim();
    let jsonText = text.match(/\{[\s\S]*\}/)?.[0] || "";
    if(!jsonText) {
      alert("Could not parse the generated rule.");
      return;
    }
    let parsed = JSON.parse(jsonText);
    let pattern = (parsed.pattern || "").toString().trim();
    let css = (parsed.css || "").toString().trim();
    if(!pattern || !css) {
      alert("Generated rule was incomplete.");
      return;
    }
    rowEl.querySelector(".themePatternInput").value = pattern;
    rowEl.querySelector(".themeCssInput").value = css;
    ensureThemeAdvancedRowCount();
  } catch(e) {
    console.error(e);
    alert("Failed to generate rule.");
  } finally {
    if(btnEl) { btnEl.textContent = originalBtnText || "⏳"; btnEl.disabled = false; }
  }

addAlphaToColor(color, alpha) =>
  let hex = color.trim();
  if(/^#([0-9a-f]{3}){1,2}$/i.test(hex)) {
    if(hex.length === 4) hex = "#" + hex[1]+hex[1]+hex[2]+hex[2]+hex[3]+hex[3];
    let r = parseInt(hex.slice(1,3),16);
    let g = parseInt(hex.slice(3,5),16);
    let b = parseInt(hex.slice(5,7),16);
    return `rgba(${r},${g},${b},${alpha})`;
  }
  if(/^rgb\(/i.test(color)) return color.replace(/^rgb\(/i, "rgba(").replace(/\)$/, `,${alpha})`);
  if(/^rgba\(/i.test(color)) return color.replace(/,[^,]+\)$/, `,${alpha})`);
  return color;

getMessagesWithSummaryReplacements(text, opts) =>
  if(!opts) opts = {};
  const minimumMessageLevel = opts.minimumMessageLevel || 0; // used by the summarization process.
  
  let messages = text.split("\n\n").map(m => m.trim()).filter(m => m);
  let messagesWithSummaryReplacements = [];
  let highestLevelSeen = 0;
  
  // it's we go backwards through the messages, and only 'collect' a message if its level is not below the highest level we've seen so far. it makes sense if you think about it for a bit.
  // said another way, we go from the end of the messages to the start while 'monotonically climbing' up a level whenever we hit a 'higher' message.
  while(messages.length > 0) {
    let m = messages.pop();
    let level = Number((m.match(/SUMMARY\^([0-9]+):/)||[])[1] || 0);
    if(level < minimumMessageLevel) continue;
    if(level >= highestLevelSeen) {
      messagesWithSummaryReplacements.unshift(m);
      highestLevelSeen = level;
    }
  }
  return messagesWithSummaryReplacements;

injectSummariesAndComputeNextSummariesInBackgroundIfNeeded() =>
  // Make summaries per-story so they don't leak across story switching:
  let storyId = window.currentStoryId ?? (window.storyManager?.currentId ?? 0);
  if(!window.summariesReadyToInjectByStoryId) window.summariesReadyToInjectByStoryId = {};
  if(!window.summariesReadyToInjectByStoryId[storyId]) window.summariesReadyToInjectByStoryId[storyId] = [];
  if(!window.alreadyDoingSummaryByStoryId) window.alreadyDoingSummaryByStoryId = {};
  
  let summariesReadyToInject = window.summariesReadyToInjectByStoryId[storyId];
  
  // inject summaries if we have any:
  if(summariesReadyToInject.length > 0) {
    // ensure logs are normalized so our message comparison checks work:
    let allMessagesOriginal = storySoFarEl.value.split(/\n{2,}/g).map(m => m.trim()).filter(m => m);
    let allMessagesNew = allMessagesOriginal.slice(0);
    for(let {summarizedMessages, lastMessageSummarizedIndex, summary, level} of summariesReadyToInject) {
      let lastSummarizedMessage = summarizedMessages[summarizedMessages.length-1];
      if(allMessagesOriginal[lastMessageSummarizedIndex] === lastSummarizedMessage) {
        allMessagesNew.splice(lastMessageSummarizedIndex + 1, 0, `SUMMARY^${level}: ${summary}`);
      } else {
        console.warn("Content of last-summmarized-message doesn't match content of message at lastMessageSummarizedIndex. Safe to ignore this warning if logs have been edited since last 'send' button click. This summary will simply be discarded and we'll compute a new one with the up-to-date story text.");
      }
    }
    storySoFarEl.value = allMessagesNew.join("\n\n");
    window.summariesReadyToInjectByStoryId[storyId] = [];
    summariesReadyToInject = window.summariesReadyToInjectByStoryId[storyId];
  }
  
  const { countTokens, idealMaxContextTokens } = ai({getMetaObject:true});
  
  const contextLengthToIdeallyStayUnder = idealMaxContextTokens*0.88;
  const numCharsToSummarizeAtATime = 1500; // don't make this bigger without testing
  
  // must get text synchronously, since storySoFarEl can be temporarily ablated during streaming for rendering performance.
  const storySoFarElText = storySoFarEl.value;
  const messagesWithSummaryReplacements = getMessagesWithSummaryReplacements(storySoFarElText);
  
  let currentlyUsedContextLength = countTokens(messagesWithSummaryReplacements.join("\n\n") + storyOverviewEl.value);
  if(currentlyUsedContextLength < contextLengthToIdeallyStayUnder) {
    console.log(`Summarization not needed. currentlyUsedContextLength=${currentlyUsedContextLength} which is less than ${contextLengthToIdeallyStayUnder}`);
    return;
  }
      
  // compute next summary in background if needed:
  (async function() {
    let thisStoryId = storyId;
    if(window.alreadyDoingSummaryByStoryId[thisStoryId]) return;
    try {
      window.alreadyDoingSummaryByStoryId[thisStoryId] = true;
      
      const allMessageObjs = storySoFarElText.split(/\n{2,}/).map(m => m.trim()).filter(m => m).map((text, i) => {
        return {
          text, // note that this `text` is trimmed in the `map` above - very important that we do this kind of normalization for summary replacement stuff, since we do actual string-match replacement.
          index: i,
          level: Number((text.match(/SUMMARY\^([0-9]+):/)||[])[1] || 0)
        };
      });

      // conceptually we treat each "level" just like the first.
      // the first level is just a bunch of messages with interspersed "SUMMARY^1: ..." messages, where the summary messages are a summary of the messages before them, up to the *previous* "SUMMARY^1: ..." message.
      // so for the next level, we just delete/ignore the ^0 messages (i.e. the *actual* messages), and do exactly the same thing - i.e. treat "SUMMARY^1: ..." as if they were "messages" and "SUMMARY^2: ..." are the summaries of those "messages".
      let summaryLevelToMessageBlocks = new Map();
      let summaryLevelBeingProcessed = 1;
      while(1) {
        // grab messages that are relevant to this 'level' (i.e. only this level and lower one):
        const thisLevelAndPreviousLevelMessageObjs = allMessageObjs.filter(m => m.level === summaryLevelBeingProcessed || m.level === summaryLevelBeingProcessed-1);
        
        if(thisLevelAndPreviousLevelMessageObjs.length === 0) {
          console.log("Finished creating summaryLevelToMessageBlocks.");
          break;
        }

        // get all summary 'blocks' (i.e. groups of messages ending with a summary message of this `level` that summarizes them, except for final block which doesn't have a summary at the end)
        const blocks = [];
        let currentBlock = [];
        currentBlock.globalMessageIndices = [];
        for(let m of thisLevelAndPreviousLevelMessageObjs) {
          currentBlock.push(m.text);
          currentBlock.globalMessageIndices.push(m.index); // this is for use in determining summary injection/placement
          if(m.level === summaryLevelBeingProcessed) {
            blocks.push(currentBlock);
            currentBlock = [];
            currentBlock.globalMessageIndices = [];
          }
        }
        if(summaryLevelBeingProcessed === 1 && currentBlock.length === 0) {
          console.warn("final block for summaryLevel==1 should have messages? if it doesn't, then we're maybe summarizing too close to the end of the story text?");
        }
        blocks.push(currentBlock); // final block doesn't have a summary at the end
        summaryLevelToMessageBlocks.set(summaryLevelBeingProcessed, blocks);
        
        summaryLevelBeingProcessed++;
      }
      
      const summaryLevelBlockEntries = [...summaryLevelToMessageBlocks.entries()].sort((a,b) => a[0]-b[0]);
      for(let [summaryLevel, blocks] of summaryLevelBlockEntries) {
        // discard if story switched
        if((window.currentStoryId ?? window.storyManager?.currentId) !== thisStoryId) {
          console.log("Discarding summary work because story changed.");
          return;
        }

        // note: a block is just an array of messages, and all of them have a summary message (i.e. higher-level message) at the end EXCEPT the last block - we're in the process of adding that summary message here.
        // but also note: the block has a globalMessageIndices property which is also an array (see above)
        let messagesToSummarizeFromFinalBlock = blocks[blocks.length-1];

        // note that we can use numCharsToSummarizeAtATime here even for the first level without worrying about summarizing too close to the end of the story because we have a currentlyUsedContextLength check before running this summarization process.
        let numCharsInFinalBlock = messagesToSummarizeFromFinalBlock.reduce((a,v) => a+v.length, 0);
        if(numCharsInFinalBlock < numCharsToSummarizeAtATime) { 
          console.log(`summaryLevel=${summaryLevel} doesn't need summarizing yet. numCharsInFinalBlock=${numCharsInFinalBlock}`);
          continue;
        }

        // remove messages from last block (which contains all messages after the last summary) until it's a good size for summarization:
        while(1) {
          if(messagesToSummarizeFromFinalBlock.length <= 2) break;
          let numChars = messagesToSummarizeFromFinalBlock.reduce((a,v) => a+v.length, 0);
          if(numChars < numCharsToSummarizeAtATime) break;

          // to speed things up, drop latter half if it's way too big:
          if(numChars > numCharsToSummarizeAtATime*10) {
            let halfOfMessagesCount = Math.floor(messagesToSummarizeFromFinalBlock.length/2);
            for(let j = 0; j < halfOfMessagesCount; j++) {
              messagesToSummarizeFromFinalBlock.pop();
              messagesToSummarizeFromFinalBlock.globalMessageIndices.pop();
            }
          } else {
            messagesToSummarizeFromFinalBlock.pop();
            messagesToSummarizeFromFinalBlock.globalMessageIndices.pop(); // this is an array of indices aligned with the messages array, for detemining summary injection location
          }
        }
  
        if(messagesToSummarizeFromFinalBlock.length === 0) {
          console.error("No messages to summarize??");
          continue;
        }
  
        // ensure per-story summariesReadyToInject exists
        if(!window.summariesReadyToInjectByStoryId[thisStoryId]) window.summariesReadyToInjectByStoryId[thisStoryId] = [];
  
        let existingSummary = window.summariesReadyToInjectByStoryId[thisStoryId].filter(s => s.summarizedMessages.join("\n\n") === messagesToSummarizeFromFinalBlock.join("\n\n"))[0];
        if(existingSummary) {
          console.error("Existing summary hasn't been injected yet?? Should have happened before this code ran.");
          return;
        }

        // Note: It may seem brittle to choose an *index* to inject the summary at, but we also check to ensure the previous message matches.
        // And if the text has since been edited, that's fine - the summary just gets thrown away and we re-do it next time the send button is clicked.
        let lastMessageSummarizedIndex = messagesToSummarizeFromFinalBlock.globalMessageIndices[messagesToSummarizeFromFinalBlock.length-1];
        if(messagesToSummarizeFromFinalBlock.globalMessageIndices.length !== messagesToSummarizeFromFinalBlock.length) { console.error("should be one index per message"); return; }
        
        let exampleBlocksForStartWith = blocks.slice(-3, -1);
        let exampleBlockSummaries = exampleBlocksForStartWith.map(b => b[b.length-1]);

        // note that we can't just remove the last two instruction summaries here - they aren't necessarily the same as the summaries from the `exampleBlocksForStartWith` because they may have been 'compressed' into a higher level, so there can actually be no overlap at all.
        // so we need to pop the instructionSummaries off based on the ones that are actually in the example blocks:
        let instructionSummaries = getMessagesWithSummaryReplacements(storySoFarElText, {minimumMessageLevel:summaryLevel});
        
        // note that we can't just remove the last two instruction summaries here - they aren't necessarily the same as the summaries from the `exampleBlocksForStartWith` because they may have been 'compressed' into a higher level, so there can actually be no overlap at all.
        // so we need to pop the instructionSummaries off based on the ones that are actually in the example blocks:
        while(1) {
          if(instructionSummaries.length === 0) break;
          if(exampleBlockSummaries.includes(instructionSummaries[instructionSummaries.length-1])) {
            instructionSummaries.pop();
            continue;
          }
          break;
        }
        instructionSummaries = instructionSummaries.map(m => m.replace(/SUMMARY\^[0-9]+:/, "").trim());
        
        let startWithBlocks = exampleBlocksForStartWith.map((block) => ({messages:block.slice(0, -1), summary:block.slice(-1)[0]}));
        startWithBlocks.push({messages:messagesToSummarizeFromFinalBlock, summary:""});
        
        if(messagesToSummarizeFromFinalBlock.join("\n").replaceAll(`SUMMARY^${summaryLevel-1}:`, "").includes("SUMMARY^")) {
          console.error("Should have only been summaryLevel-1 summaries in messagesToSummarizeText. messagesToSummarizeFromFinalBlock:", messagesToSummarizeFromFinalBlock);
        }
        
        let startWith = startWithBlocks.map(({messages, summary}, blockI) => {
          let letterLabel = "";
          if(blockI===0) letterLabel = "[A]";
          if(blockI===1) letterLabel = "[B]";
          if(blockI===2) letterLabel = "[C]";
  
          let messagesText = messages.map((message, mi) => {
            message = message.replace(`SUMMARY\^${summaryLevel-1}:`, "").replace(`SUMMARY\^${summaryLevel}:`, "").replace(/\n/g, " ").trim();
            return `${summaryLevel === 1 ? `(${mi+1}) ` : ""}${message}`;  // we prefix bottom-level messages with numbers, but not SUMMARY^N messages.
          }).join(" ");
          
          summary = summary.replace(`SUMMARY\^${summaryLevel-1}:`, "").replace(`SUMMARY\^${summaryLevel}:`, "").replace(/\n/g, " ").trim();
  
          return `>>> FULL TEXT of ${letterLabel}: ${messagesText}\n>>> SUMMARY of ${letterLabel}: ${summary}`;
        }).join("\n---\n"); // NOTE: This joiner must match the summaryPromptInstruction

        // since possible for there to be no blocks before the messages to summarize
        startWith = startWith.trim(); // this is also important to prevent whitespace at end of startWith
        // try to fix 'over-compression' bug (see 'TIP' in instruction):
        startWith = startWith.trim().slice(0, -1) + " (full, natural, readable sentences with correct grammar):";
  
        window.summaryMessagesForInstruction = instructionSummaries.length > 0 ? instructionSummaries : ["(None.)"]; // used in summaryPromptInstruction
        let instruction = root.summaryPromptInstruction.evaluateItem;
        window.summaryMessagesForInstruction = null;
  
        let promptOptions = {
          instruction,
          startWith,
          stopSequences: ["\n\n", "\n---", ">>> FULL TEXT", "FULL TEXT"],
        };
  
        // discard if story switched just before LLM call
        if((window.currentStoryId ?? window.storyManager?.currentId) !== thisStoryId) return;
  
        let data = await root.ai(promptOptions);
        if(data.stopReason === "error") continue; // could retry a few times, but this is no big deal, since every message sent triggers another attempt
        
        let summary = data.generatedText.trim().replace(/\n+/g, " ").trim().replace(/---$/, "").replace(">>> FULL TEXT", "").replace("FULL TEXT", "").trim();
        if(!summary.trim() || (instructionSummaries[instructionSummaries.length-1] || "").trim() === summary.trim()) {
          console.warn("AI copied previous summary or gave blank summary. Skipping this summary level for this 'round'. Summary:", summary);
          continue;
        }

        // Automatically fix repetition within summaries:
        if(summary.split(summary.slice(-30)).length > 5) { // detect repetition of the final 30 chars many times, earlier in the text
          console.warn("Repeition detected within summary. Fixing...");
          let result = await root.ai({
            instruction: [
              `Does the following story summary snippet shown within <story_summary_snippet>...</story_summary_snippet> include erroneous/unnecessary repetition? If so, respond with fixed, repetition-free text within <fixed_story_summary_snippet>...</fixed_story_summary_snippet>. If the text is fine, then just respond with exactly 'no_repetition', and nothing more.`,
              ``,
              `<story_summary_snippet>`,
              `${summary}`,
              `</story_summary_snippet>`,
              ``,
              `Your response should either be 'no_repetition' or start with <fixed_story_summary_snippet> and give the repetition-free summary, and then end with </fixed_story_summary_snippet>. Feel free to just respond with 'no_repetition' if there's no repetition. Don't add any extra commentary.`,
            ].join("\n"),
            stopSequences: ["</fixed_story_summary_snippet>"],
          });
          if(result.stopReason === "error") continue;
          let fixedSummary = result.generatedText.match(/<fixed_story_summary_snippet>(.+)<\/fixed_story_summary_snippet>/s)?.[1].trim();
          if(fixedSummary) summary = fixedSummary;
        }
        
        // discard if story switched before committing summary
        if((window.currentStoryId ?? window.storyManager?.currentId) !== thisStoryId) {
          console.log("Discarding computed summary because story changed.");
          return;
        }

        console.log("----------------");
        console.log("----------------");

        console.log("𝗟𝗘𝗩𝗘𝗟:", summaryLevel);
        console.log("𝗜𝗡𝗦𝗧𝗥𝗨𝗖𝗧𝗜𝗢𝗡:", instruction);
        console.log("𝗦𝗧𝗔𝗥𝗧𝗪𝗜𝗧𝗛:", startWith);
        console.log("𝗦𝗨𝗠𝗠𝗔𝗥𝗬:", summary);
        console.log("----------------");
        console.log("----------------");
        console.log("----------------");
        
        window.summariesReadyToInjectByStoryId[thisStoryId].push({summarizedMessages:messagesToSummarizeFromFinalBlock, lastMessageSummarizedIndex, summary, level:summaryLevel});
      }
    } catch(e) {
      console.error(e);
    } finally {
      window.alreadyDoingSummaryByStoryId[thisStoryId] = false;
    }
  })();
