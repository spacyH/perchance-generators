<script>
  function addOpenInNewTab() {
// Get all <a> elements in the document
const links = document.querySelectorAll('a');
// console.log(links);
// Loop through each <a> element and add the target attribute
links.forEach(link => {
  link.setAttribute('target', '_blank');
});
  }
