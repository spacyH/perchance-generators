async startButtonClick() =>
  startBtn.textContent = "⏳ loading...";
  startBtn.disabled = true;
  let generatedText = "";
  let data = await ai({
    instruction: startInstruction.evaluateItem, 
    startWith: "This planet is called",
    onChunk: (data) => {
      generatedText += data.textChunk;
      outputEl.innerHTML = generatedText;
    },
    onFinish: (data) => {
      startBtn.disabled = false;
      startBtn.textContent = "🗺️ regenerate";
      generatedText = generatedText.replace(/((?:^|\n)[0-9]+\. [^\n]+:)\n/g, "$1 "); // in case it adds titles then *newline* then description
      let generatedLines = generatedText.trim().split("\n").map(l => l.trim());
      // the world intro should be the first line:
      worldIntro = generatedLines[0];
      // regions are lines that start with "<number>.":
      let regions = generatedLines.filter(l => /^[0-9]+\./.test(l)).map(l => l.split(".").slice(1).join(".").trim());
      outputEl.innerHTML = `
        <div id="mainCtn">
          <p style="margin:0;">${worldIntro}</p>
          <p style="font-weight:bold; font-size:80%; margin:0;">Click a region to explore it:</p>
          <ul style="margin-top:0;">${regions.map(p => `<li data-item-text="${p.replace(/"/g, '&quot;').replace(/</g, "&gt;")}" class="worldItem" onclick="generateSubItems(this); event.stopPropagation();"><span class="worldItemText">${p}</span></li>`).join("")}</ul>
        </div>`.trim();
    },
  });
  
async generateSubItems(li) =>
  if(li.querySelector(".loading-placeholder") || li.querySelector("ul")) return; // already loading/loaded
  
  // collect all ancestors of this item up to 'mainCtn'
  root.ancestorChainItemTexts = [li.dataset.itemText];
  let parent = li;
  while(true) {
    parent = parent.parentElement;
    if(parent === mainCtn) break;
    if(parent.classList.contains("worldItem")) {
      root.ancestorChainItemTexts.unshift(parent.dataset.itemText); // `.unshift(blah)` means "put 'blah' at the start of the list/array"
    }
  }
  
  // generated images for the clicked world item:
  if(generateImagesCheckbox.checked) {
    let imagesHtml = "";
    for(let i = 0; i < 3; i++) {
      imagesHtml += image({
        prompt: "award-winning art, trending on artstation, masterpiece digital art, "+root.ancestorChainItemTexts.at(-1),
        negativePrompt: "bad art, low quality, worst quality, photo",
        resolution: "768x512",
        style: "height:100%;",
      });
    }
    let imagesCtn = document.createElement("div");
    imagesCtn.classList.add("imageCtn");
    imagesCtn.style.cssText = "display:flex; overflow:auto; height:180px; scrollbar-width:none;";
    imagesCtn.innerHTML = imagesHtml;
    li.append(imagesCtn);
  }
  
  // we put the generated text of the sub-items into a temporary div to start with, just so the user can see it being written, then we'll transfer it into proper bullet points in the `onFinish` function, below.
  let tmp = document.createElement("div"); 
  tmp.classList.add("loading-placeholder");
  tmp.innerHTML = "<b>Loading...</b>";
  tmp.style.cssText = "padding-left:1rem;";
  li.append(tmp);
  
  // start the generation process:
  let generatedText = "";
  let data = await ai({
    instruction: subItemsInstruction.evaluateItem,
    startWith: "1.",
    onChunk: (data) => {
      generatedText += data.textChunk;
      tmp.innerHTML = generatedText;
    },
    onFinish: (data) => {
      let generatedLines = generatedText.trim().split("\n").map(l => l.trim());
      let regions = generatedLines.filter(l => /^[0-9]+\./.test(l)).map(l => l.split(".").slice(1).join(".").trim());
      tmp.outerHTML = `<ul>${regions.map(p => `<li data-item-text="${p.replace(/"/g, '&quot;').replace(/</g, "&gt;")}" class="worldItem" onclick="generateSubItems(this); event.stopPropagation();"><span class="worldItemText">${p}</span></li>`).join("")}</ul>`;
    },
  });
