async getEvents() =>
  // let data = await fetch("https://lemmy.world/api/v3/post/list?community_name=casual_perchance&page=1").then(r => r.json());
  // let response = await data.posts;
  console.log('Hub External Card - Fetching Events...')
  let page = 1;
  let prevNum = 0;
  let retry = 0;
  let stopLoop = false;
  let eventPosts = []
  while(!stopLoop) {
    let data = await fetch("https://lemmy.world/api/v3/post/list?community_name=casual_perchance&page="+page).then(r => r.json())
    let response = await data.posts;
    response.filter(a => /^\[Community Event\]/gi.test(a.post.name)).forEach(a => eventPosts.push(a));
    if (prevNum == eventPosts.getLength && retry >= 1) stopLoop = true;
    if (prevNum == eventPosts.getLength) { retry++ } else { retry = 0 };
    // console.log(retry);
    prevNum = eventPosts.getLength;
    page++;
  }
  console.log('Hub External Card - Fetched Events. Creating Cards...')
  let eventList = eventPosts
  // console.log(eventList);
  eventList = eventList.sort((a,b) => { return new Date(b.post.published) - new Date(a.post.published) });
  eventList = eventList.filter(a => /^\[Community Event\]/gi.test(a.post.name));
  
  eventList = eventList.sort((a,b) => { return new Date(Date.parse([...a.post.body.matchAll(/end = (.+)\n/gim)][0][1])) - new Date(Date.parse([...b.post.body.matchAll(/end = (.+)\n/gim)][0][1])) });
  for (let event of eventList) {
    try {
      if(!event.post.body.match(/(?:\`\`\`\s?perchance_data(.+)\`\`\`)/gims)) continue
      let text = [...event.post.body.matchAll(/(?:\`\`\`\s?perchance_data(.+)\`\`\`)/gims)][0][1].trim() //.replace(/^\/\/(.+)\n/gim,'')
      let data = createPerchanceTree(`event\n\t${text.split('\n').map(a => `  ${a}`).join('\n')}`).event
      // console.log(text,data);
      if (!data.metadata) continue
      let metadata = data.metadata
      let now = Date.now() / 86400000
      let start = new Date(data.metadata.start)
      let end = new Date(data.metadata.end)
      let status = Date.now() < start ? 'is starting!' : ((end - Date.now())/(1000 * 60 * 60 * 24)) >= 18 ? 'has started!' : ((end - Date.now())/(1000 * 60 * 60 * 24)) >= 7 ? 'is ongoing!' : Date.now() <= end ? 'is about to end!' : 'has ended!'
      let end_date = new Date(Date.parse(metadata.end)) / 86400000
      if (status.includes('ended') && (end_date - now <= -7)) continue
      root[`${data.metadata.title}`] = data
      root[`${data.metadata.title}`].lemmy = event
      let bg = metadata.color.replaceAll(';', '')
      let isRepeating = bg.includes('repeating')
      let cardDetails = `
      <i style="font-size: 110%;">${metadata.type}</i>
      <div style="font-size: 125%;"><a style="text-underline-position: under;" href="/hub?page=events&event=${metadata.title}" target="_blank">${metadata.title}</a></div>
      <i style="font-size: 85%;">
      ${status}
      </i>
      [document.documentElement.style.setProperty('--light-gradient', '${isRepeating ? 'repeating-': ''}linear-gradient(Canvas,Canvas), ${bg}'), '']
      [document.documentElement.style.setProperty('--dark-gradient', '${isRepeating ? 'repeating-': ''}linear-gradient(CanvasText,CanvasText), ${bg}'), '']
      `
      window.events.push(cardDetails);
    } catch (err) {
      console.log(err)
    }
  }
  if (window.events.length == 0) {
    window.events.push('<div style="font-size: 125%;">No Current Events</div><i style="font-size: 85%;">Host an Event at <a href="https://lemmy.world/c/casual_perchance" target="_blank">Casual Perchance</a></i>')
  }
  if (window.events.length > 1 ) buttonRandomizer.style.display = ''
  if (!window.otherEventsScriptAdded) {
    let script = document.createElement('script');
    script.innerHTML = `
    window.oldselect = ''
    window.newselect = ''
    function norepeat() {
      while (window.newselect == window.oldselect) {
        window.newselect = window.events.selectOne
      }
      window.oldselect = window.newselect
      return window.newselect
    }
    function otherEvents() {
      let newEvent = norepeat();
      console.log('Hub External Card - Randomizing Card...')
      eventCardExternal.innerHTML = newEvent.evaluateItem
    }
    `
    document.head.append(script);
    window.otherEventsScriptAdded = true;
  }
  eventCardExternal.innerHTML = window.events.selectOne.evaluateItem
  console.log('Hub External Card - Finished Loading!')
