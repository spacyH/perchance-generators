getPluginObj() =>
  if(window.__postsPluginObj18474921) return window.__postsPluginObj18474921;
  
  let iframe = document.createElement("iframe");
  iframe.src = "https://posts-plugin.perchance.org/embed";
  iframe.style.display = "none";
  document.head.append(iframe);
  
  let requestIdToResolver = {};
  iframe.contentWindow.addEventListener("message", function(e) {
    if(requestIdToResolver[e.data.requestId]) {
      requestIdToResolver[e.data.requestId](e.data.response);
      delete requestIdToResolver[e.data.requestId]();
    }
  });
  
  window.__postsPluginObj18474921 = {
    add: async function(postObj) {
      let requestId = Math.random().toString()+Math.random().toString();
      iframe.contentWindow.postMessage({requestId, type:"add", params:{postObj}});
      return new Promise(r => requestIdToResolver[requestId]=r);
    },
    addMany: async function(postObjs) {
      let requestId = Math.random().toString()+Math.random().toString();
      iframe.contentWindow.postMessage({requestId, type:"addMany", params:{postObjs}});
      return new Promise(r => requestIdToResolver[requestId]=r);
    },
    get: async function(id) {
      let requestId = Math.random().toString()+Math.random().toString();
      iframe.contentWindow.postMessage({requestId, type:"get", params:{id}});
      return new Promise(r => requestIdToResolver[requestId]=r);
    },
    getMany: async function(ids) {
      let requestId = Math.random().toString()+Math.random().toString();
      iframe.contentWindow.postMessage({requestId, type:"getMany", params:{ids}});
      return new Promise(r => requestIdToResolver[requestId]=r);
    },
    query: async function(query) {
      let requestId = Math.random().toString()+Math.random().toString();
      iframe.contentWindow.postMessage({requestId, type:"query", params:{query}});
      return new Promise(r => requestIdToResolver[requestId]=r);
    },
    stream: async function(query) {
      let requestId = Math.random().toString()+Math.random().toString();
      iframe.contentWindow.postMessage({requestId, type:"stream", params:{query}});
      let promise = new Promise(r => requestIdToResolver[requestId]=r);
      promise.toString = function() {
        // TODO: add this as a property like '.html'? [$once:posts.stream({channel:'blog'}).html]
        return ""; // TODO: use renderPost/renderFeed/etc. in their config, or sane defaults.
      };
      return promise;
    },
  };
  
  return window.__postsPluginObj18474921;
