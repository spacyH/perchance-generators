        function makeChildrenReorderableViaDrag(container, { childSelector='*', onEnd, onStart, excludeChildren=[] } = {}) {
          let draggedEl = null;
          let ghostEl = null;
          let started = false;
          let isTouchDevice = window.matchMedia("(pointer: coarse)").matches;

          function createGhost(el) {
            const ghost = el.cloneNode(true);
            const styles = window.getComputedStyle(el);
            const rect = el.getBoundingClientRect();
            Object.assign(ghost.style, { position: 'fixed', pointerEvents: 'none',  opacity: '0.7', zIndex: '1000', width: `${rect.width}px`, height: `${rect.height}px`, margin: '0', fontSize: styles.getPropertyValue("font-size") });
            return ghost;
          }

          function handleMove(x, y) {
            if(!draggedEl || !ghostEl) return;
            ghostEl.style.left = `${x - ghostEl.getBoundingClientRect().width / 2}px`;
            ghostEl.style.top = `${y - ghostEl.getBoundingClientRect().height / 2}px`;

            const children = [...container.querySelectorAll(`:scope > ${childSelector}`)].filter(child => !excludeChildren.includes(child));

            if(children.length === 0) {
              container.appendChild(draggedEl);
              return;
            }

            // Find the closest child to the pointer
            let closestChild = children[0];
            let minDistance = Infinity;

            children.forEach(child => {
              const rect = child.getBoundingClientRect();
              const centerX = rect.left + rect.width / 2;
              const centerY = rect.top + rect.height / 2;
              const distance = Math.hypot(x - centerX, y - centerY);

              if (distance < minDistance) {
                minDistance = distance;
                closestChild = child;
              }
            });

            // Determine whether to insert before or after the closest child
            const rect = closestChild.getBoundingClientRect();
            const centerX = rect.left + rect.width / 2;

            if (x < centerX) {
              // Only move if it's not already in the correct position
              if (closestChild !== draggedEl && closestChild.previousElementSibling !== draggedEl) {
                container.insertBefore(draggedEl, closestChild);
              }
            } else {
              // Only move if it's not already in the correct position
              if (closestChild.nextElementSibling !== draggedEl) {
                container.insertBefore(draggedEl, closestChild.nextElementSibling);
              }
            }
          }

          let draggedElOriginalOpacity;
          function cleanup(timer) {
            if(timer) clearTimeout(timer);
            if(ghostEl) ghostEl.remove();
            if(draggedEl) draggedEl.style.opacity = draggedElOriginalOpacity;
            draggedEl = ghostEl = null;
            if(started) onEnd?.();
            started = false;
          }

          function initDrag(event) {
            let isTouch = event.type.includes('touch');
            let point = isTouch ? event.touches[0] : event;
            let target = event.target.closest(childSelector);
            if(!target || !container.contains(target) || excludeChildren.includes(target)) return;
            let startPos = { x: point.clientX, y: point.clientY };
            let startDelayMs = 200;

            function startDragging() {
              draggedEl = target;
              if(!started) { onStart?.(); started = true; }
              
              draggedElOriginalOpacity = draggedEl.style.opacity;
              draggedEl.style.opacity = '0.5';
              
              ghostEl = createGhost(draggedEl);
              document.body.appendChild(ghostEl);
              handleMove(point.clientX, point.clientY);
            }

            let timer = setTimeout(startDragging, startDelayMs);
            
            let moveHandlerCount = 0;
            function moveHandler(e) {
              e.preventDefault();
              let p = isTouch ? e.touches[0] : e;
              // in case they were scrolling and happened to hit the button in the process, we ignore if their pointer moves away too fast before the startDelayMs timeout:
              if(isTouch && !draggedEl && Math.hypot(p.clientX - startPos.x, p.clientY - startPos.y) > 30) {
                cleanup(timer);
                return;
              }
              if(moveHandlerCount % 2 === 0) handleMove(p.clientX, p.clientY);
              moveHandlerCount++;
            }

            function endHandler() {
              document.removeEventListener(isTouch ? 'touchmove' : 'mousemove', moveHandler);
              cleanup(timer);
            }
            document.addEventListener(isTouch ? 'touchmove' : 'mousemove', moveHandler, isTouch ? { passive: false } : undefined);
            document.addEventListener(isTouch ? 'touchend' : 'mouseup', endHandler, { once: true });
          }
          
          container.addEventListener('touchstart', initDrag);
          container.addEventListener('mousedown', initDrag);
        }
      </script>

  function createCrossfadePseudoVideoElement(opts={}) {
    const container = document.createElement('div');
    const subContainer = document.createElement('div');
    const video1 = document.createElement('video');
    const video2 = document.createElement('video');

    const fadeDuration = opts.fadeDuration ?? 1000;

    // Sub-container is so we can have relative positioning without affecting top-level container userland styles.
    container.append(subContainer);
    subContainer.style.cssText = "position:relative; width:100%; height:100%;";

    [video1, video2].forEach(video => {
      video.style.cssText = `pointer-events:none; position:absolute; top:0; left:0; width:100%; height:100%; object-fit:cover; mix-blend-mode:plus-lighter; transition:opacity ${fadeDuration}ms ease-in-out;`;
      video.autoplay = true;
      // not sure why setAttribute is needed for these (unlike autoplay):
      video.setAttribute("muted", "");
      video.setAttribute("playsinline", ""); // prevent safari from opening it in fullscreen
      video.setAttribute("disablepictureinpicture", "");
      subContainer.appendChild(video);
    });

    video2.style.opacity = '0';

    let activeVideo = video1;
    let waitingVideo = video2;
    let isTransitioning = false;
    let currentSrc = "";

    function switchVideos() {
      if (isTransitioning) return;
      isTransitioning = true;

      waitingVideo.currentTime = 0;
      waitingVideo.play();

      waitingVideo.style.opacity = '1';
      activeVideo.style.opacity = '0';

      let prevActiveVideo = activeVideo;
      setTimeout(() => {
        prevActiveVideo.pause();
        prevActiveVideo.currentTime = 0;
        isTransitioning = false;
      }, fadeDuration);

      activeVideo = waitingVideo;
      waitingVideo = prevActiveVideo;
    }

    function onTimeUpdate() {
      const timeLeft = activeVideo.duration - activeVideo.currentTime;
      if (timeLeft <= (fadeDuration / 1000)) {
        let prevActiveVideo = activeVideo;
        prevActiveVideo.removeEventListener('timeupdate', onTimeUpdate);
        switchVideos(); // caution: this function swaps active/waiting variables
        activeVideo.addEventListener('timeupdate', onTimeUpdate);
      }
    }

    // Add video-like properties and methods to container
    Object.defineProperties(container, {
      src: {
        get: () => currentSrc,
        set: (newSrc) => {
          currentSrc = newSrc;
          video1.src = video2.src = newSrc;
        },
      },
      paused: {
        get: () => activeVideo.paused,
      },
      currentTime: {
        get: () => activeVideo.currentTime,
        set: (time) => {
          activeVideo.currentTime = time;
          waitingVideo.currentTime = time;
        },
      },
      duration: {
        get: () => activeVideo.duration,
      },
    });

    container.play = () => {
      activeVideo.addEventListener('timeupdate', onTimeUpdate);
      return activeVideo.play();
    };
    container.pause = () => {
      video1.removeEventListener('timeupdate', onTimeUpdate);
      video2.removeEventListener('timeupdate', onTimeUpdate);
      video1.pause();
      video2.pause();
    };
    return container;
  }

  function toggleManualDarkMode() {
    let newColorScheme = (getCurrentColorScheme() === "dark" ? "light" : "dark");
    localStorage.forceColorScheme = newColorScheme;

    }
