<script>
(function(){
'use strict';
var pipe = null;

function el(id){return document.getElementById(id);}
function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');}

function renderPage(){
  document.title = 'Image Pipeline \u2014 Examples';
  document.getElementById('page').innerHTML =
    '<header>' +
      '<div class="breadcrumb"><a href="https://perchance.org/custom-image-pipeline-plugin" target="_blank">\u2190 custom-image-pipeline-plugin</a></div>' +
      '<h1>Examples</h1>' +
      '<p class="lead">Promise-based image generation, style presets, and batching.</p>' +
    '</header>' +

    // 01 Style explorer
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">01</span><h2>Style explorer</h2></div>' +
      '<p>Apply 4 different style presets to the same prompt. Generates sequentially to avoid clogging the queue.</p>' +
      '<input type="text" id="style-prompt" value="a forest temple at sunset">' +
      '<div class="row" style="margin-top:8px;flex-wrap:wrap">' +
        '<select id="style-a"></select>' +
        '<select id="style-b"></select>' +
        '<select id="style-c"></select>' +
        '<select id="style-d"></select>' +
      '</div>' +
      '<div class="row">' +
        '<button id="style-go" class="primary">Generate 4 styles</button>' +
      '</div>' +
      '<div class="metric-row" style="grid-template-columns:repeat(4,1fr)" id="style-grid"></div>' +
    '</section>' +

    // 02 Batch with seed variation
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">02</span><h2>Batch variation</h2></div>' +
      '<p>Generate N variations of the same prompt with sequentially varied seeds.</p>' +
      '<input type="text" id="batch-prompt" value="a cyberpunk cat with neon goggles">' +
      '<div class="row" style="margin-top:8px">' +
        '<select id="batch-style"></select>' +
        '<select id="batch-res"></select>' +
        '<input type="number" id="batch-count" value="3" min="1" max="6" style="flex:0 0 80px">' +
        '<button id="batch-go" class="primary">Generate</button>' +
      '</div>' +
      '<div id="batch-progress" class="stat"></div>' +
      '<div class="metric-row" id="batch-grid" style="grid-template-columns:repeat(3,1fr)"></div>' +
    '</section>' +

    // 03 Promise API
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">03</span><h2>Async / await API</h2></div>' +
      '<p>Single-image generation that returns a Promise. Shows the resolved data object.</p>' +
      '<input type="text" id="async-prompt" value="a glowing mushroom in a misty forest">' +
      '<div class="row" style="margin-top:8px">' +
        '<button id="async-go" class="primary">await pipe.generate(...)</button>' +
      '</div>' +
      '<div class="metric-row" style="grid-template-columns:1fr 1fr"><div id="async-img"></div><div><div class="label" style="font:10px var(--mono);color:var(--muted);letter-spacing:1px;text-transform:uppercase;margin-bottom:6px">Resolved data</div><pre id="async-data" style="font-size:11px;max-height:300px;overflow:auto">(click button)</pre></div></div>' +
    '</section>';

  // Populate style dropdowns
  var names = pipe.styleNames();
  var styleOpts = names.map(function(n){ return '<option value="'+n+'">'+n+'</option>'; }).join('');
  ['style-a','style-b','style-c','style-d','batch-style'].forEach(function(id, i){
    el(id).innerHTML = '<option value="">(no style)</option>' + styleOpts;
    if (id === 'style-a') el(id).value = 'photo';
    if (id === 'style-b') el(id).value = 'cinematic';
    if (id === 'style-c') el(id).value = 'anime';
    if (id === 'style-d') el(id).value = 'pixelart';
    if (id === 'batch-style') el(id).value = 'painting';
  });
  
  // Populate resolution dropdown
  el('batch-res').innerHTML = Object.keys(pipe.resolutions).map(function(k){
    return '<option value="'+pipe.resolutions[k]+'">'+k+' ('+pipe.resolutions[k]+')</option>';
  }).join('');
  el('batch-res').value = pipe.resolutions.square;

  // Wire events
  el('style-go').onclick = generateStyles;
  el('batch-go').onclick = generateBatch;
  el('async-go').onclick = generateAsync;
}

async function generateStyles(){
  var prompt = el('style-prompt').value.trim();
  if (!prompt) return;
  var btn = el('style-go'); btn.disabled = true; btn.textContent = 'Generating...';
  var styles = [el('style-a').value, el('style-b').value, el('style-c').value, el('style-d').value];
  
  var grid = el('style-grid');
  grid.innerHTML = styles.map(function(s, i){
    return '<div class="metric"><div class="label">'+esc(s||'no style')+'</div><div id="sg-'+i+'" style="aspect-ratio:1;background:var(--bg);border-radius:4px;display:flex;align-items:center;justify-content:center;color:var(--muted);font-size:11px">queued</div></div>';
  }).join('');
  
  for (var i = 0; i < 4; i++) {
    el('sg-'+i).textContent = 'generating...';
    btn.textContent = 'Generating ' + (i+1) + ' / 4...';
    var fullPrompt = styles[i] ? pipe.withStyle(prompt, styles[i]) : prompt;
    try {
      await pipe.generate({ prompt: fullPrompt, resolution: '512x512', into: '#sg-'+i });
    } catch (e) {
      el('sg-'+i).innerHTML = '<span class="err">' + esc(e.message) + '</span>';
    }
  }
  btn.disabled = false; btn.textContent = 'Generate 4 styles';
}

async function generateBatch(){
  var prompt = el('batch-prompt').value.trim();
  var style = el('batch-style').value;
  var res = el('batch-res').value;
  var count = parseInt(el('batch-count').value) || 3;
  if (!prompt) return;
  
  var btn = el('batch-go'); btn.disabled = true; btn.textContent = 'Generating...';
  var fullPrompt = style ? pipe.withStyle(prompt, style) : prompt;
  
  var grid = el('batch-grid');
  grid.style.gridTemplateColumns = 'repeat(' + Math.min(count, 4) + ', 1fr)';
  grid.innerHTML = '';
  for (var i = 0; i < count; i++) {
    grid.innerHTML += '<div class="metric"><div class="label">image ' + (i+1) + '</div><div id="bg-'+i+'" style="aspect-ratio:1;background:var(--bg);border-radius:4px;display:flex;align-items:center;justify-content:center;color:var(--muted);font-size:11px">queued</div></div>';
  }
  
  var startSeed = Math.floor(Math.random() * 1000000);
  for (var j = 0; j < count; j++) {
    el('bg-'+j).textContent = 'generating...';
    btn.textContent = 'Generating ' + (j+1) + ' / ' + count + '...';
    el('batch-progress').textContent = 'Seed: ' + (startSeed + j);
    try {
      await pipe.generate({ prompt: fullPrompt, seed: startSeed + j, resolution: res, into: '#bg-'+j });
    } catch (e) {
      el('bg-'+j).innerHTML = '<span class="err">' + esc(e.message) + '</span>';
    }
  }
  el('batch-progress').textContent = 'Complete. ' + count + ' images generated.';
  btn.disabled = false; btn.textContent = 'Generate';
}

async function generateAsync(){
  var prompt = el('async-prompt').value.trim();
  if (!prompt) return;
  var btn = el('async-go'); btn.disabled = true; btn.textContent = 'Generating...';
  el('async-img').innerHTML = '<div style="aspect-ratio:1;background:var(--bg);border-radius:4px;display:flex;align-items:center;justify-content:center;color:var(--muted)">generating...</div>';
  el('async-data').textContent = 'Awaiting...';
  
  try {
    var data = await pipe.generate({ prompt: prompt, into: '#async-img' });
    el('async-data').textContent = JSON.stringify(data, null, 2);
  } catch (e) {
    el('async-data').textContent = 'Error: ' + e.message;
  }
  btn.disabled = false; btn.textContent = 'await pipe.generate(...)';
}

function init(){
  // Show a "loading" placeholder while we wait for the plugin
  document.getElementById('page').innerHTML = '<p style="color:#7d8590;padding:40px 0;font:13px monospace">Loading plugin\u2026</p>';
  
  // Poll for the plugin to become available. The plugin is imported via
  // pipeline = '+String.fromCharCode(123)+'import:<slug>'+String.fromCharCode(125)+' in the lists editor.
  // Loading can take 2-5 seconds. We try every 300ms for up to 15s.
  var attempts = 0;
  var maxAttempts = 50;
  
  function attempt() {
    attempts++;
    
    // Has root.pipeline appeared yet?
    if (typeof root === 'undefined' || typeof root.pipeline !== 'function') {
      if (attempts < maxAttempts) {
        setTimeout(attempt, 300);
        return;
      }
      // Gave up
      showError('Plugin not imported. Add to your lists editor: <br><br><code style="background:var(--faint);padding:6px 10px;border-radius:4px;display:inline-block">pipeline = '+String.fromCharCode(123)+'import:custom-image-pipeline-plugin'+String.fromCharCode(125)+'</code><br><br>Replace <code>custom-image-pipeline-plugin</code> with the actual URL slug of your deployed plugin if different.');
      return;
    }
    
    // Invoke the plugin's $output to build & stash the API.
    // This is the missing piece \u2014 $output doesn't auto-run on import.
    try {
      var api = root.pipeline();
      // Prefer the window-stashed reference (set inside $output);
      // fall back to the direct return value if the stash didn't happen.
      pipe = window.__imagePipeline || api;
    } catch (e) {
      showError('Plugin imported but failed to initialize: ' + (e && e.message ? e.message : String(e)));
      return;
    }
    
    if (!pipe) {
      showError('Plugin returned no API.');
      return;
    }
    
    renderPage();
    
  }
  
  function showError(msg) {
    document.getElementById('page').innerHTML =
      '<header><h1 style="font-size:22px;font-weight:700;margin-bottom:12px">Plugin not ready</h1>' +
      '<p style="color:var(--muted);font-size:14px">' + msg + '</p>' +
      '<p style="color:var(--muted);font-size:13px;margin-top:18px">Or visit the <a href="https://perchance.org/custom-image-pipeline-plugin" target="_blank">plugin docs page</a> for installation help.</p></header>';
  }
  
  attempt();
}
if(document.readyState==='loading'){document.addEventListener('DOMContentLoaded',function(){setTimeout(init,2000);});}
else{setTimeout(init,2000);}
})();

</script>
