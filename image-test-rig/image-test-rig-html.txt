function copy(text, btn) {
  var t = btn.textContent;
  (navigator.clipboard ? navigator.clipboard.writeText(text) : Promise.reject()).catch(function() {

// \u2500\u2500 GENERATE (SEQUENTIAL) \u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500
// We fire one at a time so the queue doesn't stack up. Each call awaits
// the plugin's onFinish callback before starting the next cell.
var generationRunId = 0; // bumped to cancel in-flight runs

async function generateAll() {
  var myRunId = ++generationRunId;
  // Capture latest prompt edits
  for (var i = 0; i < 4; i++) state.prompts[i] = el('prompt-' + i).value;
  // Mark all cells as queued
  for (var j = 0; j < 4; j++) {
    el('img-' + j).innerHTML = '<div class="empty">Queued\u2026</div>';
  }
  el('ctl-go').disabled = true;
  var origText = el('ctl-go').textContent;
  try {
    for (var k = 0; k < 4; k++) {
      if (myRunId !== generationRunId) return; // newer run started, abandon
      el('ctl-go').textContent = 'Generating ' + (k + 1) + ' / 4\u2026';
      await generateCell(k);
    }
  } finally {
    if (myRunId === generationRunId) {
      el('ctl-go').disabled = false;
      el('ctl-go').textContent = origText;
    }
  }
}

function generateCell(i) {
  return new Promise(function(resolve) {
    var t = currentTest();
    var variation = t.variations[i];
    var prompt = el('prompt-' + i).value;

    if (!prompt.trim()) {
      el('img-' + i).innerHTML = '<div class="empty err">Empty prompt</div>';
      resolve(); return;
    }

    var opts = {
      prompt: prompt,
      seed: variation.seedOverride !== undefined ? variation.seedOverride : state.seed,
      resolution: variation.resOverride !== undefined ? variation.resOverride : state.resolution,
      guidanceScale: state.guidanceScale,
      onFinish: function(data) {
        if (!settled) { settled = true; resolve(data); }
      }
    };
    if (variation.neg) opts.negativePrompt = variation.neg;

    var settled = false;
    // Safety timeout \u2014 if onFinish never fires, don't block the queue forever
    var timeoutMs = 90000;
    setTimeout(function() {
      if (!settled) { settled = true; resolve(); }
    }, timeoutMs);

    el('img-' + i).innerHTML = '<div class="loading">Generating\u2026<br><span class="loading-detail">seed ' + opts.seed + '</span></div>';

    try {
      var iframeHtml = String(root.image(opts));
      el('img-' + i).innerHTML = iframeHtml;
    } catch (e) {
      el('img-' + i).innerHTML = '<div class="empty err">' + esc(e.message) + '</div>';
      if (!settled) { settled = true; resolve(); }
    }
  });
}

// Single-cell regenerate (skips the queue logic \u2014 only one request anyway)
function regenerateOne(i) {
  state.prompts[i] = el('prompt-' + i).value;
  generateCell(i);
}
