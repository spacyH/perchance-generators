(function(){
'use strict';
var tok = null;

function el(id){return document.getElementById(id);}
function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');}

function renderPage(){
  document.title = 'Token Budget \u2014 Examples';
  document.getElementById('page').innerHTML =
    '<header>' +
      '<div class="breadcrumb"><a href="https://perchance.org/accurate-token-budget-plugin" target="_blank">\u2190 accurate-token-budget-plugin</a></div>' +
      '<h1>Examples</h1>' +
      '<p class="lead">Live token counting using the DeepSeek\u2011R1 tokenizer.</p>' +
      '<div id="status" class="stat">Tokenizer status: <span class="val" id="status-val">not loaded</span></div>' +
    '</header>' +

    // 01 Live counter
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">01</span><h2>Live token counter</h2></div>' +
      '<p>Type into the box. The exact token count updates after each keystroke (with the DeepSeek\u2011R1 tokenizer).</p>' +
      '<textarea id="live-input" rows="6" placeholder="Paste any text here\u2026">The quick brown fox jumps over the lazy dog. This sentence has exactly enough common English words to demonstrate how tokenization works.</textarea>' +
      '<div class="metric-row" style="margin-top:10px">' +
        '<div class="metric"><div class="label">Characters</div><div class="value" id="m-chars">0</div></div>' +
        '<div class="metric"><div class="label">Estimate (chars/4)</div><div class="value dim" id="m-est">0</div></div>' +
        '<div class="metric"><div class="label">Exact tokens</div><div class="value" id="m-exact">\u2014</div></div>' +

      '</div>' +
    '</section>' +

    // 02 Budget check
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">02</span><h2>Budget check</h2></div>' +
      '<p>Set a token budget and see whether the text above fits.</p>' +
      '<div class="row">' +
        '<input type="number" id="budget-max" value="100" min="1" max="100000" style="flex:0 0 140px">' +
        '<span class="stat" style="line-height:32px">max tokens</span>' +
      '</div>' +
      '<div id="budget-result" class="stat" style="margin-top:8px">Enter text above and a budget here.</div>' +
    '</section>' +

    // 03 Truncate
    '<section class="demo">' +
      '<div class="demo-meta"><span class="n">03</span><h2>Truncate to fit</h2></div>' +
      '<p>Trim the text above to fit within a budget. Choose start- or end-preserving.</p>' +
      '<div class="row">' +
        '<input type="number" id="trunc-max" value="20" min="1" max="10000" style="flex:0 0 140px">' +
        '<button id="trunc-start">Keep start</button>' +
        '<button id="trunc-end">Keep end</button>' +
      '</div>' +
      '<pre id="trunc-result" style="white-space:pre-wrap;word-break:break-word">(click a button)</pre>' +
    '</section>' +

    // 04 Chunk
    '<section class="demo">' +

      '<p>Split the text above into N-token chunks for RAG-style processing.</p>' +
      '<div class="row">' +
        '<input type="number" id="chunk-size" value="15" min="1" max="2000" style="flex:0 0 140px">' +

      '</div>' +

    // 05 Chat budget

        '<button id="msg-add">Add message</button>' +
        '<button id="msg-clear">Clear all</button>' +
      '</div>' +
      '<div class="metric-row" style="margin-top:10px">' +
        '<div class="metric"><div class="label">Messages</div><div class="value" id="msg-count">0</div></div>' +
        '<div class="metric"><div class="label">Total tokens</div><div class="value" id="msg-tokens">0</div></div>' +

  // Wire events
  el('live-input').oninput = updateLive;
  el('budget-max').oninput = updateBudget;
  el('trunc-start').onclick = function(){ doTruncate('start'); };
  el('trunc-end').onclick = function(){ doTruncate('end'); };
  el('chunk-btn').onclick = doChunk;
  el('msg-input').onkeydown = function(e){ if(e.key==='Enter'&&!e.shiftKey){e.preventDefault();addMessage();} };

  el('msg-clear').onclick = clearMessages;

  updateLive();

async function updateLive(){
  var text = el('live-input').value;
  el('m-chars').textContent = text.length.toLocaleString();
  el('m-est').textContent = tok.estimate(text).toLocaleString();
  el('m-words').textContent = (text.match(/\S+/g) || []).length.toLocaleString();
  if (tok.isLoaded()) {
    el('m-exact').textContent = tok.countSync(text).toLocaleString();
  } else {
    el('m-exact').textContent = 'loading\u2026';
    try {
      var n = await tok.count(text);
      el('m-exact').textContent = n.toLocaleString();
      updateStatus();
    } catch (e) {
      el('m-exact').textContent = 'error';
    }
  }
  updateBudget();
}

async function updateBudget(){
  var text = el('live-input').value;
  var max = parseInt(el('budget-max').value) || 100;
  try {
    var b = await tok.budget(text, max);
    var cls = b.percent < 80 ? 'ok' : b.percent < 100 ? 'warn' : 'err';
    el('budget-result').innerHTML =
      '<span class="val">' + b.used + '</span> / ' + b.max + ' tokens used (' +
      '<span class="' + cls + '">' + b.percent + '%</span>). ' +
      'Remaining: <span class="val">' + b.remaining + '</span>. ' +
      'Fits: <span class="' + (b.fits ? 'ok' : 'err') + '">' + (b.fits ? 'yes' : 'no') + '</span>.';
  } catch (e) {
    el('budget-result').textContent = 'Error: ' + e.message;
  }
}

async function doTruncate(which){
  var text = el('live-input').value;
  var max = parseInt(el('trunc-max').value) || 20;
  el('trunc-result').textContent = 'Working\u2026';
  try {
    var result = which === 'start' ? await tok.truncate(text, max) : await tok.truncateStart(text, max);
    var n = await tok.count(result);
    el('trunc-result').textContent = '[' + n + ' tokens] ' + result;
  } catch (e) {
    el('trunc-result').textContent = 'Error: ' + e.message;
  }
}

async function doChunk(){
  var text = el('live-input').value;
  var size = parseInt(el('chunk-size').value) || 15;
  el('chunk-result').innerHTML = '<p class="muted">Working\u2026</p>';
  try {
    var chunks = await tok.chunk(text, size);
    var html = '<p class="muted">' + chunks.length + ' chunks of up to ' + size + ' tokens each:</p>';
    for (var i = 0; i < chunks.length; i++) {
      var n = await tok.count(chunks[i]);
      html += '<div class="metric" style="margin-bottom:6px">' +
        '<div class="label">Chunk ' + (i+1) + ' (' + n + ' tokens)</div>' +
        '<div style="font-family:var(--mono);font-size:12px;color:var(--ink);margin-top:4px;word-break:break-word">' + esc(chunks[i]) + '</div>' +
      '</div>';
    }
    el('chunk-result').innerHTML = html;
  } catch (e) {
    el('chunk-result').innerHTML = '<p class="err">' + e.message + '</p>';
  }
}

var messages = [];
async function addMessage(){
  var text = el('msg-input').value.trim();
  if (!text) return;
  messages.push({ role: messages.length % 2 ? 'assistant' : 'user', content: text });
  el('msg-input').value = '';
  await renderMessages();
}
function clearMessages(){
  messages = [];
  renderMessages();
}
async function renderMessages(){
  var html = '';
  for (var i = 0; i < messages.length; i++) {
    var n = await tok.count(messages[i].content);
    html += '<div class="metric" style="margin-bottom:6px">' +
      '<div class="label">' + esc(messages[i].role) + ' \u2014 ' + n + ' tokens</div>' +
      '<div style="font-size:13px;color:var(--ink);margin-top:4px">' + esc(messages[i].content) + '</div>' +
    '</div>';
  }
  el('msg-list').innerHTML = html;
  el('msg-count').textContent = messages.length;
  if (messages.length === 0) {
    el('msg-tokens').textContent = '0';
    el('msg-status').textContent = 'empty';
    el('msg-status').className = 'value dim';
    return;
  }
  var total = await tok.countMessages(messages);
  el('msg-tokens').textContent = total.toLocaleString();
  var pct = Math.round((total / 4096) * 100);
  el('msg-status').textContent = pct + '%';
  el('msg-status').className = 'value ' + (pct < 80 ? 'ok' : pct < 100 ? 'warn' : 'err');
}

function updateStatus(){
  var info = tok.info();
  if (info.loaded) {
    el('status-val').innerHTML = '<span class="ok">loaded</span> (vocab ' + (info.vocabSize || '?') + ')';
  } else if (info.error) {
    el('status-val').innerHTML = '<span class="err">error: ' + esc(info.error) + '</span>';
  } else {
    el('status-val').innerHTML = '<span class="warn">loading\u2026 (~7MB on first visit)</span>';
  }
}

function init(){
  // Show a "loading" placeholder while we wait for the plugin
  document.getElementById('page').innerHTML = '<p style="color:#7d8590;padding:40px 0;font:13px monospace">Loading plugin\u2026</p>';
  
  // Poll for the plugin to become available. The plugin is imported via
  // tokens = '+String.fromCharCode(123)+'import:<slug>'+String.fromCharCode(125)+' in the lists editor.
  // Loading can take 2-5 seconds. We try every 300ms for up to 15s.
  var attempts = 0;
  var maxAttempts = 50;
  
  function attempt() {
    attempts++;
    
    // Has root.tokens appeared yet?
    if (typeof root === 'undefined' || typeof root.tokens !== 'function') {
      if (attempts < maxAttempts) {
        setTimeout(attempt, 300);
        return;
      }
      // Gave up
      showError('Plugin not imported. Add to your lists editor: <br><br><code style="background:var(--faint);padding:6px 10px;border-radius:4px;display:inline-block">tokens = '+String.fromCharCode(123)+'import:accurate-token-budget-plugin'+String.fromCharCode(125)+'</code><br><br>Replace <code>accurate-token-budget-plugin</code> with the actual URL slug of your deployed plugin if different.');
      return;
    }
    
    // Invoke the plugin's $output to build & stash the API.
    // This is the missing piece \u2014 $output doesn't auto-run on import.
    try {
      var api = root.tokens();
      // Prefer the window-stashed reference (set inside $output);
      // fall back to the direct return value if the stash didn't happen.
      tok = window.__tokenBudget || api;
    } catch (e) {
      showError('Plugin imported but failed to initialize: ' + (e && e.message ? e.message : String(e)));
      return;
    }
    
    if (!tok) {
      showError('Plugin returned no API.');
      return;
    }
    
    renderPage();
    updateStatus(); tok.preload().catch(function(){}).finally(updateStatus);
  }
  
  function showError(msg) {
    document.getElementById('page').innerHTML =
      '<header><h1 style="font-size:22px;font-weight:700;margin-bottom:12px">Plugin not ready</h1>' +
      '<p style="color:var(--muted);font-size:14px">' + msg + '</p>' +
      '<p style="color:var(--muted);font-size:13px;margin-top:18px">Or visit the <a href="https://perchance.org/accurate-token-budget-plugin" target="_blank">plugin docs page</a> for installation help.</p></header>';
  }
  
  attempt();
}
if(document.readyState==='loading'){document.addEventListener('DOMContentLoaded',function(){setTimeout(init,1500);});}
else{setTimeout(init,1500);}
})();
